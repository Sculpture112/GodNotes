#include <bits/stdc++.h>

using namespace std;

long long memo[1005];

bool visited[1005];

  

long long dfs(int k)

{

    if (k == 0 || k == 1)

        return 1;

  

    if (visited[k])

        return memo[k];

    memo[k] = dfs(k - 1) + dfs(k - 2);

    visited[k] = true;

  

    return memo[k];

}

  

void solve()

{

    int m, n;

    cin >> m >> n;

  

    int distance = n - m;

  

    if (distance <= 0)

    {

        if (distance = 0)

            cout << 1 << endl;

        else

            cout << 0 << endl;

    }

    else

    {

        cout << dfs(distance) << endl;

    }

}

signed main()

{

    ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);

    solve();

    return 0;

}
