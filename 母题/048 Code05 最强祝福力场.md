// 048 Code05 最强祝福力场

// 小扣在探索丛林的过程中，将所有力场的分布都记录了下来
// forceField[i] = [x, y, side] 表示以 (x, y) 为中心，边长为 side 的正方形区域。
// 求出在这片地带中 力场强度 最强处的 力场强度（覆盖该点的力场数量）。
// 测试链接 : https://leetcode.cn/problems/xepqZ5/

#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    // 时间复杂度O(n^2)，额外空间复杂度O(n^2)，n是力场的个数
    int fieldOfGreatestBlessing(vector<vector<int>>& fields) {
        int n = fields.size();
        // 用 long long 存储坐标，坐标乘以 2 避免出现 0.5 的情况
        vector<long long> xs(n << 1);
        vector<long long> ys(n << 1);

        for (int i = 0, k = 0, p = 0; i < n; i++) {
            long long x = fields[i][0];
            long long y = fields[i][1];
            long long r = fields[i][2];
            xs[k++] = (x << 1) - r;
            xs[k++] = (x << 1) + r;
            ys[p++] = (y << 1) - r;
            ys[p++] = (y << 1) + r;
        }

        // 离散化：排序并去重
        int sizex = prepare(xs);
        int sizey = prepare(ys);

        // 二维差分数组，大小为 (离散化后的坐标数 + 2)
        // 使用 vector 动态分配防止大内存栈溢出
        vector<vector<int>> diff(sizex + 2, vector<int>(sizey + 2, 0));

        for (int i = 0; i < n; i++) {
            long long x = fields[i][0];
            long long y = fields[i][1];
            long long r = fields[i][2];
            
            // 找到离散化后的排名（从1开始）
            int a = rank(xs, (x << 1) - r, sizex);
            int b = rank(ys, (y << 1) - r, sizey);
            int c = rank(xs, (x << 1) + r, sizex);
            int d = rank(ys, (y << 1) + r, sizey);
            
            add(diff, a, b, c, d);
        }

        int ans = 0;
        // 还原二维前缀和并统计最大值
        for (int i = 1; i <= sizex; i++) {
            for (int j = 1; j <= sizey; j++) {
                diff[i][j] += diff[i - 1][j] + diff[i][j - 1] - diff[i - 1][j - 1];
                ans = max(ans, diff[i][j]);
            }
        }

        return ans;
    }

private:
    // 排序并去重，返回有效长度
    int prepare(vector<long long>& nums) {
        sort(nums.begin(), nums.end());
        nums.erase(unique(nums.begin(), nums.end()), nums.end());
        return (int)nums.size();
    }

    // 二分查找返回对应的排名（从1开始）
    int rank(const vector<long long>& nums, long long v, int size) {
        return (int)(lower_bound(nums.begin(), nums.begin() + size, v) - nums.begin()) + 1;
    }

    // 二维差分操作
    void add(vector<vector<int>>& diff, int a, int b, int c, int d) {
        diff[a][b] += 1;
        diff[c + 1][d + 1] += 1;
        diff[c + 1][b] -= 1;
        diff[a][d + 1] -= 1;
    }
};
