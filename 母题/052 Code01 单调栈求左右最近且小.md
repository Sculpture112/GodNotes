// 052 Code01 单调栈求左右最近且小
// 测试链接 : https://www.nowcoder.com/practice/2a2c00e7a88a498693568cef63a4b7bb
// 核心思路：单调栈遍历阶段 + 清算阶段 + 针对重复值的修正阶段
// 时间复杂度 O(N)，额外空间复杂度 O(N)

#include <iostream>
#include <vector>

using namespace std;

const int MAXN = 1000005;
int arr[MAXN];
int st[MAXN];
int ans[MAXN][2];
int n, r;

void compute() {
    r = 0; // 栈指针归零
    // 1. 遍历阶段
    for (int i = 0; i < n; i++) {
        // 当栈不为空且当前值小于等于栈顶值时，弹出栈顶并结算
        while (r > 0 && arr[st[r - 1]] >= arr[i]) {
            int cur = st[--r];
            // 左侧最近小：栈中它下面的那个
            ans[cur][0] = (r > 0) ? st[r - 1] : -1;
            // 右侧最近小：当前要把我踢出来的那个
            ans[cur][1] = i;
        }
        st[r++] = i;
    }

    // 2. 清算阶段
    while (r > 0) {
        int cur = st[--r];
        ans[cur][0] = (r > 0) ? st[r - 1] : -1;
        ans[cur][1] = -1;
    }

    // 3. 修正阶段 (针对有重复值的情况)
    // 从右往左修正，如果右侧小的值和当前值相等，则取那个相等值的右侧答案
    for (int i = n - 2; i >= 0; i--) {
        if (ans[i][1] != -1 && arr[ans[i][1]] == arr[i]) {
            ans[i][1] = ans[ans[i][1]][1];
        }
    }
}

int main() {
    // 快速 IO 优化
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    while (cin >> n) {
        for (int i = 0; i < n; i++) {
            cin >> arr[i];
        }
        compute();
        for (int i = 0; i < n; i++) {
            cout << ans[i][0] << " " << ans[i][1] << "\n";
        }
    }
    return 0;
}
