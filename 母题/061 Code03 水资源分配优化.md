```cpp
// 061 Code03 水资源分配优化

// 测试链接 : https://leetcode.cn/problems/optimize-water-distribution-in-a-village/
// 核心思路：超级源点（虚拟节点） + Kruskal 算法。
// 1. 巧妙地引入一个虚拟的“0号节点”代表地下水库。
// 2. 在每栋房子里建水井的成本，可以等价转换为从“0号节点”向该房子铺设一条管道的成本。
// 3. 将所有建井的选项转化为边后，与原本的房子间管道合并在一起。
// 4. 对所有边（包括真实的管道和虚拟的建井边）按成本进行升序排序。
// 5. 跑一遍标准的 Kruskal 求最小生成树，得出的总成本即为最低总成本。

#include <vector>
#include <algorithm>
#include <functional>

using namespace std;

// 封装边结构，方便后续按成本排序
struct Edge {
    int u, v, cost;
    bool operator<(const Edge& other) const {
        return cost < other.cost;
    }
};

class Solution {
public:
    int minCostToSupplyWater(int n, vector<int>& wells, vector<vector<int>>& pipes) {
        vector<Edge> edges;
        
        // 1. 将打井的成本转化为从 0 号虚拟节点到 i 号节点的边
        // 注意：房子编号从 1 开始，wells 数组索引从 0 开始
        for (int i = 0; i < n; i++) {
            edges.push_back({0, i + 1, wells[i]});
        }
        
        // 2. 加入原本的管道连通边
        for (const auto& pipe : pipes) {
            edges.push_back({pipe[0], pipe[1], pipe[2]});
        }
        
        // 3. 按照成本从小到大排序
        sort(edges.begin(), edges.end());
        
        // 4. 初始化并查集，节点编号为 0 到 n
        vector<int> father(n + 1);
        for (int i = 0; i <= n; i++) {
            father[i] = i;
        }
        
        // Lambda 实现带路径压缩的查找
        function<int(int)> find = [&](int i) {
            if (i != father[i]) {
                father[i] = find(father[i]);
            }
            return father[i];
        };
        
        // 合并操作，返回是否成功合并（即原本不在同一个集合中）
        auto unite = [&](int x, int y) {
            int fx = find(x);
            int fy = find(y);
            if (fx != fy) {
                father[fx] = fy;
                return true;
            }
            return false;
        };
        

        int ans = 0;
        for (const auto& edge : edges) {
            if (unite(edge.u, edge.v)) {
                ans += edge.cost;
            }
        }
        
        return ans;
    }
};
```

---


**核心套路:** `[超级源点 (Virtual Node) / 隐藏源点]` 将“点权”（自己打井的成本）转化为“边权”（从超级源点铺设管道的成本），从而将多源点问题转化为标准的单连通图问题，直接套用最小生成树算法。
