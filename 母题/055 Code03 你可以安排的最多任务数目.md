```cpp
// 055 Code03 你可以安排的最多任务数目

// 测试链接 : https://leetcode.cn/problems/maximum-number-of-tasks-you-can-assign/
// 核心思路：贪心 + 二分答案 + 单调队列。
// 1. 先将任务和工人的力量值从小到大排序。结论：如果能完成 m 个任务，那么一定是最容易的 m 个任务，由力量最大的 m 个工人来完成。
// 2. 二分能完成的任务数 m，二分范围是 [0, min(任务数, 工人数)]。
// 3. 验证函数 check(m)：取出最容易的 m 个任务和最强的 m 个工人。遍历这 m 个工人（从较弱到较强）。
// 4. 对于每个工人，将他不吃药就能解锁的新任务加入双端队列队尾。
//    - 若队列不为空，且队头任务（当前最容易的任务）他不吃药就能完成，贪心策略是让他不吃药完成队头任务（弹出队头）。这里要注意，队列中可能残留之前工人吃药解锁但没做的较难任务。
//    - 若队列为空，或者队头任务连他都觉得难，说明当前工人不吃药什么都做不了，必须吃药。
//    - 将当前工人吃药后能解锁的任务继续放入队尾。若此时队列仍为空，说明吃药也做不了，直接返回 false。
//    - 若吃药后队列不为空，贪心策略是让他吃药完成队尾任务（当前能解锁的最难任务，弹出队尾），把容易的任务留给后面更强的人。并记录消耗一颗药丸。
// 5. 循环结束后检查消耗的药丸数是否 <= 拥有的药丸总数。
// 时间复杂度 O(N log N + M log M)，额外空间复杂度 O(N)

#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    int maxTaskAssign(vector<int>& tasks, vector<int>& workers, int pills, int strength) {
        // 先排序
        sort(tasks.begin(), tasks.end());
        sort(workers.begin(), workers.end());
        
        int tsize = tasks.size();
        int wsize = workers.size();
        int ans = 0;
        
        // deque: 局部开辟单调队列数组，避免全局变量污染
        vector<int> deque(tsize);
        
        // Lambda 验证函数，捕获外部变量
        auto check = [&](int m) {
            int h = 0, t = 0; // 队头、队尾指针
            int cnt = 0;      // 已经使用的药丸数量
            
            // 验证的范围：最容易的 m 个任务，最强的 m 个工人
            int tl = 0, tr = m - 1;
            int wl = wsize - m, wr = wsize - 1;
            
            for (int i = wl, j = tl; i <= wr; i++) {
                // j 不重置，继承前面的进度
                // 1. 工人不吃药的情况下，去解锁任务
                for (; j <= tr && tasks[j] <= workers[i]; j++) {
                    deque[t++] = j;
                }
                
                // 检查队头任务（最容易的）不吃药能不能做
                // 注意：由于 j 是单调递增的，队列里可能包含前一个较弱工人吃药解锁但没做的任务
                if (h < t && tasks[deque[h]] <= workers[i]) {
                    h++; // 不吃药，做最容易的
                } else {
                    // 2. 必须吃药
                    for (; j <= tr && tasks[j] <= workers[i] + strength; j++) {
                        deque[t++] = j;
                    }
                    if (h < t) {
                        cnt++; // 消耗一颗药丸
                        t--;   // 吃药了，贪心做最难的（队尾），把容易的留给后面的工人
                    } else {
                        return false; // 吃药都做不了，直接认定 m 个任务无法完成
                    }
                }
            }
            return cnt <= pills;
        };
        
        // 二分答案
        int l = 0, r = min(tsize, wsize);
        while (l <= r) {
            int m = l + (r - l) / 2;
            if (check(m)) {
                ans = m;     // 能完成 m 个，记录答案并尝试更大的 m
                l = m + 1;
            } else {
                r = m - 1;   // 完不成 m 个，缩小目标
            }
        }
        
        return ans;
    }
};
```

**核心套路:** 
1. 将“求最值”转化为“二分判定”：答案具有单调性，能完成 $m$ 个必定能完成 $m-1$ 个。 2. 局部最优解构建全局最优：选定 $m$ 后，剥离出“最简单的 $m$ 个任务”和“最强的 $m$ 个工人”。3. 贪心博弈分配：不吃药就挑最简单的保底，吃药就挑战能力范围内最难的，榨干药丸价值。
