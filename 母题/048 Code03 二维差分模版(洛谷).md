// 048 Code03 二维差分模版(洛谷)

// 二维差分模版(洛谷)
// 测试链接 : https://www.luogu.com.cn/problem/P3397

#include <iostream>
#include <vector>

using namespace std;

const int MAXN = 1005;

// 全局二维数组，防止命名冲突使用 diff_arr
int diff_arr[MAXN][MAXN];
int n, q;

// 二维差分操作
void add(int a, int b, int c, int d, int k) {
    diff_arr[a][b] += k;
    diff_arr[c + 1][b] -= k;
    diff_arr[a][d + 1] -= k;
    diff_arr[c + 1][d + 1] += k;
}

// 二维前缀和还原原数组
void build() {
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            diff_arr[i][j] += diff_arr[i - 1][j] + diff_arr[i][j - 1] - diff_arr[i - 1][j - 1];
        }
    }
}

// 多组测试数据清理
void clear_arr() {
    for (int i = 1; i <= n + 1; i++) {
        for (int j = 1; j <= n + 1; j++) {
            diff_arr[i][j] = 0;
        }
    }
}

int main() {
    // 竞赛必备：解绑 C/C++ I/O 流，极大提升输入输出效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n >> q) {
        for (int i = 1; i <= q; i++) {
            int a, b, c, d;
            cin >> a >> b >> c >> d;
            add(a, b, c, d, 1);
        }
        
        build();
        
        for (int i = 1; i <= n; i++) {
            cout << diff_arr[i][1];
            for (int j = 2; j <= n; j++) {
                cout << " " << diff_arr[i][j];
            }
            cout << "\n";
        }
        
        clear_arr();
    }

    return 0;
}
