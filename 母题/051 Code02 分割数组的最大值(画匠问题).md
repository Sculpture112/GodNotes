// 051 Code02 分割数组的最大值(画匠问题)

// 给定一个非负整数数组 nums 和一个整数 m
// 你需要将这个数组分成 m 个非空的连续子数组。
// 设计一个算法使得这 m 个子数组各自和的最大值最小。
// 测试链接 : https://leetcode.cn/problems/split-array-largest-sum/

#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

class Solution {
public:
    // 时间复杂度O(n * log(sum))，额外空间复杂度O(1)
    int splitArray(vector<int>& nums, int k) {
        long long sum = 0;
        for (int num : nums) {
            sum += num;
        }
        
        long long ans = 0;
        // 在 [0, sum] 范围内二分查找可能的最小最大和
        for (long long l = 0, r = sum, m, need; l <= r; ) {
            // 中点 m 为当前尝试的限制（每部分累加和的最大值）
            m = l + ((r - l) >> 1);
            
            // 必须让数组每一部分的累加和 <= m，计算最少需要划分成几个部分
            need = f(nums, m);
            
            if (need <= k) {
                // 如果划分的部分数小于等于 k，说明 m 是达标的
                ans = m;
                // 尝试更小的最大值
                r = m - 1;
            } else {
                // 如果部分数超过了 k，说明 m 定得太小了
                l = m + 1;
            }
        }
        return (int)ans;
    }

private:
    /**
     * 贪心判断：在每一部分累加和不超过 limit 的前提下，数组最少需要划分为多少个连续子数组
     * 如果单个元素就超过了 limit，直接返回 INT_MAX 表示不可能
     */
    int f(const vector<int>& arr, long long limit) {
        int parts = 1;
        long long sum = 0;
        for (int num : arr) {
            if (num > limit) {
                return INT_MAX;
            }
            if (sum + num > limit) {
                // 当前部分已满，开启新的一份
                parts++;
                sum = num;
            } else {
                sum += num;
            }
        }
        return parts;
    }
};
