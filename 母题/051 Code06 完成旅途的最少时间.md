// 051 Code06 完成旅途的最少时间

// 有 n 辆公交车，第 i 辆公交车完成一趟旅途需要 time[i] 时间。
// 每辆公交车可以连续完成任意次旅途，且旅途之间互不干扰。
// 返回完成总共 totalTrips 趟旅途所需的最少时间。
// 测试链接 : https://leetcode.cn/problems/minimum-time-to-complete-trips/

#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

class Solution {
public:
    // 时间复杂度 O(n * log(min * totalTrips))，额外空间复杂度 O(1)
    long long minimumTime(vector<int>& time, int totalTrips) {
        // 找到公交车中最短的单趟耗时
        int min_val = INT_MAX;
        for (int x : time) {
            min_val = min(min_val, x);
        }

        long long ans = 0;
        // 二分搜索的范围：
        // 下界 l 从 1 开始（因为 totalTrips >= 1）
        // 上界 r 为单趟最快的公交车独立完成所有趟数的时间
        long long l = 1;
        long long r = (long long)min_val * totalTrips;

        while (l <= r) {
            long long m = l + ((r - l) >> 1);
            
            // 判断在 m 时间内，所有公交车能完成的旅途总数是否达标
            if (f(time, m) >= (long long)totalTrips) {
                // 达标，记录当前时间并尝试寻找更短的时间
                ans = m;
                r = m - 1;
            } else {
                // 不达标，必须增加时间
                l = m + 1;
            }
        }
        return ans;
    }

private:
    /**
     * 核心判定函数：
     * 计算在给定的 time 长度内，每辆公交车能完成的完整旅途次数之和。
     */
    long long f(const vector<int>& arr, long long time) {
        long long ans = 0;
        for (int num : arr) {
            // 每辆公交车在 time 时间内能完成的趟数为 time / num
            ans += (time / num);
        }
        return ans;
    }
};
