```cpp
// 064 Code04 获取所有钥匙的最短路径

// 给定一个二维网格 grid ，其中：
// '.' 代表一个空房间、'#' 代表一堵、'@' 是起点
// 小写字母代表钥匙、大写字母代表锁
// 从起点开始出发，一次移动是指向四个基本方向之一行走一个单位空间
// 不能在网格外面行走，也无法穿过一堵墙
// 如果途经一个钥匙，我们就把它捡起来。除非我们手里有对应的钥匙，否则无法通过锁。
// 假设 k 为 钥匙/锁 的个数，且满足 1 <= k <= 6，
// 字母表中的前 k 个字母在网格中都有自己对应的一个小写和一个大写字母
// 换言之，每个锁有唯一对应的钥匙，每个钥匙也有唯一对应的锁
// 另外，代表钥匙和锁的字母互为大小写并按字母顺序排列
// 返回获取所有钥匙所需要的移动的最少次数。如果无法获取所有钥匙，返回 -1 。
// 测试链接：https://leetcode.cn/problems/shortest-path-to-get-all-keys

#include <iostream>
#include <vector>
#include <string>
#include <cstring>

using namespace std;

const int MAXN = 31;
const int MAXM = 31;
const int MAXK = 6;
// 队列最大长度：31 * 31 * (1 << 6) = 61504，开 65005 绝对安全
const int MAXQ = 65005;

// 0:上，1:右，2:下，3:左
int mv[5] = { -1, 0, 1, 0, -1 };

bool visited[MAXN][MAXM][1 << MAXK];

// 0 : 行
// 1 : 列
// 2 : 收集钥匙的状态
int q[MAXQ][3];

int l_ptr, r_ptr, n, m, target_key;

class Solution {
public:
    void build(const vector<string>& grid) {
        l_ptr = r_ptr = target_key = 0;
        n = grid.size();
        m = grid[0].size();
        
        // C++ 中对于连续内存的多维数组，使用 memset 清空效率极高
        memset(visited, 0, sizeof(visited));
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == '@') {
                    q[r_ptr][0] = i;
                    q[r_ptr][1] = j;
                    q[r_ptr++][2] = 0;
                    // 起点入队时标记为已访问
                    visited[i][j][0] = true;
                }
                if (grid[i][j] >= 'a' && grid[i][j] <= 'f') {
                    // 利用位运算记录目标状态
                    target_key |= (1 << (grid[i][j] - 'a'));
                }
            }
        }
    }

    int shortestPathAllKeys(vector<string>& grid) {
        build(grid);
        int level = 1;
        
        while (l_ptr < r_ptr) {
            int size = r_ptr - l_ptr;
            for (int k = 0; k < size; k++) {
                int x = q[l_ptr][0];
                int y = q[l_ptr][1];
                int s = q[l_ptr++][2];
                
                for (int i = 0; i < 4; i++) {
                    int nx = x + mv[i];
                    int ny = y + mv[i + 1];
                    int ns = s;
                    
                    if (nx < 0 || nx == n || ny < 0 || ny == m || grid[nx][ny] == '#') {
                        // 越界或者障碍
                        continue;
                    }
                    if (grid[nx][ny] >= 'A' && grid[nx][ny] <= 'F' && ((ns & (1 << (grid[nx][ny] - 'A'))) == 0)) {
                        // 是锁，又没有对应的钥匙
                        continue;
                    }
                    if (grid[nx][ny] >= 'a' && grid[nx][ny] <= 'f') {
                        // 碰到了某一把钥匙，更新状态
                        ns |= (1 << (grid[nx][ny] - 'a'));
                    }
                    if (ns == target_key) {
                        // 常见剪枝
                        // 发现集齐所有钥匙直接返回，不用等整层都结束
                        return level;
                    }
                    if (!visited[nx][ny][ns]) {
                        visited[nx][ny][ns] = true;
                        q[r_ptr][0] = nx;
                        q[r_ptr][1] = ny;
                        q[r_ptr++][2] = ns;
                    }
                }
            }
            level++;
        }
        return -1;
    }
};
```

**核心套路:** 
比较简单，只是普通的BFS升维了而已
**状态扩维 BFS (平行宇宙法)**。当迷宫的连通性随着玩家收集的物品（如钥匙、消除道具）动态改变时，单纯的二维坐标 `(x, y)` 无法应对“必须原路折返去开锁”的情况，因为传统 `visited[x][y]` 会把回头路堵死。

**破局解法：** 必须引入第三维“拥有的物品状态”，即记录 `visited[x][y][state]`。由于物品数量极少（本题不超过6把钥匙），利用**二进制位掩码（Bitmask）**将物品收集状态压缩为一个整数（例如 `000101` 表示拥有第0和第2把钥匙）。在标准的二维 BFS 框架上：

1. 遇到钥匙：更新状态进入新宇宙 $\rightarrow$ `next_state = state | (1 << i)`
    
2. 遇到锁：校验当前状态有无钥匙 $\rightarrow$ `(state & (1 << i)) != 0`
