```cpp
// 058 Code01 岛屿数量

// 测试链接 : https://leetcode.cn/problems/number-of-islands/
// 核心思路：洪水填充（Flood Fill）/ DFS。
// 1. 遍历二维网格，寻找值为 '1'（陆地）的格子。
// 2. 一旦找到 '1'，说明发现了一座新岛屿，岛屿数量加 1。
// 3. 随即从该格子触发 DFS（深度优先搜索），向上下左右四个方向蔓延，将相连的 '1' 全部修改为 '0'（防止后续遍历时重复访问，俗称“沉岛”操作）。
// 4. 当整个网格遍历完毕，岛屿的总数也就是成功触发 DFS 的总次数。
// 时间复杂度 O(N * M)，额外空间复杂度 O(N * M)（最坏情况下，整个网格全为 '1'，递归调用栈的最大深度为 N * M）

#include <vector>

using namespace std;

class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        int islands = 0;
        
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == '1') {
                    islands++;
                    dfs(grid, n, m, i, j);
                }
            }
        }
        
        return islands;
    }

private:
    void dfs(vector<vector<char>>& grid, int n, int m, int i, int j) {
        // 越界或者当前格子不是 '1' 时，停止递归
        if (i < 0 || i == n || j < 0 || j == m || grid[i][j] != '1') {
            return;
        }
        
        // 修改当前格子状态，标记为已访问
        grid[i][j] = '0';
        
        // 向四个方向继续蔓延
        dfs(grid, n, m, i - 1, j); // 上
        dfs(grid, n, m, i + 1, j); // 下
        dfs(grid, n, m, i, j - 1); // 左
        dfs(grid, n, m, i, j + 1); // 右
    }
};
```
