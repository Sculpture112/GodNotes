# []

**核心套路:** 
处理 `*` 号时采取类似“完全背包”的空间复用策略：

1. **当空串使用 (匹配 0 个)**：放弃当前的 `*`，用模式串的下一个字符去应对，状态转移至 `(i, j+1)`。
    
2. **继续消耗字符 (匹配 1 个及以上)**：把当前的 `s[i]` 吃掉，但模式串 `p` 停留在原地，保留当前的 `*` 继续尝试吃掉下一个 `s[i+1]`，状态转移至 `(i+1, j)`。
    
3. **普通字符或 `?`**：老老实实匹配当前位，依赖右下角的 `(i+1, j+1)`。

// 074 Code05 通配符匹配

// 给你字符串 s、字符串 p
// s 中一定不含有 '?'、'*' 字符，p 中可能含有 '?'、'*' 字符
// '?' 表示可以变成任意字符，数量 1 个
// '*' 表示可以匹配任何字符串（包括空串）
// 测试链接 : https://leetcode.cn/problems/wildcard-matching/

#include <iostream>
#include <vector>
#include <string>

using namespace std;

class Solution {
public:
    /**
     * 1. 暴力递归版
     * 逻辑：
     * 如果 p[j] 不是 '*'，则当前字符必须匹配，然后看 (i+1, j+1)。
     * 如果 p[j] 是 '*'，有两种选择：
     *   - '*' 至少匹配一个 s[i] 字符：继续看 (i+1, j)，即让 '*' 继续在当前位置发挥作用。
     *   - '*' 匹配空串：跳过当前 '*'，看 (i, j+1)。
     */
    bool isMatch1(string str, string pat) {
        return f1(str, pat, 0, 0);
    }

    bool f1(const string& s, const string& p, int i, int j) {
        if (i == s.length()) {
            if (j == p.length()) return true;
            // s 没了，p 剩下的必须全是 '*' 才能匹配空串
            return p[j] == '*' && f1(s, p, i, j + 1);
        }
        if (j == p.length()) return false;

        if (p[j] != '*') {
            // 当前位必须匹配
            return (s[i] == p[j] || p[j] == '?') && f1(s, p, i + 1, j + 1);
        } else {
            // '*' 的决策：匹配 1 个或多个 (i+1, j) OR 匹配 0 个 (i, j+1)
            return f1(s, p, i + 1, j) || f1(s, p, i, j + 1);
        }
    }

    /**
     * 2. 记忆化搜索版
     */
    bool isMatch2(string str, string pat) {
        int n = str.length();
        int m = pat.length();
        // 0: 未计算, 1: true, 2: false
        vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
        return f2(str, pat, 0, 0, dp);
    }

    bool f2(const string& s, const string& p, int i, int j, vector<vector<int>>& dp) {
        if (dp[i][j] != 0) return dp[i][j] == 1;

        bool ans;
        if (i == s.length()) {
            if (j == p.length()) ans = true;
            else ans = (p[j] == '*' && f2(s, p, i, j + 1, dp));
        } else if (j == p.length()) {
            ans = false;
        } else {
            if (p[j] != '*') {
                ans = (s[i] == p[j] || p[j] == '?') && f2(s, p, i + 1, j + 1, dp);
            } else {
                ans = f2(s, p, i + 1, j, dp) || f2(s, p, i, j + 1, dp);
            }
        }
        dp[i][j] = ans ? 1 : 2;
        return ans;
    }

    /**
     * 3. 严格位置依赖的动态规划 (最优解)
     * 时间复杂度 O(N * M)，空间复杂度 O(N * M)
     */
    bool isMatch(string str, string pat) {
        int n = str.length();
        int m = pat.length();
        // dp[i][j] 代表 s[i...] 与 p[j...] 是否匹配
        vector<vector<bool>> dp(n + 1, vector<bool>(m + 1, false));

        // 基础情况：结尾对结尾
        dp[n][m] = true;

        // 初始化第 n 行：当 s 为空时，只有 p 后缀全是 '*' 才能匹配
        for (int j = m - 1; j >= 0; j--) {
            if (pat[j] == '*') {
                dp[n][j] = true;
            } else {
                // 一旦遇到非 '*'，前面就不可能匹配空串了
                break;
            }
        }

        // 填表过程
        for (int i = n - 1; i >= 0; i--) {
            for (int j = m - 1; j >= 0; j--) {
                if (pat[j] != '*') {
                    // 非星号情况：当前匹配且后续也匹配
                    dp[i][j] = (str[i] == pat[j] || pat[j] == '?') && dp[i + 1][j + 1];
                } else {
                    /**
                     * 星号情况：
                     * 1. dp[i + 1][j]: '*' 匹配至少一个字符，跳到 s 下一个位置，p 不动（继续复用 '*'）
                     * 2. dp[i][j + 1]: '*' 匹配空串，跳到 p 下一个位置
                     */
                    dp[i][j] = dp[i + 1][j] || dp[i][j + 1];
                }
            }
        }

        return dp[0][0];
    }
};
