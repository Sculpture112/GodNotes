// 049 Code03 最小覆盖子串

// 给你一个字符串 s 、一个字符串 t 。返回 s 中涵盖 t 所有字符的最小子串
// 如果 s 中不存在涵盖 t 所有字符的子串，则返回空字符串 "" 。
// 测试链接 : https://leetcode.cn/problems/minimum-window-substring/

#include <string>
#include <vector>
#include <climits>

using namespace std;

class Solution {
public:
    // 时间复杂度O(n)，n为字符串s的长度
    // 额外空间复杂度O(1)，因为字符集大小是固定的(256)
    string minWindow(string str, string tar) {
        if (str.length() < tar.length()) {
            return "";
        }

        // 每种字符的欠债情况
        // cnts[i] 为负数：代表字符 i 还有欠债
        // cnts[i] 为正数：代表字符 i 有多余盈余
        int cnts[256] = {0};
        for (char cha : tar) {
            cnts[(unsigned char)cha]--;
        }

        // 最小覆盖子串的长度
        int len = INT_MAX;
        // 发现的最小覆盖子串的起始位置
        int start = 0;
        // 总债务数量（还差多少个字符才能达标）
        int debt = tar.length();

        // 左右指针形成滑动窗口
        for (int l = 0, r = 0; r < (int)str.length(); r++) {
            // 窗口右边界向右扩，当前字符进入窗口
            // 如果该字符在 cnts 中是负数，说明它是在偿还债务，总债务 debt 减少
            if (cnts[(unsigned char)str[r]]++ < 0) {
                debt--;
            }

            // 当总债务为 0 时，说明当前窗口已经涵盖了 t 的所有字符
            if (debt == 0) {
                // 尝试缩紧左边界：如果左边界字符在窗口内有盈余（cnts > 0），就把它踢出窗口
                while (cnts[(unsigned char)str[l]] > 0) {
                    cnts[(unsigned char)str[l++]]--;
                }

                // 更新全局最小长度和起始位置
                if (r - l + 1 < len) {
                    len = r - l + 1;
                    start = l;
                }
            }
        }

        return len == INT_MAX ? "" : str.substr(start, len);
    }
};
