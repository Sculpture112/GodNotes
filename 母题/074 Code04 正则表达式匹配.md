// 074 Code04 正则表达式匹配

// 给你字符串 s、字符串 p
// s 中一定不含有 '.'、'*' 字符，p 中可能含有 '.'、'*' 字符
// '.' 表示可以变成任意字符，数量 1 个
// '*' 表示可以让 '*' 前面那个字符数量任意(甚至可以是 0 个)
// 测试链接 : https://leetcode.cn/problems/regular-expression-matching/

#include <iostream>
#include <vector>
#include <string>

using namespace std;

class Solution {
public:
    /**
     * 1. 暴力递归版
     * 核心逻辑：
     * 如果 p[j+1] 不是 '*'，则当前字符必须匹配，然后看 (i+1, j+1)。
     * 如果 p[j+1] 是 '*'，有两种选择：
     *   - 忽略当前 x* 组合，跳到 (i, j+2)。
     *   - 如果当前字符能匹配，则使用一个 x*，尝试匹配下一个 s[i+1]，即看 (i+1, j)。
     */
    bool isMatch1(string str, string pat) {
        return f1(str, pat, 0, 0);
    }

    bool f1(const string& s, const string& p, int i, int j) {
        if (i == s.length()) {
            if (j == p.length()) return true;
            // s 没了，p 剩下的必须是 "x*y*z*" 这种形式
            return j + 1 < p.length() && p[j + 1] == '*' && f1(s, p, i, j + 2);
        }
        if (j == p.length()) return false;

        // p[j+1] 不是 '*'
        if (j + 1 == p.length() || p[j + 1] != '*') {
            return (s[i] == p[j] || p[j] == '.') && f1(s, p, i + 1, j + 1);
        } else {
            // p[j+1] 是 '*'，这里体现了类似“完全背包”的优化思路
            // 选择 1: 让 x* 匹配 0 个，跳过
            bool p1 = f1(s, p, i, j + 2);
            // 选择 2: 如果当前匹配，让 x* 匹配 1 个，s 移动但 p 不动（下次还可以继续用这个 x*）
            bool p2 = (s[i] == p[j] || p[j] == '.') && f1(s, p, i + 1, j);
            return p1 || p2;
        }
    }

    /**
     * 2. 记忆化搜索版
     */
    bool isMatch2(string str, string pat) {
        int n = str.length();
        int m = pat.length();
        // 0: 未计算, 1: true, 2: false
        vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
        return f2(str, pat, 0, 0, dp);
    }

    bool f2(const string& s, const string& p, int i, int j, vector<vector<int>>& dp) {
        if (dp[i][j] != 0) return dp[i][j] == 1;

        bool ans;
        if (i == s.length()) {
            if (j == p.length()) ans = true;
            else ans = j + 1 < p.length() && p[j + 1] == '*' && f2(s, p, i, j + 2, dp);
        } else if (j == p.length()) {
            ans = false;
        } else {
            if (j + 1 == p.length() || p[j + 1] != '*') {
                ans = (s[i] == p[j] || p[j] == '.') && f2(s, p, i + 1, j + 1, dp);
            } else {
                ans = f2(s, p, i, j + 2, dp) || ((s[i] == p[j] || p[j] == '.') && f2(s, p, i + 1, j, dp));
            }
        }
        dp[i][j] = ans ? 1 : 2;
        return ans;
    }

    /**
     * 3. 严格位置依赖的动态规划 (最优解)
     * 时间复杂度 O(N * M)，空间复杂度 O(N * M)
     */
    bool isMatch(string str, string pat) {
        int n = str.length();
        int m = pat.length();
        // dp[i][j] 代表 s[i...] 与 p[j...] 是否匹配
        vector<vector<bool>> dp(n + 1, vector<bool>(m + 1, false));

        // 基础情况：空串匹配空串
        dp[n][m] = true;

        // 初始化第 n 行：s 为空时，p 的匹配情况
        for (int j = m - 1; j >= 0; j--) {
            dp[n][j] = j + 1 < m && pat[j + 1] == '*' && dp[n][j + 2];
        }

        // 填表
        for (int i = n - 1; i >= 0; i--) {
            for (int j = m - 1; j >= 0; j--) {
                if (j + 1 == m || pat[j + 1] != '*') {
                    // 普通匹配
                    dp[i][j] = (str[i] == pat[j] || pat[j] == '.') && dp[i + 1][j + 1];
                } else {
                    /**
                     * '*' 匹配：
                     * 1. dp[i][j + 2]: 让 x* 匹配 0 次
                     * 2. (str[i] == pat[j] || pat[j] == '.') && dp[i + 1][j]: 
                     *    如果当前匹配，让 x* 匹配 1 次，并保持在 p[j] 尝试匹配下一个 s[i+1]
                     */
                    dp[i][j] = dp[i][j + 2] || ((str[i] == pat[j] || pat[j] == '.') && dp[i + 1][j]);
                }
            }
        }

        return dp[0][0];
    }
};
