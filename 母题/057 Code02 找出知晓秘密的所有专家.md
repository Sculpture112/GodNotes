// 057 Code02 找出知晓秘密的所有专家

// 测试链接 : https://leetcode.cn/problems/find-all-people-with-secret/
// 核心思路：带属性的并查集 + 按时间分批处理。
// 1. 将所有会议按时间戳从小到大排序。
// 2. 将同一个时刻发生的所有会议作为一批一起处理。在这一批中，使用并查集将开会的人互相连通。
// 3. 赋予并查集“属性”：在合并两个集合时，只要其中一个集合包含知晓秘密的专家，合并后的整个集合代表都将标记为知晓秘密（属性“或”操作 `|`）。
// 4. 当同一时刻的一批会议全部连通完毕后，检查这一批所有参与开会的专家。如果某位专家所在的连通块最终依然不知道秘密，说明刚才这些会议里没人能把秘密传给他。
// 5. 关键撤销动作：因为秘密共享只在此时刻有效，对于没听到秘密的人，必须把他们的连通关系拆除（重置为孤立节点 `father[i] = i`），防止他们把这种“没用的连通状态”带到下一个时刻导致逻辑错乱。
// 6. 所有会议处理完毕后，遍历所有专家，凡是和知晓秘密的代表节点连通的专家，即为最终答案。
// 时间复杂度 O(M log M + N)，其中 M 为会议数量；额外空间复杂度 O(N)

#include <vector>
#include <algorithm>
#include <functional>

using namespace std;

class Solution {
public:
    vector<int> findAllPeople(int n, vector<vector<int>>& meetings, int firstPerson) {
        // father: 并查集父节点数组
        vector<int> father(n);
        // secret: 并查集代表元素的属性，记录该集合是否知晓秘密
        vector<bool> secret(n, false);
        
        // 1. 初始化并查集
        for (int i = 0; i < n; i++) {
            father[i] = i;
        }
        // 0 号专家和 firstPerson 初始就知道秘密，提前连通并赋予属性
        father[firstPerson] = 0;
        secret[0] = true;
        
        // Lambda 实现带路径压缩的查找
        function<int(int)> find = [&](int i) {
            if (i != father[i]) {
                father[i] = find(father[i]);
            }
            return father[i];
        };
        
        // 带有属性传递的合并操作
        auto unite = [&](int x, int y) {
            int fx = find(x);
            int fy = find(y);
            if (fx != fy) {
                father[fx] = fy;
                // 只要有一方知道秘密，合并后代表节点就知道秘密
                secret[fy] = secret[fy] || secret[fx];
            }
        };
        
        // 2. 会议按时间排序
        sort(meetings.begin(), meetings.end(), [](const vector<int>& a, const vector<int>& b) {
            return a[2] < b[2];
        });
        
        int m = meetings.size();
        for (int l = 0, r = 0; l < m;) {
            r = l;
            // 找出同一时刻发生的所有会议 [l, r]
            while (r + 1 < m && meetings[l][2] == meetings[r + 1][2]) {
                r++;
            }
            
            // 连通这一批会议中的所有专家
            for (int i = l; i <= r; i++) {
                unite(meetings[i][0], meetings[i][1]);
            }
            
            // 检查这一批开会的人，如果没有得知秘密，必须拆除刚刚建立的连通状态（撤销）
            for (int i = l; i <= r; i++) {
                int a = meetings[i][0];
                int b = meetings[i][1];
                if (!secret[find(a)]) {
                    father[a] = a;
                }
                if (!secret[find(b)]) {
                    father[b] = b;
                }
            }
            
            // 处理下一个时刻
            l = r + 1;
        }
        
        // 3. 收集最终知晓秘密的专家
        vector<int> ans;
        for (int i = 0; i < n; i++) {
            if (secret[find(i)]) {
                ans.push_back(i);
            }
        }
        
        return ans;
    }
};
