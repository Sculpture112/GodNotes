// 071 Code05 可以翻转1次的情况下子数组最大累加和

// 给定一个数组nums，
// 现在允许你随意选择数组连续一段进行翻转，也就是子数组逆序的调整
// 比如翻转[1,2,3,4,5,6]的[2~4]范围，得到的是[1,2,5,4,3,6]
// 返回必须随意翻转1次之后，子数组的最大累加和
// 对数器验证

#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>
#include <cstdlib>
#include <ctime>

using namespace std;

class Solution {
public:
    // 返回子数组最大累加和 (Kadane 算法)
    int maxSum(const vector<int>& nums) {
        int n = nums.size();
        int ans = nums[0];
        for (int i = 1, pre = nums[0]; i < n; i++) {
            pre = max(nums[i], pre + nums[i]);
            ans = max(ans, pre);
        }
        return ans;
    }

    // 暴力方法
    // 为了验证
    // 注意这里按值传递 nums，这样内部翻转就不会污染调用者的原数组
    int maxSumReverse1(vector<int> nums) {
        int ans = INT_MIN;
        int n = nums.size();
        for (int l = 0; l < n; l++) {
            for (int r = l; r < n; r++) {
                // C++ 标准库自带翻转函数，左闭右开区间
                reverse(nums.begin() + l, nums.begin() + r + 1);
                ans = max(ans, maxSum(nums));
                // 恢复原状，以便下一次循环
                reverse(nums.begin() + l, nums.begin() + r + 1);
            }
        }
        return ans;
    }

    // 正式方法
    // 时间复杂度O(n)
    int maxSumReverse2(const vector<int>& nums) {
        int n = nums.size();
        if (n == 0) return 0;
        
        // start[i] : 所有必须以i开头的子数组中，最大累加和是多少
        vector<int> start(n);
        start[n - 1] = nums[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            // 选自己，或者把自己和后面的连起来
            start[i] = max(nums[i], nums[i] + start[i + 1]);
        }
        
        int ans = start[0];
        
        // end : 子数组必须以i-1结尾，其中的最大累加和
        int end = nums[0];
        
        // maxEnd :
        // 0~i-1范围上，
        // 子数组必须以0结尾，其中的最大累加和
        // 子数组必须以1结尾，其中的最大累加和
        // ...
        // 子数组必须以i-1结尾，其中的最大累加和
        // 所有情况中，最大的那个累加和就是 maxEnd
        int maxEnd = nums[0];
        
        for (int i = 1; i < n; i++) {
            // maxEnd 代表前面翻转过来的部分（相当于反向的最大后缀）
            // 枚举划分点 i...
            ans = max(ans, maxEnd + start[i]);
            
            // 子数组必须以i结尾，其中的最大累加和
            end = max(nums[i], end + nums[i]);
            maxEnd = max(maxEnd, end);
        }
        
        ans = max(ans, maxEnd);//
        
        return ans;
    }
};

// 为了测试
// 生成随机数组，范围在 [-v, v] 之间
vector<int> randomArray(int n, int v) {
    vector<int> ans(n);
    for (int i = 0; i < n; i++) {
        ans[i] = rand() % (v * 2 + 1) - v;
    }
    return ans;
}

// 为了测试
// 对数器
int main() {
    srand(time(nullptr)); // 初始化随机数种子

    int n = 50;
    int v = 200;
    int testTime = 20000;
    
    cout << "测试开始\n";
    Solution sol;
    for (int i = 0; i < testTime; i++) {
        int len = (rand() % n) + 1;
        vector<int> arr = randomArray(len, v);
        
        int ans1 = sol.maxSumReverse1(arr);
        int ans2 = sol.maxSumReverse2(arr);
        
        if (ans1 != ans2) {
            cout << "出错了!\n";
            // 如果出错可以打印 arr 方便调试
            break;
        }
    }
    cout << "测试结束\n";

    return 0;
}
