```cpp
// 055 Code01 和至少为K的最短子数组

// 测试链接 : https://leetcode.cn/problems/shortest-subarray-with-sum-at-least-k/
// 核心思路：前缀和 + 单调队列。求累加和 >= K 的最短子数组长度。

// 1. 构建前缀和数组 sum，sum[i] 表示前 i 个元素的累加和（注意使用 long long 防溢出）。
// 2. 维护一个单调递增的双端队列，队列中存储的是前缀和数组的索引。
// 3. 遍历前缀和数组，若当前前缀和 sum[i] 与队头前缀和之差 >= K，说明找到了一个合法子数组。更新最短长度，并弹出队头（因为后续更靠右的元素即使与该队头构成合法子数组，长度也必定更长，所以该队头已无保留价值）。
// 4. 将当前索引 i 入队前，弹出队尾所有 >= 当前前缀和的索引（因为当前前缀和更小且位置更靠右，作为后续求差时的“减数”一定比被弹出的元素更优）。
// 时间复杂度 O(N)，额外空间复杂度 O(N)

#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

class Solution {
public:
    int shortestSubarray(vector<int>& arr, int k) {
        int n = arr.size();
        
        // sum[i] : 前 i 个数的前缀和
        // 使用 long long 防止累加过程中溢出
        vector<long long> sum(n + 1, 0);
        for (int i = 0; i < n; i++) {
            sum[i + 1] = sum[i] + arr[i];
        }
        
        // deque: 模拟双端队列，存储 sum 数组的索引
        // 局部开辟空间，随生命周期自动销毁
        vector<int> deque(n + 1);
        int h = 0, t = 0; // 头尾指针
        int ans = INT_MAX;
        
        // 遍历所有前缀和（包括前 0 个数的前缀和）
        for (int i = 0; i <= n; i++) {
            // 1. 检查队头，如果当前前缀和 - 队头前缀和 >= k，达标！
            // 结算并弹出队头，因为更长距离的没必要再看了
            while (h != t && sum[i] - sum[deque[h]] >= k) {
                ans = min(ans, i - deque[h++]);
            }
            
            // 2. 维护单调递增队列（小 -> 大）
            // 如果队尾的前缀和 >= 当前前缀和，队尾淘汰
            while (h != t && sum[deque[t - 1]] >= sum[i]) {
                t--;
            }
            
            // 3. 当前前缀和索引从尾部加入
            deque[t++] = i;
        }
        
        return ans != INT_MAX ? ans : -1;
    }
};
```
