```cpp
// 056 Code03 情侣牵手

// 测试链接 : https://leetcode.cn/problems/couples-holding-hands/
// 核心思路：并查集。
// 1. 将每对情侣看作一个节点，编号为 0 到 n/2 - 1。人 x 所属的情侣编号为 x / 2。
// 2. 遍历所有相邻的两个座位。如果坐在相邻座位上的两人不属于同一对情侣，说明这两对情侣之间产生了交错，我们将这两对情侣所在的集合进行合并（连边）。
// 3. 最终，如果一个连通分量里包含 k 对情侣，那么只需要在内部交换 k - 1 次，就能让这 k 对情侣各自牵手。
// 4. 总最少交换次数 = 所有连通分量的 (k_i - 1) 之和 = 总情侣对数 - 最终的连通分量个数。
// 时间复杂度 O(N)，额外空间复杂度 O(N)

#include <vector>
#include <functional>

using namespace std;

class Solution {
public:
    int minSwapsCouples(vector<int>& row) {
        int n = row.size();
        int m = n / 2; // 总情侣对数
        
        // father: 并查集的父节点数组
        vector<int> father(m);
        for (int i = 0; i < m; i++) {
            father[i] = i;
        }
        
        int sets = m; // 初始连通分量个数，即情侣对数
        
        // C++ 使用 std::function 配合 lambda 实现递归的路径压缩查找
        function<int(int)> find = [&](int i) {
            if (i != father[i]) {
                father[i] = find(father[i]);
            }
            return father[i];
        };
        
        // 合并操作，采用极简挂载
        auto unite = [&](int x, int y) {
            int fx = find(x);
            int fy = find(y);
            if (fx != fy) {
                father[fx] = fy;
                sets--; // 每合并一次，连通分量总数减 1
            }
        };
        
        // 每次跨越两个座位遍历
        for (int i = 0; i < n; i += 2) {
            // 将相邻两人代表的情侣集合进行合并
            unite(row[i] / 2, row[i + 1] / 2);
        }
        
        // 交换次数 = 总情侣对数 - 形成的连通分量个数
        return m - sets;
    }
};
```
