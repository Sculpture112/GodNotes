// 053 Code04 统计全1子矩形的数量

// 测试链接 : https://leetcode.cn/problems/count-submatrices-with-all-ones/
// 核心思路：二维转换一维 + 单调栈。
// 1. 遍历矩阵每一行，将当前行及以上的连续 1 压缩成一维直方图（height 数组）。
// 2. 对每一层的直方图数组使用单调递增栈求解。
// 3. 当遇到比栈顶元素小或等于的高度时，弹出栈顶并结算。为了防止重复计算，遇到高度相等时不结算，留到后面一起处理。
// 4. 结算时，计算以当前弹出高度为基准的矩形数量：
//    - 宽度范围（len）：左侧比它小的高度的索引（left）与右侧比它小的高度的索引（i）之间的距离（i - left - 1）。
//    - 独立高度区间：当前弹出的高度，减去左右两侧较高的一边（bottom）。在这个高度差内的每一层，都能形成宽度为 len 的全 1 区域。
//    - 组合数：在长度为 len 的范围内，任意两点可以组成一个底边，数量为 len * (len + 1) / 2。
// 时间复杂度 O(N * M)，额外空间复杂度 O(M)

#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    int numSubmat(vector<vector<int>>& mat) {
        int n = mat.size();
        int m = mat[0].size();
        int ans = 0;
        // height: 记录以当前行为底的连续 1 的高度
        vector<int> height(m, 0);

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                // 遇到 0 则高度断开重置为 0，否则高度累加 1
                height[j] = (mat[i][j] == 0) ? 0 : height[j] + 1;
            }
            // 计算当前行为底的子矩形数量并累加
            ans += countFromBottom(height, m);
        }
        return ans;
    }

private:
    int countFromBottom(const vector<int>& height, int m) {
        // stack: 单调栈，存储索引
        // 局部开辟，生命周期仅在当前函数内，符合 C++ 习惯
        vector<int> stack(m);
        int r = 0; // 栈顶指针
        int ans = 0;

        for (int i = 0; i < m; i++) {
            // 当新来的高度 <= 栈顶高度时，弹出并结算
            while (r > 0 && height[stack[r - 1]] >= height[i]) {
                int cur = stack[--r];
                // 只有严格大于时才结算，等于的情况留给最后的相同高度去统一结算
                if (height[cur] > height[i]) {
                    int left = (r == 0) ? -1 : stack[r - 1];
                    int len = i - left - 1;
                    int bottom = max(left == -1 ? 0 : height[left], height[i]);
                    // 高度差 * 底边组合数
                    ans += (height[cur] - bottom) * len * (len + 1) / 2;
                }
            }
            stack[r++] = i;
        }

        // 结算栈中剩余的元素
        while (r > 0) {
            int cur = stack[--r];
            int left = (r == 0) ? -1 : stack[r - 1];
            int len = m - left - 1;
            int down = (left == -1) ? 0 : height[left];
            ans += (height[cur] - down) * len * (len + 1) / 2;
        }

        return ans;
    }
};
