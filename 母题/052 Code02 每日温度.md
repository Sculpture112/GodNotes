// 052 Code02 每日温度
// 测试链接 : https://leetcode.cn/problems/daily-temperatures/
// 核心思路：单调栈。寻找右侧第一个比当前元素大的位置，计算下标差。
// 时间复杂度 O(N)，额外空间复杂度 O(N)

#include <iostream>
#include <vector>

using namespace std;

class Solution {
public:
    // 为了性能，在 LeetCode 提交时可以使用类内的成员变量模拟静态数组
    // 但为了通用性，这里展示常规的单调栈写法
    vector<int> dailyTemperatures(vector<int>& temperatures) {
        int n = temperatures.size();
        vector<int> ans(n, 0);
        // stack 内存放的是下标
        vector<int> st(n);
        int r = 0; // 栈顶指针

        for (int i = 0; i < n; i++) {
            // 如果当前温度比栈顶温度高，结算栈顶
            while (r > 0 && temperatures[st[r - 1]] < temperatures[i]) {
                int cur = st[--r];
                ans[cur] = i - cur;
            }
            st[r++] = i;
        }
        return ans;
    }
};
