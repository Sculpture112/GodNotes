// 049 Code06 K个不同整数的子数组

// 给定一个正整数数组 nums 和一个整数 k，返回 nums 中 「好子数组」 的数目。
// 如果 nums 的某个子数组中不同整数的个数恰好为 k
// 则称 nums 的这个连续、不一定不同的子数组为 「好子数组 」。
// 测试链接 : https://leetcode.cn/problems/subarrays-with-k-different-integers/

#include <vector>
#include <algorithm>

using namespace std;

// 对应 Java 中的 MAXN，根据题目数据范围调整
const int MAXN = 20005;
int cnts[MAXN];

class Solution {
public:
    int subarraysWithKDistinct(vector<int>& nums, int k) {
        return numsOfMostKinds(nums, k) - numsOfMostKinds(nums, k - 1);
    }

private:
    // arr 中有多少子数组，数字种类不超过 k
    // 时间复杂度 O(n)，额外空间复杂度 O(n)
    int numsOfMostKinds(vector<int>& arr, int k) {
        if (k <= 0) {
            return 0;
        }
        int n = arr.size();
        
        // 仿照 Java 的 Arrays.fill(cnts, 1, n + 1, 0)
        // 仅清空当前数组长度范围内可能出现的数值计数
        for (int i = 0; i <= n; i++) {
            cnts[i] = 0;
        }

        int ans = 0;
        // collect: 当前窗口内不同整数的个数
        for (int l = 0, r = 0, collect = 0; r < n; r++) {
            // 右边界进入窗口
            if (++cnts[arr[r]] == 1) {
                collect++;
            }
            
            // 如果种类超过 k，左边界右移，缩小窗口
            while (collect > k) {
                if (--cnts[arr[l++]] == 0) {
                    collect--;
                }
            }
            
            // 累加以 r 结尾，且种类数不超过 k 的子数组数量
            // 满足条件的子数组左边界可以是 [l, r] 中的任意位置
            ans += r - l + 1;
        }
        return ans;
    }
};
