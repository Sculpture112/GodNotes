// 075 Code03 观赏樱花

// 给定一个背包的容量 t，一共有 n 种货物，以及每种货物的信息：花费(cost)、价值(val)、数量(cnt)
// 如果 cnt == 0，代表这种货物可以无限选择（完全背包）
// 如果 cnt > 0，那么 cnt 代表这种货物的最大数量（多重背包）
// 挑选货物的总容量在不超过 t 的情况下，返回能得到的最大价值。
// 测试链接 : https://www.luogu.com.cn/problem/P1833

#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>

using namespace std;
跟上一题一模一样....
/**
 * 💡 核心思路：
 * 1. 时间计算：通过输入的两个时间点计算出总分钟数，即为背包容量 t。
 * 2. 类型统一：
 *    - 对于完全背包（cnt == 0），由于容量 t 最大只有 1000，且 cost > 0，
 *      所以一种物品最多只能买 t / cost 个。直接将 cnt 设为 1000 即可转为多重背包。
 *    - 对于多重背包，使用“二进制分组”优化将其拆分为多个 01 背包物品。
 * 3. 求解：运行 01 背包空间压缩算法。
 */

const int MAXN = 1000005; // 拆分后的最大物品数量
const int MAXW = 1005;    // 最大背包容量
const int ENOUGH = 1001;  // 完全背包转多重背包的足够数量

int v[MAXN];  // 拆分后物品的价值
int w[MAXN];  // 拆分后物品的重量（花费）
int dp[MAXW]; // DP 数组
int h1, m1, h2, m2, n, t, m;

/**
 * 01 背包空间压缩实现
 */
int compute() {
    for (int i = 0; i <= t; i++) dp[i] = 0;
    for (int i = 1; i <= m; i++) {
        for (int j = t; j >= w[i]; j--) {
            dp[j] = max(dp[j], dp[j - w[i]] + v[i]);
        }
    }
    return dp[t];
}

int main() {
    // 处理特殊的时间格式 HH:MM
    // 使用 scanf 可以非常方便地跳过中间的冒号
    while (scanf("%d:%d %d:%d %d", &h1, &m1, &h2, &m2, &n) != EOF) {
        // 计算总分钟数（背包容量）
        int total_minutes1 = h1 * 60 + m1;
        int total_minutes2 = h2 * 60 + m2;
        t = total_minutes2 - total_minutes1;
        
        m = 0; // 拆分物品计数器
        for (int i = 0; i < n; i++) {
            int cost, val, cnt;
            scanf("%d %d %d", &cost, &val, &cnt);
            
            // 如果是完全背包，将其转化为足够大的多重背包
            if (cnt == 0) {
                cnt = ENOUGH;
            }
            
            // 二进制分组逻辑
            for (int k = 1; k <= cnt; k <<= 1) {
                m++;
                v[m] = k * val;
                w[m] = k * cost;
                cnt -= k;
            }
            if (cnt > 0) {
                m++;
                v[m] = cnt * val;
                w[m] = cnt * cost;
            }
        }
        printf("%d\n", compute());
    }

    return 0;
}
