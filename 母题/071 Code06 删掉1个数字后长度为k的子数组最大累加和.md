// 071 Code06 删掉1个数字后长度为k的子数组最大累加和

// 给定一个数组nums，求必须删除一个数字后的新数组中
// 长度为k的子数组最大累加和，删除哪个数字随意
// 对数器验证

#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>
#include <cstdlib>
#include <ctime>

using namespace std;

class Solution {
public:
    // 暴力方法
    // 为了测试
    // 删掉 index 位置的元素，然后返回新数组
    // 注意：C++中 delete 是关键字，所以改名为 deleteElem
    vector<int> deleteElem(const vector<int>& nums, int index) {
        vector<int> ans;
        ans.reserve(nums.size() - 1);
        for (int j = 0; j < nums.size(); j++) {
            if (j != index) {
                ans.push_back(nums[j]);
            }
        }
        return ans;
    }

    // 暴力方法
    // 为了测试
    // 枚举每一个子数组找到最大累加和
    int lenKmaxSum(const vector<int>& nums, int k) {
        int n = nums.size();
        int ans = INT_MIN;
        for (int i = 0; i <= n - k; i++) {
            int cur = 0;
            for (int j = i, cnt = 0; cnt < k; j++, cnt++) {
                cur += nums[j];
            }
            ans = max(ans, cur);
        }
        return ans;
    }

    // 暴力方法
    // 为了测试
    int maxSum1(const vector<int>& nums, int k) {
        int n = nums.size();
        if (n <= k) {
            return 0;
        }
        int ans = INT_MIN;
        for (int i = 0; i < n; i++) {
            vector<int> rest = deleteElem(nums, i);
            ans = max(ans, lenKmaxSum(rest, k));
        }
        return ans;
    }

    // 正式方法
    // 时间复杂度 O(N)
    int maxSum2(const vector<int>& nums, int k) {
        int n = nums.size();
        if (n <= k) {
            return 0;
        }
        
        // 单调队列 : 维持窗口内最小值的更新结构
        // window 中存储的是数组的下标
        vector<int> window(n);
        int l = 0;
        int r = 0;
        
        // 窗口累加和
        long long sum = 0;
        int ans = INT_MIN;
        
        for (int i = 0; i < n; i++) {
            // 单调队列 : i 位置进入单调队列，保持队列单调递增（队首是最小值）
            while (l < r && nums[window[r - 1]] >= nums[i]) {
                r--;
            }
            window[r++] = i;
            sum += nums[i];
            
            // 当窗口大小达到 k + 1 时，开始计算答案并向右滑动
            if (i >= k) {
                // 窗口内总共有 k + 1 个数，减去单调队列队首记录的最小值，即为删掉 1 个数后的最大累加和
                ans = max(ans, (int)(sum - nums[window[l]]));
                
                // 单调队列 : 如果单调队列最左侧的位置过期了（即当前窗口最左侧的元素要出窗口了），从队列中弹出
                if (window[l] == i - k) {
                    l++;
                }
                // 真实窗口滑动，减去最左侧过期元素的值
                sum -= nums[i - k];
            }
        }
        return ans;
    }
};

// 为了测试
// 生成长度为 n，值在 [-v, +v] 之间的随机数组
vector<int> randomArray(int n, int v) {
    vector<int> ans(n);
    for (int i = 0; i < n; i++) {
        ans[i] = rand() % (2 * v + 1) - v;
    }
    return ans;
}

// 为了测试
// 对数器
int main() {
    srand(time(nullptr)); // 初始化随机数种子

    int n = 200;
    int v = 1000;
    int testTimes = 10000;
    
    cout << "测试开始\n";
    Solution sol;
    for (int i = 0; i < testTimes; i++) {
        int len = (rand() % n) + 1;
        vector<int> nums = randomArray(len, v);
        int k = (rand() % n) + 1;
        
        int ans1 = sol.maxSum1(nums, k);
        int ans2 = sol.maxSum2(nums, k);
        
        if (ans1 != ans2) {
            cout << "出错了!\n";
            break;
        }
    }
    cout << "测试结束\n";

    return 0;
}
