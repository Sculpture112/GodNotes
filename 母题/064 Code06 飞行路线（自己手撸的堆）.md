// 064 Code06 飞行路线（自己手撸的堆）

// Alice和Bob现在要乘飞机旅行，他们选择了一家相对便宜的航空公司
// 该航空公司一共在n个城市设有业务，设这些城市分别标记为0 ~ n−1
// 一共有m种航线，每种航线连接两个城市，并且航线有一定的价格
// Alice 和 Bob 现在要从一个城市沿着航线到达另一个城市，途中可以进行转机
// 航空公司对他们这次旅行也推出优惠，他们可以免费在最多k种航线上搭乘飞机
// 那么 Alice 和 Bob 这次出行最少花费多少
// 测试链接 : https://www.luogu.com.cn/problem/P4568
// 请同学们务必参考如下代码中关于输入、输出的处理
// 这是输入输出处理效率很高的写法
// C++ 提交时直接全部复制提交即可通过

#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

const int MAXN = 10005;
// 无向图，m <= 50000，存双向边最大需要 100000，开 100005
const int MAXM = 100005;
const int MAXK = 15;

// 链式前向星建图需要
int head[MAXN];
int nxt[MAXM];
int to_node[MAXM];
int weight[MAXM];
int cnt;

// Dijkstra需要
int dist[MAXN][MAXK];
bool visited[MAXN][MAXK];

// C++ 中用 struct 来整合堆中的元素，内存连续且交换效率极高
struct Element {
    int u;
    int use;
    int cost;
};

// 自己写的普通堆，静态结构，推荐
// 因为(点编号，使用免费路线的次数)，两个参数的组合才是图中的点
// 两个参数的组合对应一个点(一个堆的下标)，所以反向索引堆不好写
Element heap_arr[MAXM * MAXK];
int heapSize;

int n, m, k, s, t;

// 提取出的全局变量，模拟 Java 中复用的全局 u, use, cost
int curr_u, curr_use, curr_cost;

void build() {
    cnt = 1;
    heapSize = 0;
    for (int i = 0; i < n; i++) {
        head[i] = 0;
        for (int j = 0; j <= k; j++) {
            dist[i][j] = INT_MAX;
            visited[i][j] = false;
        }
    }
}

void addEdge(int u, int v, int w) {
    nxt[cnt] = head[u];
    to_node[cnt] = v;
    weight[cnt] = w;
    head[u] = cnt++;
}

void push_heap(int u, int t, int c) {
    heap_arr[heapSize] = {u, t, c};
    int i = heapSize++;
    while (heap_arr[i].cost < heap_arr[(i - 1) / 2].cost) {
        swap(heap_arr[i], heap_arr[(i - 1) / 2]);
        i = (i - 1) / 2;
    }
}

void heapify(int i) {
    int l = i * 2 + 1;
    while (l < heapSize) {
        int best = l + 1 < heapSize && heap_arr[l + 1].cost < heap_arr[l].cost ? l + 1 : l;
        best = heap_arr[best].cost < heap_arr[i].cost ? best : i;
        if (best == i) {
            break;
        }
        swap(heap_arr[best], heap_arr[i]);
        i = best;
        l = i * 2 + 1;
    }
}

void pop_heap() {
    curr_u = heap_arr[0].u;
    curr_use = heap_arr[0].use;
    curr_cost = heap_arr[0].cost;
    swap(heap_arr[0], heap_arr[--heapSize]);
    heapify(0);
}

int dijkstra() {
    dist[s][0] = 0;
    push_heap(s, 0, 0);
    
    while (heapSize > 0) {
        pop_heap();
        if (visited[curr_u][curr_use]) {
            continue;
        }
        visited[curr_u][curr_use] = true;
        
        if (curr_u == t) {
            // 常见剪枝
            // 发现终点直接返回
            // 不用等都结束
            return curr_cost;
        }
        
        for (int ei = head[curr_u]; ei > 0; ei = nxt[ei]) {
            int v = to_node[ei];
            int w = weight[ei];
            
            if (curr_use < k && dist[v][curr_use + 1] > dist[curr_u][curr_use]) {
                // 使用免费
                dist[v][curr_use + 1] = dist[curr_u][curr_use];
                push_heap(v, curr_use + 1, dist[v][curr_use + 1]);
            }
            if (dist[v][curr_use] > dist[curr_u][curr_use] + w) {
                // 不用免费
                dist[v][curr_use] = dist[curr_u][curr_use] + w;
                push_heap(v, curr_use, dist[v][curr_use]);
            }
        }
    }
    return -1;
}

int main() {
    // C++ 的快速 I/O 模板，极大提升读写效率
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    // 持续读取输入，直到 EOF
    while (cin >> n >> m >> k >> s >> t) {
        build();
        for (int i = 0, a, b, c; i < m; i++) {
            cin >> a >> b >> c;
            addEdge(a, b, c);
            addEdge(b, a, c);
        }
        cout << dijkstra() << "\n";
    }

    return 0;
}
