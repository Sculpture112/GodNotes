```cpp
// 059 Code01 图的三种建立方式
// 核心重点：链式前向星。不仅省空间，而且访问极其丝滑，是手写图论算法的首选。

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 假设点的最大数量和边的最大数量
const int MAXN = 100;
const int MAXM = 200;

// 1. 邻接矩阵方式
int graphMatrix[MAXN][MAXN];

// 2. 邻接表方式 (C++ 风格)
struct Edge {
    int to;
    int weight;
};
vector<Edge> graphList[MAXN];

// 3. 链式前向星方式 (最快，左神最推崇)
int head[MAXN];
int nxt[MAXM]; // next是C++关键字，通常写成nxt或nxe
int to[MAXM];
int weight[MAXM];
int cnt; // 当前边的编号

// 初始化
void build(int n) {
    // 矩阵清空
    for (int i = 0; i <= n; i++) {
        for (int j = 0; j <= n; j++) graphMatrix[i][j] = 0;
        graphList[i].clear();
    }
    // 链式前向星清空
    cnt = 1;
    fill(head, head + n + 1, 0);
}

// 链式前向星加边 (有向边)
void addEdge(int u, int v, int w) {
    to[cnt] = v;
    weight[cnt] = w;
    nxt[cnt] = head[u]; // 当前边的下一条边指向旧的head
    head[u] = cnt++;    // 更新head为当前边的编号
}

// 遍历链式前向星
void traverse(int n) {
    for (int i = 1; i <= n; i++) {
        cout << i << " (邻居, 边权): ";
        for (int ei = head[i]; ei > 0; ei = nxt[ei]) {
            cout << "(" << to[ei] << ", " << weight[ei] << ") ";
        }
        cout << endl;
    }
}

int main() {
    int n = 4;
    build(n);
    // 假设有一组边 {u, v, w}
    int edges[2][3] = {{2, 2, 10}, {1, 3, 20}};
    
    for(auto& edge : edges) {
        addEdge(edge[0], edge[1], edge[2]);
    }

    cout << "--- 链式前向星遍历结果 ---" << endl;
    traverse(n);

    return 0;
}
```
