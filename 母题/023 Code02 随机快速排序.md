// 023 Code02 随机快速排序，填函数练习风格

// 测试链接 : https://leetcode.cn/problems/sort-an-array/
// 核心思路：随机快速排序。
// 针对算法题，我为你保留了“一题多解”的完整实现。这不仅能应对不同场景，也是面试中展示扎实基本功的加分项：
// 
// 解法一：经典随机快排（双路划分）
// - 划分规则：只区分 <=x 和 >x。
// - 缺点：当数组中存在大量重复元素时，划分极其不均，时间复杂度退化至 O(N^2)，容易在 LeetCode 上导致超时或爆栈。
//
// 解法二：改进版随机快排（三路划分 / 荷兰国旗问题）- 强烈推荐！
// - 划分规则：区分 <x、==x 和 >x。
// - 优点：一次性锁定所有等于枢轴 x 的元素，大幅缩小下一轮递归的区间，完美解决大量重复数据导致的性能退化。
// - 时间复杂度 O(N log N)，额外空间复杂度 O(log N)（递归栈深度）

#include <vector>
#include <cstdlib>
#include <ctime>

using namespace std;

class Solution {
public:
    vector<int> sortArray(vector<int>& nums) {
        // 设置随机种子，保证每次运行随机性
        srand(time(nullptr));
        if (nums.size() > 1) {
            // 你可以自由切换 quickSort1 或 quickSort2 进行测试
            // 推荐使用改进版的 quickSort2
            quickSort2(nums, 0, nums.size() - 1);
        }
        return nums;
    }

private:
    // ==========================================
    // 解法一：随机快速排序经典版（双路划分）
    // ==========================================
    void quickSort1(vector<int>& arr, int l, int r) {
        if (l >= r) {
            return;
        }
        // 随机选取枢轴
        int x = arr[l + rand() % (r - l + 1)];
        int mid = partition1(arr, l, r, x);
        quickSort1(arr, l, mid - 1);
        quickSort1(arr, mid + 1, r);
    }

    // 已知 arr[l....r] 范围上一定有 x 这个值
    // 划分数组 <=x 放左边，>x 放右边
    int partition1(vector<int>& arr, int l, int r, int x) {
        int a = l, xi = 0;
        for (int i = l; i <= r; i++) {
            if (arr[i] <= x) {
                swap(arr[a], arr[i]);
                if (arr[a] == x) {
                    xi = a;
                }
                a++;
            }
        }
        // 确保划分完成后 <=x 区域的最后一个数字是 x
        swap(arr[xi], arr[a - 1]);
        return a - 1;
    }

    // ==========================================
    // 解法二：随机快速排序改进版（三路划分）
    // ==========================================
    void quickSort2(vector<int>& arr, int l, int r) {
        if (l >= r) {
            return;
        }
        // 随机选取枢轴，保证概率上收敛到 O(N log N)
        int x = arr[l + rand() % (r - l + 1)];
        
        // C++ 中使用 std::pair 完美替代原版中的全局变量 first 和 last
        pair<int, int> bounds = partition2(arr, l, r, x);
        
        quickSort2(arr, l, bounds.first - 1);
        quickSort2(arr, bounds.second + 1, r);
    }

    // 荷兰国旗问题
    // 划分数组 <x 放左边，==x 放中间，>x 放右边
    // 返回 ==x 区域的左右边界
    pair<int, int> partition2(vector<int>& arr, int l, int r, int x) {
        int first = l;
        int last = r;
        int i = l;
        while (i <= last) {
            if (arr[i] == x) {
                i++;
            } else if (arr[i] < x) {
                swap(arr[first++], arr[i++]);
            } else {
                swap(arr[i], arr[last--]);
            }
        }
        return {first, last};
    }
};
