```cpp
// 022 Code02 翻转对数量

// 测试链接 : https://leetcode.cn/problems/reverse-pairs/
// 核心思路：分治法（归并排序的经典变种）。
// 1. 求翻转对数量（i < j 且 nums[i] > 2 * nums[j]），可以利用归并排序的过程，在合并两个有序子数组时顺手完成统计。
// 2. 将数组递归地分为左右两半。当左右两部分都各自内部有序后，对于左组的每一个元素 arr[i]，由于右组也是升序的，满足 arr[i] > 2 * arr[j] 的指针 j 会单调向右移动，不会回退。
// 3. 因此，在正常合并（merge）之前，先用双指针遍历一次，统计出跨越左右两部分的翻转对数量。
// 4. 统计时需要特别注意乘法溢出问题，必须将比较元素强制转换为 64 位整型（long long）。
// 5. 统计完毕后，执行标准的归并排序合并操作，使得当前区间整体变有序，供上一层继续使用。
// 时间复杂度 O(N log N)，额外空间复杂度 O(N)

#include <vector>

using namespace std;

class Solution {
private:
    int merge(vector<int>& arr, vector<int>& help, int l, int m, int r) {
        int ans = 0;
        
        // 1. 核心统计部分
        // 利用双指针单调不退回的特性，时间复杂度 O(N)
        for (int i = l, j = m + 1; i <= m; i++) {
            // 防溢出：强制转换为 long long 后再进行比较
            while (j <= r && (long long)arr[i] > (long long)arr[j] * 2) {
                j++;
            }
            // 此时 j - 1 是满足条件的最远右端点，数量为 j - (m + 1)
            ans += j - m - 1;
        }
        
        // 2. 正常 merge 排序部分
        int idx = l;
        int a = l;
        int b = m + 1;
        while (a <= m && b <= r) {
            help[idx++] = arr[a] <= arr[b] ? arr[a++] : arr[b++];
        }
        while (a <= m) {
            help[idx++] = arr[a++];
        }
        while (b <= r) {
            help[idx++] = arr[b++];
        }
        
        // 写回原数组
        for (idx = l; idx <= r; idx++) {
            arr[idx] = help[idx];
        }
        
        return ans;
    }

    // 统计 l...r 范围上翻转对的数量，同时将范围排好序
    int counts(vector<int>& arr, vector<int>& help, int l, int r) {
        if (l == r) {
            return 0;
        }
        int m = l + (r - l) / 2;
        // 答案 = 左侧内部翻转对 + 右侧内部翻转对 + 跨越左右的翻转对
        return counts(arr, help, l, m) + counts(arr, help, m + 1, r) + merge(arr, help, l, m, r);
    }

public:
    int reversePairs(vector<int>& nums) {
        if (nums.empty()) return 0;
        int n = nums.size();
        
        // 局部开辟辅助数组，交由 C++ 自动管理生命周期，避免全局变量污染
        vector<int> help(n);
        return counts(nums, help, 0, n - 1);
    }
};
```
