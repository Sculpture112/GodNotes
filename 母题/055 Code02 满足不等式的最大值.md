```cpp
// 055 Code02 满足不等式的最大值

// 测试链接 : https://leetcode.cn/problems/max-value-of-equation/
// 核心思路：等式转换 + 单调队列。
// 1. 题目要求最大化 yi + yj + |xi - xj|。因为已知点是按 x 排序的，所以当 i < j 时，xi < xj，等式可转化为：(xj + yj) + (yi - xi)。
// 2. 对于当前遍历到的点 j (xj, yj)，(xj + yj) 是固定的。我们需要在它前面找一个满足 (xj - xi) <= k 的点 i，使得 (yi - xi) 尽可能大。
// 3. 维护一个单调递减的双端队列。队列中存储点的坐标信息，队头始终保持着当前窗口内 (yi - xi) 的最大值。
// 4. 遍历每个点：
//    - 首先弹出队头所有不满足距离限制（xi + k < xj）的过期点。
//    - 若队列不为空，则用队头点与当前点计算指标并更新全局最大值。
//    - 在将当前点入队前，弹出队尾所有 (yi - xi) 小于等于当前点 (yj - xj) 的点，以维持队列单调递减的特性。
// 时间复杂度 O(N)，额外空间复杂度 O(N)

#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

class Solution {
public:
    int findMaxValueOfEquation(vector<vector<int>>& points, int k) {
        int n = points.size();
        
        // deque: 模拟双端队列，存储点的结构 pair<x, y>
        // C++ 使用 std::pair 既能满足存储两项数据的需求，内存局部性也更好
        // 局部开辟空间，随生命周期自动回收
        vector<pair<int, int>> deque(n);
        int h = 0, t = 0; // 头尾指针
        int ans = INT_MIN;
        
        for (int i = 0; i < n; i++) {
            int x = points[i][0];
            int y = points[i][1];
            
            // 1. 检查队头，若队头点的 x 与当前点 x 的距离超过了 k，过期淘汰
            while (h < t && deque[h].first + k < x) {
                h++;
            }
            
            // 2. 此时队头就是距离合法且 yi - xi 最大的点，收集答案
            if (h < t) {
                ans = max(ans, x + y + deque[h].second - deque[h].first);
            }
            
            // 3. 维护单调递减队列（按照 yi - xi 从大到小排列）
            // 若队尾点的 yi - xi <= 当前点的 y - x，队尾淘汰
            while (h < t && deque[t - 1].second - deque[t - 1].first <= y - x) {
                t--;
            }
            
            // 4. 当前点入队尾
            deque[t].first = x;
            deque[t++].second = y;
        }
        
        return ans;
    }
};
```
