**核心套路:** 
- **[降维打击 / 前后缀分解（空间换时间）]**：
    
    处理 3 个（或多个）无重叠子数组组合求最值的问题时，直接暴力枚举会导致 $O(N^3)$ 甚至更高的复杂度。这里的核心套路是“固定中间，预处理两端”。
    
    - 第一步：利用滑动窗口或前缀和，把所有长度为 $k$ 的子数组的“块总和”求出来。
        
    - 第二步：左右开弓。从左向右扫一遍预处理出 `prefix` 数组（代表 $0 \dots i$ 范围内最大的子数组起始点）；从右向左扫一遍预处理出 `suffix` 数组（代表 $i \dots N-1$ 范围内最大的子数组起始点）。
        
    - 第三步：只用一个指针遍历“中间子数组”的合法位置，左边挑 `prefix` 的最优解，右边挑 `suffix` 的最优解，三者相加即可在 $O(N)$ 时间内找到全局最优。
        
- **[状态传递 / 同步滑动窗口（空间压缩）]**：
    
    如果面试官要求 $O(1)$ 空间，则需要将问题抽象为“流水线上的状态机”。
    
    - 维护三个紧挨着的、长度同为 $k$ 的滑动窗口，让它们同步向右滑动。
        
    - 状态传递逻辑：`最优的窗口1` 决定 `最优的组合1+2`，`最优的组合1+2` 决定 `最优的最终组合1+2+3`。
        
    - 每次滑动时，用当前的新值去挑战历史最大值，只要严格大于（`>`）就更新记录并保存坐标，这样顺位滑到数组结尾时，手里攥着的一定是字典序最小的全局最优解。


```cpp
// 071 Code04 三个无重叠子数组的最大和

// 给你一个整数数组 nums 和一个整数 k
// 找出三个长度为 k 、互不重叠、且全部数字和（3 * k 项）最大的子数组
// 并返回这三个子数组
// 以下标的数组形式返回结果，数组中的每一项分别指示每个子数组的起始位置
// 如果有多个结果，返回字典序最小的一个
// 测试链接 : https://leetcode.cn/problems/maximum-sum-of-3-non-overlapping-subarrays/

#include <iostream>
#include <vector>

using namespace std;

class Solution {
public:
    vector<int> maxSumOfThreeSubarrays(const vector<int>& nums, int k) {
        int n = nums.size();
        
        // sums[i] : 以i开头并且长度为k的子数组的累加和
        vector<int> sums(n, 0);
        for (int l = 0, r = 0, sum = 0; r < n; r++) {
            // l....r
            sum += nums[r];
            if (r - l + 1 == k) {
                sums[l] = sum;
                sum -= nums[l];
                l++;
            }
        }
        
        // prefix[i] :
        // 0~i范围上所有长度为k的子数组中，拥有最大累加和的子数组，是以什么位置开头的
        vector<int> prefix(n, 0);
        for (int l = 1, r = k; r < n; l++, r++) {
            if (sums[l] > sums[prefix[r - 1]]) {
                // 注意 >，为了同样最大累加和的情况下，保证最小的字典序（越靠左越好）
                prefix[r] = l;
            } else {
                prefix[r] = prefix[r - 1];
            }
        }
        
        // suffix[i] :
        // i~n-1范围上所有长度为k的子数组中，拥有最大累加和的子数组，是以什么位置开头的
        vector<int> suffix(n, 0);
        suffix[n - k] = n - k;
        for (int l = n - k - 1; l >= 0; l--) {
            if (sums[l] >= sums[suffix[l + 1]]) {
                // 注意 >=，由于是从右往左遍历，当相等时，当前的 l 更靠左，所以要更新以保证字典序最小
                suffix[l] = l;
            } else {
                suffix[l] = suffix[l + 1];
            }
        }
        
        int a = 0, b = 0, c = 0, max_sum = 0;
        // 0...i-1    i...j    j+1...n-1
        //   左     中(长度为k)     右
        for (int i = k, j = 2 * k - 1; j < n - k; i++, j++) {
            // 0.....i-1   i.....j  j+1.....n-1
            // 最好开头p      i开头     最好开头s
            int p = prefix[i - 1];
            int s = suffix[j + 1];
            int sum = sums[p] + sums[i] + sums[s];
            
            if (sum > max_sum) {
                // 注意 >，为了同样最大累加和的情况下，保证最小的字典序
                max_sum = sum;
                a = p;
                b = i;
                c = s;
            }
        }
        
        return {a, b, c};
    }
};
```
