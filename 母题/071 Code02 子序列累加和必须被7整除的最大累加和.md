**核心套路:** 
- **将“余数”作为状态的独立维度**
    
    当题目出现“子序列和必须被 $K$ 整除”或“模 $K$ 等于某数”的限制时，标准套路是把 $0$ 到 $K-1$ 的所有余数作为 DP 数组的一维。
    
    在这道题里就是 `dp[i][j]` 中的 `j`，它代表考虑前 `i` 个数时，累加和模 $7$ 余 `j` 的最大值。这样就把无限的累加和压缩到了 $7$ 个有限的状态中。
    
- **无效状态的初始化与拦截**
    
    - **初始化**：必须明确区分“和为 0”与“不存在”。除了 `dp[0][0] = 0`（0个数和为0，余数也是0，是合法起点）之外，其他 `dp[0][1...6]` 必须初始化为 `-1`（或 $-\infty$），表示该状态**不可达**。
        
    - **拦截**：在进行状态转移时，必须加一层判断 `if (dp[i - 1][need] != -1)`，防止把无效的、根本不存在的子序列拉过来强行加上当前数字。
        
- **利用同余性质逆推“前置依赖状态”（核心转移方程）**
    
    当我们决定把当前数字 `x` 加入子序列时，当前数字自带的余数是 `cur = x % 7`。为了让加上 `x` 后的总和余数为 `j`，我们需要算出上一层应该提供什么余数（`need`）。
    
    - 逆推公式套路：**`need = (j - cur + K) % K`**（此题 $K=7$）。
        
    - _注意：公式里加上 $K$ 是为了抵消 `j - cur` 可能产生的负数，防止 C++ 中负数取模产生错误结果。_ 算出合法的 `need` 后，再去上一层取 `dp[i - 1][need]` 进行状态更新。

```cpp
// 071 Code02 子序列累加和必须被7整除的最大累加和

// 给定一个非负数组nums，
// 可以任意选择数字组成子序列，但是子序列的累加和必须被7整除
// 返回最大累加和
// 对数器验证

#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdlib>
#include <ctime>

using namespace std;

class Solution {
public:
    // 暴力方法中的辅助递归函数
    int f(const vector<int>& nums, int i, int s) {
        if (i == nums.size()) {
            return s % 7 == 0 ? s : 0;
        }
        return max(f(nums, i + 1, s), f(nums, i + 1, s + nums[i]));
    }

    // 暴力方法
    // 为了验证
    int maxSum1(const vector<int>& nums) {
        // nums形成的所有子序列的累加和都求出来
        // 其中 % 7 == 0 的那些累加和中，返回最大的
        // 就是 f 函数的功能
        return f(nums, 0, 0);
    }

    // 正式方法
    // 时间复杂度O(n)
    int maxSum2(const vector<int>& nums) {
        int n = nums.size();
        // dp[i][j] : nums[0...i-1]
        // nums前i个数形成的子序列一定要做到，子序列累加和 % 7 == j
        // 这样的子序列最大累加和是多少
        // 注意 : dp[i][j] == -1 代表不存在这样的子序列
        vector<vector<int>> dp(n + 1, vector<int>(7, -1));
        dp[0][0] = 0;
        
        for (int i = 1; i <= n; i++) {
            int x = nums[i - 1];
            int cur = x % 7;
            for (int j = 0; j < 7; j++) {
                dp[i][j] = dp[i - 1][j];
                // 求 need 是核心
                // 或者写成 need = cur <= j ? (j - cur) : (j - cur + 7);
                int need = (7 + j - cur) % 7;
                
                if (dp[i - 1][need] != -1) {
                    dp[i][j] = max(dp[i][j], dp[i - 1][need] + x);
                }
            }
        }
        return dp[n][0];
    }
};

// 为了测试
// 生成随机数组
vector<int> randomArray(int n, int v) {
    vector<int> ans(n);
    for (int i = 0; i < n; i++) {
        ans[i] = rand() % v;
    }
    return ans;
}

// 为了测试
// 对数器
int main() {
    srand(time(nullptr)); // 初始化随机数种子

    int n = 15;
    int v = 30;
    int testTime = 20000;
    
    cout << "测试开始\n";
    Solution sol;
    for (int i = 0; i < testTime; i++) {
        int len = (rand() % n) + 1;
        vector<int> nums = randomArray(len, v);
        
        int ans1 = sol.maxSum1(nums);
        int ans2 = sol.maxSum2(nums);
        
        if (ans1 != ans2) {
            cout << "出错了!\n";
            break;
        }
    }
    cout << "测试结束\n";

    return 0;
}
```
