// 074 Code01 分组背包(模版)

// 给定一个正数 m 表示背包的容量，有 n 个货物可供挑选
// 每个货物有自己的体积(容量消耗)、价值(获得收益)、组号(分组)
// 同一个组的物品只能挑选 1 件，所有挑选物品的体积总和不能超过背包容量
// 怎么挑选货物能达到价值最大，返回最大的价值
// 测试链接 : https://www.luogu.com.cn/problem/P1757

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 根据题目数据范围设置最大常量
const int MAXN = 1005;
const int MAXM = 1005;

struct Item {
    int weight;
    int value;
    int team;
};

Item items[MAXN];
int dp[MAXM];
int m, n;

/**
 * 方案一：严格位置依赖的动态规划 (二维 DP)
 * 时间复杂度 O(n * m)，空间复杂度 O(组数 * m)
 */
int compute1(int teams_count) {
    // dp[i][j] 表示前 i 组物品，在容量不超过 j 的情况下的最大价值
    // 组的编号从 1 到 teams_count
    vector<vector<int>> dp_table(teams_count + 1, vector<int>(m + 1, 0));

    // 处理分组：按照排好序的 items 数组进行逻辑分组
    for (int start = 1, end = 2, i = 1; start <= n; i++) {
        while (end <= n && items[end].team == items[start].team) {
            end++;
        }
        // 当前组的范围是 [start, end-1]
        for (int j = 0; j <= m; j++) {
            // 可能性 1：当前组 i 一个物品都不选，继承上一组的结果
            dp_table[i][j] = dp_table[i - 1][j];
            
            // 可能性 2：在当前组 [start, end-1] 中挑选一个物品
            for (int k = start; k < end; k++) {
                if (j >= items[k].weight) {
                    dp_table[i][j] = max(dp_table[i][j], dp_table[i - 1][j - items[k].weight] + items[k].value);
                }
            }
        }
        start = end++;
    }
    return dp_table[teams_count][m];
}

/**
 * 方案二：空间压缩版 (一维 DP)
 * 时间复杂度 O(n * m)，空间复杂度 O(m)
 */
int compute2() {
    // 初始化 DP 数组，dp[0...m] = 0
    for (int i = 0; i <= m; i++) dp[i] = 0;

    for (int start = 1, end = 2; start <= n; ) {
        // 找到当前组的边界 [start, end-1]
        while (end <= n && items[end].team == items[start].team) {
            end++;
        }
        
        // 分组背包的空间压缩核心：容量 j 必须从大到小遍历
        // 这样在计算 dp[j] 时，dp[j - items[k].weight] 存储的还是“上一组”的结果
        // 从而保证了“每组只选一个”的限制
        for (int j = m; j >= 0; j--) {
            for (int k = start; k < end; k++) {
                if (j >= items[k].weight) {
                    dp[j] = max(dp[j], items[k].value + dp[j - items[k].weight]);
                }
            }
        }
        start = end++;
    }
    return dp[m];
}

int main() {
    // 提升 I/O 效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> m >> n) {
        for (int i = 1; i <= n; i++) {
            cin >> items[i].weight >> items[i].value >> items[i].team;
        }

        // 关键：按照组号进行排序，方便后续按组处理
        sort(items + 1, items + n + 1, [](const Item& a, const Item& b) {
            return a.team < b.team;
        });

        // 计算总组数 (用于 compute1)
        int teams_count = 0;
        if (n > 0) {
            teams_count = 1;
            for (int i = 2; i <= n; i++) {
                if (items[i].team != items[i - 1].team) {
                    teams_count++;
                }
            }
        }

        // 通常提交空间压缩版本
        cout << compute2() << "\n";
    }

    return 0;
}
