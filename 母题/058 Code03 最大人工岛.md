```cpp
// 058 Code03 最大人工岛

// 测试链接 : https://leetcode.cn/problems/making-a-large-island/
// 核心思路：洪水填充（Flood Fill）/ DFS + 连通块合并。
// 1. 编号：第一遍遍历网格，使用 DFS 发现所有岛屿，并将每个岛屿的陆地格子标记为唯一的编号（从 2 开始，避免与 0 和 1 混淆）。
// 2. 测算：再次遍历网格，统计每个编号岛屿的面积，记录在 sizes 数组中，并同步记录当前已有的最大岛屿面积（应对全 1 无法填海的情况）。
// 3. 填海：遍历网格中所有的 '0'（海洋）。对于每个 '0'，检查其上下左右四个方向相邻的岛屿编号。
// 4. 合并：将相邻且不重复的岛屿面积相加，再加上当前变成 '1' 的这一格，即为填海后形成的新岛屿面积。用这个新面积去挑战全局最大值。
// 时间复杂度 O(N * M)，额外空间复杂度 O(N * M)（最坏情况下递归栈的深度及 sizes 数组开销）

#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    int largestIsland(vector<vector<int>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        int id = 2; // 岛屿编号从 2 开始
        
        // 1. 遍历所有陆地，标记属于哪个岛屿
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    dfs(grid, n, m, i, j, id++);
                }
            }
        }
        
        // 2. 统计每个岛屿的面积
        // id 此时代表一共分配到了哪个编号，直接开辟 id 大小的空间即可
        vector<int> sizes(id, 0);
        int ans = 0;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] > 1) {
                    ans = max(ans, ++sizes[grid[i][j]]);
                }
            }
        }
        
        // 3. 尝试填海造陆，将每个 0 变成 1
        // visited 数组用于去重，防止把同一个岛屿的面积加两次
        vector<bool> visited(id, false);
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 0) {
                    // 获取上下左右相邻格子的岛屿编号（越界或海洋则视为编号 0）
                    int up = (i > 0) ? grid[i - 1][j] : 0;
                    int down = (i + 1 < n) ? grid[i + 1][j] : 0;
                    int left = (j > 0) ? grid[i][j - 1] : 0;
                    int right = (j + 1 < m) ? grid[i][j + 1] : 0;
                    
                    // 当前海洋格填平，贡献 1 点面积
                    int merge = 1; 
                    
                    // 将上面的岛屿面积加入（0 号代表无岛屿，sizes[0] 就是 0）
                    visited[up] = true;
                    merge += sizes[up];
                    
                    // 如果下面的岛屿和上面不是同一个，加入其面积
                    if (!visited[down]) {
                        merge += sizes[down];
                        visited[down] = true;
                    }
                    if (!visited[left]) {
                        merge += sizes[left];
                        visited[left] = true;
                    }
                    if (!visited[right]) {
                        merge += sizes[right];
                        visited[right] = true;
                    }
                    
                    // 挑战最大值
                    ans = max(ans, merge);
                    
                    // 恢复现场，供下一个 0 的判断使用
                    visited[up] = false;
                    visited[down] = false;
                    visited[left] = false;
                    visited[right] = false;
                }
            }
        }
        
        return ans;
    }

private:
    void dfs(vector<vector<int>>& grid, int n, int m, int i, int j, int id) {
        if (i < 0 || i == n || j < 0 || j == m || grid[i][j] != 1) {
            return;
        }
        // 将属于该岛屿的格子全部标记为当前 id
        grid[i][j] = id;
        dfs(grid, n, m, i - 1, j, id); // 上
        dfs(grid, n, m, i + 1, j, id); // 下
        dfs(grid, n, m, i, j - 1, id); // 左
        dfs(grid, n, m, i, j + 1, id); // 右
    }
};
```
