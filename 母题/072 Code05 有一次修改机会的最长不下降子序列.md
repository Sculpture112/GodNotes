// 072 Code05 有一次修改机会的最长不下降子序列

// 给定一个长度为 n 的数组 arr，和一个整数 k
// 只有一次机会可以将其中连续的 k 个数全修改成任意一个值
// 返回最长不下降子序列长度
// 测试链接 : https://www.luogu.com.cn/problem/P8776

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 根据题目范围设置最大空间
const int MAXN = 100005;
int arr[MAXN];
int right_len[MAXN];
int ends_buf[MAXN];
int n, k;

/**
 * 求最长不上升子序列长度的二分搜索
 * ends_buf[0...len-1] 是降序的，找到 < num 的最左位置
 */
int bs1(int len, int num) {
    int l = 0, r = len - 1, m, ans = -1;
    while (l <= r) {
        m = l + ((r - l) >> 1);
        if (ends_buf[m] < num) {
            ans = m;
            r = m - 1;
        } else {
            l = m + 1;
        }
    }
    return ans;
}

/**
 * 求最长不下降子序列长度的二分搜索
 * ends_buf[0...len-1] 是升序的，找到 > num 的最左位置
 */
int bs2(int len, int num) {
    int l = 0, r = len - 1, m, ans = -1;
    while (l <= r) {
        m = l + ((r - l) >> 1);
        if (ends_buf[m] > num) {
            ans = m;
            r = m - 1;
        } else {
            l = m + 1;
        }
    }
    return ans;
}

/**
 * 生成辅助数组 right_len
 * right_len[j] 表示：在 arr[j...n-1] 范围内，且必须以 arr[j] 开头的最长不下降子序列长度
 * 算法逻辑：从右往左遍历，求以 arr[i] 结尾的最长不上升子序列
 */
void calculate_right() {
    int len = 0;
    for (int i = n - 1; i >= 0; i--) {
        int find = bs1(len, arr[i]);
        if (find == -1) {
            ends_buf[len++] = arr[i];
            right_len[i] = len;
        } else {
            ends_buf[find] = arr[i];
            right_len[i] = find + 1;
        }
    }
}
	
/**
 * 核心计算逻辑：
 * 假设我们将 arr[i+1 ... i+k] 这一段连续 k 个数修改
 * 那么总长度 = (左侧子序列长度，且结尾数值 <= arr[i+k+1]) + k + (以 arr[i+k+1] 开头的右侧子序列长度)
 */
int compute() {
    calculate_right();
    int len = 0;
    int ans = 0;
    
    // 清空 ends_buf 准备计算左侧 LIS 结构
    for (int i = 0; i < n; i++) ends_buf[i] = 0;
    
    // j 是修改区间后的第一个元素的下标
    // i 是待加入左侧 LIS 结构的元素下标
    for (int i = 0, j = k; j < n; i++, j++) {
        // 在左侧已处理的序列中，找到数值 <= arr[j] 的元素个数
        int find = bs2(len, arr[j]);
        int left_part = (find == -1) ? len : find;
        
        ans = max(ans, left_part + k + right_len[j]);
        
        // 将 arr[i] 放入左侧 LIS 结构，为下一轮滑动窗口做准备
        find = bs2(len, arr[i]);
        if (find == -1) {
            ends_buf[len++] = arr[i];
        } else {
            ends_buf[find] = arr[i];
        }
    }
    
    // 考虑修改区间在数组最末尾的情况
    ans = max(ans, len + k);
    return ans;
}

int main() {
    // 提升 I/O 效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n >> k) {
        for (int i = 0; i < n; i++) {
            cin >> arr[i];
        }
        
        if (k >= n) {
            cout << n << "\n";
        } else {
            cout << compute() << "\n";
        }
    }

    return 0;
}
