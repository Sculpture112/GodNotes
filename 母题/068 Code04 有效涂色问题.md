// 068 Code04 有效涂色问题

// 给定n、m两个参数
// 一共有n个格子，每个格子可以涂上一种颜色，颜色在m种里选
// 当涂满n个格子，并且m种颜色都使用了，叫一种有效方法
// 求一共有多少种有效的涂色方法
// 1 <= n, m <= 5000
// 结果比较大请 % 1000000007 之后返回
// 对数器验证

#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>

using namespace std;

const int MAXN = 5005;
const int MOD = 1000000007;

// 动态规划使用的全局数组
int dp[MAXN][MAXN];

// 暴力方法中的核心递归
// 把所有填色的方法暴力枚举
// 然后一个一个验证是否有效
// 这是一个带路径的递归，无法改成动态规划
int f(vector<int>& path, vector<bool>& set, int i, int n, int m) {
    if (i == n) {
        fill(set.begin(), set.end(), false);
        int colors = 0;
        for (int c : path) {
            if (!set[c]) {
                set[c] = true;
                colors++; 
            }
        }
        return colors == m ? 1 : 0;
    } else {
        int ans = 0;
        for (int j = 1; j <= m; j++) {
            path[i] = j;
            ans += f(path, set, i + 1, n, m);
        }
        return ans;
    }
}

// 暴力方法
// 为了验证
int ways1(int n, int m) {
    vector<int> path(n, 0);
    vector<bool> set(m + 1, false);
    return f(path, set, 0, n, m);
}

// 正式方法
// 时间复杂度 O(n * m)
int ways2(int n, int m) {
    // 为了防止多次调用时全局数组存在脏数据，这里对其边界进行初始化
    // 实际竞赛中如果单次用例只需跑到 n, m，可以省略全盘 memset 以提高速度
    for (int j = 1; j <= m; j++) {
        dp[0][j] = 0;
        dp[1][j] = 0; // 虽然下面会覆盖 dp[i][1]，但清理一下更安全
    }

    // dp[i][j]:
    // 一共有 m 种颜色
    // 前 i 个格子涂满 j 种颜色的方法数
    for (int i = 1; i <= n; i++) {
        dp[i][1] = m;
    }
    
    for (int i = 2; i <= n; i++) {
        for (int j = 2; j <= m; j++) {
            // C++ 中务必使用 1LL 强转为 long long，防止乘法溢出
            long long p1 = (1LL * dp[i - 1][j] * j) % MOD;
            long long p2 = (1LL * dp[i - 1][j - 1] * (m - j + 1)) % MOD;
            dp[i][j] = (p1 + p2) % MOD;
        }
    }
    return dp[n][m];
}

int main() {
    // 测试的数据量比较小
    // 那是因为数据量大了，暴力方法过不了
    // 但是这个数据量足够说明正式方法是正确的
    int N = 9;
    int M = 9;
    cout << "功能测试开始\n";
    for (int n = 1; n <= N; n++) {
        for (int m = 1; m <= M; m++) {
            int ans1 = ways1(n, m);
            int ans2 = ways2(n, m);
            if (ans1 != ans2) {
                cout << "出错了!\n";
            }
        }
    }
    cout << "功能测试结束\n\n";

    cout << "性能测试开始\n";
    int n = 5000;
    int m = 4877;
    cout << "n : " << n << "\n";
    cout << "m : " << m << "\n";
    
    auto start_time = chrono::high_resolution_clock::now();
    int ans = ways2(n, m);
    auto end_time = chrono::high_resolution_clock::now();
    
    auto duration = chrono::duration_cast<chrono::milliseconds>(end_time - start_time).count();
    
    cout << "取模之后的结果 : " << ans << "\n";
    cout << "运行时间 : " << duration << " 毫秒\n";
    cout << "性能测试结束\n";

    return 0;
}
