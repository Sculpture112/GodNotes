// 051 Code05 同时运行N台电脑的最长时间

// 你有 n 台电脑。给你整数 n 和一个下标从 0 开始的整数数组 batteries
// 其中第 i 个电池可以让一台电脑 运行 batteries[i] 分钟
// 你想使用这些电池让 全部 n 台电脑 同时 运行。
// 一开始，你可以给每台电脑连接 至多一个电池
// 然后在任意整数时刻，你都可以将一台电脑与它的电池断开连接，并连接另一个电池，你可以进行这个操作 任意次
// 请你返回你可以让 n 台电脑同时运行的 最长 分钟数。
// 测试链接 : https://leetcode.cn/problems/maximum-running-time-of-n-computers/

#include <vector>
#include <algorithm>
#include <numeric>

using namespace std;

class Solution {
public:
    /**
     * 二分答案法 + 贪心分析
     * 时间复杂度 O(n * log(max))，额外空间复杂度 O(1)
     */
    long long maxRunTime(int n, vector<int>& batteries) {
        long long sum = 0;
        int max_val = 0;
        for (int x : batteries) {
            sum += x;
            max_val = max(max_val, x);
        }

        // 核心贪心结论：
        // 如果所有电池的总电量 sum 能够支撑每台电脑运行的时间超过了最强电池的寿命 max_val，
        // 那么对于这个平均时间来说，所有电池都可以看作是可以无限拆分的“碎片电量”。
        // 此时，最大运行时间就是总电量除以电脑台数。
        if (sum >= (long long)max_val * n) {
            return sum / n;
        }

        // 如果不满足上述条件，说明最终的运行时间一定小于最大电池的电量（即 max_val）。
        // 此时在该范围内进行二分搜索。
        long long l = 0, r = max_val, ans = 0;
        while (l <= r) {
            long long mid = l + (r - l) / 2;
            if (check(batteries, n, mid)) {
                ans = mid;
                l = mid + 1;
            } else {
                r = mid - 1;
            }
        }
        return ans;
    }

private:
    /**
     * 判断让 n 台电脑共同运行 time 分钟是否可行
     * @param batteries 电池数组
     * @param n 电脑数量
     * @param time 目标运行时间
     */
    bool check(const vector<int>& batteries, int n, long long time) {
        long long fragment_sum = 0;
        for (int x : batteries) {
            if (x >= time) {
                // 如果电池电量大于等于目标时间，它能独立负责一台电脑
                n--;
            } else {
                // 如果电池电量小于目标时间，它作为“碎片电量”贡献给剩余电脑
                fragment_sum += x;
            }
            
            // 只要碎片电量总和能填补剩下所有电脑在 time 时间内的缺口，即为可行
            if (n <= 0 || fragment_sum >= (long long)n * time) {
                return true;
            }
        }
        return false;
    }
};
