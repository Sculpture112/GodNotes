// 054 Code01 滑动窗口最大值（单调队列经典用法模版）

// 测试链接 : https://leetcode.cn/problems/sliding-window-maximum/
// 核心思路：单调队列。维护一个大小为 k 的滑动窗口内的最大值。
// 1. 使用手写双端队列（数组加头尾指针）存储数组元素的索引，队列内对应的元素值从队头到队尾严格单调递减。
// 2. 先预处理前 k-1 个元素，构建初始状态的单调队列。
// 3. 窗口右边界 r 每次向右滑动一步：新元素入队前，将队尾所有小于等于新元素的索引弹出，保证队列的单调性，然后将 r 入队。
// 4. 收集当前窗口 [l, r] 的最大值，即为此时队头索引对应的值。
// 5. 检查队头索引是否已经滑出当前窗口（即队头索引是否等于窗口左边界 l），若是，则弹出队头（头指针 h++）。
// 时间复杂度 O(N)，额外空间复杂度 O(N)

#include <vector>

using namespace std;

class Solution {
public:
    vector<int> maxSlidingWindow(vector<int>& arr, int k) {
        int n = arr.size();
        // deque: 手写双端队列，存储索引
        // ans: 收集结果
        // 局部开辟空间，交由 C++ 自动管理
        vector<int> deque(n);
        int h = 0, t = 0; // 队头、队尾指针
        
        // 先形成长度为 k - 1 的窗口
        for (int i = 0; i < k - 1; i++) {
            // 维持队列从大到小的单调性
            while (h < t && arr[deque[t - 1]] <= arr[i]) {
                t--;
            }
            deque[t++] = i;
        }
        
        int m = n - k + 1;
        vector<int> ans(m);
        
        // 窗口开始滑动：[l, r] 为当前窗口的左右边界
        for (int l = 0, r = k - 1; l < m; l++, r++) {
            // 少一个元素，让 r 位置的数进来
            while (h < t && arr[deque[t - 1]] <= arr[r]) {
                t--;
            }
            deque[t++] = r;
            
            // 收集答案：队头保存的就是当前窗口最大值的索引
            ans[l] = arr[deque[h]];
            
            // 检查 l 位置的数是否过期，如果队头刚好是即将滑出的 l，则弹出队头
            if (deque[h] == l) {
                h++;
            }
        }
        
        return ans;
    }
};
