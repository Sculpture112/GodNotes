// 059 Code04 火星词典
// 测试链接 : https://leetcode.cn/problems/Jf1JuT/
// 核心思路：相邻单词对比建图 + 拓扑排序

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

class Solution {
public:
    string alienOrder(vector<string>& words) {
        // 1. 初始化入度表，-1 表示该字母未出现
        vector<int> indegree(26, -1);
        for (const string& w : words) {
            for (char c : w) indegree[c - 'a'] = 0;
        }

        // 2. 建图：比较相邻单词
        vector<vector<int>> graph(26);
        for (int i = 0; i < words.size() - 1; ++i) {
            string cur = words[i];
            string next = words[i + 1];
            int len = min(cur.length(), next.length());
            int j = 0;
            for (; j < len; ++j) {
                if (cur[j] != next[j]) {
                    graph[cur[j] - 'a'].push_back(next[j] - 'a');
                    indegree[next[j] - 'a']++;
                    break;
                }
            }
            // 边界检查：如果 cur 是 next 的前缀且 cur 更长，非法
            if (j == len && cur.length() > next.length()) return "";
        }

        // 3. 统计出现的字母种类并把入度为 0 的入队
        int kinds = 0;
        vector<int> q(26);
        int l = 0, r = 0;
        for (int i = 0; i < 26; ++i) {
            if (indegree[i] != -1) {
                kinds++;
                if (indegree[i] == 0) q[r++] = i;
            }
        }

        // 4. 拓扑排序过程
        string ans = "";
        while (l < r) {
            int cur = q[l++];
            ans += (char)(cur + 'a');
            for (int nxt : graph[cur]) {
                if (--indegree[nxt] == 0) {
                    q[r++] = nxt;
                }
            }
        }

        // 5. 校验结果长度，不相等说明有环
        return ans.length() == kinds ? ans : "";
    }
};
