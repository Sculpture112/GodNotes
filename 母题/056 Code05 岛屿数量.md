```cpp
// 056 Code05 岛屿数量

// 测试链接 : https://leetcode.cn/problems/number-of-islands/
// 核心思路：并查集。计算二维网格中相连陆地（'1'）形成的岛屿数量。
// 1. 将二维网格映射到一维：一维索引 index = i * cols + j。
// 2. 初始化：遍历整个网格，遇到 '1' 则将其视作一个独立的连通分量（sets++），其父节点指向自己。
// 3. 连通：再次遍历网格，对于每个 '1'，检查其左侧和上方相邻的格子是否也是 '1'。若是，则将它们所在的集合合并。
// 4. 每发生一次有效的合并（原本不在同一个集合），独立的岛屿数量 sets 就减 1。
// 5. 遍历结束后，剩下的 sets 也就是互不相连的岛屿总数。
// 时间复杂度 O(N * M)，额外空间复杂度 O(N * M)

#include <vector>
#include <functional>

using namespace std;

class Solution {
public:
    int numIslands(vector<vector<char>>& grid) {
        int n = grid.size();
        int m = grid[0].size();
        int sets = 0;
        
        // father: 并查集父节点数组，大小为二维网格的一维映射总长
        vector<int> father(n * m);
        
        // 1. 初始化并查集
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == '1') {
                    int index = i * m + j;
                    father[index] = index; // 陆地指向自己
                    sets++;                // 初始假设每个陆地格子都是孤岛
                }
            }
        }
        
        // Lambda 实现递归的路径压缩查找
        function<int(int)> find = [&](int i) {
            if (i != father[i]) {
                father[i] = find(father[i]);
            }
            return father[i];
        };
        
        // 极简合并操作
        auto unite = [&](int x, int y) {
            int fx = find(x);
            int fy = find(y);
            if (fx != fy) {
                father[fx] = fy;
                sets--; // 两个独立陆地块合并，岛屿总数减 1
            }
        };
        
        // 2. 遍历并连通陆地
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == '1') {
                    int currentIndex = i * m + j;
                    // 从上到下、从左到右遍历，所以只看左和上即可覆盖所有相邻连通情况
                    if (j > 0 && grid[i][j - 1] == '1') {
                        unite(currentIndex, currentIndex - 1);
                    }
                    if (i > 0 && grid[i - 1][j] == '1') {
                        unite(currentIndex, (i - 1) * m + j);
                    }
                }
            }
        }
        
        return sets;
    }
};
```
