```cpp
// 064 Code02 最小体力消耗路径

// 你准备参加一场远足活动。给你一个二维 rows x columns 的地图 heights
// 其中 heights[row][col] 表示格子 (row, col) 的高度
// 一开始你在最左上角的格子 (0, 0) ，且你希望去最右下角的格子 (rows-1, columns-1) 
// （注意下标从 0 开始编号）。你每次可以往 上，下，左，右 四个方向之一移动
// 你想要找到耗费 体力 最小的一条路径
// 一条路径耗费的体力值是路径上，相邻格子之间高度差绝对值的最大值
// 请你返回从左上角走到右下角的最小 体力消耗值
// 测试链接 ：https://leetcode.cn/problems/path-with-minimum-effort/

#include <iostream>
#include <vector>
#include <queue>
#include <cmath>
#include <algorithm>
#include <climits>

using namespace std;

// 0:上，1:右，2:下，3:左
// 利用长度为 5 的数组巧妙表示 4 个方向的坐标偏移
int mv[5] = { -1, 0, 1, 0, -1 };

// 定义一个结构体存放节点信息
struct Cell {
    int x;
    int y;
    int c; // 源点到当前格子的代价
    
    // 重载小于号。C++ 默认大根堆，为了实现小根堆，反转比较逻辑：代价大的反而优先级低
    bool operator<(const Cell& other) const {
        return c > other.c;
    }
};

class Solution {
public:
    int minimumEffortPath(vector<vector<int>>& heights) {
        // (0,0)源点
        // -> (x,y)
        int n = heights.size();
        int m = heights[0].size();
        
        vector<vector<int>> distance(n, vector<int>(m, INT_MAX));
        distance[0][0] = 0;
        
        vector<vector<bool>> visited(n, vector<bool>(m, false));
        
        // C++ 使用自定义的 Cell 结构体，因为内部重载了 <，直接声明即可变成小根堆
        priority_queue<Cell> heap;
        heap.push({0, 0, 0});
        
        while (!heap.empty()) {
            Cell record = heap.top();
            heap.pop();
            
            int x = record.x;
            int y = record.y;
            int c = record.c;
            
            if (visited[x][y]) {
                continue;
            }
            
            if (x == n - 1 && y == m - 1) {
                // 常见剪枝
                // 发现终点直接返回
                // 不用等都结束
                return c;
            }
            visited[x][y] = true;
            
            for (int i = 0; i < 4; i++) {
                int nx = x + mv[i];
                int ny = y + mv[i + 1];
                
                if (nx >= 0 && nx < n && ny >= 0 && ny < m && !visited[nx][ny]) {
                    int nc = max(c, abs(heights[x][y] - heights[nx][ny]));
                    if (nc < distance[nx][ny]) {
                        distance[nx][ny] = nc;
                        heap.push({nx, ny, nc});
                    }
                }
            }
        }
        
        return -1;
    }
};
```

---


**核心套路:** 
- **Dijkstra 变形（贪心扩展）**：将传统的最短路中的“路径权值累加 `dist[u] + w`” 修改为“取路径上的最大边权 `max(dist[u], w)`”。利用小根堆维护当前所有可达节点中的最小代价，首次弹出终点时，得到的必然是全局最优解（最小的最大耗费）。
    
- **二分答案 + 连通性检测（判定性问题转化）**：遇到“求最大值的最小化”问题，标准的通用解法是二分。二分枚举体力消耗的上限 `mid`，然后在图中只允许走高度差绝对值 `<= mid` 的边，利用 BFS 或 DFS 判断起点能否到达终点。
    
- **Kruskal 最小生成树思想（并查集）**：将所有相邻格子视为一条边，边的权重就是高度差的绝对值。将所有边按权重从小到大排序，依次加入并查集。当起点 `(0,0)` 和终点 `(n-1, m-1)` 在并查集中连通的那一刻，当前加入的边的权重就是答案。
