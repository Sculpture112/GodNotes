// 068 Code02 编辑距离

// 给你两个单词 word1 和 word2
// 请返回将 word1 转换成 word2 所使用的最少代价
// 你可以对一个单词进行如下三种操作：
// 插入一个字符，代价a
// 删除一个字符，代价b
// 替换一个字符，代价c
// 测试链接 : https://leetcode.cn/problems/edit-distance/

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <climits>

using namespace std;

class Solution {
public:
    // 已经展示太多次从递归到动态规划了
    // 直接写动态规划吧
    int minDistance(string word1, string word2) {
        // 三种方法任选其一均可
        // 这里以空间压缩版本为例，代价都是 1
        return editDistance3(word1, word2, 1, 1, 1);
    }

    // 原初尝试版
    // a : s1中插入1个字符的代价
    // b : s1中删除1个字符的代价
    // c : s1中改变1个字符的代价
    // 返回从s1转化成s2的最低代价
    int editDistance1(const string& s1, const string& s2, int a, int b, int c) {
        int n = s1.length();
        int m = s2.length();
        
        // dp[i][j] :
        // s1[前缀长度为i]想变成s2[前缀长度为j]，至少付出多少代价
        vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
        
        for (int i = 1; i <= n; i++) {
            dp[i][0] = i * b;
        }
        for (int j = 1; j <= m; j++) {
            dp[0][j] = j * a;
        }
        
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                int p1 = INT_MAX;
                if (s1[i - 1] == s2[j - 1]) {
                    p1 = dp[i - 1][j - 1];
                }
                
                int p2 = INT_MAX;
                if (s1[i - 1] != s2[j - 1]) {
                    p2 = dp[i - 1][j - 1] + c;
                }
                
                int p3 = dp[i][j - 1] + a;
                int p4 = dp[i - 1][j] + b;
                
                // C++11 支持使用初始化列表求多个值的最小值
                dp[i][j] = min({p1, p2, p3, p4});
            }
        }
        return dp[n][m];
    }

    // 枚举小优化版
    int editDistance2(const string& s1, const string& s2, int a, int b, int c) {
        int n = s1.length();
        int m = s2.length();
        
        vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
        
        for (int i = 1; i <= n; i++) {
            dp[i][0] = i * b;
        }
        for (int j = 1; j <= m; j++) {
            dp[0][j] = j * a;
        }
        
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                if (s1[i - 1] == s2[j - 1]) {
                    dp[i][j] = dp[i - 1][j - 1];
                } else {
                    dp[i][j] = min({dp[i - 1][j] + b, dp[i][j - 1] + a, dp[i - 1][j - 1] + c});
                }
            }
        }
        return dp[n][m];
    }

    // 空间压缩
    int editDistance3(const string& s1, const string& s2, int a, int b, int c) {
        int n = s1.length();
        int m = s2.length();
        
        vector<int> dp(m + 1, 0);
        
        for (int j = 1; j <= m; j++) {
            dp[j] = j * a;
        }
        
        for (int i = 1; i <= n; i++) {
            int leftUp = (i - 1) * b;
            dp[0] = i * b;
            for (int j = 1; j <= m; j++) {
                int backUp = dp[j];
                if (s1[i - 1] == s2[j - 1]) {
                    dp[j] = leftUp;
                } else {
                    dp[j] = min({dp[j] + b, dp[j - 1] + a, leftUp + c});
                }
                leftUp = backUp;
            }
        }
        return dp[m];
    }
};
