```cpp
// 062 Code05 二维接雨水

// 给你一个 m * n 的矩阵，其中的值均为非负整数，代表二维高度图每个单元的高度
// 请计算图中形状最多能接多少体积的雨水。
// 测试链接 : https://leetcode.cn/problems/trapping-rain-water-ii/

#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

// 题目规定 m, n <= 200
const int MAXN = 205;
const int MAXM = 205;

bool visited[MAXN][MAXM];

// 0:上，1:右，2:下，3:左 的坐标偏移
int mv[5] = { -1, 0, 1, 0, -1 };

// 定义单元格结构体
struct Cell {
    int r;
    int c;
    int w; // 水线 (water line)

    // C++ 的 priority_queue 默认是大根堆
    // 为了实现小根堆，我们需要反转比较逻辑 (w > other.w)
    bool operator<(const Cell& other) const {
        return w > other.w;
    }
};

class Solution {
public:
    int trapRainWater(vector<vector<int>>& height) {
        int n = height.size();
        int m = height[0].size();
        
        // LeetCode 会重用对象，必须重置全局状态
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                visited[i][j] = false;
            }
        }
        
        priority_queue<Cell> pq;
        
        // 将所有四周边界加入小根堆，并标记为已访问
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (i == 0 || i == n - 1 || j == 0 || j == m - 1) {
                    pq.push({i, j, height[i][j]});
                    visited[i][j] = true;
                }
            }
        }
        
        int ans = 0;
        
        // 从最外围最矮的“桶壁”开始往内部灌水
        while (!pq.empty()) {
            Cell record = pq.top();
            pq.pop();
            
            int r = record.r;
            int c = record.c;
            int w = record.w;
            
            // 累加当前格子能接的水量
            // (初始边界格子的 w 等于其 height，差值为 0，不会多加)
            ans += w - height[r][c];
            
            // 尝试向 4 个方向扩展
            for (int i = 0; i < 4; i++) {
                int nr = r + mv[i];
                int nc = c + mv[i + 1];
                
                if (nr >= 0 && nr < n && nc >= 0 && nc < m && !visited[nr][nc]) {
                    // 相邻格子的水线取决于当前水线和它自身高度的最大值（木桶效应）
                    pq.push({nr, nc, max(height[nr][nc], w)});
                    visited[nr][nc] = true;
                }
            }
        }
        
        return ans;
    }
};
```

---


**核心套路:** 
1. 外围入堆建墙：将矩阵最外围一圈的格子作为初始边界，加入小根堆（按水线高度升序排列），并标记为已访问。 
2. 木桶效应收缩：每次弹出堆顶元素（即当前边界中最矮的格子，代表“最短的木板”），向其四周未访问的邻居试探灌水。
3. 计算积水：如果邻居的高度小于当前“短板”的水线，说明找到了一个坑，该邻居的积水量为 (当前水线 - 邻居自身高度)。
4. 边界推进：将邻居作为新边界加入小根堆。它作为新木板的有效挡水高度为 max(自身物理高度, 漫过来的当前水线)。
