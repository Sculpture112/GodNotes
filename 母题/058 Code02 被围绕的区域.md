```cpp
// 058 Code02 被围绕的区域

// 测试链接 : https://leetcode.cn/problems/surrounded-regions/
// 核心思路：洪水填充（Flood Fill）/ DFS。从边界突破的逆向思维。
// 1. 任何被 'X' 围绕的 'O' 都不可能与网格边界上的 'O' 相连。反之，与边界上的 'O' 相连的所有 'O' 都绝对安全，不会被围绕。
// 2. 遍历矩阵的四条边界，一旦发现 'O'，就以此为起点触发 DFS，将相连的所有 'O' 标记为一种临时状态（例如 'F'，代表 Free/安全）。
// 3. 边界相关的连通块全部蔓延标记完后，遍历整个矩阵进行状态结算：
//    - 剩下还没被修改的 'O' 必然是被彻底包围的“死子”，直接将其修改为 'X'。
//    - 被标记为 'F' 的“活子”是与边界连通的，将其恢复为正常的 'O'。
// 时间复杂度 O(N * M)，额外空间复杂度 O(N * M)（最坏情况下递归调用栈的最大深度）

#include <vector>

using namespace std;

class Solution {
public:
    void solve(vector<vector<char>>& board) {
        if (board.empty() || board[0].empty()) return;
        int n = board.size();
        int m = board[0].size();
        
        // 1. 处理上下两行边界
        for (int j = 0; j < m; j++) {
            if (board[0][j] == 'O') {
                dfs(board, n, m, 0, j);
            }
            if (board[n - 1][j] == 'O') {
                dfs(board, n, m, n - 1, j);
            }
        }
        
        // 2. 处理左右两列边界（避开四个角，避免重复处理）
        for (int i = 1; i < n - 1; i++) {
            if (board[i][0] == 'O') {
                dfs(board, n, m, i, 0);
            }
            if (board[i][m - 1] == 'O') {
                dfs(board, n, m, i, m - 1);
            }
        }
        
        // 3. 遍历整个网格，结算最终状态
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (board[i][j] == 'O') {
                    board[i][j] = 'X'; // 没被标记过，说明被完全包围了，沉没！
                } else if (board[i][j] == 'F') {
                    board[i][j] = 'O'; // 标记过的是安全的，恢复原状
                }
            }
        }
    }

private:
    void dfs(vector<vector<char>>& board, int n, int m, int i, int j) {
        // 越界或者当前格子不是 'O' 时，停止递归
        if (i < 0 || i == n || j < 0 || j == m || board[i][j] != 'O') {
            return;
        }
        
        // 标记为 'F' (安全)
        board[i][j] = 'F';
        
        // 向四个方向继续蔓延
        dfs(board, n, m, i + 1, j); // 下
        dfs(board, n, m, i - 1, j); // 上
        dfs(board, n, m, i, j + 1); // 右
        dfs(board, n, m, i, j - 1); // 左
    }
};
```
