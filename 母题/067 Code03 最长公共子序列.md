**核心套路:** 
1. 样本对应模型（双字符串比较模型）：当遇到两个字符串/数组求某种最优解时，通常使用二维 DP，`dp[i][j]` 表示 `s1` 的前 `i` 个字符和 `s2` 的前 `j` 个字符的答案。 
2. 状态转移的本质是对当前字符选与不选的分类讨论： - 如果 `s1[i-1] == s2[j-1]`，必然可以增加公共子序列长度，转移自左上角：`dp[i][j] = dp[i-1][j-1] + 1` - 如果 `s1[i-1] != s2[j-1]`，当前字符不可能同时出现在公共子序列中，要么舍弃 `s1` 的当前字符，要么舍弃 `s2` 的当前字符，转移自上方和左方的最大值：`dp[i][j] = max(dp[i-1][j], dp[i][j-1])` 
3. 空间压缩：因为 `dp[i][j]` 只依赖于其上方、左方和左上角的值，可以使用一维数组配合两个临时变量（记录原本的左上角状态）进行滚动更新，从而将空间复杂度从 O(N*M) 降至 O(min(N, M))。

```cpp
// 067 Code03 最长公共子序列

// 给定两个字符串text1和text2
// 返回这两个字符串的最长 公共子序列 的长度
// 如果不存在公共子序列，返回0
// 两个字符串的 公共子序列 是这两个字符串所共同拥有的子序列
// 测试链接 : https://leetcode.cn/problems/longest-common-subsequence/

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

class Solution {
public:
    int longestCommonSubsequence(string text1, string text2) {
        // 五种方法任选其一均可
        // return longestCommonSubsequence1(text1, text2);
        // return longestCommonSubsequence2(text1, text2);
        // return longestCommonSubsequence3(text1, text2);
        // return longestCommonSubsequence4(text1, text2);
        return longestCommonSubsequence5(text1, text2);
    }

    // 暴力尝试（使用下标定义尝试）
    int longestCommonSubsequence1(const string& s1, const string& s2) {
        return f1(s1, s2, s1.length() - 1, s2.length() - 1);
    }

    // s1[0....i1]与s2[0....i2]最长公共子序列长度
    int f1(const string& s1, const string& s2, int i1, int i2) {
        if (i1 < 0 || i2 < 0) {
            return 0;
        }
        int p1 = f1(s1, s2, i1 - 1, i2 - 1);
        int p2 = f1(s1, s2, i1 - 1, i2);
        int p3 = f1(s1, s2, i1, i2 - 1);
        int p4 = s1[i1] == s2[i2] ? (p1 + 1) : 0;
        
        // C++11 支持初始化列表求多个值的最大值
        return max({p1, p2, p3, p4});
    }

    // 为了避免很多边界讨论
    // 很多时候往往不用下标来定义尝试，而是用长度来定义尝试
    // 因为长度最短是0，而下标越界的话讨论起来就比较麻烦
    int longestCommonSubsequence2(const string& s1, const string& s2) {
        return f2(s1, s2, s1.length(), s2.length());
    }

    // s1[前缀长度为len1]对应s2[前缀长度为len2]
    // 最长公共子序列长度
    int f2(const string& s1, const string& s2, int len1, int len2) {
        if (len1 == 0 || len2 == 0) {
            return 0;
        }
        int ans;
        if (s1[len1 - 1] == s2[len2 - 1]) {
            ans = f2(s1, s2, len1 - 1, len2 - 1) + 1;
        } else {
            ans = max(f2(s1, s2, len1 - 1, len2), f2(s1, s2, len1, len2 - 1));
        }
        return ans;
    }

    // 记忆化搜索
    int longestCommonSubsequence3(const string& s1, const string& s2) {
        int n = s1.length();
        int m = s2.length();
        vector<vector<int>> dp(n + 1, vector<int>(m + 1, -1));
        return f3(s1, s2, n, m, dp);
    }

    int f3(const string& s1, const string& s2, int len1, int len2, vector<vector<int>>& dp) {
        if (len1 == 0 || len2 == 0) {
            return 0;
        }
        if (dp[len1][len2] != -1) {
            return dp[len1][len2];
        }
        int ans;
        if (s1[len1 - 1] == s2[len2 - 1]) {
            ans = f3(s1, s2, len1 - 1, len2 - 1, dp) + 1;
        } else {
            ans = max(f3(s1, s2, len1 - 1, len2, dp), f3(s1, s2, len1, len2 - 1, dp));
        }
        dp[len1][len2] = ans;
        return ans;
    }

    // 严格位置依赖的动态规划
    int longestCommonSubsequence4(const string& s1, const string& s2) {
        int n = s1.length();
        int m = s2.length();
        vector<vector<int>> dp(n + 1, vector<int>(m + 1, 0));
        
        for (int len1 = 1; len1 <= n; len1++) {
            for (int len2 = 1; len2 <= m; len2++) {
                if (s1[len1 - 1] == s2[len2 - 1]) {
                    dp[len1][len2] = 1 + dp[len1 - 1][len2 - 1];
                } else {
                    dp[len1][len2] = max(dp[len1 - 1][len2], dp[len1][len2 - 1]);
                }
            }
        }
        return dp[n][m];
    }

    // 严格位置依赖的动态规划 + 空间压缩
    int longestCommonSubsequence5(const string& str1, const string& str2) {
        // 使用指针巧妙替换长短串，避免了 C++ 内部的 string 拷贝开销
        const string* p1 = &str1;
        const string* p2 = &str2;
        if (str1.length() < str2.length()) {
            p1 = &str2;
            p2 = &str1;
        }
        // 解引用绑定到引用上方便后续操作
        const string& s1 = *p1;
        const string& s2 = *p2;
        
        int n = s1.length();
        int m = s2.length();
        vector<int> dp(m + 1, 0);
        
        for (int len1 = 1; len1 <= n; len1++) {
            int leftUp = 0, backup;
            for (int len2 = 1; len2 <= m; len2++) {
                backup = dp[len2];
                if (s1[len1 - 1] == s2[len2 - 1]) {
                    dp[len2] = 1 + leftUp;
                } else {
                    dp[len2] = max(dp[len2], dp[len2 - 1]);
                }
                leftUp = backup;
            }
        }
        return dp[m];
    }
};
```
