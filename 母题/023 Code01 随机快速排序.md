```cpp
// 023 Code01 随机快速排序，acm练习风格
// 测试链接 : https://www.luogu.com.cn/problem/P1177
// 请同学们务必参考如下代码中关于输入、输出的处理
// 这是输入输出处理效率很高的写法
// C++ 提交时直接全部复制提交即可通过

#include <iostream>
#include <vector>
#include <cstdlib>
#include <algorithm>

using namespace std;

const int MAXN = 100005;

int arr[MAXN];
int n;

// 荷兰国旗问题边界变量
int first_idx, last_idx;

// 已知 arr[l....r] 范围上一定有 x 这个值
// 划分数组 <=x 放左边，>x 放右边
// 并且确保划分完成后 <=x 区域的最后一个数字是 x
int partition1(int l, int r, int x) {
    // a : arr[l....a-1] 范围是 <=x 的区域
    // xi : 记录在 <=x 的区域上任何一个 x 的位置，哪一个都可以
    int a = l, xi = 0;
    for (int i = l; i <= r; i++) {
        if (arr[i] <= x) {
            swap(arr[a], arr[i]);
            if (arr[a] == x) {
                xi = a;
            }
            a++;
        }
    }
    swap(arr[xi], arr[a - 1]);
    return a - 1;
}

// 随机快速排序经典版(不推荐)
// 甚至在洛谷上测试因为递归开太多层会爆栈导致出错
void quickSort1(int l, int r) {
    // l == r，只有一个数
    // l > r，范围不存在，不用管
    if (l >= r) {
        return;
    }
    // 随机这一下，常数时间比较大
    // 但只有这一下随机，才能在概率上把快速排序的时间复杂度收敛到 O(n * logn)
    // l......r 随机选一个位置，x 这个值，做划分
    int x = arr[l + rand() % (r - l + 1)];
    int mid = partition1(l, r, x);
    quickSort1(l, mid - 1);
    quickSort1(mid + 1, r);
}


// 已知 arr[l....r] 范围上一定有 x 这个值
// 划分数组 <x 放左边，==x 放中间，>x 放右边
// 把全局变量 first_idx, last_idx，更新成 ==x 区域的左右边界
void partition2(int l, int r, int x) {
    first_idx = l;
    last_idx = r;
    int i = l;
    while (i <= last_idx) {
        if (arr[i] == x) {
            i++;
        } else if (arr[i] < x) {
            swap(arr[first_idx++], arr[i++]);
        } else {
            swap(arr[i], arr[last_idx--]);
        }
    }
}

// 随机快速排序改进版(推荐)
// 可以通过所有测试用例
void quickSort2(int l, int r) {
    if (l >= r) {
        return;
    }
    // 随机这一下，常数时间比较大
    // 但只有这一下随机，才能在概率上把快速排序的时间复杂度收敛到 O(n * logn)
    int x = arr[l + rand() % (r - l + 1)];
    partition2(l, r, x);
    // 为了防止底层的递归过程覆盖全局变量
    // 这里用临时变量记录 first、last
    int left = first_idx;
    int right = last_idx;
    quickSort2(l, left - 1);
    quickSort2(right + 1, r);
}


int main() {
    // C++ 的快速 I/O 模板，能够极大提升读写效率
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    // 持续读取输入
    while (cin >> n) {
        for (int i = 0; i < n; i++) {
            cin >> arr[i];
        }
        
        quickSort2(0, n - 1);
        
        for (int i = 0; i < n - 1; i++) {
            cout << arr[i] << " ";
        }
        cout << arr[n - 1] << "\n";
    }

    return 0;
}
```

**核心套路:** 
- **随机化主元 (Random Pivot)**：通过 `arr[l + rand() % (r - l + 1)]` 随机选择基准值，打破特定数据分布（如基本有序或完全逆序）带来的最坏情况，使时间复杂度在概率上稳定收敛于 $O(N \log N)$。
    
- **三路划分 (3-way Partition)**：将数组严格划分为 `<x`, `==x`, `>x` 三个区域。其核心优势在于，当数组中存在大量重复元素时，一次划分就能将所有等于基准值的元素一次性排定位置，极大地减少了后续递归的规模，防止算法退化为 $O(N^2)$。
    
- **防越界与全局变量保护**：在递归调用进入下一层前，必须用局部变量（如代码中的 `left` 和 `right`）接住全局变量 `first_idx` 和 `last_idx` 的值，防止底层的递归操作将其覆盖。
