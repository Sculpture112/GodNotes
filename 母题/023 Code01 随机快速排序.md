// 023 Code01 随机快速排序

// 测试链接 : https://www.luogu.com.cn/problem/P1177
// 核心思路：随机选取枢轴 + 划分（Partition）。

// 针对这道经典题目，这里提供“一题多解”的 C++ 实现，包含经典双路划分和优化的三路划分：
// 
// 解法一：经典随机快排（双路划分）
// - 划分规则：将数组划分为 <=x 和 >x 两部分。
// - 缺点：在面对大量重复元素的极端测试用例时，划分极度不平衡，会导致递归树过深，进而爆栈（Stack Overflow）。
//
// 解法二：改进版随机快排（三路划分 / 荷兰国旗问题）- 推荐！
// - 划分规则：将数组划分为 <x、==x 和 >x 三部分。
// - 优点：一次性将所有等于 x 的元素全部安排到最终位置，大幅缩减后续递归的区间范围。完美解决了大量重复元素导致的退化问题。
// - 时间复杂度 O(N log N)，额外空间复杂度 O(log N)（递归调用栈深度）

#include <iostream>
#include <vector>
#include <cstdlib>
#include <ctime>

using namespace std;

// 针对洛谷等平台的快速读入模板
inline bool read(int &x) {
    x = 0; int f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == EOF) return false;
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    x *= f;
    return true;
}

class Solution {
public:
    // ==========================================
    // 解法一：随机快速排序经典版（双路划分，可能爆栈）
    // ==========================================
    void quickSort1(vector<int>& arr, int l, int r) {
        if (l >= r) {
            return;
        }
        // 随机选取枢轴，使得时间复杂度在概率上收敛于 O(N log N)
        int x = arr[l + rand() % (r - l + 1)];
        int mid = partition1(arr, l, r, x);
        quickSort1(arr, l, mid - 1);
        quickSort1(arr, mid + 1, r);
    }

private:
    int partition1(vector<int>& arr, int l, int r, int x) {
        int a = l, xi = 0;
        for (int i = l; i <= r; i++) {
            if (arr[i] <= x) {
                swap(arr[a], arr[i]);
                if (arr[a] == x) {
                    xi = a;
                }
                a++;
            }
        }
        // 确保划分完成后 <=x 区域的最后一个数字是 x
        swap(arr[xi], arr[a - 1]);
        return a - 1;
    }

public:
    // ==========================================
    // 解法二：随机快速排序改进版（三路划分，强烈推荐）
    // ==========================================
    void quickSort2(vector<int>& arr, int l, int r) {
        if (l >= r) {
            return;
        }
        int x = arr[l + rand() % (r - l + 1)];
        // 接收荷兰国旗划分返回的 ==x 区域的左右边界
        pair<int, int> bounds = partition2(arr, l, r, x);
        quickSort2(arr, l, bounds.first - 1);
        quickSort2(arr, bounds.second + 1, r);
    }

private:
    // 荷兰国旗问题划分逻辑
    // 返回 pair<int, int> 替代全局变量，避免状态污染
    pair<int, int> partition2(vector<int>& arr, int l, int r, int x) {
        int first = l;
        int last = r;
        int i = l;
        while (i <= last) {
            if (arr[i] == x) {
                i++;
            } else if (arr[i] < x) {
                swap(arr[first++], arr[i++]);
            } else {
                swap(arr[i], arr[last--]);
            }
        }
        return {first, last};
    }
};

int main() {
    // 初始化随机数种子
    srand(time(nullptr));
    
    int n;
    if (read(n)) {
        vector<int> arr(n);
        for (int i = 0; i < n; i++) {
            read(arr[i]);
        }
        
        Solution sol;
        // 使用推荐的改进版（三路划分）进行排序
        sol.quickSort2(arr, 0, n - 1);
        
        for (int i = 0; i < n - 1; i++) {
            printf("%d ", arr[i]);
        }
        if (n > 0) {
            printf("%d\n", arr[n - 1]);
        }
    }
    return 0;
}
