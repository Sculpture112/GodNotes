// 075 Code01 多重背包不进行枚举优化

// 宝物筛选
// 一共有 n 种货物, 背包容量为 t
// 每种货物的价值(v[i])、重量(w[i])、数量(c[i])都给出
// 请返回选择货物不超过背包容量的情况下，能得到的最大的价值
// 测试链接 : https://www.luogu.com.cn/problem/P1776

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * 逻辑核心分析：
 * 多重背包问题中，每种物品有一个有限的数量 c[i]。
 * 如果不进行优化，最直观的思路是在 01 背包的基础上，针对每种物品多套一层循环，
 * 枚举“当前物品到底选几个（0 ~ c[i]）”。
 */

const int MAXN = 105;
const int MAXW = 40005;

int v[MAXN];
int w[MAXN];
int c[MAXN];
int dp[MAXW];
int n, t;

/**
 * 方案一：严格位置依赖的动态规划 (二维 DP)
 * 时间复杂度 O(n * t * 每种商品的平均个数)
 * 空间复杂度 O(n * t)
 */
int compute1() {
    // dp[i][j] 表示考虑前 i 种货物，容量不超过 j 时的最大价值
    vector<vector<int>> dp_table(n + 1, vector<int>(t + 1, 0));

    for (int i = 1; i <= n; i++) {
        for (int j = 0; j <= t; j++) {
            // 可能性 0: 一个也不要
            dp_table[i][j] = dp_table[i - 1][j];
            // 枚举选 k 个
            for (int k = 1; k <= c[i] && k * w[i] <= j; k++) {
                dp_table[i][j] = max(dp_table[i][j], dp_table[i - 1][j - k * w[i]] + k * v[i]);
            }
        }
    }
    return dp_table[n][t];
}

/**
 * 方案二：空间压缩版 (一维 DP)
 * 时间复杂度 O(n * t * 每种商品的平均个数)
 * 空间复杂度 O(t)
 * 注：由于没有优化枚举，大数据量下此方案在洛谷等平台可能会超时。
 */
int compute2() {
    // 重置 DP 数组（应对多组测试数据）
    for (int i = 0; i <= t; i++) dp[i] = 0;

    for (int i = 1; i <= n; i++) {
        // 01 背包空间压缩的核心：容量逆序遍历
        for (int j = t; j >= 0; j--) {
            // 在当前容量 j 下，尝试放入 k 个物品 i
            for (int k = 1; k <= c[i] && k * w[i] <= j; k++) {
                dp[j] = max(dp[j], dp[j - k * w[i]] + k * v[i]);
            }
        }
    }
    return dp[t];
}

int main() {
    // 提升 I/O 效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n >> t) {
        for (int i = 1; i <= n; i++) {
            cin >> v[i] >> w[i] >> c[i];
        }
        
        // 此题如果不优化，compute2 大概率会超时，但逻辑上是正确的
        cout << compute2() << "\n";
    }

    return 0;
}
