// 074 Code03 完全背包(从二维推导到一维)

// 给定一个正数 t，表示背包的容量
// 有 m 种货物，每种货物可以选择任意个
// 每种货物都有体积 cost[i] 和价值 val[i]
// 返回在不超过总容量的情况下，怎么挑选货物能达到价值最大
// 测试链接 : https://www.luogu.com.cn/problem/P1616

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 注意：本题容量 t 高达 10^7，物品数 m 为 10^4
// 在实际测评中，二维数组 int[10001][10000001] 会直接撑爆内存（需要约 400GB）
// 但为了学习算法逻辑，我们必须理解它的转移过程
const int MAXM = 10005;
const int MAXT = 10000005;

int cost_arr[MAXM];
int val_arr[MAXM];
long long dp_1d[MAXT]; // 实际跑通测试需要这个
int t, m;

/**
 * 方案一：严格位置依赖的动态规划 (二维 DP)
 * 
 * 核心逻辑：
 * 1. 01 背包：dp[i][j] 依赖上一行 dp[i-1][j]（不选）和 上一行左侧 dp[i-1][j-c]（选）。
 * 2. 完全背包：dp[i][j] 依赖上一行 dp[i-1][j]（不选）和 【当前行】左侧 dp[i][j-c]（选）。
 * 
 * 为什么看当前行？
 * 因为选了一个物品后，你还可以继续选第 i 种物品，所以你依然可以参考第 i 行的状态。
 */
long long compute1() {
    // 此代码仅用于教学逻辑理解，直接运行会 MLE (Memory Limit Exceeded)
    vector<vector<long long>> dp(m + 1, vector<long long>(t + 1, 0));
    
     
    for (int i = 1; i <= m; i++) {
        for (int j = 0; j <= t; j++) {
            // 可能性 1：不选当前第 i 种物品，直接继承上一行的结果
            dp[i][j] = dp[i - 1][j];
            // 可能性 2：选一个当前物品，并继续在当前行寻找最大可能
            if (j >= cost_arr[i]) {
                dp[i][j] = max(dp[i][j], dp[i][j - cost_arr[i]] + val_arr[i]);
            }
        }
    }
    return dp[m][t];
    
    return 0;
}

/**
 * 方案二：空间压缩优化 (一维 DP)
 * 
 * 逻辑演变：
 * 观察方案一的转移方程：dp[i][j] = max(dp[i-1][j], dp[i][j-c] + v)
 * - 我们需要上一行相同位置的值（dp[i-1][j]）。
 * - 我们需要【这一行】左边已经算好的值（dp[i][j-c]）。
 * 
 * 在一维数组中，如果我们【正序遍历】（从左往右）：
 * 当计算到 dp[j] 时，dp[j-c] 已经在这一轮被更新过了，它正好代表 dp[i][j-c]！
 * 这就是完全背包正序遍历的数学美感所在。
 */
long long compute2() {
    // 重置数组（多组测试用例需要）
    for (int j = 0; j <= t; j++) dp_1d[j] = 0;

    for (int i = 1; i <= m; i++) {
        // 【关键点】必须正序遍历
        // 这样计算 dp[j] 时，参考的 dp[j - cost_arr[i]] 已经是本轮更新后的“新值”了
        for (int j = cost_arr[i]; j <= t; j++) {
            dp_1d[j] = max(dp_1d[j], dp_1d[j - cost_arr[i]] + val_arr[i]);
        }
    }
    return dp_1d[t];
}

int main() {
    // 提升 I/O 效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> t >> m) {
        for (int i = 1; i <= m; i++) {
            cin >> cost_arr[i] >> val_arr[i];
        }
        // 在学习阶段，心里想着 compute1，手上写出 compute2
        cout << compute2() << "\n";
    }

    return 0;
}
