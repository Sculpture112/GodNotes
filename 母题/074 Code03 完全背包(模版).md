// 074 Code03 完全背包(模版)

// 给定一个正数 t，表示背包的容量
// 有 m 种货物，每种货物可以选择任意个
// 每种货物都有体积 cost[i] 和价值 val[i]
// 返回在不超过总容量的情况下，怎么挑选货物能达到价值最大
// 测试链接 : https://www.luogu.com.cn/problem/P1616

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * 核心逻辑分析：
 * 完全背包与 01 背包的区别在于：每种物品可以无限件使用。
 * 
 * 1. 严格位置依赖 (二维 DP):
 *    dp[i][j] 的最大值来自：
 *    - 可能性 1: 不选第 i 种物品，即 dp[i-1][j]
 *    - 可能性 2: 至少选一个第 i 种物品。
 *      因为可以无限选，选了一个后，我们依然处在“可以选第 i 种物品”的状态，
 *      所以是 dp[i][j - cost[i]] + val[i]
 *    公式：dp[i][j] = max(dp[i-1][j], dp[i][j - cost[i]] + val[i])
 * 
 * 2. 空间压缩 (一维 DP):
 *    在 01 背包中，为了使用“上一行”的数据，我们需要从右往左（逆序）遍历。
 *    在完全背包中，我们需要使用“当前行”左侧已经更新过的数据，
 *    因此，我们必须从左往右（正序）遍历。
 */

// 注意：本题容量 T 高达 10^7，物品 M 高达 10^4
// 必须使用全局数组，且 dp 数组需使用 long long 防止价值总和溢出
const int MAXT = 10000005;
const int MAXM = 10005;

int cost[MAXM];
int val[MAXM];
long long dp[MAXT];
int t, m;

/**
 * 方案一：严格位置依赖的动态规划 (二维 DP)
 * 复杂度：时间 O(M*T)，空间 O(M*T)
 * 注意：由于 T=10^7, M=10^4，二维数组会造成内存超限 (MLE)
 */

}

/**
 * 方案二：空间压缩版 (一维 DP)
 * 复杂度：时间 O(M*T)，空间 O(T)
 * 能够通过全部测试用例
 */
long long compute2() {
    // 初始化：由于是全局变量，默认已为 0，若多组数据则需 fill
    for (int j = 0; j <= t; j++) dp[j] = 0;

    for (int i = 1; i <= m; i++) {
        // 正序遍历：确保计算 dp[j] 时，dp[j - cost[i]] 已经是
        // 这一行（即考虑了当前物品 i）更新后的结果
        for (int j = cost[i]; j <= t; j++) {
            dp[j] = max(dp[j], dp[j - cost[i]] + val[i]);
        }
    }
    return dp[t];
}

int main() {
    // 提升 C++ I/O 效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> t >> m) {
        for (int i = 1; i <= m; i++) {
            cin >> cost[i] >> val[i];
        }
        cout << compute2() << "\n";
    }

    return 0;
}
