// 072 Code02 俄罗斯套娃信封问题

// 给你一个二维整数数组 envelopes ，其中 envelopes[i] = [wi, hi]
// 表示第 i 个信封的宽度和高度
// 当另一个信封的宽度和高度都比这个信封大的时候
// 这个信封就可以放进另一个信封里，如同俄罗斯套娃一样
// 请计算 最多能有多少个信封能组成一组“俄罗斯套娃”信封
// 测试链接 : https://leetcode.cn/problems/russian-doll-envelopes/

#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    /**
     * 时间复杂度 O(n * logn)：排序复杂度 O(n * logn)，LIS 过程复杂度 O(n * logn)。
     * 额外空间复杂度 O(n)：用于 ends 数组存储。
     */
    int maxEnvelopes(vector<vector<int>>& envelopes) {
        int n = envelopes.size();
        if (n == 0) return 0;

        /**
         * 关键排序策略：
         * 1. 宽度 (w) 从小到大排序。
         * 2. 如果宽度相同，高度 (h) 从大到小排序。
         * 
         * 为什么？
         * 因为宽度从小到大，我们只需要关注高度是否也递增。
         * 当宽度相同时，高度从大到小排列，可以保证在计算最长递增子序列（LIS）时，
         * 相同宽度的信封由于高度是递减的，最多只能选出一个（因为我们要的是严格递增）。
         * 这样就巧妙地把二维问题转化为了高度数组的一维 LIS 问题。
         */
        sort(envelopes.begin(), envelopes.end(), [](const vector<int>& a, const vector<int>& b) {
            if (a[0] != b[0]) {
                return a[0] < b[0];
            }
            return a[1] > b[1];
        });

        // 对高度数组求最长严格递增子序列
        vector<int> ends(n);
        int len = 0;
        for (int i = 0; i < n; i++) {
            int num = envelopes[i][1];
            int find = bs(ends, len, num);
            if (find == -1) {
                ends[len++] = num;
            } else {
                ends[find] = num;
            }
        }
        return len;
    }

private:
    /**
     * 在 ends[0...len-1] 中找到第一个 >= num 的位置
     * 等同于 std::lower_bound
     */
    int bs(const vector<int>& ends, int len, int num) {
        int l = 0, r = len - 1, m, ans = -1;
        while (l <= r) {
            m = l + ((r - l) >> 1);
            if (ends[m] >= num) {
                ans = m;
                r = m - 1;
            } else {
                l = m + 1;
            }
        }
        return ans;
    }
};
