// 073 Code01 01背包(模版)

// 给定一个正数 t，表示背包的容量
// 有 m 个货物，每个货物可以选择一次
// 每个货物有自己的体积 cost[i] 和价值 val[i]
// 返回在不超过总容量的情况下，怎么挑选货物能达到价值最大
// 测试链接 : https://www.luogu.com.cn/problem/P1048

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 根据题目描述设置最大范围
const int MAXM = 105;
const int MAXT = 1005;

int cost[MAXM];
int val[MAXM];
int dp_table[MAXT];
int t, n;

/**
 * 方案一：严格位置依赖的动态规划 (二维 DP)
 * 时间复杂度 $O(n \times t)$
 * 空间复杂度 $O(n \times t)$
 */
int compute1() {
    // dp[i][j] 表示前 i 个物品在容量为 j 的情况下的最大价值
    vector<vector<int>> dp(n + 1, vector<int>(t + 1, 0));
    
    for (int i = 1; i <= n; i++) {
        for (int j = 0; j <= t; j++) {
            // 可能性 1：不要当前 i 号物品
            dp[i][j] = dp[i - 1][j];
            // 可能性 2：要当前 i 号物品（前提是容量足够）
            if (j >= cost[i]) {
                dp[i][j] = max(dp[i][j], dp[i - 1][j - cost[i]] + val[i]);
            }
        }
    }
    return dp[n][t];
}

/**
 * 方案二：空间压缩优化 (一维 DP)
 * 时间复杂度 $O(n \times t)$
 * 空间复杂度 $O(t)$
 */
int compute2() {
    // 每次处理新物品前，清空当前容量数组（对应 Java 的 Arrays.fill）
    for (int i = 0; i <= t; i++) dp_table[i] = 0;

    for (int i = 1; i <= n; i++) {
        // 必须从右往左遍历，保证 dp[j - cost[i]] 使用的是“上一层” i-1 的数据
        for (int j = t; j >= cost[i]; j--) {
            dp_table[j] = max(dp_table[j], dp_table[j - cost[i]] + val[i]);
        }
    }
    return dp_table[t];
}

int main() {
    // 提升 I/O 效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> t >> n) {
        for (int i = 1; i <= n; i++) {
            cin >> cost[i] >> val[i];
        }
        // 在洛谷等平台提交时，通常采用空间压缩版本 compute2
        cout << compute2() << "\n";
    }

    return 0;
}
