**核心套路:** 
对于非负数组求前 $k$ 小子序列和，由于 $N$ 和 $K$ 极大，传统的动态规划（01 背包）会直接超时超空间。标准解法是“排序 + 堆驱动的状态树搜索（类似 Dijkstra 找最短路径）”。必须先对数组升序排序，确保子序列和的增长具有单调边界。第 1 小必定是空集（和为 0）。将只选第一个数的状态 {index: 0, sum: nums[0]} 初始化入小根堆。每次弹出堆顶（当前最小），并向下进行不重不漏的两次裂变派生：追加 (买买买)：保留当前状态，加上下一个数 $\rightarrow$ sum + nums[i+1]替换 (吃后悔药)：去掉当前状态里最后加入的数，换成下一个更大的数 $\rightarrow$ sum - nums[i] + nums[i+1]

// 073 Code06 非负数组前 k 个最小的子序列累加和

// 给定一个数组 nums，含有 n 个数字，都是非负数
// 给定一个正数 k，返回所有子序列中累加和最小的前 k 个累加和
// 子序列包含空集
// 1 <= n <= 10^5, 1 <= nums[i] <= 10^6, 1 <= k <= 10^5
// 注意：由于 k 和 n 很大，01 背包方案会超时。最优解为堆 (Heap) 路径搜索。
// 时间复杂度：O(n * log n) + O(k * log k)
// 额外空间复杂度：O(n) 或 O(k) 视实现而定

#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>

using namespace std;

/**
 * 结构体：用于堆中的节点
 * index: 当前考虑到的 nums 中的下标
 * sum: 当前子序列的累加和
 */
struct Node {
    int index;
    long long sum;

    // 小根堆：累加和小的在前
    bool operator>(const Node& other) const {
        return sum > other.sum;
    }
};

class Solution {
public:
    /**
     * 最优解：使用堆进行搜索
     * 核心逻辑：
     * 1. 空集的累加和为 0，是最小的。
     * 2. 对数组进行升序排序。
     * 3. 初始将 {0, nums[0]} 加入小根堆。
     * 4. 每次弹出堆顶 {i, s}，它是当前的第 x 小累加和。
     * 5. 通过 {i, s} 派生出两个后续可能：
     *    - 包含下一个数：{i + 1, s + nums[i + 1]}
     *    - 不包含当前数，但包含下一个数（替换）：{i + 1, s - nums[i] + nums[i + 1]}
     */
    vector<long long> topKSum(vector<int>& nums, int k) {
        int n = nums.size();
        sort(nums.begin(), nums.end());

        vector<long long> ans;
        ans.push_back(0); // 第 1 小：空集累加和为 0
        if (k == 1) return ans;

        priority_queue<Node, vector<Node>, greater<Node>> heap;
        heap.push({0, (long long)nums[0]});

        // 还需要找 k-1 个
        while (ans.size() < (size_t)k && !heap.empty()) {
            Node cur = heap.top();
            heap.pop();

            ans.push_back(cur.sum);

            if (cur.index + 1 < n) {
                // 派生方式 1：包含下一个数 (Include next)
                heap.push({cur.index + 1, cur.sum + nums[cur.index + 1]});
                // 派生方式 2：替换当前数为下一个数 (Replace current with next)
                // 这种方式确保了每个子序列只被以一种唯一的“最右索引”方式加入堆中，避免重复
                heap.push({cur.index + 1, cur.sum - nums[cur.index] + nums[cur.index + 1]});
            }
        }

        return ans;
    }
};

// ==========================
// 以下为对数器验证相关的辅助代码
// ==========================

// 暴力递归获取所有子序列和
void dfs(const vector<int>& nums, int index, long long sum, vector<long long>& all) {
    if (index == nums.size()) {
        all.push_back(sum);
    } else {
        dfs(nums, index + 1, sum, all);
        dfs(nums, index + 1, sum + nums[index], all);
    }
}

vector<long long> topKSum_BruteForce(vector<int> nums, int k) {
    vector<long long> all;
    dfs(nums, 0, 0, all);
    sort(all.begin(), all.end());
    if (all.size() > (size_t)k) {
        all.resize(k);
    }
    return all;
}

int main() {
    Solution sol;
    
    // 简单测试
    vector<int> nums = {3, 1, 4, 2};
    int k = 10;
    vector<long long> result = sol.topKSum(nums, k);
    
    cout << "Top " << k << " subsequence sums: ";
    for (long long s : result) {
        cout << s << " ";
    }
    cout << endl;

    return 0;
}
