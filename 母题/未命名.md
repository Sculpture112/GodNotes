// 061 Code04 检查边长度限制的路径是否存在

// 测试链接 : https://leetcode.cn/problems/checking-existence-of-edge-length-limited-paths/
// 核心思路：离线查询 + 并查集。严格对标 Java 版逻辑。
// 1. 将边按权重从小到大排序。
// 2. 将查询数组也按 limit 从小到大排序，但同时必须绑定原查询的索引，以便最终按原顺序返回结果。
// 3. 遍历排序后的查询数组：由于 limit 是递增的，可以直接用一个递增的指针 j 将所有满足 权重 < limit 的边加入并查集连通起来。
// 4. 连通完当前满足条件的边后，直接使用并查集检查查询的起点和终点是否连通即可。
// 时间复杂度 O(E log E + Q log Q + E + Q)，其中 E 是边数，Q 是查询数。额外空间复杂度 O(N + Q)

#include <vector>
#include <algorithm>
#include <functional>

using namespace std;

class Solution {
public:
    vector<bool> distanceLimitedPathsExist(int n, vector<vector<int>>& edgeList, vector<vector<int>>& queries) {
        // 1. 边集按权重升序排序
        sort(edgeList.begin(), edgeList.end(), [](const vector<int>& a, const vector<int>& b) {
            return a[2] < b[2];
        });

        int m = edgeList.size();
        int k = queries.size();

        // 2. 重构查询数组，附加原始索引位置
        // questions[i] = [u, v, limit, 原始索引]
        vector<vector<int>> questions(k, vector<int>(4));
        for (int i = 0; i < k; i++) {
            questions[i][0] = queries[i][0];
            questions[i][1] = queries[i][1];
            questions[i][2] = queries[i][2];
            questions[i][3] = i;
        }

        // 查询集按 limit 升序排序
        sort(questions.begin(), questions.end(), [](const vector<int>& a, const vector<int>& b) {
            return a[2] < b[2];
        });

        // 3. 初始化并查集
        vector<int> father(n);
        for (int i = 0; i < n; i++) {
            father[i] = i;
        }

        // Lambda 递归查找 + 路径压缩
        function<int(int)> find = [&](int i) {
            if (i != father[i]) {
                father[i] = find(father[i]);
            }
            return father[i];
        };

        // 判断是否同一集合
        auto isSameSet = [&](int x, int y) {
            return find(x) == find(y);
        };

        // 极简合并
        auto unite = [&](int x, int y) {
            father[find(x)] = find(y);
        };

        vector<bool> ans(k);
        int j = 0; // 边集遍历指针，随着 limit 变大，j 不会回退

        // 4. 离线处理每一个查询
        for (int i = 0; i < k; i++) {
            // 当前问题需要的 limit = questions[i][2]
            // 将所有严格小于 limit 的边加入并查集
            while (j < m && edgeList[j][2] < questions[i][2]) {
                unite(edgeList[j][0], edgeList[j][1]);
                j++;
            }
            // 检查当前查询的两点是否在同一个连通块，并将结果写回原始索引处
            ans[questions[i][3]] = isSameSet(questions[i][0], questions[i][1]);
        }

        return ans;
    }
};
