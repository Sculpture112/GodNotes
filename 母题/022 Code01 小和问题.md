// 022 Code01 小和问题

// 测试链接 : https://www.nowcoder.com/practice/edfe05a1d45c4ea89101d936cac32469
// 核心思路：分治法（归并排序的经典变种）。
// 1. 求数组的小和，等价于求每个元素左侧比它小的元素之和。利用归并排序的过程，在合并两个有序子数组时，可以高效地批量计算跨越左右两部分的小和。
// 2. 将数组递归地分为左右两半。递归返回时，保证左半部分和右半部分内部都已经是有序的，并且已经求出了内部的小和。
// 3. 在合并（merge）阶段，利用双指针统计跨越左右的小和：维护一个运行时的 `sum`，记录左组目前所有小于等于右组当前元素的累加和。每当右侧指针 `j` 向右移动，就把当前的 `sum` 累加到最终答案中。
// 4. 统计完毕后，执行标准的归并排序合并操作，使得整体变有序，供上一层继续使用。
// 注意事项：小和的结果可能非常大，必须使用 64 位整型（long long）以防溢出，这是此类笔试题的常见考点。
// 时间复杂度 O(N log N)，额外空间复杂度 O(N)

#include <iostream>
#include <vector>

using namespace std;

// 针对牛客网多组测试数据及严苛时间限制的快读模板
inline bool read(int &x) {
    x = 0; int f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == EOF) return false;
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    x *= f;
    return true;
}

class Solution {
private:
    // 返回跨左右产生的小和累加和，同时将左右两侧整体排序
    long long merge(vector<int>& arr, vector<int>& help, int l, int m, int r) {
        long long ans = 0;
        long long sum = 0; // 左组累加和也要防溢出
        
        // 1. 核心统计部分
        int i = l;
        for (int j = m + 1; j <= r; j++) {
            // 只要左组的数小于等于右组当前的数，就累加到 sum 里
            while (i <= m && arr[i] <= arr[j]) {
                sum += arr[i++];
            }
            // 将当前的 sum 计入总答案
            ans += sum;
        }
        
        // 2. 正常 merge 部分
        int idx = l;
        int a = l;
        int b = m + 1;
        while (a <= m && b <= r) {
            help[idx++] = arr[a] <= arr[b] ? arr[a++] : arr[b++];
        }
        while (a <= m) {
            help[idx++] = arr[a++];
        }
        while (b <= r) {
            help[idx++] = arr[b++];
        }
        for (idx = l; idx <= r; idx++) {
            arr[idx] = help[idx];
        }
        
        return ans;
    }

    // 递归求小和
    long long smallSum(vector<int>& arr, vector<int>& help, int l, int r) {
        if (l == r) {
            return 0;
        }
        int m = l + (r - l) / 2;
        // 左侧小和 + 右侧小和 + 跨越左右的小和
        return smallSum(arr, help, l, m) + smallSum(arr, help, m + 1, r) + merge(arr, help, l, m, r);
    }

public:
    long long calculate(vector<int>& arr) {
        if (arr.empty()) return 0;
        int n = arr.size();
        // 局部开辟辅助数组，避免全局变量污染
        vector<int> help(n);
        return smallSum(arr, help, 0, n - 1);
    }
};

int main() {
    int n;
    // 持续读取直到 EOF
    while (read(n)) {
        vector<int> arr(n);
        for (int i = 0; i < n; i++) {
            read(arr[i]);
        }
        
        Solution sol;
        // 使用 printf 输出 long long，避免 cout 的耗时
        printf("%lld\n", sol.calculate(arr));
    }
    return 0;
}
