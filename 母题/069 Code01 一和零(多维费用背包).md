**核心套路:** 
#### 1. 模型抽象转换（万物皆可背包）

要将看似不相关的题目套入背包模型，需要明确三个核心要素：

- **物品（Items）：** 数组 `strs` 中的每一个字符串。
    
- **背包容量（Capacity）：** 限制条件 `m`（最多提供的 0 的数量）和 `n`（最多提供的 1 的数量）。
    
- **物品成本（Cost）与价值（Value）：** 当前字符串所需 0 和 1 的数量是“成本”；因为要求最大子集的长度，所以每个字符串的“价值”固定为 1。
    

#### 2. 状态定义与转移方程

- **定义：** $dp[z][o]$ 表示在使用 $z$ 个 0 和 $o$ 个 1 的容量限制下，能够“装入”的最大字符串数量。
    
- **转移方程：** 面对一个新的字符串（假设需要 $zeros$ 个 0 和 $ones$ 个 1），我们要么**不选**它，要么在容量允许的情况下**选**它。
    
    $dp[z][o] = \max(dp[z][o], dp[z - zeros][o - ones] + 1)$
    

#### 3. 空间压缩的“倒序遍历”铁律（最核心考点）

在方法四（2D-DP）中，舍弃了代表物品数量的维度（$i$）进行空间压缩时，**内层的容量循环必须严格倒序遍历**

```c++
for (int z = m; z >= zeros; z--) {
    for (int o = n; o >= ones; o--) {
         // ...
    }
}
```

- **原因：** 如果正序遍历，容量小的格子会先被更新为“选了当前字符串”的状态。当推导容量大的格子时，如果依赖到了前面刚更新的格子，就会导致**同一个字符串被重复选择了多次**（这就变成了“完全背包”）。
    
- **倒序的意义：** 保证计算 $dp[z][o]$ 时，所依赖的 $dp[z - zeros][o - ones]$ 绝对是上一轮（上一个字符串）留下的干净数据。
    

#### 4. 边界控制与天然剪枝

正如你在方法四代码中注释的那样，内层循环的下界直接设置到 `zeros` 和 `ones`（即 `z >= zeros` 和 `o >= ones`）：

- **防止越界：** 避免出现 $z - zeros < 0$ 导致数组越界报错。
    
- **天然剪枝：** 如果当前剩余的容量 $z$ 或 $o$ 已经小于当前字符串的成本，那无论如何也装不下，直接跳过转移即可，节省了大量的无用计算。


```cpp
// 069 Code01 一和零(多维费用背包)

// 给你一个二进制字符串数组 strs 和两个整数 m 和 n
// 请你找出并返回 strs 的最大子集的长度
// 该子集中 最多 有 m 个 0 和 n 个 1
// 如果 x 的所有元素也是 y 的元素，集合 x 是集合 y 的 子集
// 测试链接 : https://leetcode.cn/problems/ones-and-zeroes/

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

class Solution {
public:
    int findMaxForm(vector<string>& strs, int m, int n) {
        // 四种方法任选其一均可
        // return findMaxForm1(strs, m, n);
        // return findMaxForm2(strs, m, n);
        // return findMaxForm3(strs, m, n);
        return findMaxForm4(strs, m, n);
    }

    // 统计一个字符串中0和1的数量
    // 返回 pair<0的数量, 1的数量>
    pair<int, int> zerosAndOnes(const string& str) {
        int zeros = 0;
        int ones = 0;
        for (char c : str) {
            if (c == '0') {
                zeros++;
            } else {
                ones++;
            }
        }
        return {zeros, ones};
    }

    // 暴力尝试
    int findMaxForm1(const vector<string>& strs, int m, int n) {
        return f1(strs, 0, m, n);
    }

    // strs[i....]自由选择，希望零的数量不超过z、一的数量不超过o
    // 最多能选多少个字符串
    int f1(const vector<string>& strs, int i, int z, int o) {
        if (i == strs.size()) {
            // 没有字符串了
            return 0;
        }
        // 不使用当前的strs[i]字符串
        int p1 = f1(strs, i + 1, z, o);
        // 使用当前的strs[i]字符串
        int p2 = 0;
        pair<int, int> counts = zerosAndOnes(strs[i]);
        int zeros = counts.first;
        int ones = counts.second;
        
        if (zeros <= z && ones <= o) {
            p2 = 1 + f1(strs, i + 1, z - zeros, o - ones);
        }
        return max(p1, p2);
    }

    // 记忆化搜索
    int findMaxForm2(const vector<string>& strs, int m, int n) {
        int len = strs.size();
        vector<vector<vector<int>>> dp(len, vector<vector<int>>(m + 1, vector<int>(n + 1, -1)));
        return f2(strs, 0, m, n, dp);
    }

    int f2(const vector<string>& strs, int i, int z, int o, vector<vector<vector<int>>>& dp) {
        if (i == strs.size()) {
            return 0;
        }
        if (dp[i][z][o] != -1) {
            return dp[i][z][o];
        }
        int p1 = f2(strs, i + 1, z, o, dp);
        int p2 = 0;
        pair<int, int> counts = zerosAndOnes(strs[i]);
        int zeros = counts.first;
        int ones = counts.second;
        
        if (zeros <= z && ones <= o) {
            p2 = 1 + f2(strs, i + 1, z - zeros, o - ones, dp);
        }
        int ans = max(p1, p2);
        dp[i][z][o] = ans;
        return ans;
    }

    // 严格位置依赖的动态规划
    int findMaxForm3(const vector<string>& strs, int m, int n) {
        int len = strs.size();
        vector<vector<vector<int>>> dp(len + 1, vector<vector<int>>(m + 1, vector<int>(n + 1, 0)));
        
        for (int i = len - 1; i >= 0; i--) {
            pair<int, int> counts = zerosAndOnes(strs[i]);
            int zeros = counts.first;
            int ones = counts.second;
            
            for (int z = 0; z <= m; z++) {
                for (int o = 0; o <= n; o++) {
                    int p1 = dp[i + 1][z][o];
                    int p2 = 0;
                    if (zeros <= z && ones <= o) {
                        p2 = 1 + dp[i + 1][z - zeros][o - ones];
                    }
                    dp[i][z][o] = max(p1, p2);
                }
            }
        }
        return dp[0][m][n];
    }

    // 严格位置依赖的动态规划 + 空间压缩
    int findMaxForm4(const vector<string>& strs, int m, int n) {
        // 代表 i == len
        vector<vector<int>> dp(m + 1, vector<int>(n + 1, 0));
        
        for (const string& s : strs) {
            // 每个字符串逐渐遍历即可
            // 更新每一层的表
            // 和之前的遍历没有区别，务必从右下角往左上角更新防止数据污染
            pair<int, int> counts = zerosAndOnes(s);
            int zeros = counts.first;
            int ones = counts.second;
            
            for (int z = m; z >= zeros; z--) {//这里遍历到zero.因为过程出现z-zero，提前剪枝判断，one同理
                for (int o = n; o >= ones; o--) {
                    dp[z][o] = max(dp[z][o], 1 + dp[z - zeros][o - ones]);
                }
            }
        }
        return dp[m][n];
    }
};
```
