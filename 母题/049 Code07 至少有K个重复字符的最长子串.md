// 049 Code07 至少有K个重复字符的最长子串

// 给你一个字符串 s 和一个整数 k ，请你找出 s 中的最长子串
// 要求该子串中的每一字符出现次数都不少于 k 。返回这一子串的长度
// 如果不存在这样的子字符串，则返回 0。
// 测试链接 : https://leetcode.cn/problems/longest-substring-with-at-least-k-repeating-characters/

#include <vector>
#include <string>
#include <algorithm>
#include <cstring>

using namespace std;

class Solution {
public:
    // 时间复杂度 O(26 * n)，n 是字符串长度
    // 额外空间复杂度 O(1)，字符集大小固定为 256
    int longestSubstring(string str, int k) {
        int n = str.length();
        int cnts[256];
        int ans = 0;
        
        // 核心思路：枚举子串中正好包含 require 种字符的情况
        // require 从 1 种到 26 种（小写字母范围）
        for (int require = 1; require <= 26; require++) {
            memset(cnts, 0, sizeof(cnts));
            
            // collect : 窗口内收集到的字符种类数
            // satisfy : 窗口内出现次数 >= k 的字符种类数
            for (int l = 0, r = 0, collect = 0, satisfy = 0; r < n; r++) {
                unsigned char curR = (unsigned char)str[r];
                if (++cnts[curR] == 1) {
                    collect++;
                }
                if (cnts[curR] == k) {
                    satisfy++;
                }
                
                // 如果窗口内种类超过了当前的限制 require，缩小左边界
                while (collect > require) {
                    unsigned char curL = (unsigned char)str[l];
                    if (cnts[curL] == 1) {
                        collect--;
                    }
                    if (cnts[curL] == k) {
                        satisfy--;
                    }
                    cnts[curL]--;
                    l++;
                }
                
                // 当窗口内种类正好等于 require，且所有种类都达标（次数 >= k）时
                if (satisfy == require) {
                    ans = max(ans, r - l + 1);
                }
            }
        }
        return ans;
    }
};
