```cpp
// 062 Code01 地图分析

// 测试链接 : https://leetcode.cn/problems/as-far-from-land-as-possible/
// 核心思路：多源 BFS（广度优先搜索）。
// 1. 将所有陆地单元格作为起点，全部加入队列，同时统计海洋的数量。
// 2. 如果网格全为海洋或全为陆地，按题意直接返回 -1。
// 3. 像水波扩散一样，以所有陆地为第 0 层，逐层向海洋扩展（BFS），每次扩展距离加 1。
// 4. 最后一次扩展到达的海洋，就是距离陆地最远的海洋。真正的最大曼哈顿距离为总扩展层数减 1。
// 时间复杂度 O(N * M)，空间复杂度 O(N * M)

#include <iostream>
#include <vector>

using namespace std;

const int MAXN = 101;
const int MAXM = 101;

// 全局静态数组，避免动态分配开销，保持 ACM 竞赛风格
int q[MAXN * MAXM][2];
bool visited[MAXN][MAXM];
int l_ptr, r_ptr; // 避免与 std 的 l/r 命名冲突

// 0:上，1:右，2:下，3:左
// 利用长度为 5 的数组巧妙表示 4 个方向的坐标偏移
int mv[5] = { -1, 0, 1, 0, -1 };

class Solution {
public:
    int maxDistance(vector<vector<int>>& grid) {
        // LeetCode 会使用同一个对象多次调用，需重置全局指针
        l_ptr = r_ptr = 0;
        int n = grid.size();
        int m = grid[0].size();
        int seas = 0;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (grid[i][j] == 1) {
                    visited[i][j] = true;
                    q[r_ptr][0] = i;
                    q[r_ptr++][1] = j;
                } else {
                    visited[i][j] = false;
                    seas++;
                }
            }
        }

        if (seas == 0 || seas == n * m) {
            return -1;
        }

        int level = 0;
        while (l_ptr < r_ptr) {
            level++;
            int size = r_ptr - l_ptr;
            for (int k = 0; k < size; k++) {
                int x = q[l_ptr][0];
                int y = q[l_ptr++][1];
                for (int i = 0; i < 4; i++) {
                    int nx = x + mv[i];
                    int ny = y + mv[i + 1];
                    if (nx >= 0 && nx < n && ny >= 0 && ny < m && !visited[nx][ny]) {
                        visited[nx][ny] = true;
                        q[r_ptr][0] = nx;
                        q[r_ptr++][1] = ny;
                    }
                }
            }
        }
        
        return level - 1;
    }
};
```

---


**核心套路:** 多源 BFS。与其遍历每个海洋去找最近的陆地（极易超时），不如反向思考：把所有的陆地（1）全部作为起点放入队列，让它们同时像“水波纹”一样向外一圈圈扩散（涨潮）。最后被波纹覆盖到的海洋格子，就是距离所有陆地最远的海洋。距离 = 扩散的总层数 - 1。
