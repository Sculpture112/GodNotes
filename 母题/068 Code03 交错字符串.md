**核心套路:** 
解决此类问题的核心在于：**“自顶向下”思考状态转移，再“自底向上”进行填表。** #### 1. 状态定义 设 `dp[i][j]` 表示：`s1` 的前 `i` 个字符和 `s2` 的前 `j` 个字符，能否交错拼接成 `s3` 的前 `i+j` 个字符。

#### 2. 状态转移方程（核心推导）

当前状态 `dp[i][j]` 对应的 `s3` 的最后一个字符是 `s3[i+j-1]`。这个字符只能由 `s1[i-1]` 带来，或者由 `s2[j-1]` 带来。 由此产生两条转移路径：

- **路径一（选 s1）**：如果 `s1[i-1] == s3[i+j-1]`，那么能否交错就取决于 `dp[i-1][j]`（即 `s1` 少用一个字符时的状态）。
    
- **路径二（选 s2）**：如果 `s2[j-1] == s3[i+j-1]`，那么能否交错就取决于 `dp[i][j-1]`（即 `s2` 少用一个字符时的状态）。
    

两者只要有一个为真即可，由此得出： `dp[i][j] = (s1[i - 1] == s3[i + j - 1] && dp[i - 1][j]) || (s2[j - 1] == s3[i + j - 1] && dp[i][j - 1])`

```cpp
// 068 Code03 交错字符串

// 给定三个字符串 s1、s2、s3
// 请帮忙验证s3是否由s1和s2交错组成
// 测试链接 : https://leetcode.cn/problems/interleaving-string/

#include <iostream>
#include <vector>
#include <string>

using namespace std;

class Solution {
public:
    bool isInterleave(string s1, string s2, string s3) {
        // 两种方法任选其一均可
        // return isInterleave1(s1, s2, s3);
        return isInterleave2(s1, s2, s3);
    }

    // 二维动态规划
    bool isInterleave1(const string& s1, const string& s2, const string& s3) {
        if (s1.length() + s2.length() != s3.length()) {
            return false;
        }
        
        int n = s1.length();
        int m = s2.length();
        
        // dp[i][j]:
        // s1[前缀长度为i]和s2[前缀长度为j]，能否交错组成出s3[前缀长度为i+j]
        vector<vector<bool>> dp(n + 1, vector<bool>(m + 1, false));
        dp[0][0] = true;
        
        for (int i = 1; i <= n; i++) {
            if (s1[i - 1] != s3[i - 1]) {
                break;
            }
            dp[i][0] = true;
        }
        
        for (int j = 1; j <= m; j++) {
            if (s2[j - 1] != s3[j - 1]) {
                break;
            }
            dp[0][j] = true;
        }
        
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                dp[i][j] = (s1[i - 1] == s3[i + j - 1] && dp[i - 1][j]) || 
                           (s2[j - 1] == s3[i + j - 1] && dp[i][j - 1]);
            }
        }
        
        return dp[n][m];
    }

    // 空间压缩
    bool isInterleave2(const string& s1, const string& s2, const string& s3) {
        if (s1.length() + s2.length() != s3.length()) {
            return false;
        }
        
        int n = s1.length();
        int m = s2.length();
        
        vector<bool> dp(m + 1, false);
        dp[0] = true;
        
        for (int j = 1; j <= m; j++) {
            if (s2[j - 1] != s3[j - 1]) {
                break;
            }
            dp[j] = true;
        }
        
        for (int i = 1; i <= n; i++) {
            dp[0] = s1[i - 1] == s3[i - 1] && dp[0];
            for (int j = 1; j <= m; j++) {
                dp[j] = (s1[i - 1] == s3[i + j - 1] && dp[j]) || 
                        (s2[j - 1] == s3[i + j - 1] && dp[j - 1]);
            }
        }
        
        return dp[m];
    }
};
```
