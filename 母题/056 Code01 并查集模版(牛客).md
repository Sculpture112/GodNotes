// 056 Code01 并查集模版(牛客)

// 测试链接 : https://www.nowcoder.com/practice/e7ed657974934a30b2010046536a5372
// 核心思路：并查集（Union-Find）。高效支持合并两个集合，以及查询两个元素是否属于同一个集合。
// 1. 初始化：每个元素的父节点指向自己，初始所在集合大小为 1。
// 2. 查找 (find)：通过迭代向上寻找代表节点（父节点是自己的节点）。寻找过程中使用局部数组模拟栈收集沿途节点，并在找到代表节点后进行“路径压缩”，将它们直接挂在代表节点下，极大降低后续查找深度。
// 3. 合并 (union)：找到两个元素各自的代表节点。若不相同，则采用“小挂大”策略（按大小合并），将节点数较少的集合的代表节点挂到节点数较多的集合的代表节点下，维持树的相对平衡。
// 时间复杂度：单次查找与合并的均摊时间复杂度为 O(1)（严格来说是反阿克曼函数），额外空间复杂度 O(N)

#include <iostream>
#include <vector>

using namespace std;

// 针对牛客网的快速读入模板，包含 EOF 判断
inline bool read(int &x) {
    x = 0; int f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == EOF) return false;
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    x *= f;
    return true;
}

// 将并查集逻辑封装在类中，避免全局变量污染，生命周期由实例控制
class UnionFind {
private:
    vector<int> father;
    vector<int> size;
    vector<int> stack;

public:
    // 构造函数：初始化并查集
    UnionFind(int n) {
        father.resize(n + 1);
        size.resize(n + 1, 1);
        stack.resize(n + 1);
        for (int i = 0; i <= n; i++) {
            father[i] = i;
        }
    }

    // i 号节点，往上一直找，找到代表节点并返回（带路径压缩）
    int find(int i) {
        int s = 0; // 模拟栈的大小
        while (i != father[i]) {
            stack[s++] = i;
            i = father[i];
        }
        // 沿途节点收集好了，i 已经跳到代表节点了
        // 开始路径压缩，全部直接挂在代表节点下
        while (s > 0) {
            father[stack[--s]] = i;
        }
        return i;
    }

    // 判断两个节点是否属于同一个集合
    bool isSameSet(int x, int y) {
        return find(x) == find(y);
    }

    // 合并两个节点所在的集合
    void unite(int x, int y) {
        int fx = find(x);
        int fy = find(y);
        if (fx != fy) {
            // fx 是集合的代表，fy 是集合的代表
            // 小挂大策略
            if (size[fx] >= size[fy]) {
                size[fx] += size[fy];
                father[fy] = fx;
            } else {
                size[fy] += size[fx];
                father[fx] = fy;
            }
        }
    }
};

int main() {
    int n, m;
    // 多组测试数据循环读取
    while (read(n) && read(m)) {
        // 根据当前的 n 实例化并查集
        UnionFind uf(n);
        
        for (int i = 0; i < m; i++) {
            int op, x, y;
            read(op);
            read(x);
            read(y);
            
            if (op == 1) {
                if (uf.isSameSet(x, y)) {
                    printf("Yes\n");
                } else {
                    printf("No\n");
                }
            } else if (op == 2) {
                uf.unite(x, y);
            }
        }
    }
    return 0;
}
