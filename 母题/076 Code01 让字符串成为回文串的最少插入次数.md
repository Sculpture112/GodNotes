// 076 Code01 让字符串成为回文串的最少插入次数

// 给你一个字符串 s
// 每一次操作你都可以在字符串的任意位置插入任意字符
// 请你返回让 s 成为回文串的最少操作次数
// 测试链接 : https://leetcode.cn/problems/minimum-insertion-steps-to-make-a-string-palindrome/

#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

class Code01_MinimumInsertionToPalindrome {
public:
    // ==========================================
    // 解法一：暴力尝试
    // ==========================================
    static int minInsertions1(string str) {
        int n = str.length();
        return f1(str, 0, n - 1);
    }

    // s[l....r] 这个范围上的字符串，整体都变成回文串
    // 返回至少插入几个字符
    static int f1(const string& s, int l, int r) {
        if (l == r) {
            return 0; // 单个字符本身就是回文
        }
        if (l + 1 == r) {
            return s[l] == s[r] ? 0 : 1; // 两个字符，相等不用插，不等插1个
        }
        
        // 超过两个字符的情况
        if (s[l] == s[r]) {
            return f1(s, l + 1, r - 1);
        } else {
            return min(f1(s, l, r - 1), f1(s, l + 1, r)) + 1;
        }
    }

    // ==========================================
    // 解法二：记忆化搜索 (Top-Down DP)
    // ==========================================
    static int minInsertions2(string str) {
        int n = str.length();
        // 初始化 dp 表为 -1，表示尚未计算
        vector<vector<int>> dp(n, vector<int>(n, -1));
        return f2(str, 0, n - 1, dp);
    }

    static int f2(const string& s, int l, int r, vector<vector<int>>& dp) {
        if (dp[l][r] != -1) {
            return dp[l][r];
        }
        int ans;
        if (l == r) {
            ans = 0;
        } else if (l + 1 == r) {
            ans = s[l] == s[r] ? 0 : 1;
        } else {
            if (s[l] == s[r]) {
                ans = f2(s, l + 1, r - 1, dp);
            } else {
                ans = min(f2(s, l, r - 1, dp), f2(s, l + 1, r, dp)) + 1;
            }
        }
        dp[l][r] = ans;
        return ans;
    }

    // ==========================================
    // 解法三：严格位置依赖的动态规划 (Bottom-Up DP)
    // ==========================================
    static int minInsertions3(string str) {
        const string& s = str;
        int n = s.length();
        if (n < 2) return 0;
        
        vector<vector<int>> dp(n, vector<int>(n, 0));
        
        // 填写对角线相邻的两条线
        for (int l = 0; l < n - 1; l++) {
            dp[l][l + 1] = s[l] == s[l + 1] ? 0 : 1;
        }
        
        // 从下往上，从左往右填表
        for (int l = n - 3; l >= 0; l--) {
            for (int r = l + 2; r < n; r++) {
                if (s[l] == s[r]) {
                    dp[l][r] = dp[l + 1][r - 1]; // 依赖左下角
                } else {
                    dp[l][r] = min(dp[l][r - 1], dp[l + 1][r]) + 1; // 依赖左边和下边
                }
            }
        }
        return dp[0][n - 1];
    }

    // ==========================================
    // 解法四：空间压缩优化
    // ==========================================
    static int minInsertions4(string str) {
        const string& s = str;
        int n = s.length();
        if (n < 2) {
            return 0;
        }
        
        // 只需要一个一维数组滚动更新
        vector<int> dp(n, 0);
        dp[n - 1] = s[n - 2] == s[n - 1] ? 0 : 1;
        
        // leftDown 用于保存被覆盖前的 dp[l+1][r-1]（即左下角的值）
        // backUp 用于临时周转
        for (int l = n - 3, leftDown, backUp; l >= 0; l--) {
            leftDown = dp[l + 1];
            dp[l + 1] = s[l] == s[l + 1] ? 0 : 1;
            for (int r = l + 2; r < n; r++) {
                backUp = dp[r]; // 更新当前格子前，先把原来的值存下来，作为下一个格子的 leftDown
                if (s[l] == s[r]) {
                    dp[r] = leftDown;
                } else {
                    dp[r] = min(dp[r - 1], dp[r]) + 1;
                }
                leftDown = backUp;
            }
        }
        return dp[n - 1];
    }
};
