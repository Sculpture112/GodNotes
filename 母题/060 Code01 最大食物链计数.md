[[拓扑排序]] [[链式前向星]]
```c++
// 060 Code01 最大食物链计数
// 测试链接 : https://www.luogu.com.cn/problem/P4017
// 核心思路：拓扑排序 + 动态规划递推。
// 复杂度：O(N + M)

#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>

using namespace std;

const int MAXN = 5001;
const int MAXM = 500001;
const int MOD = 80112002;

// 链式前向星
int head[MAXN], nxt[MAXM], to[MAXM], cnt;
int indegree[MAXN], outdegree[MAXN]; // 需要额外记录出度
int lines[MAXN]; // 记录到达每个点的路径数
int q[MAXN];

void build(int n) {
    cnt = 1;
    memset(head, 0, sizeof(int) * (n + 1));
    memset(indegree, 0, sizeof(int) * (n + 1));
    memset(outdegree, 0, sizeof(int) * (n + 1));
    memset(lines, 0, sizeof(int) * (n + 1));
}

void addEdge(int u, int v) {
    nxt[cnt] = head[u];
    to[cnt] = v;
    head[u] = cnt++;
    outdegree[u]++; // 统计出度
    indegree[v]++;  // 统计入度
}

int solve(int n) {
    int l = 0, r = 0;
    // 1. 找到所有最初级动物（入度为 0）
    for (int i = 1; i <= n; i++) {
        if (indegree[i] == 0) {
            q[r++] = i;
            lines[i] = 1;
        }
    }

    int ans = 0;
    while (l < r) {
        int u = q[l++];
        
        // 2. 如果是顶级捕食者（出度为 0），其结果累加到答案
        if (outdegree[u] == 0) {
            ans = (ans + lines[u]) % MOD;
        }

        // 3. 遍历邻居，传递路径计数
        for (int ei = head[u]; ei > 0; ei = nxt[ei]) {
            int v = to[ei];
            lines[v] = (lines[v] + lines[u]) % MOD;
            if (--indegree[v] == 0) {
                q[r++] = v;
            }
        }
    }
    return ans;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    while (cin >> n >> m) {
        build(n);
        for (int i = 0, u, v; i < m; i++) {
            cin >> u >> v;
            addEdge(u, v);
        }
        cout << solve(n) << endl;
    }
    return 0;
}
```
