// 076 Code06 布尔运算

// 给定一个布尔表达式和一个期望的布尔结果 result
// 布尔表达式由 0 (false)、1 (true)、& (AND)、 | (OR) 和 ^ (XOR) 符号组成
// 布尔表达式一定是正确的，不需要检查有效性
// 但是其中没有任何括号来表示优先级
// 你可以随意添加括号来改变逻辑优先级
// 目的是让表达式能够最终得出result的结果
// 返回最终得出result有多少种不同的逻辑计算顺序
// 测试链接 : https://leetcode.cn/problems/boolean-evaluation-lcci/

#include <iostream>
#include <string>
#include <vector>

using namespace std;

class Code06_BooleanEvaluation {
public:
    // ==========================================
    // 解法：记忆化搜索 (区间 DP)
    // ==========================================
    static int countEval(string str, int result) {
        int n = str.length();
        if (n == 0) return 0;
        
        // dp[i][j][0] 存 false 的方法数
        // dp[i][j][1] 存 true 的方法数
        // 初始化为 -1，表示尚未计算过
        vector<vector<vector<int>>> dp(n, vector<vector<int>>(n, vector<int>(2, -1)));
        
        vector<int> ft = f(str, 0, n - 1, dp);
        return ft[result];
    }

private:
    // s[l...r] 是表达式的一部分，且一定符合范式
    // 0/1  逻  0/1   逻       0/1
    //  l  l+1  l+2  l+3........r
    // s[l...r]  0 : ?
    //           1 : ?
    // 返回值 : vector<int>，ans[0] = false方法数, ans[1] = true方法数
    static vector<int> f(const string& s, int l, int r, vector<vector<vector<int>>>& dp) {
        // 如果已经计算过，直接返回缓存的结果
        if (dp[l][r][0] != -1) {
            return dp[l][r];
        }
        
        int f_cnt = 0;
        int t_cnt = 0;
        
        if (l == r) {
            // 只剩一个字符，0/1
            f_cnt = (s[l] == '0') ? 1 : 0;
            t_cnt = (s[l] == '1') ? 1 : 0;
        } else {
            // 枚举每一个逻辑符号最后执行，k = l+1 ... r-1，步长为 2
            for (int k = l + 1; k < r; k += 2) {
                // l ... r
                vector<int> left = f(s, l, k - 1, dp);
                int a = left[0]; // 左侧为 false 的方法数
                int b = left[1]; // 左侧为 true 的方法数
                
                vector<int> right = f(s, k + 1, r, dp);
                int c = right[0]; // 右侧为 false 的方法数
                int d = right[1]; // 右侧为 true 的方法数
                
                if (s[k] == '&') {
                    f_cnt += a * c + a * d + b * c;
                    t_cnt += b * d;
                } else if (s[k] == '|') {
                    f_cnt += a * c;
                    t_cnt += a * d + b * c + b * d;
                } else {
                    // s[k] == '^' (异或运算)
                    f_cnt += a * c + b * d;
                    t_cnt += a * d + b * c;
                }
            }
        }
        
        // 存入 dp 缓存
        dp[l][r][0] = f_cnt;
        dp[l][r][1] = t_cnt;
        
        return dp[l][r];
    }
};
