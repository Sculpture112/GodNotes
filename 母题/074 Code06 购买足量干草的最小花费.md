// 074 Code06 购买足量干草的最小花费

// 有 n 个提供干草的公司，每个公司都有两个信息
// cost[i] 代表购买 1 次产品需要花的钱
// val[i] 代表购买 1 次产品所获得的干草数量
// 每个公司的产品都可以购买任意次
// 你一定要至少购买 h 数量的干草，返回最少要花多少钱
// 测试链接 : https://www.luogu.com.cn/problem/P2918

#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

/**
 * 逻辑核心分析：
 * 1. 这是一个“完全背包”的变种。标准完全背包是“不超过容量，求最大价值”，
 *    本题是“不低于目标，求最小花费”。
 * 
 * 2. 目标是“至少”购买 h 磅干草。由于干草是成捆卖的，买恰好 h 磅的开销可能
 *    比买稍微超过 h 磅（比如 h + 2）的开销更大。
 * 
 * 3. 搜索范围的确定：
 *    最大的单捆干草数量为 maxv。如果在某次决策中我们已经得到了超过 h 的重量，
 *    那么重量上限设为 h + maxv 已经足够。因为任何超过 h + maxv 的方案，
 *    都可以通过减去一捆干草来得到一个重量仍然 >= h 且花费更低的方案。
 */

const int MAXN = 105;
const int MAXM = 55005; // h(50000) + maxv(5000)

int val[MAXN];
int cost[MAXN];
int dp[MAXM];
int n, h, m;

/**
 * 空间压缩版的完全背包实现
 * 时间复杂度 O(n * m)，空间复杂度 O(m)
 */
int compute() {
    // 初始化 DP 数组：
    // dp[j] 表示凑齐恰好 j 磅干草的最小花费。
    // 起始时 dp[0] = 0，其余为无穷大。
    for (int j = 1; j <= m; j++) {
        dp[j] = INT_MAX;
    }
    dp[0] = 0;

    for (int i = 1; i <= n; i++) {
        // 完全背包正序遍历
        for (int j = val[i]; j <= m; j++) {
            if (dp[j - val[i]] != INT_MAX) {
                dp[j] = min(dp[j], dp[j - val[i]] + cost[i]);
            }
        }
    }

    // 最终答案在 [h, m] 范围内寻找最小花费
    int ans = INT_MAX;
    for (int j = h; j <= m; j++) {
        ans = min(ans, dp[j]);
    }
    return ans;
}

int main() {
    // 提升 I/O 效率，对应 Java 的 StreamTokenizer
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n >> h) {
        int maxv = 0;
        for (int i = 1; i <= n; i++) {
            cin >> val[i] >> cost[i];
            if (val[i] > maxv) {
                maxv = val[i];
            }
        }
        
        // 关键点：将背包容量上限扩充到 h + maxv
        m = h + maxv;
        
        cout << compute() << "\n";
    }

    return 0;
}
