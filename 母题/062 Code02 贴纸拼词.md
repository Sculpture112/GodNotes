// 062 Code02 贴纸拼词

// 我们有 n 种不同的贴纸。每个贴纸上都有一个小写的英文单词。
// 您想要拼写出给定的字符串 target ，方法是从收集的贴纸中切割单个字母并重新排列它们
// 如果你愿意，你可以多次使用每个贴纸，每个贴纸的数量是无限的。
// 返回你需要拼出 target 的最小贴纸数量。如果任务不可能，则返回 -1
// 注意：在所有的测试用例中，所有的单词都是从 1000 个最常见的美国英语单词中随机选择的
// 并且 target 被选择为两个随机单词的连接。
// 测试链接 : https://leetcode.cn/problems/stickers-to-spell-word/

#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <algorithm>

using namespace std;

// 依 Java 代码设定队列最大容量
const int MAXN = 401;

// 全局静态数组和指针，保持 ACM 竞赛风格
string q[MAXN];
int l_ptr, r_ptr;

// 下标0 -> a
// 下标1 -> b
// 下标2 -> c
// ...
// 下标25 -> z
vector<string> graph[26];
unordered_set<string> visited;

class Solution {
public:
    // 宽度优先遍历的解法
    // 也可以使用动态规划
    // 后续课程会有动态规划专题讲解
    int minStickers(vector<string>& stickers, string target) {
        // LeetCode 会使用同一个对象多次调用，需重置全局状态
        for (int i = 0; i < 26; i++) {
            graph[i].clear();
        }
        visited.clear();

        for (string str : stickers) {
            sort(str.begin(), str.end());
            for (int i = 0; i < str.length(); i++) {
                // 只在字母第一次出现时，将该贴纸加入对应的图中
                if (i == 0 || str[i] != str[i - 1]) {
                    graph[str[i] - 'a'].push_back(str);
                }
            }
        }

        sort(target.begin(), target.end());
        visited.insert(target);
        l_ptr = r_ptr = 0;
        q[r_ptr++] = target;
        int level = 1;

        // 使用队列的形式是整层弹出
        while (l_ptr < r_ptr) {
            int size = r_ptr - l_ptr;
            for (int i = 0; i < size; i++) {
                string cur = q[l_ptr++];
                // 根据当前需要的第一个字符，去寻找包含该字符的贴纸集合
                for (const string& s : graph[cur[0] - 'a']) {
                    string nxt = get_next(cur, s);
                    if (nxt == "") {
                        return level;
                    } else if (!visited.count(nxt)) {
                        visited.insert(nxt);
                        // 循环数组防止溢出越界（原 Java 代码未取模是因为状态数较少直接用 401，
                        // 为保证严格转换，此处遵循原逻辑直接使用 r_ptr++，
                        // 若实际提交出现越界可改为 q[r_ptr++ % MAXN]）
                        q[r_ptr++ % MAXN] = nxt; 
                    }
                }
            }
            level++;
        }
        return -1;
    }

    // 命名为 get_next 避免与 std::next 发生冲突
    string get_next(const string& t, const string& s) {
        string builder = "";
        for (int i = 0, j = 0; i < t.length();) {
            if (j == s.length()) {
                builder += t[i++];
            } else {
                if (t[i] < s[j]) {
                    builder += t[i++];
                } else if (t[i] > s[j]) {
                    j++;
                } else {
                    i++;
                    j++;
                }
            }
        }
        return builder;
    }
};
