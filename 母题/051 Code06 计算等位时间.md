// 051 Code06 计算等位时间

// 给定一个数组arr长度为n，表示n个服务员，每服务一个人的时间
// 给定一个正数m，表示有m个人等位，如果你是刚来的人，请问你需要等多久？
// 假设m远远大于n，比如n <= 10^3, m <= 10^9
// 该题目是二分答案法在实际面试中的经典应用（Google 高频题）

#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <climits>
#include <ctime>

using namespace std;

// 堆模拟：由于需要对数器验证，故保留此方法
// 结构体用于堆存储服务员状态：醒来时间，服务单人耗时
struct Server {
    long long awake_time;
    int service_time;

    // 醒来时间早的排在堆顶
    bool operator>(const Server& other) const {
        return awake_time > other.awake_time;
    }
};

// 堆模拟方法（验证用，m 很大时会超时）
long long waitingTime1(const vector<int>& arr, int m) {
    priority_queue<Server, vector<Server>, greater<Server>> pq;
    for (int x : arr) {
        pq.push({0, x});
    }
    for (int i = 0; i < m; i++) {
        Server cur = pq.top();
        pq.pop();
        cur.awake_time += cur.service_time;
        pq.push(cur);
    }
    return pq.top().awake_time;
}

// 核心判定函数：如果每个服务员工作 time 时间，总共可以接待多少位客人？
//（包括正在开始服务的客人）
long long f(const vector<int>& arr, long long time) {
    long long ans = 0;
    for (int num : arr) {
        // time / num 表示已经服务完的客人，+1 表示在 time 时刻刚开始服务的客人
        ans += (time / num) + 1;
    }
    return ans;
}

// 二分答案法（最优解）
// 时间复杂度 O(n * log(min_service * m))，额外空间复杂度 O(1)
long long waitingTime2(const vector<int>& arr, int m) {
    if (m == 0) return 0;
    int min_service = INT_MAX;
    for (int x : arr) {
        min_service = min(min_service, x);
    }

    long long ans = 0;
    // 二分范围：[0, 最快服务员独自处理完所有人所需的时间]
    // 注意：m 为 10^9 时，乘积会溢出 int，必须使用 long long
    long long l = 0;
    long long r = (long long)min_service * m;

    while (l <= r) {
        long long mid = l + ((r - l) >> 1);
        // 如果在 mid 时间内能接待的人数 >= m + 1 (包括排在第 m+1 位的自己)
        if (f(arr, mid) >= (long long)m + 1) {
            ans = mid;
            r = mid - 1;
        } else {
            l = mid + 1;
        }
    }
    return ans;
}

// 对数器辅助：生成随机数组
vector<int> randomArray(int n, int v) {
    vector<int> arr(n);
    for (int i = 0; i < n; i++) {
        arr[i] = (rand() % v) + 1;
    }
    return arr;
}

int main() {
    // 竞赛/测试效率优化
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cout << "测试开始" << endl;
    srand(time(NULL));
    int N = 50;
    int V = 30;
    int M = 3000;
    int testTime = 20000;

    for (int i = 0; i < testTime; i++) {
        int n = (rand() % N) + 1;
        vector<int> arr = randomArray(n, V);
        int m = rand() % M;
        long long ans1 = waitingTime1(arr, m);
        long long ans2 = waitingTime2(arr, m);
        if (ans1 != ans2) {
            cout << "出错了!" << endl;
            break;
        }
    }
    cout << "测试结束" << endl;

    return 0;
}
