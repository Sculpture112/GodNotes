**核心套路:** 
- **基础 DP 转移 ($O(N^2)$)**：定义 `dp[i]` 为必须以 `nums[i]` 结尾的最长严格递增子序列的长度。对于每个 `i`，枚举它前面的所有 `j` ($0 \le j < i$)，只要满足 `nums[j] < nums[i]`，即可进行状态转移：$dp[i] = \max(dp[i], dp[j] + 1)$。全局最大值即为所求。
    
- **贪心 + 二分查找 ($O(N \log N)$)**：维护一个 `ends` 数组，`ends[i]` 代表目前所有长度为 `i + 1` 的递增子序列中，**最小的结尾数值**。
    
    - 遍历 `nums`，如果当前数比 `ends` 里的最后一个有效数（即当前最大长度的最小结尾）还要大，说明它能扩展递增序列的长度，直接加到 `ends` 有效区的末尾。
        
    - 否则，利用二分查找在 `ends` 有效区中找到第一个 $\ge nums[i]$ 的数字，用 `nums[i]` 将其替换。**核心目的**：让相同长度的递增子序列的结尾变得尽可能小，从而更容易在未来接上新的数字。
        
- **衍生考点**：如果是求“最长不下降子序列”，只需修改二分查找的条件，去找第一个 $> nums[i]$ 的位置进行替换即可（即允许存在相同的数字）。

// 072 Code01 最长递增子序列和最长不下降子序列

// 给定一个整数数组 nums
// 找到其中最长严格递增子序列长度、最长不下降子序列长度
// 测试链接 : https://leetcode.cn/problems/longest-increasing-subsequence/

#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    /**
     * 普通解法的动态规划
     * 时间复杂度 O(n^2)，额外空间复杂度 O(n)
     */
    int lengthOfLIS1(vector<int>& nums) {
        int n = nums.size();
        if (n == 0) return 0;
        vector<int> dp(n);
        int ans = 0;
        for (int i = 0; i < n; i++) {
            dp[i] = 1;
            for (int j = 0; j < i; j++) {
                if (nums[j] < nums[i]) {
                    dp[i] = max(dp[i], dp[j] + 1);
                }
            }
            ans = max(ans, dp[i]);
        }
        return ans;
    }

    /**
     * 最优解：二分搜索法
     * 时间复杂度 O(n * logn)，额外空间复杂度 O(n)
     */
    int lengthOfLIS(vector<int>& nums) {
        int n = nums.size();
        if (n == 0) return 0;
        // ends[i] 表示目前所有长度为 i+1 的递增子序列中，最小的结尾数值
        vector<int> ends(n);
        int len = 0;
        for (int i = 0; i < n; i++) {
            // 在 ends 有效区中找 >= nums[i] 的最左位置
            int find = bs1(ends, len, nums[i]);
            if (find == -1) {
                // 如果没找到，说明 nums[i] 比当前所有结尾都大，扩展有效区
                ends[len++] = nums[i];
            } else {
                // 如果找到了，更新对应长度的最小结尾
                ends[find] = nums[i];
            }
        }
        return len;
    }

    /**
     * 如果求的是“最长不下降子序列” (Longest Non-decreasing Subsequence)
     * 只需要将主循环中的 bs1 换成 bs2 即可
     */
    int lengthOfNonDecreasing(vector<int>& nums) {
        int n = nums.size();
        if (n == 0) return 0;
        vector<int> ends(n);
        int len = 0;
        for (int i = 0; i < n; i++) {
            // 找 > nums[i] 的最左位置
            int find = bs2(ends, len, nums[i]);
            if (find == -1) {
                ends[len++] = nums[i];
            } else {
                ends[find] = nums[i];
            }
        }
        return len;
    }

private:
    /**
     * 最长递增子序列用的二分搜索：
     * ends[0...len-1] 是严格升序的，找到第一个 >= num 的位置
     * C++ STL 等价函数：std::lower_bound
     */
    int bs1(const vector<int>& ends, int len, int num) {
        int l = 0, r = len - 1, m, ans = -1;
        while (l <= r) {
            m = l + ((r - l) >> 1);
            if (ends[m] >= num) {
                ans = m;
                r = m - 1;
            } else {
                l = m + 1;
            }
        }
        return ans;
    }

    /**
     * 最长不下降子序列用的二分搜索：
     * ends[0...len-1] 是非降序的，找到第一个 > num 的位置
     * C++ STL 等价函数：std::upper_bound
     */
    int bs2(const vector<int>& ends, int len, int num) {
        int l = 0, r = len - 1, m, ans = -1;
        while (l <= r) {
            m = l + ((r - l) >> 1);
            if (ends[m] > num) {
                ans = m;
                r = m - 1;
            } else {
                l = m + 1;
            }
        }
        return ans;
    }
};
