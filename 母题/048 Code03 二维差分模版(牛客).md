// 048 Code03 二维差分模版(牛客)

// 二维差分模版(牛客)
// 测试链接 : https://www.nowcoder.com/practice/50e1a93989df42efb0b1dec386fb4ccc

#include <iostream>

using namespace std;

const int MAXN = 1005;
const int MAXM = 1005;

// 全局二维数组，防止命名冲突使用 diff_arr
long long diff_arr[MAXN][MAXM];
int n, m, q;

// 二维差分操作
void add(int a, int b, int c, int d, long long k) {
    diff_arr[a][b] += k;
    diff_arr[c + 1][b] -= k;
    diff_arr[a][d + 1] -= k;
    diff_arr[c + 1][d + 1] += k;
}

// 二维前缀和还原原数组
void build() {
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= m; j++) {
            diff_arr[i][j] += diff_arr[i - 1][j] + diff_arr[i][j - 1] - diff_arr[i - 1][j - 1];
        }
    }
}

// 多组测试数据清理
void clear_arr() {
    for (int i = 1; i <= n + 1; i++) {
        for (int j = 1; j <= m + 1; j++) {
            diff_arr[i][j] = 0;
        }
    }
}

int main() {
    // 竞赛必备：解绑 C/C++ I/O 流，极大提升输入输出效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n >> m >> q) {
        // 读取初始矩阵，视作在 (i, j) 到 (i, j) 范围加上对应的值
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                long long val;
                cin >> val;
                add(i, j, i, j, val);
            }
        }
        
        // 处理 q 次修改操作
        for (int i = 1; i <= q; i++) {
            int a, b, c, d;
            long long k;
            cin >> a >> b >> c >> d >> k;
            add(a, b, c, d, k);
        }
        
        build();
        
        for (int i = 1; i <= n; i++) {
            cout << diff_arr[i][1];
            for (int j = 2; j <= m; j++) {
                cout << " " << diff_arr[i][j];
            }
            cout << "\n";
        }
        
        clear_arr();
    }

    return 0;
}
