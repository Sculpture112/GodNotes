// 049 Code05 替换子串得到平衡字符串

// 有一个只含有 'Q', 'W', 'E', 'R' 四种字符，且长度为 n 的字符串。
// 假如在该字符串中，这四个字符都恰好出现 n/4 次，那么它就是一个「平衡字符串」。
// 给你一个这样的字符串 s，请通过「替换一个子串」的方式，使原字符串 s 变成一个「平衡字符串」。
// 你可以用和「待替换子串」长度相同的 任何 其他字符串来完成替换。
// 请返回待替换子串的最小可能长度。
// 如果原字符串自身就是一个平衡字符串，则返回 0。
// 测试链接 : https://leetcode.cn/problems/replace-the-substring-for-balanced-string/

#include <vector>
#include <string>
#include <algorithm>
#include <climits>

using namespace std;

class Solution {
public:
    // 时间复杂度 O(n)，其中 n 是字符串的长度
    // 额外空间复杂度 O(n)，用于存储映射后的数组 s；若直接在遍历时转换可优化至 O(1)
    int balancedString(string str) {
        int n = str.length();
        // 将字符映射为数字 0, 1, 2, 3 方便数组下标访问
        vector<int> s(n);
        int cnts[4] = {0};
        for (int i = 0; i < n; i++) {
            char c = str[i];
            s[i] = (c == 'W' ? 1 : (c == 'E' ? 2 : (c == 'R' ? 3 : 0)));
            cnts[s[i]]++;
        }

        int target = n / 4;
        int debt = 0;
        for (int i = 0; i < 4; i++) {
            if (cnts[i] <= target) {
                // 如果字符数量没有超过目标值，不需要被替换，对应债务设为 0
                cnts[i] = 0;
            } else {
                // 如果字符超标，计算其超出的“负债量”（用负数表示）
                cnts[i] = target - cnts[i];
                debt -= cnts[i];
            }
        }

        // 如果初始状态就是平衡的，直接返回 0
        if (debt == 0) {
            return 0;
        }

        int ans = INT_MAX;
        // 滑动窗口：[l, r] 代表待替换的子串
        for (int l = 0, r = 0; r < n; r++) {
            // 窗口右边界向右扩：如果进入窗口的字符是“超标字符”（cnts < 0），则总债务减少
            if (cnts[s[r]]++ < 0) {
                debt--;
            }
            
            // 当总债务为 0 时，说明当前窗口 [l, r] 内的字符通过替换，足以让整个字符串平衡
            if (debt == 0) {
                // 尝试收缩左边界：如果左边界字符在窗口内是多余的（cnts > 0），就将其踢出窗口
                while (cnts[s[l]] > 0) {
                    cnts[s[l++]]--;
                }
                // 更新最小可能长度
                ans = min(ans, r - l + 1);
            }
        }
        
        return ans;
    }
};
