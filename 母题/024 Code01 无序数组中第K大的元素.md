```cpp
// 024 Code01 无序数组中第K大的元素

// 测试链接 : https://leetcode.cn/problems/kth-largest-element-in-an-array/

#include <iostream>
#include <vector>
#include <cstdlib>
#include <algorithm>

using namespace std;

// 荷兰国旗问题边界变量
int first_idx, last_idx;

class Solution {
public:
    // 随机选择算法，时间复杂度 O(n)
    int findKthLargest(vector<int>& nums, int k) {
        // 第 K 大元素，对应如果数组升序排序后，索引为 size - k 的元素
        return randomizedSelect(nums, nums.size() - k);
    }

    // 如果 arr 排序的话，在 i 位置的数字是什么
    int randomizedSelect(vector<int>& arr, int i) {
        int ans = 0;
        for (int l = 0, r = arr.size() - 1; l <= r;) {
            // 随机这一下，常数时间比较大
            // 但只有这一下随机，才能在概率上把时间复杂度收敛到 O(n)
            int x = arr[l + rand() % (r - l + 1)];
            partition(arr, l, r, x);
            
            // 因为左右两侧只需要走一侧
            // 所以不需要像快排那样用临时变量记录全局的 first_idx、last_idx
            // 直接用即可
            if (i < first_idx) {
                r = first_idx - 1;
            } else if (i > last_idx) {
                l = last_idx + 1;
            } else {
                // 命中了 == x 的区域，直接返回
                ans = arr[i];
                break;
            }
        }
        return ans;
    }

    // 已知 arr[l....r] 范围上一定有 x 这个值
    // 划分数组 <x 放左边，==x 放中间，>x 放右边
    // 把全局变量 first_idx, last_idx，更新成 ==x 区域的左右边界
    void partition(vector<int>& arr, int l, int r, int x) {
        first_idx = l;
        last_idx = r;
        int i = l;
        while (i <= last_idx) {
            if (arr[i] == x) {
                i++;
            } else if (arr[i] < x) {
                // C++ 中直接使用 std::swap
                swap(arr[first_idx++], arr[i++]);
            } else {
                swap(arr[i], arr[last_idx--]);
            }
        }
    }
};
```

---


**核心套路:** 
快速选择的核心在于“舍弃”：与快速排序需要递归处理左右两侧不同，快速选择在每次划分（Partition）后，通过判断目标下标落在 `<x`、`==x` 还是 `>x` 区域，**只进入其中一侧**继续搜索。这样就能把时间复杂度从快排的 $O(N \log N)$ 降维打击到期望 $O(N)$。三路划分（荷兰国旗）的加入，完美解决了数组中存在“大量重复元素”时算法性能退化的问题。
