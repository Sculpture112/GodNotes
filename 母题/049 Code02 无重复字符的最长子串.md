// 049 Code02 无重复字符的最长子串

// 给定一个字符串 s ，请你找出其中不含有重复字符的 最长子串 的长度。
// 测试链接 : https://leetcode.cn/problems/longest-substring-without-repeating-characters/

#include <string>
#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    int lengthOfLongestSubstring(string str) {
        int n = str.length();
        // char -> int -> 0 ~ 255
        // 每一种字符上次出现的位置，初始化为 -1
        vector<int> last(256, -1);
        
        // 不含有重复字符的 最长子串 的长度
        int ans = 0;
        for (int l = 0, r = 0; r < n; r++) {
            // 当前字符的 ASCII 值
            // 使用 unsigned char 防止某些环境下 char 为负数导致越界
            unsigned char cur = (unsigned char)str[r];
            
            // 如果当前字符在之前出现过，左边界 l 跳到其上次出现位置的下一个位置
            // 使用 max 是为了确保 l 只向右移动（窗口不回缩）
            l = max(l, last[cur] + 1);
            
            // 计算当前窗口长度并更新最大值
            ans = max(ans, r - l + 1);
            
            // 更新当前字符上一次出现的位置
            last[cur] = r;
        }
        
        return ans;
    }
};
