// 053 Code01 最大宽度坡

// 测试链接 : https://leetcode.cn/problems/maximum-width-ramp/
// 核心思路：单调栈。寻找 A 中的坡的最大宽度，即最大的 j - i 满足 A[i] <= A[j]。

// 1. 正序遍历，维护一个从栈底到栈顶严格单调递减的索引栈。因为只有比之前元素更小的值，才可能在更短的距离内造就更大的坡度起点。

// 2. 逆序遍历数组，若当前值大于等于栈顶索引对应的值，说明构成了坡。

// 3. 弹出栈顶并更新答案，直到栈为空或当前值不再大于等于栈顶值，此过程保证了每个起点匹配到最远的终点。
// 时间复杂度 O(N)，额外空间复杂度 O(N)
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    int maxWidthRamp(vector<int>& arr) {
        int n = arr.size();
        // stack: 模拟栈，存储索引
        // C++ 直接在局部开辟定长 vector 即可，随生命周期自动销毁
        vector<int> stack(n);
        int r = 0; // 栈顶指针
        
        // 1. 将 0 位置优先入栈
        stack[r++] = 0;
        for (int i = 1; i < n; i++) {
            // 构建单调递减栈
            if (arr[stack[r - 1]] > arr[i]) {
                stack[r++] = i;
            }
        }
        
        int ans = 0;
        // 2. 逆序遍历，结算最大坡度
        for (int j = n - 1; j >= 0; j--) {
            // 当当前数字 >= 栈顶数字时，说明构成了坡，结算并弹出栈顶
            while (r > 0 && arr[stack[r - 1]] <= arr[j]) {
                ans = max(ans, j - stack[--r]);
            }
        }
        
        return ans;
    }
};
