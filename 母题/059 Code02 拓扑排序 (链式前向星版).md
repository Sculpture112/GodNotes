// 059 Code02 拓扑排序 (链式前向星版)
// 核心思路：静态数组链式前向星 + Kahn 算法
// 时间复杂度 O(N + M)，空间复杂度 O(N + M)，极致常数优化

#include <iostream>
#include <algorithm>
#include <cstring>

using namespace std;

const int MAXN = 200005;
const int MAXM = 200005;

// 链式前向星
int head[MAXN];
int nxt[MAXM];
int to[MAXM];
int cnt;

// 拓扑排序辅助
int indegree[MAXN];
int q[MAXN];
int ans[MAXN];
int l, r;

void build(int n) {
    cnt = 1; // 边编号从1开始
    memset(head, 0, sizeof(int) * (n + 1));
    memset(indegree, 0, sizeof(int) * (n + 1));
}

void addEdge(int u, int v) {
    nxt[cnt] = head[u];
    to[cnt] = v;
    head[u] = cnt++;
}

bool topoSort(int n) {
    l = r = 0;
    for (int i = 1; i <= n; i++) {
        if (indegree[i] == 0) {
            q[r++] = i;
        }
    }

    int fill = 0;
    while (l < r) {
        int cur = q[l++];
        ans[fill++] = cur;
        // 关键：链式前向星遍历
        for (int ei = head[cur]; ei != 0; ei = nxt[ei]) {
            int v = to[ei];
            if (--indegree[v] == 0) {
                q[r++] = v;
            }
        }
    }
    return fill == n;
}

int main() {
    // 快速 IO
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    while (cin >> n >> m) {
        build(n);
        for (int i = 0; i < m; i++) {
            int u, v;
            cin >> u >> v;
            addEdge(u, v);
            indegree[v]++;
        }

        if (topoSort(n)) {
            for (int i = 0; i < n; i++) {
                cout << ans[i] << (i == n - 1 ? "" : " ");
            }
            cout << "\n";
        } else {
            cout << "-1\n";
        }
    }
    return 0;
}
