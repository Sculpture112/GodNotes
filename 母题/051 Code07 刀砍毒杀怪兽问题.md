// 051 Code07 刀砍毒杀怪兽问题

// 怪兽的初始血量是一个整数hp，给出每一回合刀砍和毒杀的数值cuts和poisons
// 第i回合如果用刀砍，怪兽在这回合会直接损失cuts[i]的血，不再有后续效果
// 第i回合如果用毒杀，怪兽在这回合不会损失血量，但是之后每回合都损失poisons[i]的血量
// 并且你选择的所有毒杀效果，在之后的回合都会叠加
// 两个数组cuts、poisons，长度都是n，代表你一共可以进行n回合
// 每一回合你只能选择刀砍或者毒杀中的一个动作
// 返回至少多少回合，怪兽会死掉
// 测试链接 : 暂无（大厂笔试题），可用代码中的逻辑进行验证

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * 核心判定函数：判断在 limit 回合内是否能杀死怪兽
 * @param cuts 每一回合刀砍的伤害
 * @param poisons 每一回合毒杀每回合叠加的伤害
 * @param hp 怪兽初始血量
 * @param limit 限制的回合数
 * @return 是否达标
 */
bool f(const vector<int>& cuts, const vector<int>& poisons, long long hp, int limit) {
    int n = cuts.size();
    // 你只能在前 n 回合做出选择，或者在 limit 回合内结束
    int can_act = min(n, limit);
    
    for (int i = 0; i < can_act; i++) {
        int current_round = i + 1;
        // 贪心策略：
        // 这一回合如果刀砍，伤害是 cuts[i]
        // 这一回合如果毒杀，伤害会从第 i+2 回合持续到第 limit 回合
        // 持续的回合数是 (limit - current_round)
        long long cut_damage = cuts[i];
        long long poison_total_damage = (long long)(limit - current_round) * poisons[i];
        
        // 每一回合选择当前能带来的最大总收益
        hp -= max(cut_damage, poison_total_damage);
        
        if (hp <= 0) {
            return true;
        }
    }
    return hp <= 0;
}

/**
 * 二分答案法
 * 时间复杂度 O(n * log(hp))，额外空间复杂度 O(1)
 */
int fast2(const vector<int>& cuts, const vector<int>& poisons, int hp) {
    int ans = hp; // 最差情况每回合只打1点血
    int l = 1, r = hp + 1;
    
    while (l <= r) {
        int m = l + ((r - l) >> 1);
        if (f(cuts, poisons, (long long)hp, m)) {
            ans = m;
            r = m - 1;
        } else {
            l = m + 1;
        }
    }
    return ans;
}

int main() {
    // 竞赛必备：解绑 C/C++ I/O 流
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, hp;
    while (cin >> n >> hp) {
        vector<int> cuts(n);
        vector<int> poisons(n);
        for (int i = 0; i < n; i++) cin >> cuts[i];
        for (int i = 0; i < n; i++) cin >> poisons[i];
        
        cout << fast2(cuts, poisons, hp) << "\n";
    }

    return 0;
}
