```cpp
// 021 Code02 归并排序，填函数练习风格

// 测试链接 : https://leetcode.cn/problems/sort-an-array/
// 核心思路：归并排序（非递归版）。
// 1. 采用自底向上的迭代方式，避免递归深度造成的栈溢出风险和额外函数调用开销，是 LeetCode 填函数模式下的绝佳模板。
// 2. 初始步长 step 设为 1，每次将相邻的、长度为 step 的两个有序序列合并，合并后步长翻倍，直到步长覆盖整个数组。
// 3. 在合并过程中，利用局部开辟的辅助数组 help 暂存合并好的元素，合并完成后统一拷贝回原数组 nums。
// 4. 注意边界条件的处理，当左半边序列结束后，如果不存在右半边序列（m + 1 >= n），则本轮合并提前结束。
// 时间复杂度 O(N log N)，额外空间复杂度 O(N)

#include <vector>
#include <algorithm>

using namespace std;

class Solution {
private:
    // 将 nums[l...m] 和 nums[m+1...r] 两个有序区间合并
    void merge(vector<int>& nums, vector<int>& help, int l, int m, int r) {
        int i = l;
        int a = l;
        int b = m + 1;
        
        while (a <= m && b <= r) {
            help[i++] = nums[a] <= nums[b] ? nums[a++] : nums[b++];
        }
        while (a <= m) {
            help[i++] = nums[a++];
        }
        while (b <= r) {
            help[i++] = nums[b++];
        }
        // 将排好序的区间写回原数组
        for (i = l; i <= r; i++) {
            nums[i] = help[i];
        }
    }

public:
    vector<int> sortArray(vector<int>& nums) {
        int n = nums.size();
        if (n < 2) {
            return nums;
        }
        
        // 局部开辟辅助数组，生命周期随函数自动销毁
        vector<int> help(n);
        
        // step 为步长，每次翻倍
        for (int step = 1; step < n; step <<= 1) {
            int l = 0;
            while (l < n) {
                int m = l + step - 1;
                // 如果没有右侧区间，本组不需要合并，直接跳出处理下一轮步长
                if (m + 1 >= n) {
                    break;
                }
                // 确定右边界，防止越界
                int r = min(l + (step << 1) - 1, n - 1);
                merge(nums, help, l, m, r);
                // 移动到下一组的起点
                l = r + 1;
            }
        }
        
        return nums;
    }
};
```
