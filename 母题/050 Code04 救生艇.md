// 050 Code04 救生艇

// 给定数组 people，people[i] 表示第 i 个人的体重
// 船的数量不限，每艘船可以承载的最大重量为 limit
// 每艘船最多可同时载两人，但条件是这些人的重量之和最多为 limit
// 返回 承载所有人所需的最小船数
// 测试链接 : https://leetcode.cn/problems/boats-to-save-people/

#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    /**
     * 时间复杂度 O(n * logn)：主要开销在排序上。
     * 额外空间复杂度 O(1)：排序如果是原地排序则为 O(1)，忽略递归栈空间。
     */
    int numRescueBoats(vector<int>& people, int limit) {
        // 先对体重进行排序，以便使用双指针贪心策略
        sort(people.begin(), people.end());
        
        int ans = 0;
        int l = 0;
        int r = people.size() - 1;
        
        while (l <= r) {
            // 如果只剩一个人，重量就是他自己；否则尝试最轻的 + 最重的
            int sum = (l == r) ? people[l] : people[l] + people[r];
            
            if (sum > limit) {
                // 如果最重的和最轻的加起来都超重，那最重的人只能单独坐一艘船
                r--;
            } else {
                // 如果不超重，两人可以同乘一艘船，左右指针都向内移动
                l++;
                r--;
            }
            // 无论哪种情况，都需要消耗一艘船
            ans++;
        }
        
        return ans;
    }
};
