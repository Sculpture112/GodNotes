```cpp
// 050 Code07 缺失的第一个正数

// 给你一个未排序的整数数组 nums ，请你找出其中没有出现的最小的正整数。
// 请你实现时间复杂度为 O(n) 并且只使用常数级别额外空间的解决方案。
// 测试链接 : https://leetcode.cn/problems/first-missing-positive/

#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    /**
     * 时间复杂度 O(n)：每个元素要么被归位，要么进入垃圾区，每个元素最多处理两次。
     * 额外空间复杂度 O(1)：原地修改数组。
     */
    int firstMissingPositive(vector<int>& arr) {
        // l 的左边是已经处理好的区域，即满足 arr[i] == i + 1 的区域
        // 我们永远盯着 l 位置的数字看，看能不能扩充这个区域
        int l = 0;
        
        // [r...] 为“垃圾区”
        // 最理想的情况下，我们希望收集到 1~n 所有的正整数
        // 一旦发现某个数没法合法收集（重复、越界、负数），预期就会变差，r 左移
        int r = arr.size();
        
        while (l < r) {
            if (arr[l] == l + 1) {
                // 情况 1：已经在正确位置
                l++;
            } else if (arr[l] <= l || arr[l] > r || arr[arr[l] - 1] == arr[l]) {
                // 情况 2：数字不合法或者是重复值，扔进垃圾区
                // 1) arr[l] <= l: 可能是负数、0，或者是已经出现在左侧区域的数
                // 2) arr[l] > r: 超过了当前可能收集到的最大连续范围
                // 3) arr[arr[l] - 1] == arr[l]: 目标位置已经有正确的数了，当前是重复值
                swap(arr[l], arr[--r]);
            } else {
                // 情况 3：合法数字但不在正确位置，将其交换到目标下标 arr[l] - 1 处
                swap(arr[l], arr[arr[l] - 1]);
            }
        }
        
        // 最后 l 停下的位置，说明 1...l 都是对的，缺失的第一个正数就是 l + 1
        return l + 1;
    }
};
```
