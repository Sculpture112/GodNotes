// 052 Code03 子数组的最小值之和
// 测试链接 : https://leetcode.cn/problems/sum-of-subarray-minimums/
// 核心思路：单调栈 + 贡献法。
// 时间复杂度 O(N)，空间复杂度 O(N)

#include <iostream>
#include <vector>

using namespace std;

class Solution {
public:
    int sumSubarrayMins(vector<int>& arr) {
        int n = arr.size();
        long long ans = 0;
        int mod = 1e9 + 7;// n,ans,mod
        
        
        // 静态数组模拟栈，性能更好
        vector<int> st(n); //q,r
        int r = 0;

        for (int i = 0; i < n; i++) {
            // 当遇到更小的数时，结算栈顶
            // 注意这里使用 >= 是为了处理重复值，防止重复计数
            while (r > 0 && arr[st[r - 1]] >= arr[i]) {// 想象cur向两边扩张，一遇到比自己小的就停下。向右扩张时遇到相同元素停下。向左时不会遇到相同元素，因为之前踢出了
                int cur = st[--r];
                int left = (r == 0) ? -1 : st[r - 1];
                int right = i;
                // 计算贡献：(当前位置 - 左边界) * (右边界 - 当前位置) * 当前值
                ans = (ans + (long long)(cur - left) * (right - cur) % mod * arr[cur]) % mod;
            }
            st[r++] = i;
        }

        // 清算阶段：栈中剩余元素右侧没有比它们小的数了
        while (r > 0) {
            int cur = st[--r];
            int left = (r == 0) ? -1 : st[r - 1];
            int right = n;
            ans = (ans + (long long)(cur - left) * (right - cur) % mod * arr[cur]) % mod;
        }

        return (int)ans;
    }
};
