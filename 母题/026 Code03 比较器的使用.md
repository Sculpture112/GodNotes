// 026 Code03 比较器的使用

#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <string>

using namespace std;

// C++ 中用 struct 来定义，默认全 public，非常适合 ACM 竞赛
struct Employee {
    int company;
    int age;

    Employee(int c, int a) {
        company = c;
        age = a;
    }
};

// 在 C++ 中，自定义比较器通常通过重载 () 运算符的结构体（仿函数）来实现
struct EmployeeComparator {
    bool operator()(const Employee& o1, const Employee& o2) const {
        // C++ 与 Java 的比较器机制有本质不同！
        // Java 返回负数表示 o1 优先级高。
        // C++ 要求返回 bool 值：如果 o1 必须排在 o2 严格前面，则返回 true；否则返回 false。
        // 不再有大小的概念，就是优先级的概念。你想怎么排，就直接写期望的小于关系。
        return o1.age < o2.age;
    }
};

int main() {
    Employee s1(2, 27);
    Employee s2(1, 60);
    Employee s3(4, 19);
    Employee s4(3, 23);
    Employee s5(1, 35);
    Employee s6(3, 55);
    
    // 使用 vector 替代 Java 的数组
    vector<Employee> arr = { s1, s2, s3, s4, s5, s6 };
    
    // 使用仿函数进行排序
    sort(arr.begin(), arr.end(), EmployeeComparator());
    for (const Employee& e : arr) {
        cout << e.company << " , " << e.age << "\n";
    }

    cout << "=====\n";

    // C++ 中等价的 Lambda 表达式排序（按年龄降序，即 b.age - a.age 的效果）
    // 注意：如果是降序，期望 a 比 b 大时排前面，所以写 a.age > b.age
    sort(arr.begin(), arr.end(), [](const Employee& a, const Employee& b) {
        return a.age > b.age;
    });
    for (const Employee& e : arr) {
        cout << e.company << " , " << e.age << "\n";
    }

    cout << "=====\n";
    
    // 所有员工，先按照谁的公司编号小，谁在前；如果公司编号一样，谁年龄小谁在前
    sort(arr.begin(), arr.end(), [](const Employee& a, const Employee& b) {
        return a.company != b.company ? (a.company < b.company) : (a.age < b.age);
    });
    for (const Employee& e : arr) {
        cout << e.company << " , " << e.age << "\n";
    }

    // TreeSet 的 C++ 对应实现是 std::set
    set<Employee, EmployeeComparator> treeSet1;
    for (const Employee& e : arr) {
        treeSet1.insert(e);
    }
    cout << treeSet1.size() << "\n";

    // 会去重，因为 EmployeeComparator 中只比较 age，age 为 27 的已经存在了
    // C++ 的 set 会认为 !comp(a, b) && !comp(b, a) 即为相等，从而去重
    treeSet1.insert(Employee(2, 27));
    cout << treeSet1.size() << "\n";

    cout << "===\n";

    // 如果不想去重，Java 中的做法是增加更多比较（如 toString 内存地址）。
    // 在 C++ 中，最标准的做法是直接使用 std::multiset（允许重复元素的有序表）。
    // 这里我们依然按照复合条件排序，但放入 multiset 中。
    auto cmp = [](const Employee& a, const Employee& b) {
        return a.company != b.company ? (a.company < b.company) : (a.age < b.age);
    };
    
    // decltype 获取 lambda 的类型并传入 multiset
    multiset<Employee, decltype(cmp)> treeSet2(cmp);
    for (const Employee& e : arr) {
        treeSet2.insert(e);
    }
    cout << treeSet2.size() << "\n";

    // 不会去重，因为 multiset 原生支持相同元素共存
    treeSet2.insert(Employee(2, 27));
    cout << treeSet2.size() << "\n";

    cout << "===\n";

    // PriorityQueue (C++ 中是 priority_queue) 不会去重，不再展示了

    // 字典序
    string str1 = "abcde";
    string str2 = "ks";
    // C++ 的 string::compare 与 Java 的 compareTo 完全一致，返回 <0, 0, >0 的整数
    cout << str1.compare(str2) << "\n";
    cout << str2.compare(str1) << "\n";

    return 0;
}
