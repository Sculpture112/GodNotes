// 051 Code03 机器人跳跃问题

// 机器人正在玩一个古老的基于DOS的游戏
// 游戏中有N+1座建筑，从0到N编号，从左到右排列
// 编号为0的建筑高度为0个单位，编号为i的建筑的高度为H(i)个单位
// 起初机器人在编号为0的建筑处
// 每一步，它跳到下一个（右边）建筑。假设机器人在第k个建筑，且它现在的能量值是E
// 下一步它将跳到第个k+1建筑
// 它将会得到或者失去正比于与H(k+1)与E之差的能量
// 如果 H(k+1) > E 那么机器人就失去H(k+1)-E的能量值，否则它将得到E-H(k+1)的能量值
// 游戏目标是到达第个N建筑，在这个过程中，能量值不能为负数个单位
// 现在的问题是机器人以多少能量值开始游戏，才可以保证成功完成游戏
// 测试链接 : https://www.nowcoder.com/practice/7037a3d57bbd4336856b8e16a9cafd71

#include <iostream>
#include <algorithm>

using namespace std;

// 题目给出 N <= 10^5
const int MAXN = 100005;

int arr[MAXN];
int n;

/**
 * 核心判断函数：初始能量为 energy，是否能通关
 * @param energy 当前尝试的初始能量
 * @param max_h  所有建筑中的最大高度
 * @return 是否能顺利到达终点且能量不为负
 */
bool f(long long energy, int max_h) {
    // 逻辑推导：
    // 如果 H > E，新能量 = E - (H - E) = 2E - H
    // 如果 H <= E，新能量 = E + (E - H) = 2E - H
    // 结论：无论高度如何，能量更新公式统一为 2E - H
    for (int i = 1; i <= n; i++) {
        energy = 2 * energy - arr[i];

        // 关键保护：
        // 如果能量一旦超过了所有建筑的最大高度，
        // 那么 2E - H 必然大于等于 E (因为 E >= H)，能量只会增加不会减少
        // 这也防止了 energy 在 2E 迭代过程中超出 long long 的范围
        if (energy >= max_h) {
            return true;
        }
        // 能量为负，通关失败
        if (energy < 0) {
            return false;
        }
    }
    return true;
}

/**
 * 二分搜索通关所需的最小能量
 * 时间复杂度 O(N * log(max_h))
 */
int compute(int l, int r, int max_h) {
    int ans = -1;
    while (l <= r) {
        int m = l + ((r - l) >> 1);
        if (f(m, max_h)) {
            // 达标，记录答案并尝试更小的能量
            ans = m;
            r = m - 1;
        } else {
            // 不达标，增加初始能量
            l = m + 1;
        }
    }
    return ans;
}

int main() {
    // 竞赛必备：解绑 C/C++ I/O 流，极大提升输入输出效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n) {
        int max_h = 0;
        for (int i = 1; i <= n; i++) {
            cin >> arr[i];
            if (arr[i] > max_h) {
                max_h = arr[i];
            }
        }
        
        // 初始能量最小可能是 0，最大不回超过建筑中的最高高度
        cout << compute(0, max_h, max_h) << "\n";
    }

    return 0;
}
