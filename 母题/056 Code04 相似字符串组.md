```cpp
// 056 Code04 相似字符串组

// 测试链接 : https://leetcode.cn/problems/similar-string-groups/
// 核心思路：并查集。求相似字符串形成的连通分量个数。
// 1. 将每个字符串视作一个独立的节点，初始化并查集，初始时连通分量个数 sets 等于字符串总数 n。
// 2. 双重循环遍历所有的字符串对 (i, j)。若它们目前不在同一个集合中（即 find(i) != find(j)），则判断它们是否相似。
// 3. 相似的判断依据：两个字符串对应位置的字符不同的个数为 0（完全相同）或 2（仅交换一次即可相同）。若遍历过程中发现不同字符数超过 2（即 diff >= 3），则直接剪枝判定为不相似。
// 4. 若 (i, j) 相似，则合并它们所在的集合，并将连通分量个数 sets 减 1。
// 5. 最终遍历完成后，返回 sets 即为关联组的数量。
// 时间复杂度 O(N^2 * M)，其中 N 是字符串数量，M 是字符串长度；额外空间复杂度 O(N)

#include <vector>
#include <string>
#include <functional>

using namespace std;

class Solution {
public:
    int numSimilarGroups(vector<string>& strs) {
        int n = strs.size();
        int m = strs[0].length();
        
        // father: 并查集父节点数组
        vector<int> father(n);
        for (int i = 0; i < n; i++) {
            father[i] = i;
        }
        
        int sets = n; // 初始连通分量个数
        
        // Lambda 实现递归路径压缩的查找
        function<int(int)> find = [&](int i) {
            if (i != father[i]) {
                father[i] = find(father[i]);
            }
            return father[i];
        };
        
        // 极简合并操作
        auto unite = [&](int x, int y) {
            int fx = find(x);
            int fy = find(y);
            if (fx != fy) {
                father[fx] = fy;
                sets--; // 合并一次，独立分量少一个
            }
        };
        
        // 遍历所有字符串对
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                // 如果它们已经在同一个集合中，就没必要再去费时检查相似性了
                if (find(i) != find(j)) {
                    int diff = 0;
                    for (int k = 0; k < m && diff < 3; k++) {
                        if (strs[i][k] != strs[j][k]) {
                            diff++;
                        }
                    }
                    // 完全相等(0)或只差两个位置(2)即为相似
                    if (diff == 0 || diff == 2) {
                        unite(i, j);
                    }
                }
            }
        }
        
        return sets;
    }
};
```
