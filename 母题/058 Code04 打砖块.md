// 058 Code04 打砖块

// 测试链接 : https://leetcode.cn/problems/bricks-falling-when-hit/
// 核心思路：逆向思维 + 洪水填充（DFS）。
// 1. 正向敲击砖块会导致大面积碎裂掉落，且可能引发连锁反应，极难正向模拟。因此采用“时光倒流”的逆向思维。
// 2. 先将所有 hits 指定的砖块全部“敲掉”（对应网格值减 1）。
// 3. 从网格第一行（顶部）出发进行 DFS，将此时依然稳定、与顶部相连的砖块标记为 2。
// 4. 逆序遍历 hits 数组，尝试将砖块“时光倒流”补回（对应网格值加 1）。
// 5. 如果补回的砖块是“有价值”的（即它本身是 1，且它位于顶部，或者与目前标记为 2 的稳定砖块相邻），说明这块砖的出现能救活一片砖。
// 6. 从该位置开始触发 DFS，统计能新“救活”多少块砖。单次掉落的砖块数即为（新救活的砖块总数 - 1，减去被敲击的那块砖本身）。
// 时间复杂度 O(H + N * M)，其中 H 为 hits 数组的长度；额外空间复杂度 O(N * M)（最坏情况下递归栈的最大深度）

#include <vector>

using namespace std;

class Solution {
private:
    int n, m;
    // 使用指针保存网格引用，避免复制，也避免了全局变量污染
    vector<vector<int>>* g;

    // 从 (i, j) 出发，遇到 1 就感染成 2
    // 返回本次感染新增了多少个 2
    int dfs(int i, int j) {
        if (i < 0 || i == n || j < 0 || j == m || (*g)[i][j] != 1) {
            return 0;
        }
        (*g)[i][j] = 2;
        // 向上下左右蔓延
        return 1 + dfs(i + 1, j) + dfs(i, j + 1) + dfs(i - 1, j) + dfs(i, j - 1);
    }

    // 检查 (i, j) 位置补回的砖块是否有价值触发 DFS
    // 有价值的条件：它是 1，且它挨着顶部(i == 0)，或者四周有已经稳定的砖块(值为 2)
    bool worth(int i, int j) {
        return (*g)[i][j] == 1 &&
               (i == 0 ||
               (i > 0 && (*g)[i - 1][j] == 2) ||
               (i < n - 1 && (*g)[i + 1][j] == 2) ||
               (j > 0 && (*g)[i][j - 1] == 2) ||
               (j < m - 1 && (*g)[i][j + 1] == 2));
    }

public:
    vector<int> hitBricks(vector<vector<int>>& grid, vector<vector<int>>& hits) {
        n = grid.size();
        m = grid[0].size();
        g = &grid;
        
        vector<int> ans(hits.size(), 0);
        // 如果只有一行，任何砖块掉落都不会发生，因为所有砖块都直接连接着顶部
        if (n == 1) {
            return ans;
        }
        
        // 1. 时光走到尽头：把所有要敲击的砖块全部敲掉（减 1）
        for (const auto& hit : hits) {
            (*g)[hit[0]][hit[1]]--;
        }
        
        // 2. 统计残局：从顶部开始 DFS，将所有依然挂在上面的稳定砖块标记为 2
        for (int j = 0; j < m; j++) {
            dfs(0, j);
        }
        
        // 3. 时光倒流：逆序把砖块一块块补回来
        for (int i = hits.size() - 1; i >= 0; i--) {
            int row = hits[i][0];
            int col = hits[i][1];
            
            // 补回当前砖块
            (*g)[row][col]++;
            
            // 如果补回的这块砖能和稳定结构连上
            if (worth(row, col)) {
                // 它救活的砖块数减去它自己，就是当初它被敲掉时掉落的砖块数
                ans[i] = dfs(row, col) - 1;
            }
        }
        
        return ans;
    }
};
