// 050 Code02 寻找重复数

// 给定一个包含 n + 1 个整数的数组 nums ，其数字都在 [1, n] 范围内
// 可知至少存在一个重复的整数。
// 假设 nums 只有 一个重复的整数 ，返回 这个重复的数 。
// 你设计的解决方案必须 不修改 数组 nums 且只用常量级 O(1) 的额外空间。
// 测试链接 : https://leetcode.cn/problems/find-the-duplicate-number/

#include <vector>

using namespace std;

class Solution {
public:
    /**
     * 快慢指针（弗洛伊德判圈算法）
     * 时间复杂度 O(n)，额外空间复杂度 O(1)
     * 将数组值看作指向下一个下标的指针，因为有重复数字，必然会形成环
     */
    int findDuplicate(vector<int>& nums) {
        if (nums.size() < 2) {
            return -1;
        }

        // 第一阶段：寻找快慢指针在环中的相遇点
        // 慢指针走一步，快指针走两步
        int slow = nums[0];
        int fast = nums[nums[0]];
        while (slow != fast) {
            slow = nums[slow];
            fast = nums[nums[fast]];
        }

        // 第二阶段：寻找环的入口（即重复的数）
        // 快指针回到出发点，改为每次走一步
        fast = 0;
        while (slow != fast) {
            fast = nums[fast];
            slow = nums[slow];
        }

        return slow;
    }
};
