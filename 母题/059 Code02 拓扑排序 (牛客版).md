// 059 Code02 拓扑排序 (牛客版)
// 核心思路：Kahn 算法 + 静态数组 + 快速 IO
// 时间复杂度 O(N + M)，空间复杂度 O(N + M)

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 使用静态数组，防止频繁分配内存
const int MAXN = 200001;
int indegree[MAXN];
int q[MAXN];
int ans[MAXN];
int head, tail;

bool topoSort(int n, const vector<vector<int>>& graph) {
    head = tail = 0;
    // 1. 初始入度为 0 的点入队
    for (int i = 1; i <= n; i++) {
        if (indegree[i] == 0) {
            q[tail++] = i;
        }
    }

    int fill = 0;
    while (head < tail) {
        int cur = q[head++];
        ans[fill++] = cur;
        // 2. 遍历邻居，入度减 1
        for (int nxt : graph[cur]) {
            if (--indegree[nxt] == 0) {
                q[tail++] = nxt;
            }
        }
    }
    // 3. 计数判断是否有环
    return fill == n;
}

int main() {
    // C++ 快速 IO 优化
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    while (cin >> n >> m) {
        // 动态建图（邻接表）
        vector<vector<int>> graph(n + 1);
        // 重置入度数组
        fill(indegree, indegree + n + 1, 0);

        for (int i = 0; i < m; i++) {
            int u, v;
            cin >> u >> v;
            graph[u].push_back(v);
            indegree[v]++;
        }

        if (topoSort(n, graph)) {
            for (int i = 0; i < n; i++) {
                cout << ans[i] << (i == n - 1 ? "" : " ");
            }
            cout << "\n";
        } else {
            cout << "-1\n";
        }
    }
    return 0;
}
