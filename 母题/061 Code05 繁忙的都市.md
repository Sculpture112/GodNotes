// 061 Code05 繁忙的都市

// 测试链接 : https://www.luogu.com.cn/problem/P2330
// 核心思路：严格对标 Java 版逻辑，单纯的 Kruskal 算法求最小生成树。
// 1. 改造的道路尽量少，且连通所有路口，即选出 n-1 条边。
// 2. 改造的道路中分值最大的道路分值尽量小，这就是典型的最小生成树（MST）性质（最小瓶颈路）。
// 3. 将所有边按权重从小到大排序，使用并查集依次加入边，直到加入 n-1 条边为止，此时的最大边权即为所求。

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 针对洛谷等平台的快读模板
inline bool read(int &x) {
    x = 0; int f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == EOF) return false;
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    x *= f;
    return true;
}

// 封装边结构，方便排序
struct Edge {
    int u, v, w;
    bool operator<(const Edge& other) const {
        return w < other.w;
    }
};

class Solution {
private:
    vector<int> father;

    int find(int i) {
        if (i != father[i]) {
            father[i] = find(father[i]);
        }
        return father[i];
    }

    // 如果 x 和 y 本来就是一个集合，返回 false
    // 如果 x 和 y 不是一个集合，合并之后返回 true
    bool unite(int x, int y) {
        int fx = find(x);
        int fy = find(y);
        if (fx != fy) {
            father[fx] = fy;
            return true;
        }
        return false;
    }

public:
    void solve(int n, int m, vector<Edge>& edges) {
        // 初始化并查集
        father.resize(n + 1);
        for (int i = 1; i <= n; i++) {
            father[i] = i;
        }

        // 按权重从小到大排序
        sort(edges.begin(), edges.end());

        int ans = 0;
        int edgeCnt = 0;
        
        for (const auto& edge : edges) {
            if (unite(edge.u, edge.v)) {
                edgeCnt++;
                ans = max(ans, edge.w); // 记录当前加入的最大边权
                
                // 找齐 n-1 条边即可提前退出
                if (edgeCnt == n - 1) {
                    break;
                }
            }
        }
        
        // 输出格式：选出的道路数(n-1) 分值最大的那条道路的分值
        printf("%d %d\n", n - 1, ans);
    }
};

int main() {
    int n, m;
    // 持续读取直到 EOF
    while (read(n) && read(m)) {
        vector<Edge> edges(m);
        for (int i = 0; i < m; i++) {
            read(edges[i].u);
            read(edges[i].v);
            read(edges[i].w);
        }
        
        Solution().solve(n, m, edges);
    }
    return 0;
}
