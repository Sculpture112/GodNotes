// 053 Code02 去除重复字母保证剩余字符串的字典序最小

// 测试链接 : https://leetcode.cn/problems/remove-duplicate-letters/
// 核心思路：单调栈 + 贪心。
// 1. 统计每个字符在字符串中出现的总频次。
// 2. 遍历字符串，若当前字符未在栈中：
//    - 比较当前字符与栈顶字符，若栈顶字符字典序更大，且后续还会出现（cnts > 0），则弹出栈顶并标记为不在栈中。
//    - 将当前字符压入栈，并标记为已在栈中。
// 3. 无论当前字符是否入栈，其总频次均需减 1。
// 时间复杂度 O(N)，额外空间复杂度 O(K)，K 为字符集大小（26）

#include <string>
#include <vector>

using namespace std;

// 解法一：严格对应 Java 原逻辑的单调栈模拟
class Solution {
public:
    string removeDuplicateLetters(string s) {
        // cnts: 每种字符剩余待处理的频次
        // enter: 记录字符是否已在栈内
        // stack: 模拟单调栈
        vector<int> cnts(26, 0);
        vector<bool> enter(26, false);
        vector<char> stack(26);
        int r = 0; // 栈顶指针

        // 1. 预处理词频
        for (char ch : s) {
            cnts[ch - 'a']++;
        }

        // 2. 贪心策略维护单调栈
        for (char cur : s) {
            int curIdx = cur - 'a';
            // 如果当前字符没进栈，尝试入栈
            if (!enter[curIdx]) {
                // 栈不为空 && 栈顶字符字典序大于当前字符 && 栈顶字符后面还会出现
                while (r > 0 && stack[r - 1] > cur && cnts[stack[r - 1] - 'a'] > 0) {
                    enter[stack[r - 1] - 'a'] = false;
                    r--;
                }
                stack[r++] = cur;
                enter[curIdx] = true;
            }
            // 每处理一个字符（无论是否进栈），剩余频次都要减 1
            cnts[curIdx]--;
        }

        // 3. 利用迭代器范围直接构造结果字符串
        return string(stack.begin(), stack.begin() + r);
    }
};

// 解法二：C++ 特性优化版（直接使用 std::string 作为单调栈）
// C++ 中的 std::string 本身支持动态扩容和尾部操作（push_back, pop_back），
// 因此可以直接将其作为单调栈使用，省略额外的 char 数组和指针 r，代码更简洁。
class Solution2 {
public:
    string removeDuplicateLetters(string s) {
        vector<int> cnts(26, 0);
        vector<bool> enter(26, false);
        string res = ""; // 直接复用字符串作为单调栈

        for (char ch : s) {
            cnts[ch - 'a']++;
        }

        for (char cur : s) {
            int curIdx = cur - 'a';
            if (!enter[curIdx]) {
                while (!res.empty() && res.back() > cur && cnts[res.back() - 'a'] > 0) {
                    enter[res.back() - 'a'] = false;
                    res.pop_back(); // 直接弹出字符串末尾字符
                }
                res.push_back(cur); // 将当前字符追加到末尾
                enter[curIdx] = true;
            }
            cnts[curIdx]--;
        }

        return res;
    }
};
