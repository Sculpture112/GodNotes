```cpp
// 072 Code03 使数组K递增的最少操作次数

// 给你一个下标从0开始包含n个正整数的数组arr，和一个正整数k
// 如果对于每个满足 k <= i <= n-1 的下标 i
// 都有 arr[i-k] <= arr[i] ，那么称 arr 是K递增的
// 每一次操作中，你可以选择一个下标i并将arr[i]改成任意正整数
// 请你返回使数组变成K递增的最少操作次数
// 测试链接 : https://leetcode.cn/problems/minimum-operations-to-make-the-array-k-increasing/

#include <vector>
#include <algorithm>

using namespace std;

// 根据题目数据范围，数组长度最大为 10^5
const int MAXN = 100005;
int nums_buf[MAXN];
int ends_buf[MAXN];

class Solution {
public:
    /**
     * 核心思路：
     * K递增意味着数组被分成了 k 个独立的组：
     * 组0: arr[0], arr[k], arr[2k]...
     * 组1: arr[1], arr[k+1], arr[2k+1]...
     * 每一组内部必须是非递减（不下降）的。
     * 要使修改次数最少，等价于让每一组保留尽可能多的数字，即求每一组的“最长不下降子序列”。
     * 
     * 时间复杂度 O(n * log(n/k))，由于 k 个组总长度为 n，整体接近 O(n * logn)
     * 额外空间复杂度 O(n)
     */
    int kIncreasing(vector<int>& arr, int k) {
        int n = arr.size();
        int ans = 0;
        
        for (int i = 0; i < k; i++) {
            int size = 0;
            // 1. 提取当前 k-距离 构成的子序列
            for (int j = i; j < n; j += k) {
                nums_buf[size++] = arr[j];
            }
            // 2. 当前组需要修改的最少次数 = 总长度 - 最长不下降子序列长度
            ans += (size - lengthOfNonDecreasing(size));
        }
        return ans;
    }

private:
    /**
     * 求 nums_buf[0...size-1] 中的最长不下降子序列长度
     */
    int lengthOfNonDecreasing(int size) {
        int len = 0;
        for (int i = 0; i < size; i++) {
            // 在 ends 数组中寻找第一个大于当前数的位置
            int find = bs(len, nums_buf[i]);
            if (find == -1) {
                // 如果没找到，说明当前数可以接在所有递增序列后面
                ends_buf[len++] = nums_buf[i];
            } else {
                // 如果找到了，用更小的数替换它，增加后续扩展的可能性
                ends_buf[find] = nums_buf[i];
            }
        }
        return len;
    }

    /**
     * 二分查找：在 ends_buf[0...len-1] 中找到第一个 > num 的位置
     * 如果是求最长严格递增，这里应该是 >= num
     */
    int bs(int len, int num) {
        int l = 0, r = len - 1, m, ans = -1;
        while (l <= r) {
            m = l + ((r - l) >> 1);
            if (ends_buf[m] > num) {
                ans = m;
                r = m - 1;
            } else {
                l = m + 1;
            }
        }
        return ans;
    }
};
```
