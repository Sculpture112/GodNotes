// 052 Code05 最大矩形
// 测试链接 : https://leetcode.cn/problems/maximal-rectangle/

// 核心思路：将二维矩阵问题转化为每一行作为底部的“柱状图中最大的矩形”问题。
// 1. 利用压缩数组（heights）维护当前行往上连续 '1' 的高度。
// 2. 遍历每一行时，更新 heights 数组：若当前格为 '0'，高度置 0；若为 '1'，高度累加。
// 3. 对每一行更新后的 heights 数组，利用单调栈求出该行作为底部的最大矩形面积。

// 时间复杂度 O(N * M)，其中 N 为行数，M 为列数
// 额外空间复杂度 O(M)，用于存储高度数组和单调栈
#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    int maximalRectangle(vector<vector<char>>& matrix) {
        if (matrix.empty() || matrix[0].empty()) {
            return 0;
        }
        int n = matrix.size();
        int m = matrix[0].size();
        
        // 压缩数组：记录以当前行为底时，每个位置上方连续 '1' 的高度
        vector<int> heights(m, 0);
        // 静态数组模拟栈
        vector<int> stack(m);
        int ans = 0;

        for (int i = 0; i < n; i++) {
            // 更新高度数组
            for (int j = 0; j < m; j++) {
                heights[j] = (matrix[i][j] == '0') ? 0 : heights[j] + 1;
            }
            // 调用单调栈结算当前行的最大矩形
            ans = max(ans, largestRectangleArea(heights, stack, m));
        }
        return ans;
    }

private:
    // 逻辑等同于“柱状图中最大的矩形”
    int largestRectangleArea(const vector<int>& heights, vector<int>& stack, int m) {
        int r = 0;
        int ans = 0;

        for (int i = 0; i < m; i++) {
            // 当高度减小时，结算栈中比当前高度大的位置
            while (r > 0 && heights[stack[r - 1]] >= heights[i]) {
                int cur = stack[--r];
                int left = (r == 0) ? -1 : stack[r - 1];
                ans = max(ans, heights[cur] * (i - left - 1));
            }
            stack[r++] = i;
        }

        // 清算栈中剩余元素
        while (r > 0) {
            int cur = stack[--r];
            int left = (r == 0) ? -1 : stack[r - 1];
            ans = max(ans, heights[cur] * (m - left - 1));
        }

        return ans;
    }
};
