// 054 Code03 接取落水的最小花盆

// 测试链接 : https://www.luogu.com.cn/problem/P2698
// 核心思路：排序 + 滑动窗口 + 双单调队列。
// 1. 将所有水滴按 x 坐标升序排序。
// 2. 维护两个单调队列：一个记录窗口内水滴的最大高度（从大到小），一个记录最小高度（从小到大）。
// 3. 窗口左右边界分别为 l 和 r，不断向右扩张 r，将新水滴的高度加入双单调队列，直到窗口内的最大高度差（max - min）>= d。
// 4. 一旦高度差满足条件，说明当前窗口 [l, r) 能接住水滴并满足时间差，记录此时的花盆宽度（即 arr[r-1].x - arr[l].x）。
// 5. 然后左边界 l 向右滑动，尝试寻找更小的花盆宽度，并在 l 滑出窗口时检查队首元素是否过期并弹出。
// 时间复杂度 O(N log N) (排序的时间复杂度主导，滑动窗口过程为 O(N))，额外空间复杂度 O(N)

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 针对洛谷的快速读入模板，增加 EOF 判断
inline bool read(int &x) {
    x = 0; int f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == EOF) return false;
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    x *= f;
    return true; 
}

// 定义水滴结构体
struct Drop {
    int x, y;
};

int main() {
    int n, d;
    // 持续读取直到 EOF
    while (read(n) && read(d)) {
        vector<Drop> arr(n);
        for (int i = 0; i < n; i++) {
            read(arr[i].x);
            read(arr[i].y);
        }

        // 所有水滴根据 x 排序，谁小谁在前
        sort(arr.begin(), arr.end(), [](const Drop& a, const Drop& b) {
            return a.x < b.x;
        });

        // 局部开辟单调队列，随生命周期自动释放
        vector<int> maxDeque(n);
        vector<int> minDeque(n);
        int maxh = 0, maxt = 0, minh = 0, mint = 0;

        // 使用 Lambda 表达式捕获局部变量，完美替代 Java 中的全局变量和辅助函数
        auto ok = [&]() {
            int max_val = (maxh < maxt) ? arr[maxDeque[maxh]].y : 0;
            int min_val = (minh < mint) ? arr[minDeque[minh]].y : 0;
            return max_val - min_val >= d;
        };

        auto push = [&](int r) {
            while (maxh < maxt && arr[maxDeque[maxt - 1]].y <= arr[r].y) {
                maxt--;
            }
            maxDeque[maxt++] = r;
            while (minh < mint && arr[minDeque[mint - 1]].y >= arr[r].y) {
                mint--;
            }
            minDeque[mint++] = r;
        };

        auto pop = [&](int l) {
            if (maxh < maxt && maxDeque[maxh] == l) {
                maxh++;
            }
            if (minh < mint && minDeque[minh] == l) {
                minh++;
            }
        };

        int ans = 2147483647; // 等同于 Integer.MAX_VALUE
        for (int l = 0, r = 0; l < n; l++) {
            // [l,r) : 水滴的编号
            // l : 当前花盘的左边界，arr[l].x
            while (!ok() && r < n) {
                push(r++);
            }
            if (ok()) {
                ans = min(ans, arr[r - 1].x - arr[l].x);
            }
            pop(l);
        }

        // 使用 printf 快速输出
        if (ans == 2147483647) {
            printf("-1\n");
        } else {
            printf("%d\n", ans);
        }
	    }

    return 0;
}
