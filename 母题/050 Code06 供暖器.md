// 050 Code06 供暖器

// 冬季已经来临。 你的任务是设计一个有固定加热半径的供暖器向所有房屋供暖。
// 在加热器的加热半径范围内的每个房屋都可以获得供暖。
// 现在，给出位于一条水平线上的房屋 houses 和供暖器 heaters 的位置
// 请你找出并返回可以覆盖所有房屋的最小加热半径。
// 说明：所有供暖器都遵循你的半径标准，加热的半径也一样。
// 测试链接 : https://leetcode.cn/problems/heaters/

#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

class Solution {
public:
    /**
     * 时间复杂度 O(n * logn + m * logm)：主要开销在排序上。
     * 双指针遍历过程的时间复杂度为 O(n + m)。
     * 额外空间复杂度 O(1)：仅使用了常数个指针变量。
     */
    int findRadius(vector<int>& houses, vector<int>& heaters) {
        // 先对房屋和供暖器进行排序
        sort(houses.begin(), houses.end());
        sort(heaters.begin(), heaters.end());
        
        int ans = 0;
        // i 号房屋，j 号供暖器
        for (int i = 0, j = 0; i < (int)houses.size(); i++) {
            // 通过 best 函数判断：对于当前房屋 i，当前的供暖器 j 是否比下一个供暖器 j+1 更优
            // 如果不是更优，说明供暖器 j 可以被淘汰，指针 j 右移
            while (!best(houses, heaters, i, j)) {
                j++;
            }
            // 更新所有房屋所需的最小半径的最大值
            ans = max(ans, abs(heaters[j] - houses[i]));
        }
        return ans;
    }

private:
    /**
     * 判断对于第 i 个房屋，第 j 个供暖器是否是当前最好的选择。
     * 1. 如果 j 已经是最后一个供暖器，无法再选下一个，返回 true。
     * 2. 如果第 j 个供暖器到房屋的距离 < 第 j+1 个供暖器到房屋的距离，说明 j 更优，返回 true。
     * 3. 否则返回 false，表示应该尝试下一个供暖器。
     */
    bool best(const vector<int>& houses, const vector<int>& heaters, int i, int j) {
        if (j == (int)heaters.size() - 1) {
            return true;
        }
        // 使用 long long 转换防止两个 int 相减后 abs 溢出（虽然坐标通常在 int 范围内）
        return abs((long long)heaters[j] - houses[i]) < abs((long long)heaters[j + 1] - houses[i]);
    }
};
