// 052 Code06 单调栈（洛谷模板题）
// 测试链接 : https://www.luogu.com.cn/problem/P5788

// 核心思路：单调栈。寻找每个位置右侧第一个大于它的数。
// 1. 维护一个从栈底到栈顶单调递减（非严格）的索引栈。
// 2. 遍历数组，若当前值大于栈顶索引对应的值，说明找到了栈顶元素的“右侧最近大值”。
// 3. 弹出栈顶并记录答案，直到栈为空或当前值不再大于栈顶值，最后将当前索引入栈。

// 时间复杂度 O(N)，额外空间复杂度 O(N)
#include <cstdio>
#include <vector>

using namespace std;

// 针对洛谷严苛的空间和时间限制，使用快速读入
inline int read() {
    int x = 0, f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    return x * f;
}

int main() {
    int n = read();
    
    // arr: 原始数组 (1~n)
    // stack: 模拟栈，存储索引
    // ans: 结果数组
    // C++ 内存效率较高，直接开辟数组即可，无需像 Java 那样极限复用
    vector<int> arr(n + 1);
    vector<int> stack(n + 1);
    vector<int> ans(n + 1, 0);
    
    for (int i = 1; i <= n; i++) {
        arr[i] = read();
    }

    int r = 0; // 栈顶指针
    for (int i = 1; i <= n; i++) {
        // 当当前数字 > 栈顶数字时，结算栈顶
        while (r > 0 && arr[stack[r - 1]] < arr[i]) {
            ans[stack[--r]] = i;
        }
        stack[r++] = i;
    }

    // 栈中剩余元素右侧没有更大的值，ans 已初始化为 0

    // 使用 printf 快速输出
    for (int i = 1; i <= n; i++) {
        printf("%d%c", ans[i], (i == n ? '\n' : ' '));
    }

    return 0;
}
