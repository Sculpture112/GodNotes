```cpp
// 066 Code06 最长有效括号

// 给你一个只包含 '(' 和 ')' 的字符串
// 找出最长有效（格式正确且连续）括号子串的长度。
// 测试链接 : https://leetcode.cn/problems/longest-valid-parentheses/

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>

using namespace std;

class Solution {
public:
    // 时间复杂度 O(n)，n 是 s 字符串的长度
    int longestValidParentheses(string s) {
        int n = s.length();
        if (n == 0) {
            return 0;
        }
        
        // dp[0...n-1]
        // dp[i] : 子串必须以 i 位置的字符结尾的情况下，往左整体有效的最大长度
        // 初始化全为 0
        vector<int> dp(n, 0);
        int ans = 0;
        
        for (int i = 1; i < n; i++) {
            if (s[i] == ')') {
                // p 代表如果和前面的有效括号部分拼起来，还需要匹配的前一个字符的位置
                int p = i - dp[i - 1] - 1;
                
                //  ?         )
                //  p         i
                if (p >= 0 && s[p] == '(') {
                    // 当前这一对 () 贡献 2 的长度，加上它内部的 dp[i-1]，
                    // 再看看 p 前面是否还有连着的有效括号 dp[p-1]
                    dp[i] = dp[i - 1] + 2 + (p - 1 >= 0 ? dp[p - 1] : 0);
                }
            }
            ans = max(ans, dp[i]);
        }
        
        return ans;
    }
};
```
