// 050 Code03 接雨水

// 给定 n 个非负整数表示每个宽度为 1 的柱子的高度图，计算按此排列的柱子，下雨之后能接多少雨水
// 测试链接 : https://leetcode.cn/problems/trapping-rain-water/

#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    // 辅助数组的解法（不是最优解）
    // 时间复杂度O(n)，额外空间复杂度O(n)
    int trap1(vector<int>& nums) {
        int n = nums.size();
        if (n <= 2) return 0;
        
        vector<int> lmax(n);
        vector<int> rmax(n);
        
        // 0~i范围上的最大值，记录在lmax[i]
        lmax[0] = nums[0];
        for (int i = 1; i < n; i++) {
            lmax[i] = max(lmax[i - 1], nums[i]);
        }
        
        // i~n-1范围上的最大值，记录在rmax[i]
        rmax[n - 1] = nums[n - 1];
        for (int i = n - 2; i >= 0; i--) {
            rmax[i] = max(rmax[i + 1], nums[i]);
        }
        
        int ans = 0;
        // 遍历中间的柱子，计算每个位置能接的雨水
        for (int i = 1; i < n - 1; i++) {
            // 当前位置的水位取决于左右两侧最大高度的较小值
            ans += max(0, min(lmax[i - 1], rmax[i + 1]) - nums[i]);
        }
        return ans;
    }

    // 双指针的解法（最优解）
    // 时间复杂度O(n)，额外空间复杂度O(1)
    int trap2(vector<int>& nums) {
        int n = nums.size();
        if (n <= 2) return 0;

        // l从1开始，r从n-2开始，两侧边缘(0和n-1)不可能接到水
        int l = 1, r = n - 2;
        // lmax代表 0...l-1 范围内的最大高度
        // rmax代表 r+1...n-1 范围内的最大高度
        int lmax = nums[0], rmax = nums[n - 1];
        int ans = 0;

        while (l <= r) {
            if (lmax <= rmax) {
                // 如果左侧最大高度小于等于右侧，则左侧l位置的水位由lmax决定
                ans += max(0, lmax - nums[l]);
                lmax = max(lmax, nums[l++]);
            } else {
                // 如果右侧最大高度小于左侧，则右侧r位置的水位由rmax决定
                ans += max(0, rmax - nums[r]);
                rmax = max(rmax, nums[r--]);
            }
        }
        return ans;
    }
};
