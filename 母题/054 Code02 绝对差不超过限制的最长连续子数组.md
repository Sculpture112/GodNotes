// 054 Code02 绝对差不超过限制的最长连续子数组

// 测试链接 : https://leetcode.cn/problems/longest-continuous-subarray-with-absolute-diff-less-than-or-equal-to-limit/
// 核心思路：滑动窗口 + 双单调队列。
// 1. 维护两个单调队列：一个递减队列用于获取窗口内的最大值，一个递增队列用于获取窗口内的最小值。
// 2. 窗口右边界 r 不断向右探索：如果新加入的数字使得（新窗口最大值 - 新窗口最小值 <= limit），则将 r 加入窗口（更新两个单调队列），并 r++。
// 3. 当 r 无法继续向右扩张时，此时 [l, r) 就是以 l 为左边界的最长合法子数组，记录并更新最大长度。
// 4. 左边界 l 向右收缩一位，检查过期：如果 l 位置的元素是当前队头，则将其从队列中弹出。
// 时间复杂度 O(N)，额外空间复杂度 O(N)

#include <vector>
#include <algorithm>

using namespace std;

class Solution {
private:
    // 窗口内最大值的更新结构（单调递减队列）
    vector<int> maxDeque;
    // 窗口内最小值的更新结构（单调递增队列）
    vector<int> minDeque;
    int maxh, maxt, minh, mint;
    
    // 使用指针避免拷贝整个数组，方便辅助函数直接访问
    const vector<int>* arr;

    // 判断如果加入数字 number，窗口最大值 - 窗口最小值是否依然 <= limit
    bool ok(int limit, int number) {
        int current_max = (maxh < maxt) ? max((*arr)[maxDeque[maxh]], number) : number;
        int current_min = (minh < mint) ? min((*arr)[minDeque[minh]], number) : number;
        return current_max - current_min <= limit;
    }

    // r 位置的数字进入窗口，修改窗口内最大值和最小值的更新结构
    void push(int r) {
        // 维护最大值单调队列（大 -> 小）
        while (maxh < maxt && (*arr)[maxDeque[maxt - 1]] <= (*arr)[r]) {
            maxt--;
        }
        maxDeque[maxt++] = r;
        
        // 维护最小值单调队列（小 -> 大）
        while (minh < mint && (*arr)[minDeque[mint - 1]] >= (*arr)[r]) {
            mint--;
        }
        minDeque[mint++] = r;
    }

    // 窗口要吐出 l 位置的数了，检查过期
    void pop(int l) {
        if (maxh < maxt && maxDeque[maxh] == l) {
            maxh++;
        }
        if (minh < mint && minDeque[minh] == l) {
            minh++;
        }
    }

public:
    int longestSubarray(vector<int>& nums, int limit) {
        int n = nums.size();
        // 局部开辟定长空间，交由 C++ 自动管理
        maxDeque.assign(n, 0);
        minDeque.assign(n, 0);
        maxh = maxt = minh = mint = 0;
        arr = &nums;
        
        int ans = 0;
        for (int l = 0, r = 0; l < n; l++) {
            // [l, r)，r 永远是没有进入窗口的、下一个数所在的位置
            while (r < n && ok(limit, nums[r])) {
                push(r++);
            }
            // 从 while 出来的时候，[l, r) 是以 l 开头的子数组能向右延伸的最大范围
            ans = max(ans, r - l);
            pop(l);
        }
        
        return ans;
    }
};
