**核心套路:** 
- 任意子序列和等于“最大可能和（所有正数之和）”减去“未被选中的正数与被选中的负数的绝对值之和”。
    
- 将原数组全部转换为绝对值并排序，原问题等价于求：`最大正数和` - `绝对值数组的第 k 小子序列和`。

```cpp
// 073 Code07 找出数组的第 K 大和

// 给你一个数组 nums 和一个整数 k 。你可以选择数组的任意子序列并对其元素求和。
// 数组中的值可以是正数、负数或 0 。空子序列的和视为 0 。
// 请你返回第 k 大的子序列和（包含重复项）。
// 测试链接 : https://leetcode.cn/problems/find-the-k-sum-of-an-array/

#include <vector>
#include <algorithm>
#include <queue>

using namespace std;

/**
 * 逻辑转换推导：
 * 1. 数组的最大子序列和 maxSum 等于所有正数之和。
 * 2. 任何其他子序列和都可以看作是从 maxSum 中：
 *    - 减去某些已经包含的正数
 *    - 加上某些未包含的负数（等价于减去负数的绝对值）
 * 3. 因此，所有的子序列和都可以表示为：maxSum - (nums 绝对值数组的某个子序列和)。
 * 4. 找第 k 大的子序列和，等价于求：maxSum - (nums 绝对值数组的第 k 小子序列和)。
 * 5. 绝对值数组的第 1 小子序列和是空集，即 0。
 */
class Solution {
public:
    struct Node {
        long long sum;
        int idx;

        // 小根堆：累加和小的在前
        bool operator>(const Node& other) const {
            return sum > other.sum;
        }
    };

    long long kSum(vector<int>& nums, int k) {
        int n = nums.size();
        long long maxSum = 0;
        
        // 1. 计算最大可能的和，并将原数组转为绝对值数组
        for (int i = 0; i < n; i++) {
            if (nums[i] > 0) {
                maxSum += nums[i];
            } else {
                nums[i] = -nums[i]; // 负数变正数，0 不动
            }
        }
        
        // 2. 对绝对值数组进行升序排序，准备找前 k 小子序列和
        sort(nums.begin(), nums.end());

        // 3. 使用堆找绝对值数组的第 k 小子序列和
        // 第 1 小一定是 0（空集），我们从第 2 小开始通过堆寻找
        if (k == 1) return maxSum;

        priority_queue<Node, vector<Node>, greater<Node>> minHeap;
        // 放入第一个基础子序列：只包含绝对值最小的那个数
        minHeap.push({(long long)nums[0], 0});

        long long kthSmallestSubsetSum = 0;
        // 循环 k-1 次，弹出的第 k-1 个元素就是绝对值数组中排名第 k 的子序列和
        // （注意：空集 0 是第 1 小，所以堆里弹出的是第 2 小到第 k 小）
        for (int i = 1; i < k; i++) {
            Node cur = minHeap.top();
            minHeap.pop();
            
            kthSmallestSubsetSum = cur.sum;
            
            if (cur.idx + 1 < n) {
                // 派生策略：
                // 1. 包含下一个数：当前和 + 下一个数
                minHeap.push({cur.sum + nums[cur.idx + 1], cur.idx + 1});
                // 2. 替换当前数：当前和 - 当前数 + 下一个数
                minHeap.push({cur.sum - nums[cur.idx] + nums[cur.idx + 1], cur.idx + 1});
            }
        }

        return maxSum - kthSmallestSubsetSum;
    }
};
```
