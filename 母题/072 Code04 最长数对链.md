// 072 Code04 最长数对链

// 给你一个由 n 个数对组成的数对数组 pairs
// 其中 pairs[i] = [lefti, righti] 且 lefti < righti
// 现在，我们定义一种 跟随 关系，当且仅当 b < c 时
// 数对 p2 = [c, d] 才可以跟在 p1 = [a, b] 后面
// 找出并返回能够形成的最长数对链的长度
// 测试链接 : https://leetcode.cn/problems/maximum-length-of-pair-chain/

#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

class Solution {
public:
    /**
     * 方法一：类似 LIS 的二分查找解法
     * 时间复杂度 O(n * logn)，额外空间复杂度 O(n)
     * 这种方法虽然可行，但在处理此类区间覆盖问题时，通常不如贪心解法直观。
     */
    int findLongestChain1(vector<vector<int>>& pairs) {
        int n = pairs.size();
        // 根据数对的开始位置排序，从小到大
        sort(pairs.begin(), pairs.end(), [](const vector<int>& a, const vector<int>& b) {
            return a[0] < b[0];
        });

        // ends[i] 表示长度为 i+1 的数对链中，最小的结尾数值
        vector<int> ends(n);
        int len = 0;
        for (auto& pair : pairs) {
            // 在 ends 有效区中找到第一个 >= 当前数对开始位置的结尾
            int find = bs(ends, len, pair[0]);
            if (find == -1) {
                // 如果没找到，说明当前数对可以接到最长链的后面
                ends[len++] = pair[1];
            } else {
                // 如果找到了，说明当前数对可以替代现有的结尾，
                // 我们选择较小的结尾数值，以便给后面留出更多空间
                ends[find] = min(ends[find], pair[1]);
            }
        }
        return len;
    }

    /**
     * 方法二：贪心解法（最优解，区间调度问题经典思路）
     * 时间复杂度 O(n * logn)，额外空间复杂度 O(1)
     * 策略：每次选择结尾最早的数对，这样能为后面留出最多的空间。
     */
    int findLongestChain2(vector<vector<int>>& pairs) {
        // 根据数对的结束位置排序，从小到大
        sort(pairs.begin(), pairs.end(), [](const vector<int>& a, const vector<int>& b) {
            return a[1] < b[1];
        });

        int pre = INT_MIN;
        int ans = 0;
        for (auto& pair : pairs) {
            // 如果当前数对的开始位置大于前一个选择数对的结束位置
            if (pre < pair[0]) {
                pre = pair[1];
                ans++;
            }
        }
        return ans;
    }

private:
    /**
     * 二分查找：找到 ends[0...len-1] 中第一个 >= num 的位置
     */
    int bs(const vector<int>& ends, int len, int num) {
        int l = 0, r = len - 1, m, ans = -1;
        while (l <= r) {
            m = l + ((r - l) >> 1);
            if (ends[m] >= num) {
                ans = m;
                r = m - 1;
            } else {
                l = m + 1;
            }
        }
        return ans;
    }
};
