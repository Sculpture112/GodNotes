// 072 Code04 多重背包单调队列优化

// 宝物筛选
// 一共有 n 种货物, 背包容量为 t
// 每种货物的价值(v[i])、重量(w[i])、数量(c[i])都给出
// 请返回选择货物不超过背包容量的情况下，能得到的最大的价值
// 测试链接 : https://www.luogu.com.cn/problem/P1776

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * 💡 核心逻辑：单调队列优化 (Monotonic Queue Optimization)
 * 
 * 1. 为什么需要优化？
 *    多重背包的转移方程：dp[i][j] = max(dp[i-1][j - k*w] + k*v)，其中 0 <= k <= c。
 *    对于同一个重量 w，我们可以根据容量 j 对 w 的余数进行分组（同余分组）。
 *    比如 w=3，容量 j 分为：
 *    组0: 0, 3, 6, 9...
 *    组1: 1, 4, 7, 10...
 *    组2: 2, 5, 8, 11...
 * 
 * 2. 转换方程：
 *    令 j = q*w + mod (mod 是余数)。
 *    dp[i][q*w + mod] = max{ dp[i-1][p*w + mod] + (q-p)*v } (其中 q-c <= p <= q)
 *    将与 p 无关的项提取出来：
 *    dp[i][q*w + mod] = max{ dp[i-1][p*w + mod] - p*v } + q*v
 * 
 * 3. 结论：
 *    括号内的项 {dp[i-1][p*w + mod] - p*v} 仅取决于下标 p。
 *    这就是一个“滑动窗口最大值”问题，窗口大小为 c+1，可以用单调队列在 O(1) 时间内处理。
 *    总时间复杂度降至 O(n * t)。
 */

const int MAXN = 105;
const int MAXW = 40005;

int v_arr[MAXN];
int w_arr[MAXN];
int c_arr[MAXN];
int dp[MAXW];
int q[MAXW]; // 单调队列，存储下标
int l, r;
int n, t;

/**
 * 计算当前位置的“指标值” (Metric)
 * 即上文推导出的：dp[p*w + mod] - p*v
 */
inline int get_value(int j, int weight, int val) {
    return dp[j] - (j / weight) * val;
}

/**
 * 空间压缩的单调队列优化多重背包
 * 时间复杂度 O(n * t)，空间复杂度 O(t)
 */
int compute() {
    // 初始化 DP 数组为 0
    for (int j = 0; j <= t; j++) dp[j] = 0;

    for (int i = 1; i <= n; i++) {
        int weight = w_arr[i];
        int val = v_arr[i];
        int count = c_arr[i];

        // 限制：完全背包（如果数量足够多，可以直接走完全背包流程，这里统一处理）
        count = min(count, t / weight);

        // 按余数分组
        for (int mod = 0; mod < min(t + 1, weight); mod++) {
            l = r = 0;
            
            /**
             * 空间压缩技巧：
             * 为了保证使用的是“上一行”的数据，我们需要从大到小（右到左）处理容量。
             * 窗口先从右侧最远端开始初始化。
             */
            
            // 1. 初始化单调队列：先把右侧前 count 个指标放入队列
            for (int j = t - mod, cnt = 1; j >= 0 && cnt <= count; j -= weight, cnt++) {
                while (l < r && get_value(q[r - 1], weight, val) <= get_value(j, weight, val)) {
                    r--;
                }
                q[r++] = j;
            }

            // 2. 滑动窗口：从右往左移动
            // j 是当前要求解的 dp 位置
            // enter 是随着窗口向左移动，即将进入窗口左侧（低容量端）的新位置
            for (int j = t - mod, enter = j - weight * count; j >= 0; j -= weight, enter -= weight) {
                // 如果左侧还有新元素，让其进入单调队列
                if (enter >= 0) {
                    while (l < r && get_value(q[r - 1], weight, val) <= get_value(enter, weight, val)) {
                        r--;
                    }
                    q[r++] = enter;
                }
                
                // 此时队头 q[l] 就是 [j-count*w, j] 范围内指标最大的下标
                dp[j] = get_value(q[l], weight, val) + (j / weight) * val;

                // 随着 j 向左移，如果队头正好是当前的 j，说明它下一轮就出界了
                if (q[l] == j) {
                    l++;
                }
            }
        }
    }
    return dp[t];
}

int main() {
    // 极致 I/O 优化
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n >> t) {
        for (int i = 1; i <= n; i++) {
            cin >> v_arr[i] >> w_arr[i] >> c_arr[i];
        }
        cout << compute() << "\n";
    }

    return 0;
}
