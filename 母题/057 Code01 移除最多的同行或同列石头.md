```cpp
// 057 Code01 移除最多的同行或同列石头

// 测试链接 : https://leetcode.cn/problems/most-stones-removed-with-same-row-or-column/
// 核心思路：并查集。
// 1. 将每一块石头视为一个节点。如果两块石头同行或同列，它们就属于同一个连通分量。
// 2. 在一个包含 k 块石头的连通分量中，我们可以按照特定的顺序一块一块地移除，直到只剩下最后 1 块石头。因此该连通分量最多可移除 k - 1 块石头。
// 3. 假设共有 n 块石头，最终形成了 sets 个连通分量，那么总共可移除的石头最大数量 = 所有连通分量的 (k_i - 1) 之和 = n - sets。
// 4. 为了高效合并，使用哈希表记录某行或某列“第一次”遇到的石头编号。后续在此行/列遇到的石头，直接与第一次遇到的石头所在集合进行合并即可。
// 时间复杂度 O(N)，额外空间复杂度 O(N)

#include <vector>
#include <unordered_map>
#include <functional>

using namespace std;

class Solution {
public:
    int removeStones(vector<vector<int>>& stones) {
        int n = stones.size();
        
        // father: 并查集父节点数组
        vector<int> father(n);
        for (int i = 0; i < n; i++) {
            father[i] = i;
        }
        
        int sets = n; // 初始假设每块石头都是孤立的连通分量
        
        // Lambda 实现带路径压缩的查找
        function<int(int)> find = [&](int i) {
            if (i != father[i]) {
                father[i] = find(father[i]);
            }
            return father[i];
        };
        
        // 极简合并操作
        auto unite = [&](int x, int y) {
            int fx = find(x);
            int fy = find(y);
            if (fx != fy) {
                father[fx] = fy;
                sets--; // 发生合并，连通分量减少
            }
        };
        
        // 记录某行、某列第一次遇到的石头编号
        // key : 行号或列号
        // value : 第一次遇到的石头编号
        unordered_map<int, int> rowFirst;
        unordered_map<int, int> colFirst;
        
        for (int i = 0; i < n; i++) {
            int row = stones[i][0];
            int col = stones[i][1];
            
            // 处理行
            if (rowFirst.find(row) == rowFirst.end()) {
                rowFirst[row] = i; // 第一次遇到该行，记录下来
            } else {
                unite(i, rowFirst[row]); // 之前遇到过，和该行的第一块石头合并
            }
            
            // 处理列
            if (colFirst.find(col) == colFirst.end()) {
                colFirst[col] = i; // 第一次遇到该列，记录下来
            } else {
                unite(i, colFirst[col]); // 之前遇到过，和该列的第一块石头合并
            }
        }
        
        // 总石头数 - 独立连通分量数 = 最多可移除的石头数
        return n - sets;
    }
};
```
