// 047 Code02 等差数列差分

// 一开始1~n范围上的数字都是0，一共有m个操作，每次操作为(l,r,s,e,d)
// 表示在l~r范围上依次加上首项为s、末项为e、公差为d的数列
// m个操作做完之后，统计1~n范围上所有数字的最大值和异或和
// 测试链接 : https://www.luogu.com.cn/problem/P4231

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int MAXN = 10000005;

// 全局大数组，自动初始化为 0，且分配在堆区防止栈溢出
long long arr[MAXN];
int n, m;

// 在 [l, r] 区间上加上首项为 s，末项为 e，公差为 d 的等差数列
// 二阶差分的 4 个关键点修改
void set_diff(int l, int r, long long s, long long e, long long d) {
    arr[l] += s;
    arr[l + 1] += d - s;
    arr[r + 1] -= d + e;
    arr[r + 2] += e;
}//这里将需要的写出来，倒着推即可！用加数等于后面全部都加这个数即可轻松推出（推两遍

// 两次前缀和还原原数组
void build() {
    // 第一次前缀和：将二阶差分还原为一阶差分
    for (int i = 1; i <= n; i++) {
        arr[i] += arr[i - 1];
    }
    // 第二次前缀和：将一阶差分还原为原数组
    for (int i = 1; i <= n; i++) {
        arr[i] += arr[i - 1];
    }
}

int main() {
    // 竞赛必备：解绑 C/C++ I/O 流，极大提升输入输出效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n >> m) {
        // 如果题目有多组数据，这里需要清空数组。
        // 但根据题意及原 Java 代码，本题属于单组测试用例
        
        for (int i = 0; i < m; i++) {
            int l, r;
            long long s, e;
            cin >> l >> r >> s >> e;
            
            // 保护机制：防止 l == r 时除以 0 导致运行时异常 (RE)
            long long d = (l == r) ? 0 : (e - s) / (r - l);
            set_diff(l, r, s, e, d);
        }
        
        build();
        
        long long max_val = 0;
        long long xor_sum = 0;
        
        for (int i = 1; i <= n; i++) {
            max_val = max(max_val, arr[i]);
            xor_sum ^= arr[i];
        }
        
        cout << xor_sum << " " << max_val << "\n";
    }

    return 0;
}
