// 074 Code02 从栈中取出 K 个硬币的最大面值和

// 一张桌子上总共有 n 个硬币栈。每个栈有正整数个带面值的硬币。
// 每一次操作中，你可以从任意一个栈的顶部取出 1 个硬币，移除它并放入钱包。
// 给你一个列表 piles，其中 piles[i] 是一个整数数组，表示第 i 个栈从顶到底的面值。
// 返回在恰好进行 k 次操作的前提下，你钱包里硬币面值之和最大为多少。
// 测试链接 : https://leetcode.cn/problems/maximum-value-of-k-coins-from-piles/

#include <vector>
#include <algorithm>

using namespace std;

/**
 * 逻辑分析：
 * 这是一个典型的“分组背包”问题。
 * 每一个栈（pile）代表一个物品组。
 * 在每一个组内，你可以有多种选择方案：
 * 1. 拿走 0 个硬币（即不选该组）。
 * 2. 拿走顶部 1 个硬币，价值为 pile[0]。
 * 3. 拿走顶部 2 个硬币，价值为 pile[0] + pile[1]。
 * ...
 * x. 拿走顶部 x 个硬币，价值为该栈的前 x 个元素之和。
 * 
 * 限制：在每一个组内只能选择其中一种方案，总拿取的硬币数不能超过 k。
 */

class Solution {
public:
    /**
     * 方案一：严格位置依赖的动态规划 (二维 DP)
     * 时间复杂度 O(k * 硬币总数)，空间复杂度 O(n * k)
     */
    int maxValueOfCoins1(vector<vector<int>>& piles, int k) {
        int n = piles.size();
        // dp[i][j] 表示前 i 个栈中，一共拿走 j 个硬币获得的最大价值
        vector<vector<int>> dp(n + 1, vector<int>(k + 1, 0));

        for (int i = 1; i <= n; i++) {
            const vector<int>& stack = piles[i - 1];
            // 当前栈最多能贡献的硬币数
            int stackSize = stack.size();
            
            // 预处理当前组的前缀和，preSum[x] 表示拿走前 x 个硬币的价值
            vector<int> preSum(stackSize + 1, 0);
            for (int x = 1; x <= stackSize; x++) {
                preSum[x] = preSum[x - 1] + stack[x - 1];
            }

            for (int j = 0; j <= k; j++) {
                // 方案 1：当前组一个都不拿
                dp[i][j] = dp[i - 1][j];
                
                // 方案 2：枚举从当前组拿走 count 个硬币的所有可能
                // count 不能超过当前栈的大小，也不能超过背包总容量 j
                int maxCount = min(stackSize, j);
                for (int count = 1; count <= maxCount; count++) {
                    dp[i][j] = max(dp[i][j], dp[i - 1][j - count] + preSum[count]);
                }
            }
        }
        return dp[n][k];
    }

    /**
     * 方案二：空间压缩优化 (一维 DP)
     * 时间复杂度 O(k * 硬币总数)，空间复杂度 O(k)
     */
    int maxValueOfCoins(vector<vector<int>>& piles, int k) {
        // dp[j] 表示一共拿走 j 个硬币的最大价值
        vector<int> dp(k + 1, 0);

        for (const auto& stack : piles) {
            int stackSize = stack.size();
            
            // 计算前缀和，为了节省空间可以不用数组，直接在枚举时累加
            // 但为了代码清晰，这里展示 logic
            vector<int> preSum(stackSize + 1, 0);
            for (int x = 1; x <= stackSize; x++) {
                preSum[x] = preSum[x - 1] + stack[x - 1];
            }

            // 分组背包空间压缩：必须从大到小遍历容量 j
            for (int j = k; j > 0; j--) {
                // 枚举当前栈贡献 count 个硬币
                int maxCount = min(stackSize, j);
                for (int count = 1; count <= maxCount; count++) {
                    // dp[j] 取当前的 dp[j]（不选该栈的新方案）
                    // 与 dp[j - count] + preSum[count]（在之前基础上选 count 个）中的最大值
                    dp[j] = max(dp[j], dp[j - count] + preSum[count]);
                }
            }
        }
        return dp[k];
    }
};
