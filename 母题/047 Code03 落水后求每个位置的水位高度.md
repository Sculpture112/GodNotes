// 047 Code03 落水后求每个位置的水位高度

// 一群人落水后求每个位置的水位高度
// 问题描述比较复杂，见测试链接
// 测试链接 : https://www.luogu.com.cn/problem/P5026

#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring> // 用于 memset 初始化

using namespace std;

// 题目说了m <= 10^6，代表湖泊宽度
// 这就是MAXN的含义，湖泊最大宽度的极限
const int MAXN = 1000005;

// 数值保护，因为题目中v最大是10000
// 所以左侧影响最远的位置到达了x - 3 * v + 1
// 所以右侧影响最远的位置到达了x + 3 * v - 1
// x如果是正式的位置(1~m)，那么左、右侧可能超过正式位置差不多30000的规模
// 这就是OFFSET的含义
const int OFFSET = 30005;
//
// 全局大数组，自动初始化为 0，且分配在堆区防止栈溢出
// 准备一个长度为OFFSET + MAXN + OFFSET的数组，省去越界讨论
long long arr[OFFSET + MAXN + OFFSET];
int n, m;

// 在 [l, r] 区间上加上首项为 s，末项为 e，公差为 d 的等差数列
// 二阶差分的 4 个关键点修改，加入 OFFSET 防止越界
void set_diff(int l, int r, long long s, long long e, long long d) {
    arr[l + OFFSET] += s;
    arr[l + 1 + OFFSET] += d - s;
    arr[r + 1 + OFFSET] -= d + e;
    arr[r + 2 + OFFSET] += e;
}

// v体积的朋友，在x处落水，修改差分数组
void fall(long long v, int x) {
    set_diff(x - 3 * v + 1, x - 2 * v, 1, v, 1);
    set_diff(x - 2 * v + 1, x, v - 1, -v, -1);
    set_diff(x + 1, x + 2 * v, -v + 1, v, 1);
    set_diff(x + 2 * v + 1, x + 3 * v - 1, v - 1, 1, -1);
}

// 两次前缀和还原原数组
void build() {
    // 第一次前缀和：将二阶差分还原为一阶差分
    for (int i = 1; i <= m + OFFSET; i++) {
        arr[i] += arr[i - 1];
    }
    // 第二次前缀和：将一阶差分还原为原数组
    for (int i = 1; i <= m + OFFSET; i++) {
        arr[i] += arr[i - 1];
    }
}

int main() {
    // 竞赛必备：解绑 C/C++ I/O 流，极大提升输入输出效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n >> m) {
        // 如果题目有多组数据，这里需要清空数组。
        // 虽然洛谷 P5026 通常为单组测试用例，但由于加入了 while 循环，严谨起见加上 memset
        memset(arr, 0, sizeof(arr));

        for (int i = 0; i < n; i++) {
            long long v;
            int x;
            cin >> v >> x;
            fall(v, x);
        }

        // 生成最终的水位数组
        build();

        // 开始收集答案
        // 0...OFFSET这些位置是辅助位置，为了防止越界设计的
        // 从OFFSET+1开始往下数m个，才是正式的位置1...m
        int start = OFFSET + 1;
        cout << arr[start++];
        for (int i = 2; i <= m; i++) {
            cout << " " << arr[start++];
        }
        cout << "\n";
    }

    return 0;
}
