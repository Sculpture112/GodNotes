// 073 Code05 有依赖的背包(模版)

// 物品分为两大类：主件和附件
// 若想买附件，必须先购买其所属的主件。
// 每个主件最多有 2 个附件，附件不再有附件。
// 所有的物品编号在 1~m 之间。每个物品有：价格 v、重要度 p、归属 q。
// 价格即花费，价格 * 重要度 即收益。归属 q=0 表示主件，q>0 表示该主件的编号。
// 测试链接 : https://www.luogu.com.cn/problem/P1064
// 测试链接 : https://www.nowcoder.com/practice/f9c6f980eeec43ef85be20755ddbeaf4

#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>

using namespace std;

// 根据题目数据范围
// 总钱数 n <= 32000，商品数 m <= 60
const int MAXN = 33005;
const int MAXM = 65;

int cost[MAXM];      // 价格
int val[MAXM];       // 收益 (v * p)
bool king[MAXM];     // 是否为主件
int fans[MAXM];      // 主件拥有的附件数量
int follows[MAXM][2]; // 记录附件的编号
int dp[MAXN];        // 空间压缩的 DP 数组
int n, m;

/**
 * 清理函数，每次测试重置状态
 */
void clean() {
    memset(fans, 0, sizeof(fans));
    memset(dp, 0, sizeof(dp));
}

/**
 * 方案一：严格位置依赖的动态规划
 * 时间复杂度 $O(m \times n)$，空间复杂度 $O(m \times n)$
 */
int compute1() {
    // dp[i][j] 表示考虑前 i 个商品，在花费不超过 j 的情况下的最大收益
    // 因为附件必须依赖主件，我们可以只在遇到主件时进行“决策展开”
    vector<vector<int>> dp_table(m + 1, vector<int>(n + 1, 0));
    int p = 0; // 上一个处理过的主件编号
    
    for (int i = 1; i <= m; i++) {
        if (king[i]) {
            int fan1 = (fans[i] >= 1) ? follows[i][0] : -1;
            int fan2 = (fans[i] >= 2) ? follows[i][1] : -1;
            
            for (int j = 0; j <= n; j++) {
                // 1. 不买当前主件（继承上一个主件的结果）
                dp_table[i][j] = dp_table[p][j];
                
                // 2. 只买主件
                if (j >= cost[i]) {
                    dp_table[i][j] = max(dp_table[i][j], dp_table[p][j - cost[i]] + val[i]);
                }
                // 3. 主件 + 附件1
                if (fan1 != -1 && j >= cost[i] + cost[fan1]) {
                    dp_table[i][j] = max(dp_table[i][j], dp_table[p][j - cost[i] - cost[fan1]] + val[i] + val[fan1]);
                }
                // 4. 主件 + 附件2
                if (fan2 != -1 && j >= cost[i] + cost[fan2]) {
                    dp_table[i][j] = max(dp_table[i][j], dp_table[p][j - cost[i] - cost[fan2]] + val[i] + val[fan2]);
                }
                // 5. 主件 + 附件1 + 附件2
                if (fan1 != -1 && fan2 != -1 && j >= cost[i] + cost[fan1] + cost[fan2]) {
                    dp_table[i][j] = max(dp_table[i][j], dp_table[p][j - cost[i] - cost[fan1] - cost[fan2]] + val[i] + val[fan1] + val[fan2]);
                }
            }
            p = i; // 更新上一个主件指针
        }
    }
    return dp_table[p][n];
}

/**
 * 方案二：空间压缩版 (最优解)
 * 时间复杂度 $O(m \times n)$，空间复杂度 $O(n)$
 */
int compute2() {
    for (int i = 1; i <= m; i++) {
        if (king[i]) {
            int f1 = (fans[i] >= 1) ? follows[i][0] : -1;
            int f2 = (fans[i] >= 2) ? follows[i][1] : -1;
            
            // 01 背包逆序遍历，确保 j - cost 拿的是上一层主件的结果
            for (int j = n; j >= cost[i]; j--) {
                // 决策 1: 只买主件
                dp[j] = max(dp[j], dp[j - cost[i]] + val[i]);
                
                // 决策 2: 主 + 附1
                if (f1 != -1 && j >= cost[i] + cost[f1]) {
                    dp[j] = max(dp[j], dp[j - cost[i] - cost[f1]] + val[i] + val[f1]);
                }
                // 决策 3: 主 + 附2
                if (f2 != -1 && j >= cost[i] + cost[f2]) {
                    dp[j] = max(dp[j], dp[j - cost[i] - cost[f2]] + val[i] + val[f2]);
                }
                // 决策 4: 主 + 附1 + 附2
                if (f1 != -1 && f2 != -1 && j >= cost[i] + cost[f1] + cost[f2]) {
                    dp[j] = max(dp[j], dp[j - cost[i] - cost[f1] - cost[f2]] + val[i] + val[f1] + val[f2]);
                }
            }
        }
    }
    return dp[n];
}

int main() {
    // 提升 I/O 效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n >> m) {
        clean();
        for (int i = 1; i <= m; i++) {
            int v, p, q;
            cin >> v >> p >> q;
            cost[i] = v;
            val[i] = v * p;
            if (q == 0) {
                king[i] = true;
            } else {
                king[i] = false;
                // q 号主件增加一个附件
                follows[q][fans[q]++] = i;
            }
        }
        cout << compute2() << "\n";
    }

    return 0;
}
