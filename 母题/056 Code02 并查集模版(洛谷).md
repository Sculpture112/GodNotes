// 056 Code02 并查集模版(洛谷)

// 测试链接 : https://www.luogu.com.cn/problem/P3367
// 核心思路：并查集（Union-Find）。本实现用递归函数实现路径压缩，并且省掉了按秩合并（小挂大）的优化，代码更轻量简洁。
// 1. 初始化：每个元素的父节点指向自己。
// 2. 查找 (find)：递归查找代表节点，并在回溯时将沿途节点的父节点直接更新为代表节点（路径压缩）。
// 3. 合并 (unite)：找到两个元素各自的代表节点，直接将其中一个的父节点指向另一个。
// 时间复杂度：单次操作均摊接近 O(1)，额外空间复杂度 O(N)（主要为递归调用栈和父节点数组）

#include <iostream>
#include <vector>

using namespace std;

// 针对洛谷等平台的快速读入模板，增加 EOF 判断
inline bool read(int &x) {
    x = 0; int f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == EOF) return false;
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    x *= f;
    return true;
}

// 依然推荐封装成类，管理生命周期，避免多组测试数据间的全局变量污染
class UnionFind {
private:
    vector<int> father;

public:
    // 构造函数：初始化并查集
    UnionFind(int n) {
        father.resize(n + 1);
        for (int i = 0; i <= n; i++) {
            father[i] = i;
        }
    }

    // 递归实现的查找与路径压缩
    int find(int i) {
        if (i != father[i]) {
            father[i] = find(father[i]); // 递归寻根并直接赋值给当前节点的父节点
        }
        return father[i];
    }

    // 判断两个节点是否属于同一个集合
    bool isSameSet(int x, int y) {
        return find(x) == find(y);
    }

    // 合并两个节点所在的集合
    void unite(int x, int y) {
        // C++ 中 union 是关键字，所以习惯用 unite 命名
        // 极简合并：省去了判断树大小（秩）的过程，直接挂载
        father[find(x)] = find(y);
    }
};

int main() {
    int n, m;
    // 持续读取直到 EOF
    while (read(n) && read(m)) {
        UnionFind uf(n);
        
        for (int i = 0; i < m; i++) {
            int z, x, y;
            read(z);
            read(x);
            read(y);
            
            if (z == 1) {
                uf.unite(x, y);
            } else if (z == 2) {
                if (uf.isSameSet(x, y)) {
                    printf("Y\n");
                } else {
                    printf("N\n");
                }
            }
        }
    }
    return 0;
}
