// 075 Code05 混合背包 + 多重背包普通窗口优化

// 能成功找零的钱数种类
// 每一种货币都给定面值val[i]，和拥有的数量cnt[i]
// 想知道目前拥有的货币，在钱数为1、2、3...m时
// 能找零成功的钱数有多少
// 测试链接 : http://poj.org/problem?id=1742

#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>

using namespace std;

/**
 * 💡 核心思路：
 * 1. 混合背包分类处理：
 *    - 当 cnt == 1 时，按 01背包 处理（空间压缩从右往左更新，防止重复使用）。
 *    - 当 val * cnt > m 时，货币总价值已超过最大容量，按 完全背包 处理（空间压缩从左往右更新）。
 *    - 其他情况按 多重背包 处理，采用 同余分组的滑动窗口 优化。
 * 2. 同余分组窗口优化：
 *    - 按对 val 取余的结果（0 到 val-1）将所有的状态分为 val 组。
 *    - 因为只有余数相同的状态之间才会发生状态转移（比如 val=3，那么状态 1, 4, 7, 10 是一组）。
 *    - 每一组内部，维护一个大小为 cnt 的滑动窗口，通过 trueCnt 记录窗口中为 true 的数量来实现 $O(1)$ 的状态转移。
 */

const int MAXN = 105;
const int MAXM = 100005;

int val[MAXN];
int cnt[MAXN];
bool dp[MAXM];
int n, m;

/**
 * 空间压缩版混合背包计算
 */
int compute() {
    // 初始化 dp 数组，除 dp[0] 为 true 外，其余均为 false
    for (int i = 1; i <= m; i++) dp[i] = false;
    dp[0] = true;

    for (int i = 1; i <= n; i++) {
        if (cnt[i] == 1) {
            // 01背包：从右往左更新
            for (int j = m; j >= val[i]; j--) {
                if (dp[j - val[i]]) {
                    dp[j] = true;
                }
            }
        } else if (val[i] * cnt[i] > m) {
            // 完全背包：从左往右更新
            for (int j = val[i]; j <= m; j++) {
                if (dp[j - val[i]]) {
                    dp[j] = true;
                }
            }
        } else {
            // 多重背包：同余分组 + 滑动窗口优化
            // mod 为余数，将状态分为 val[i] 组
            for (int mod = 0; mod < val[i]; mod++) {
                int trueCnt = 0;
                // 1. 初始化窗口：先统计最右侧初始窗口内的 true 数量
                for (int j = m - mod, size = 0; j >= 0 && size <= cnt[i]; j -= val[i], size++) {
                    trueCnt += dp[j] ? 1 : 0;
                }
                // 2. 窗口整体向左滑动
                // l 为移出窗口的最左端下标
                for (int j = m - mod, l = j - val[i] * (cnt[i] + 1); j >= 1; j -= val[i], l -= val[i]) {
                    if (dp[j]) {
                        // 如果当前状态原本就是 true，不需要更新，但它即将从后续考量中移除当前格子的影响
                        trueCnt--;
                    } else {
                        // 如果窗口内存在前置的 true (trueCnt != 0)，说明可以通过加上若干个 val[i] 抵达当前 j
                        if (trueCnt != 0) {
                            dp[j] = true;
                        }
                    }
                    // 窗口左侧移入新元素，右侧（上面的逻辑）实质已处理并抛弃旧元素
                    if (l >= 0) {
                        trueCnt += dp[l] ? 1 : 0;
                    }
                }
            }
        }
    }

    // 统计能找零成功的钱数
    int ans = 0;
    for (int i = 1; i <= m; i++) {
        if (dp[i]) {
            ans++;
        }
    }
    return ans;
}

int main() {
    // 循环读取直到 EOF，且 n 和 m 不能同时为 0
    while (scanf("%d %d", &n, &m) != EOF && (n != 0 || m != 0)) {
        for (int i = 1; i <= n; i++) {
            scanf("%d", &val[i]);
        }
        for (int i = 1; i <= n; i++) {
            scanf("%d", &cnt[i]);
        }
        printf("%d\n", compute());
    }
    return 0;
}
