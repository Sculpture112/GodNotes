#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>

using namespace std;

/**
 * 075 Code05 混合背包 + 多重背包窗口优化
 * 
 * 核心思路：
 * 1. 这是一个可行性 DP。dp[j] 表示金额 j 是否可以被找零。
 * 2. 窗口优化：对于面值为 v，数量为 c 的硬币。
 *    在同余分组 mod 下，金额 j 是否可达，取决于 j-v, j-2v ... j-cv 中是否有可达金额。
 *    我们维护一个计数器 trueCnt，记录当前窗口（大小为 c+1）内有多少个 true。
 */

const int MAXN = 105;
const int MAXM = 100005;

int val[MAXN];
int cnt[MAXN];
bool dp[MAXM];
int n, m;

int compute() {
    // 初始化 dp 数组，只有金额 0 是默认可达的
    memset(dp, 0, sizeof(dp));
    dp[0] = true;

    for (int i = 1; i <= n; i++) {
        if (cnt[i] == 1) {
            // 1. 01 背包：从右往左遍历，确保每个硬币只用一次
            for (int j = m; j >= val[i]; j--) {
                if (dp[j - val[i]]) dp[j] = true;
            }
        } 
        else if ((long long)val[i] * cnt[i] >= m) {
            // 2. 完全背包：当硬币多到足以覆盖总额时，从左往右遍历
            for (int j = val[i]; j <= m; j++) {
                if (dp[j - val[i]]) dp[j] = true;
            }
        } 
        else {
            // 3. 多重背包滑动窗口优化 (针对可行性问题简化)
            // 按面值 val[i] 的余数进行同余分组
            for (int mod = 0; mod < val[i]; mod++) {
                int trueCnt = 0;
                // 初始化初始窗口（从高位开始）
                // 窗口范围是 [j - cnt[i]*val[i], j]
                for (int j = m - mod, size = 0; j >= 0 && size <= cnt[i]; j -= val[i], size++) {
                    if (dp[j]) trueCnt++;
                }

                // 滑动窗口：j 是当前要更新的位置，l 是即将滑入窗口的新位置
                for (int j = m - mod, l = j - val[i] * (cnt[i] + 1); j >= 1; j -= val[i], l -= val[i]) {
                    if (dp[j]) {
                        // 如果当前 j 已经是 true，它对后面更小的 j 没有“贡献”
                        // 且窗口向左滑时，j 会移出窗口
                        trueCnt--;
                    } else {
                        // 如果当前 j 是 false，但窗口内有其他的 true，则当前 j 变为 true
                        if (trueCnt > 0) dp[j] = true;
                    }
                    // 将 l 位置（更小的金额）滑入窗口统计
                    if (l >= 0 && dp[l]) {
                        trueCnt++;
                    }
                }
            }
        }
    }

    // 统计 1~m 范围内有多少个金额是可达的
    int ans = 0;
    for (int i = 1; i <= m; i++) {
        if (dp[i]) ans++;
    }
    return ans;
}

int main() {
    // 提升 I/O 效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n >> m && (n != 0 || m != 0)) {
        for (int i = 1; i <= n; i++) cin >> val[i];
        for (int i = 1; i <= n; i++) cin >> cnt[i];
        cout << compute() << "\n";
    }

    return 0;
}
