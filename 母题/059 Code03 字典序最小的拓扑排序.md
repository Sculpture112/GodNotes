// 059 Code03 字典序最小的拓扑排序
// 核心思路：链式前向星 + 小根堆 (priority_queue)
// 时间复杂度 O(M + N * logN)，空间复杂度 O(M + N)

#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <cstring>

using namespace std;

const int MAXN = 100005;
const int MAXM = 100005;

// 链式前向星
int head[MAXN], nxt[MAXM], to[MAXM], cnt;
int indegree[MAXN];
int ans[MAXN];

void build(int n) {
    cnt = 1;
    memset(head, 0, sizeof(int) * (n + 1));
    memset(indegree, 0, sizeof(int) * (n + 1));
}

void addEdge(int u, int v) {
    nxt[cnt] = head[u];
    to[cnt] = v;
    head[u] = cnt++;
}

void topoSort(int n) {
    // C++ 小根堆：优先弹出数值小的点
    priority_queue<int, vector<int>, greater<int>> pq;
    
    for (int i = 1; i <= n; i++) {
        if (indegree[i] == 0) pq.push(i);
    }

    int fill = 0;
    while (!pq.empty()) {
        int cur = pq.top();
        pq.pop();
        ans[fill++] = cur;

        for (int ei = head[cur]; ei != 0; ei = nxt[ei]) {
            int v = to[ei];
            if (--indegree[v] == 0) {
                pq.push(v);
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    while (cin >> n >> m) {
        build(n);
        for (int i = 0; i < m; i++) {
            int u, v;
            cin >> u >> v;
            addEdge(u, v);
            indegree[v]++;
        }

        topoSort(n);

        for (int i = 0; i < n; i++) {
            cout << ans[i] << (i == n - 1 ? "" : " ");
        }
        cout << "\n";
    }
    return 0;
}
