```c++
// 060 Code02 喧闹和富有
// 核心思路：拓扑排序递推最小安静值。
// 时间复杂度 O(N + M)，空间复杂度 O(N + M)

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    vector<int> loudAndRich(vector<vector<int>>& richer, vector<int>&quiet) {
        int n = quiet.size();
        // 1. 建图与入度统计
        vector<vector<int>> graph(n);
        vector<int> indegree(n, 0);
        for (auto& r : richer) {
            // richer[0] 比 richer[1] 更有钱: richer[0] -> richer[1]
            graph[r[0]].push_back(r[1]);
            indegree[r[1]]++;
        }
        // 2. 队列初始化与 ans 数组初始化
        vector<int> q(n);
        int l = 0, r = 0;
        vector<int> ans(n);
        for (int i = 0; i < n; i++) {
            ans[i] = i; // 初始认为最安静的是自己
            if (indegree[i] == 0) {
                q[r++] = i;
            }
        }

        // 3. 拓扑排序更新
        while (l < r) {
            int cur = q[l++];
            for (int next : graph[cur]) {
                // 如果 cur 掌握的最安静的人，比 next 掌握的最安静的人还要安静
                if (quiet[ans[cur]] < quiet[ans[next]]) {
                    ans[next] = ans[cur];
                }
                if (--indegree[next] == 0) {
                    q[r++] = next;
                }
            }
        }

        return ans;
    }
};
```
