// 021 Code01 归并排序，acm练习风格

// 测试链接 : https://www.luogu.com.cn/problem/P1177
// 核心思路：归并排序（非递归版）。
// 1. 采用自底向上的迭代方式，避免了递归造成的函数调用栈开销，是更为极致的最优解。
// 2. 设置步长 step，从 1 开始，每次翻倍，直到步长大于等于数组长度。
// 3. 在每个步长下，将数组按步长分成若干组，相邻两组进行合并（merge）。
// 4. 合并时，使用局部开辟的辅助数组 help 暂存合并后的有序序列，然后再拷贝回原数组。
// 时间复杂度 O(N log N)，额外空间复杂度 O(N)

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 针对洛谷等平台的快读模板
inline bool read(int &x) {
    x = 0; int f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == EOF) return false;
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    x *= f;
    return true;
}

class Solution {
private:
    // l....r 一共有 n 个数
    // 时间复杂度 O(N)
    void merge(vector<int>& arr, vector<int>& help, int l, int m, int r) {
        int i = l;
        int a = l;
        int b = m + 1;
        
        // 两个区间合并
        while (a <= m && b <= r) {
            help[i++] = arr[a] <= arr[b] ? arr[a++] : arr[b++];
        }
        // 左侧指针、右侧指针，必有一个越界、另一个不越界
        while (a <= m) {
            help[i++] = arr[a++];
        }
        while (b <= r) {
            help[i++] = arr[b++];
        }
        // 拷贝回原数组
        for (i = l; i <= r; i++) {
            arr[i] = help[i];
        }
    }

public:
    // 归并排序非递归版（严格遵循单一最优解要求，舍弃递归版）
    void mergeSort(vector<int>& arr) {
        int n = arr.size();
        if (n < 2) return;
        
        // 局部开辟辅助数组，随生命周期自动释放
        vector<int> help(n);
        
        // 一共发生 O(log N) 次步长翻倍
        for (int l, m, r, step = 1; step < n; step <<= 1) {
            // 内部分组 merge，单层循环的时间复杂度 O(N)
            l = 0;
            while (l < n) {
                m = l + step - 1;
                // 如果左组的末尾已经达到或超过数组边界，说明没有右组，直接退出当前层
                if (m + 1 >= n) {
                    break;
                }
                // 确定右边界，防止越界
                r = min(l + (step << 1) - 1, n - 1);
                merge(arr, help, l, m, r);
                // 推进到下一组的起点
                l = r + 1;
            }
        }
    }
};

int main() {
    int n;
    if (read(n)) {
        vector<int> arr(n);
        for (int i = 0; i < n; i++) {
            read(arr[i]);
        }
        
        Solution().mergeSort(arr);
        
        // 使用 printf 快速输出
        for (int i = 0; i < n - 1; i++) {
            printf("%d ", arr[i]);
        }
        if (n > 0) {
            printf("%d\n", arr[n - 1]);
        }
    }
    return 0;
}
