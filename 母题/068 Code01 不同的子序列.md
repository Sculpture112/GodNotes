**核心套路:** 
面对双字符串求匹配数/编辑距离的问题，核心是构建基于“前缀”的二维状态矩阵，并通过分析“最后一个字符”的选与不选来进行状态转移。

1. **状态定义:** $dp[i][j]$ 表示在字符串 $s$ 的前 $i$ 个字符中，有多少个子序列等于字符串 $t$ 的前 $j$ 个字符。
    
2. **边界初始化 (Base Case):** * $dp[i][0] = 1$：当目标串 $t$ 为空串时，无论 $s$ 多长，通过“全不选”构成空串的方案数仅有 1 种。
    
    - $dp[0][j] = 0$ ($j > 0$)：当 $s$ 为空串而 $t$ 非空时，永远无法匹配，方案数为 0。
        
3. **状态转移方程 (核心分类讨论):**
    
    - **当 $s[i-1] \neq t[j-1]$ 时:** $s$ 的当前字符无法参与匹配，只能靠 $s$ 的前 $i-1$ 个字符去匹配。
        
        - 方程: $dp[i][j] = dp[i-1][j]$
            
    - **当 $s[i-1] == t[j-1]$ 时:** 面临岔路口。你可以选择**用**它匹配（依赖 $dp[i-1][j-1]$），也可以选择**不用**它匹配，把机会留给 $s$ 前面的字符（依赖 $dp[i-1][j]$）。求总数，需将两者相加。
        
        - 方程: $dp[i][j] = dp[i-1][j] + dp[i-1][j-1]$
            
4. **空间压缩技巧:** 观察状态方程可知，当前行只依赖“正上方”和“左上方”的元素。因此可以压缩为一维数组 $dp[m+1]$。**关键点**：内层循环必须**倒序遍历**（从 $m$ 到 $1$），以防止左侧上一行的旧状态被提前覆盖。


```cpp
// 068 Code01 不同的子序列

// 给你两个字符串s和t ，统计并返回在s的子序列中t出现的个数
// 答案对 1000000007 取模
// 测试链接 : https://leetcode.cn/problems/distinct-subsequences/

#include <iostream>
#include <vector>
#include <string>

using namespace std;

class Solution {
public:
    int numDistinct(string s, string t) {
        // 三种方法任选其一均可
        // return numDistinct1(s, t);
        // return numDistinct2(s, t);
        return numDistinct3(s, t);
    }

    // 二维动态规划
    int numDistinct1(const string& s, const string& t) {
        int n = s.length();
        int m = t.length();
        
        // dp[i][j] :
        // s[前缀长度为i]的所有子序列中，有多少个子序列等于t[前缀长度为j]
        // 使用 unsigned long long 防止中间无用状态在累加时造成溢出报错
        vector<vector<unsigned long long>> dp(n + 1, vector<unsigned long long>(m + 1, 0));
        
        for (int i = 0; i <= n; i++) {
            dp[i][0] = 1;
        }
        
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                dp[i][j] = dp[i - 1][j];
                if (s[i - 1] == t[j - 1]) {
                    dp[i][j] += dp[i - 1][j - 1];
                }
            }
        }
        return dp[n][m];
    }

    // 空间压缩
    int numDistinct2(const string& s, const string& t) {
        int n = s.length();
        int m = t.length();
        
        // 同样使用 unsigned long long 防止中间状态溢出
        vector<unsigned long long> dp(m + 1, 0);
        dp[0] = 1;
        
        for (int i = 1; i <= n; i++) {
            // 根据空间依赖关系，必须从右往左更新
            for (int j = m; j >= 1; j--) {
                if (s[i - 1] == t[j - 1]) {
                    dp[j] += dp[j - 1];
                }
            }
        }
        return dp[m];
    }

    // 本题说了要取模，所以增加取模的逻辑
    // 由于每次累加都取模，所以普通的 int 即可满足，不会溢出
    int numDistinct3(const string& s, const string& t) {
        int mod = 1000000007;
        int n = s.length();
        int m = t.length();
        
        vector<int> dp(m + 1, 0);
        dp[0] = 1;
        
        for (int i = 1; i <= n; i++) {
            for (int j = m; j >= 1; j--) {
                if (s[i - 1] == t[j - 1]) {
                    dp[j] = (dp[j] + dp[j - 1]) % mod;
                }
            }
        }
        return dp[m];
    }
};
```
