// 052 Code04 柱状图中最大的矩形
// 测试链接 : https://leetcode.cn/problems/largest-rectangle-in-histogram
// 核心思路：单调栈。对每个柱子求其作为最小高度时的最大宽度。
// 时间复杂度 O(N)，额外空间复杂度 O(N)

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
	int largestRectangleArea(vector<int>& heights) {
		int n = heights.size();
		int ans = 0;
		// 静态数组模拟栈
		vector<int> st(n);
		int r = 0;

		for (int i = 0; i < n; i++) {
			// 当前高度破坏了单调性（递增），开始结算
			while (r > 0 && heights[st[r - 1]] >= heights[i]) {
				int cur = st[--r];
				int left = (r == 0) ? -1 : st[r - 1];
				int right = i;
				// 宽度计算：右边界 - 左边界 - 1
				ans = max(ans, heights[cur] * (right - left - 1));
			}
			st[r++] = i;
		}

		// 清算阶段：栈中剩下的元素，右侧没有比它们矮的了
		while (r > 0) {
			int cur = st[--r];
			int left = (r == 0) ? -1 : st[r - 1];
			int right = n;
			ans = max(ans, heights[cur] * (right - left - 1));
		}

		return ans;
	}
};
