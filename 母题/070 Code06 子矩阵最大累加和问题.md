```cpp
// 070 Code06 子矩阵最大累加和问题

// 给定一个二维数组grid，找到其中子矩阵的最大累加和
// 返回拥有最大累加和的子矩阵左上角和右下角坐标
// 如果有多个子矩阵都有最大累加和，返回哪一个都可以
// 测试链接 : https://leetcode.cn/problems/max-submatrix-lcci/

#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>

using namespace std;

class Solution {
public:
    // 如果行和列的规模都是n，时间复杂度O(n^3)，最优解了
    vector<int> getMaxMatrix(const vector<vector<int>>& grid) {
        int n = grid.size();
        if (n == 0) {
            return {};
        }
        int m = grid[0].size();
        
        int max_sum = INT_MIN;
        int a = 0, b = 0, c = 0, d = 0;
        
        // 用于压缩多行数据的一维数组
        vector<int> nums(m, 0);
        
        for (int up = 0; up < n; up++) {
            // 每次更换上边界，都需要清空压缩数组
            fill(nums.begin(), nums.end(), 0);
            
            for (int down = up; down < n; down++) {
                // 如下代码就是题目1的附加问题 :
                // 子数组中找到拥有最大累加和的子数组
                for (int l = 0, r = 0, pre = INT_MIN; r < m; r++) {
                    // 将当前行的数据累加到压缩数组中
                    nums[r] += grid[down][r];
                    
                    if (pre >= 0) {
                        pre += nums[r];
                    } else {
                        pre = nums[r];
                        l = r; // 换开头
                    }
                    
                    if (pre > max_sum) {
                        max_sum = pre;
                        a = up;    // 左上角行
                        b = l;     // 左上角列
                        c = down;  // 右下角行
                        d = r;     // 右下角列
                    }
                }
            }
        }
        
        // C++11 支持直接返回初始化列表来构造 vector
        return {a, b, c, d};
    }
};
```
