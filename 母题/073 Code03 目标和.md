**核心套路:** 
动态规划与数学推导的结合。最核心的是将“添加正负号”问题，通过代数等式转化为“求特定累加和的子集个数”问题（即标准的 0-1 背包问题），从而实现降维打击，大幅优化时间和空间复杂度

// 073 Code03 目标和

// 给你一个非负整数数组 nums 和一个整数 target 。
// 向数组中的每个整数前添加 '+' 或 '-' ，然后串联起所有整数，可以构造一个表达式。
// 返回可以通过上述方法构造的，运算结果等于 target 的不同表达式的数目。
// 测试链接 : https://leetcode.cn/problems/target-sum/

#include <vector>
#include <unordered_map>
#include <numeric>
#include <cmath>

using namespace std;

class Solution {
public:
    /**
     * 方法一：暴力递归版
     * 时间复杂度 O(2^n)
     */
    int findTargetSumWays1(vector<int>& nums, int target) {
        return f1(nums, target, 0, 0);
    }

    int f1(vector<int>& nums, int target, int i, int sum) {
        if (i == nums.size()) {
            return sum == target ? 1 : 0;
        }
        return f1(nums, target, i + 1, sum + nums[i]) + f1(nums, target, i + 1, sum - nums[i]);
    }

    /**
     * 方法二：记忆化搜索版
     * 因为累加和存在负数，使用 unordered_map 模拟 Java 的 HashMap
     */
    int findTargetSumWays2(vector<int>& nums, int target) {
        unordered_map<int, unordered_map<int, int>> dp;
        return f2(nums, target, 0, 0, dp);
    }

    int f2(vector<int>& nums, int target, int i, int j, unordered_map<int, unordered_map<int, int>>& dp) {
        if (i == nums.size()) {
            return j == target ? 1 : 0;
        }
        if (dp.count(i) && dp[i].count(j)) {
            return dp[i][j];
        }
        int ans = f2(nums, target, i + 1, j + nums[i], dp) + f2(nums, target, i + 1, j - nums[i], dp);
        dp[i][j] = ans;
        return ans;
    }

    /**
     * 方法三：严格位置依赖的动态规划 + 下标平移技巧
     * 时间复杂度 O(n * sum)，空间复杂度 O(n * sum)
     */
    int findTargetSumWays3(vector<int>& nums, int target) {
        int s = 0;
        for (int num : nums) s += num;
        if (abs(target) > s) return 0;

        int n = nums.size();
        int m = 2 * s + 1; // 范围从 -s 到 +s，总长度为 2s+1
        
        // dp[i][j + s] 表示从第 i 个数开始到最后，能凑出累加和 j 的方法数
        // s 作为偏移量（Offset），使得下标范围变为 [0, 2s]
        vector<vector<int>> dp(n + 1, vector<int>(m, 0));
        
        // 基础情况：所有数都考虑完，且目标达成
        dp[n][target + s] = 1;

        for (int i = n - 1; i >= 0; i--) {
            for (int j = -s; j <= s; j++) {
                // 原逻辑：dp[i][j] = dp[i+1][j+nums[i]] + dp[i+1][j-nums[i]]
                if (j + nums[i] + s < m) {
                    dp[i][j + s] += dp[i + 1][j + nums[i] + s];
                }
                if (j - nums[i] + s >= 0) {
                    dp[i][j + s] += dp[i + 1][j - nums[i] + s];
                }
            }
        }
        return dp[0][s]; // 返回从下标 0 开始，已形成累加和为 0 的情况（平移后为 s）
    }

    /**
     * 方法四：转化为 01 背包问题（最优解）
     * 逻辑推导：
     * 设正数集合为 P，负数集合为 N
     * sum(P) - sum(N) = target
     * sum(P) + sum(N) = sum(nums)
     * 两式相加得：2 * sum(P) = target + sum(nums)
     * 得到：sum(P) = (target + sum(nums)) / 2
     * 问题转化为：在数组中寻找有多少个子集的累加和等于这个新目标。
     * 
     * 时间复杂度 O(n * target)，空间复杂度 O(target)
     */
    int findTargetSumWays4(vector<int>& nums, int target) {
        long long sum = 0;
        for (int n : nums) sum += n;

        // 排除不可能的情况：
        // 1. 目标绝对值大于总和
        // 2. 根据公式，(target + sum) 必须是偶数且非负
        if (sum < abs(target) || ((target + sum) & 1)) {
            return 0;
        }
        
        return subsets(nums, (int)((target + sum) >> 1));
    }

private:
    // 标准 01 背包空间压缩版：求累加和为 t 的子集个数
    int subsets(const vector<int>& nums, int t) {
        if (t < 0) return 0;
        vector<int> dp(t + 1, 0);
        dp[0] = 1; // 累加和为 0 的方法只有一种：什么都不选
        for (int num : nums) {
            for (int j = t; j >= num; j--) {
                dp[j] += dp[j - num];
            }
        }
        return dp[t];
    }
};
