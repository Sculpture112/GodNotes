// 073 Code02 夏季特惠

// 某公司游戏平台的夏季特惠开始了，你决定入手一些游戏
// 现在你一共有 X 元的预算，平台上所有的 n 个游戏均有折扣
// 标号为 i 的游戏的原价 a_i 元，现价只要 b_i 元
// 获得的总优惠金额不低于超过预算的总金额，心理上就不会觉得吃亏。
// 在心理上不觉得吃亏的前提下，获得尽可能多的快乐值。
// 测试链接 : https://leetcode.cn/problems/tJau2o/

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * 题目数学逻辑转换：
 * 设购买的游戏集合为 S
 * 总预算：x
 * 现价总和：sum(b_i)
 * 优惠总和：sum(a_i - b_i)
 * 条件：sum(a_i - b_i) >= sum(b_i) - x
 * 移项：x + sum(a_i - b_i - b_i) >= 0
 * 令 well_i = a_i - 2 * b_i
 * 条件变为：x + sum(well_i) >= 0
 * 
 * 策略：
 * 1. 如果 well_i >= 0，这件商品不仅送快乐值，还“增加”了你的预算，必买。
 * 2. 如果 well_i < 0，这件商品会消耗你的预算，消耗量为 |well_i|，属于 01 背包选择。
 */

const int MAXN = 505;
const int MAXX = 100005;

int cost[MAXN];
long long val[MAXN];
long long dp[MAXX];
int n, m, x;

// 01背包空间压缩实现
long long compute(int current_x, int item_count) {
    // 初始化 DP 数组
    for (int i = 0; i <= current_x; i++) dp[i] = 0;
    
    for (int i = 1; i < item_count; i++) {
        for (int j = current_x; j >= cost[i]; j--) {
            dp[j] = max(dp[j], dp[j - cost[i]] + val[i]);
        }
    }
    return dp[current_x];
}

int main() {
    // 提升 I/O 效率
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    while (cin >> n >> x) {
        m = 1; // 记录待选商品的数量
        long long initial_happy = 0;
        long long current_budget = x;

        for (int i = 0; i < n; i++) {
            int pre, cur;
            long long happy;
            cin >> pre >> cur >> happy;
            
            int well = pre - 2 * cur;
            if (well >= 0) {
                // 必买商品：增加预算空间并累加初始快乐值
                current_budget += well;
                initial_happy += happy;
            } else {
                // 选买商品：放入背包，其“花费”为 well 的绝对值
                cost[m] = -well;
                val[m] = happy;
                m++;
            }
        }

        // 注意：计算后的 current_budget 可能非常大（原预算 + 所有正 well 值）
        // 但根据题目 MAXX 限制，我们可以安全处理
        if (current_budget < 0) {
            // 如果初始预算加必买商品依然凑不够，这种 case 理论上不会在合法输入中导致问题
            // 但为了严谨，若预算为负且无可选商品，结果为 initial_happy
            cout << initial_happy << "\n";
        } else {
            // 限制预算上限，防止数组越界（MAXX 为题目给定的 X 可能达到的最大范围）
            // 在实际比赛中，如果 current_budget 超过了所有 cost 之和，可以取 cost 总和
            cout << initial_happy + compute((int)current_budget, m) << "\n";
        }
    }

    return 0;
}
