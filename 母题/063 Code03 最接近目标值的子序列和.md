// 063 Code03 最接近目标值的子序列和

// 给你一个整数数组 nums 和一个目标值 goal
// 你需要从 nums 中选出一个子序列，使子序列元素总和最接近 goal
// 也就是说，如果子序列元素和为 sum ，你需要 最小化绝对差 abs(sum - goal)
// 返回 abs(sum - goal) 可能的 最小值
// 注意，数组的子序列是通过移除原始数组中的某些元素（可能全部或无）而形成的数组。
// 数据量描述:
// 1 <= nums.length <= 40
// -10^7 <= nums[i] <= 10^7
// -10^9 <= goal <= 10^9
// 测试链接 : https://leetcode.cn/problems/closest-subsequence-sum/

#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

// 最大一半规模为 20，2^20 个状态
const int MAXN = 1 << 20;

// 全局静态数组，避免动态分配开销
int lsum[MAXN];
int rsum[MAXN];

// 避免与 std::fill 命名冲突，改名为 fill_ptr
int fill_ptr;

class Solution {
public:
    int minAbsDifference(vector<int>& nums, int goal) {
        int n = nums.size();
        long long min_val = 0;
        long long max_val = 0;
        
        // 剪枝 1：提前判断正负极值
        for (int i = 0; i < n; i++) {
            if (nums[i] >= 0) {
                max_val += nums[i];
            } else {
                min_val += nums[i];
            }
        }
        if (max_val < goal) {
            return abs(max_val - goal);
        }
        if (min_val > goal) {
            return abs(min_val - goal);
        }
        
        // 原始数组排序，为了后面递归的时候，还能剪枝
        // 常数优化：把相同数字聚在一起批量处理
        sort(nums.begin(), nums.end());
        
        fill_ptr = 0;
        collect(nums, 0, n >> 1, 0, lsum);
        int lsize = fill_ptr;
        
        fill_ptr = 0;
        collect(nums, n >> 1, n, 0, rsum);
        int rsize = fill_ptr;
        
        sort(lsum, lsum + lsize);
        sort(rsum, rsum + rsize);
        
        int ans = abs(goal);
        
        // 双指针合并查询
        // i 从左半侧最小值开始，j 从右半侧最大值开始
        for (int i = 0, j = rsize - 1; i < lsize; i++) {
            // 当往左移动能够使得绝对差变小或者相等时，j 持续左移
            while (j > 0 && abs(goal - lsum[i] - rsum[j - 1]) <= abs(goal - lsum[i] - rsum[j])) {
                j--;
            }
            ans = min(ans, abs(goal - lsum[i] - rsum[j]));
        }
        return ans;
    }

    void collect(const vector<int>& nums, int i, int e, int s, int sum[]) {
        if (i == e) {
            sum[fill_ptr++] = s;
        } else {
            // nums[i.....]这一组，相同的数字有几个
            int j = i + 1;
            while (j < e && nums[j] == nums[i]) {
                j++;
            }
            // nums[ 1 1 1 1 1 2....
            //       i         j
            for (int k = 0; k <= j - i; k++) {
                // k = 0 个
                // k = 1 个
                // k = 2 个
                collect(nums, j, e, s + k * nums[i], sum);
            }
        }
    }
};
