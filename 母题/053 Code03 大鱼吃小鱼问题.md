// 053 Code03 大鱼吃小鱼问题

// 测试链接 : https://www.nowcoder.com/practice/77199defc4b74b24b8ebf6244e1793de
// 测试链接 : https://leetcode.cn/problems/steps-to-make-array-non-decreasing/
// 核心思路：单调栈。求使数组变为非递减数组所需的步数，等价于每条鱼吃掉右侧比它小的鱼需要的最大轮数。

// 1. 从右向左逆序遍历数组。维护一个单调递减栈，栈中记录两个信息：元素的值，以及该元素经历的轮数。
// 2. 对于当前遇到的鱼 arr[i]，如果它比栈顶的鱼大，说明它可以吃掉栈顶的鱼。
// 3. 吃掉栈顶鱼所需的新轮数 = max(当前鱼之前吃别的鱼已经花费的轮数 + 1, 栈顶鱼自身的轮数)。
// 4. 不断弹出栈顶直到栈为空或当前鱼不再大于栈顶鱼，然后将当前鱼的值和它的轮数入栈。更新全局最大轮数。
// 时间复杂度 O(N)，额外空间复杂度 O(N)

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 针对多组测试数据及严苛耗时限制的快读模板（增加 EOF 判断以适配牛客网）
inline bool read(int &x) {
    x = 0; int f = 1;
    char ch = getchar();
    while (ch < '0' || ch > '9') {
        if (ch == EOF) return false;
        if (ch == '-') f = -1;
        ch = getchar();
    }
    while (ch >= '0' && ch <= '9') {
        x = x * 10 + ch - '0';
        ch = getchar();
    }
    x *= f;
    return true;
}

int main() {
    int n;
    // 牛客网支持多组输入测试，使用 while 持续读取直到 EOF
    while (read(n)) {
        // arr: 原始数组
        vector<int> arr(n);
        for (int i = 0; i < n; i++) {
            read(arr[i]);
        }

        // stack_val: 模拟栈，存鱼的体重
        // stack_turns: 模拟栈，存该位置花费的轮数
        // C++ 局部 vector 生命周期内自动管理内存，无需全局复用
        vector<int> stack_val(n);
        vector<int> stack_turns(n);
        int r = 0;   // 栈顶指针
        int ans = 0; // 全局最大轮数

        for (int i = n - 1; i >= 0; i--) {
            int curTurns = 0;
            // 当当前鱼体重 > 栈顶鱼体重时，结算栈顶
            while (r > 0 && stack_val[r - 1] < arr[i]) {
                curTurns = max(curTurns + 1, stack_turns[--r]);
            }
            // 当前鱼和它的轮数入栈
            stack_val[r] = arr[i];
            stack_turns[r++] = curTurns;
            
            ans = max(ans, curTurns);
        }

        // 快速输出
        printf("%d\n", ans);
    }
    return 0;
}
