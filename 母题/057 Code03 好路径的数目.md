```cpp
// 057 Code03 好路径的数目

// 测试链接 : https://leetcode.cn/problems/number-of-good-paths/
// 核心思路：并查集 + 边排序。
// 1. 题目要求好路径的起点和终点值相同，且路径上的所有节点值都不超过该值。
// 2. 将所有边按照两个端点中较大的节点值进行升序排序。这样我们在从小到大处理边时，当前处理的边所连接的连通块中的所有节点值，必然都小于等于当前边端点的最大值。
// 3. 维护一个并查集。每个连通分量的代表节点，必须是该连通块中节点值最大的那个。同时用 maxcnt 数组记录该最大值在连通块中出现的次数。
// 4. 合并两个连通块时：
//    - 若最大值不等，较小的一方直接挂到较大的一方下面（保证代表节点始终是最大值）。
//    - 若最大值相等，说明找到了满足条件的好路径。新增的好路径数量为两个连通块最大值个数的乘积（maxcnt[fx] * maxcnt[fy]），并将它们合并，同步更新 maxcnt。
// 5. 最终总好路径数为新增路径数加上单个节点本身构成的路径数（即节点总数 n）。
// 时间复杂度 O(M log M + N)，其中 M 为边数，N 为节点数；额外空间复杂度 O(N)

#include <vector>
#include <algorithm>
#include <functional>

using namespace std;

class Solution {
public:
    int numberOfGoodPaths(vector<int>& vals, vector<vector<int>>& edges) {
        int n = vals.size();
        
        // father: 并查集父节点数组
        // maxcnt: 连通块中最大值的出现次数
        vector<int> father(n);
        vector<int> maxcnt(n, 1);
        
        for (int i = 0; i < n; i++) {
            father[i] = i;
        }
        
        // Lambda 实现带路径压缩的查找
        function<int(int)> find = [&](int i) {
            if (i != father[i]) {
                father[i] = find(father[i]);
            }
            return father[i];
        };
        
        // 将边按照两端点中较大的值进行升序排序
        sort(edges.begin(), edges.end(), [&](const vector<int>& a, const vector<int>& b) {
            return max(vals[a[0]], vals[a[1]]) < max(vals[b[0]], vals[b[1]]);
        });
        
        int ans = n; // 单个节点本身也是一条合法路径，初始值为 n
        
        // 遍历排序后的边，进行合并并统计好路径
        for (const auto& edge : edges) {
            int fx = find(edge[0]);
            int fy = find(edge[1]);
            
            if (fx != fy) {
                if (vals[fx] > vals[fy]) {
                    father[fy] = fx;
                } else if (vals[fx] < vals[fy]) {
                    father[fx] = fy;
                } else {
                    // 两个连通块的最大值相等，产生跨越连通块的好路径
                    ans += maxcnt[fx] * maxcnt[fy];
                    father[fy] = fx;
                    maxcnt[fx] += maxcnt[fy];
                }
            }
        }
        
        return ans;
    }
};
```
