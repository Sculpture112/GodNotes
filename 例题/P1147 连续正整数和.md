#include <bits/stdc++.h>

using namespace std;

// #define int long long

// const int N = ;

void solve()

{

    int m;

    cin >> m;

    int l = 1, r = 2, sum = 3;

  

    while(l<r){

        if(sum == m){

            cout << l << " " << r << "\n";

            sum -= l;

            l++;

        }

        else if(sum<m){

            r++;

            sum += r;

        }

        else{

            sum -= l;

            l++;

        }

    }

}

signed main()

{

    ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);

    solve();

    return 0;

}
