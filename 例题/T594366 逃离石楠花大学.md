#include <iostream>
#include <vector>
#include <queue>

using namespace std;

const int INF = 1e9; // 定义无穷大

// 边的结构体
struct Edge {
    int to, weight;
};

// 优先队列中的状态结构体
struct State {
    int flowers; // 经过的石楠花数量
    int length;  // 路径长度
    int u;       // 当前节点编号

    // 重载大于号，用于优先队列构成小根堆
    // 优先比较石楠花数量，其次比较路径长度
    bool operator>(const State& other) const {
        if (flowers != other.flowers) {
            return flowers > other.flowers;
        }
        return length > other.length;
    }
};

int main() {
    // 优化输入输出速度
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m, k, s, t;
    if (!(cin >> n >> m >> k >> s >> t)) return 0;

    // 记录节点是否有石楠花
    vector<bool> is_flower(n + 1, false);
    for (int i = 0; i < k; ++i) {
        int node;
        cin >> node;
        is_flower[node] = true;
    }

    // 邻接表存图
    vector<vector<Edge>> adj(n + 1);
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
    }

    // 距离数组，初始化为无穷大
    vector<pair<int, int>> dist(n + 1, {INF, INF});
    // 小根堆优先队列
    priority_queue<State, vector<State>, greater<State>> pq;

    // 初始化起点
    dist[s] = {is_flower[s], 0};
    pq.push({is_flower[s], 0, s});

    while (!pq.empty()) {
        State curr = pq.top();
        pq.pop();

        int u = curr.u;

        // 如果当前取出的状态已经不是最优的，直接跳过（懒惰删除机制）
        if (curr.flowers > dist[u].first || 
           (curr.flowers == dist[u].first && curr.length > dist[u].second)) {
            continue;
        }

        // 如果到达终点，因为是 Dijkstra 贪心出来的，必定是最优解，可以直接结束
        if (u == t) break; 

        // 遍历相邻节点
        for (const auto& edge : adj[u]) {
            int v = edge.to;
            int w = edge.weight;
            
            // 计算到达下一个节点的新状态
            int next_flowers = curr.flowers + is_flower[v];
            int next_length = curr.length + w;

            // 如果新状态更优，则更新并压入队列
            if (next_flowers < dist[v].first || 
               (next_flowers == dist[v].first && next_length < dist[v].second)) {
                dist[v] = {next_flowers, next_length};
                pq.push({next_flowers, next_length, v});
            }
        }
    }

    // 输出结果
    if (dist[t].first == INF) {
        cout << -1 << "\n";
    } else {
        cout << dist[t].second << "\n";
    }

    return 0;
}
