    {
        // 必须先把左边 <= 右边的元素跳过！
        while (i <= m && arr[i] <= arr[j])
        {
            i++;
        }
        // 此时从 i 到 m 的所有元素，都严格大于 arr[j]
        sum += (m - i + 1);
    }
    int idx = l;
    int a = l;
    int b = m + 1;

    {
        help[idx++] = arr[a] <= arr[b] ? arr[a++] : arr[b++];
    }
    while (a <= m)
    {
        help[idx++] = arr[a++];
    }
    while (b <= r)
    {
        help[idx++] = arr[b++];
    }
    for (int i = l; i <= r; i++)
    {
        arr[i] = help[i];
    }
    return sum;
}

long long sum(int l, int r)
{
    if (l == r)
    {
        return 0;
    }

    int m = l + (r - l) / 2;
    return sum(l, m) + sum(m + 1, r) + merge(l, r, m);
}

void solve()
{
    cin >> n;
    arr.resize(n);
    for (int i = 0; i < n; i++)
    {
        cin >> arr[i];
    }
    help.resize(n, 0);
    cout << sum(0, n - 1);
}
signed main()

    ios::sync_with_stdio(false), cin.tie(0), cout.tie(0);
    solve();
