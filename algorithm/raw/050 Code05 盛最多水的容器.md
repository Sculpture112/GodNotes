// 050 Code05 盛最多水的容器
// 测试链接 : https://leetcode.cn/problems/container-with-most-water/
// 核心思路：双指针贪心。每次舍弃高度较短的那一端，尝试寻找更高的边界。
// 时间复杂度 O(N)，额外空间复杂度 O(1)

# include <iostream>

# include <vector>

# include <algorithm>

using namespace std;

class Solution {
public:
    int maxArea(vector<int>& height) {
        int ans = 0;
        int l = 0;
        int r = height.size() - 1;

        while (l < r) {
            // 当前木桶原理：水面高度由短板决定
            int h = min(height[l], height[r]);
            // 更新最大面积
            ans = max(ans, h * (r - l));

            // 谁小移动谁
            // 如果左侧短，l 右移尝试寻找更高板；反之 r 左移
            if (height[l] <= height[r]) {
                l++;
            } else {
                r--;
            }
        }
        return ans;
    }
};

int main() {
    Solution sol;
    vector<int> h = {1, 8, 6, 2, 5, 4, 8, 3, 7};
    // 预期结果：49 (下标 1 的 8 和下标 8 的 7 构成的容器)
    printf("盛最多水的容器面积: %d\n", sol.maxArea(h));
    return 0;
}
