// 049 Code06 K个不同整数的子数组
// 测试链接 : https://leetcode.cn/problems/subarrays-with-k-different-integers/
// 核心思路：将“恰好K个”转化为“最多K个 - 最多K-1个”

# include <iostream>

# include <vector>

# include <cstring>

using namespace std;

const int MAXN = 20005;
int cnts[MAXN];

// 计算 arr 中有多少个子数组，其不同数字的种类不超过 k
int numsOfMostKinds(const vector<int>& arr, int k) {
    if (k == 0) return 0;

    // 重置频次数组（LeetCode中数值范围通常是1~n）
    memset(cnts, 0, sizeof(int) * (arr.size() + 1));

    int n = arr.size();
    int ans = 0;
    // collect 记录当前窗口内不同数字的种类数
    for (int l = 0, r = 0, collect = 0; r < n; r++) {
        // 进窗
        if (++cnts[arr[r]] == 1) {
            collect++;
        }

        // 种类超过了k，缩窗
        while (collect > k) {
            if (--cnts[arr[l++]] == 0) {
                collect--;
            }
        }

        // 关键：累加以 r 结尾的合法子数组数量
        ans += (r - l + 1);
    }
    return ans;
}

int subarraysWithKDistinct(vector<int>& nums, int k) {
    // 恰好 k = 最多 k - 最多 k-1
    return numsOfMostKinds(nums, k) - numsOfMostKinds(nums, k - 1);
}

int main() {
    vector<int> nums = {1, 2, 1, 2, 3};
    int k = 2;
    // 结果应为 7
    printf("好子数组数目: %d\n", subarraysWithKDistinct(nums, k));
    return 0;
}
