#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,m;cin>>n>>m;
    if(m==0){
        cout<<0<<endl;
        return 0;
    }

    vector<int> s(n+1,0);
    for(int i =1;i<=n;i++){
        int a;cin>>a;
        s[i] = s[i-1] +a;
    }
    int min_sum = s[m] -s[0];

    for(int i= m;i<=n;i++){
        min_sum = min(min_sum,s[i] - s[i-m]);
    }
    cout<<min_sum<<endl;
    return 0;
}

        current_sum = current_sum + a[i] -a[i-m];
