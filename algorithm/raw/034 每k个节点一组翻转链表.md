// 034 每k个节点一组翻转链表

# include <iostream>

struct ListNode { int val; ListNode *next; }; // 节点定义

class Solution {
public:
    // 找到从s开始第k个节点并返回，若不足k个则返回nullptr
    ListNode* teamEnd(ListNode* s, int k) {
        while (--k > 0 && s) s = s.next;
        return s;
    }

    // 翻转从s到e的局部链表，并让原头部s指向下一组的开头
    void reverse(ListNode* s, ListNode* e) {
        e = e->next; // 暂存下一组的开头
        ListNode *pre = nullptr, *cur = s, *nxt = nullptr;
        while (cur != e) { // 标准三指针翻转
            nxt = cur->next;
            cur->next = pre;
            pre = cur;
            cur = nxt;
        }
        s->next = e; // 翻转后s成了尾部，接上下一组
    }

    ListNode* reverseKGroup(ListNode* head, int k) {
        ListNode *start = head, *end = teamEnd(start, k);
        if (!end) return head; // 第一组都不够k个，直接返回
        head = end; // 第一组翻转后的新头就是全链表的新头
        reverse(start, end);
        ListNode* lastEnd = start; // 记录上一组翻转后的结尾
        while (lastEnd->next) {
            start = lastEnd->next;
            end = teamEnd(start, k);
            if (!end) return head; // 剩余部分不足k个，停止翻转
            reverse(start, end);
            lastEnd->next = end; // 重要：上一组结尾指向本组翻转后的新头
            lastEnd = start; // 更新上一组结尾
        }
        return head;
    }
};
