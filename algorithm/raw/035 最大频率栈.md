// 035 最大频率栈

# include <unordered_map>

# include <vector>

# include <algorithm>

using namespace std;

class FreqStack {
    int topTimes = 0; // 当前出现的最大频率
    unordered_map<int, vector<int>> cntValues; // 频率 -> 该频率下的元素栈
    unordered_map<int, int> valueTimes; // 元素 -> 当前出现的次数
public:
    // 入栈：更新频率，并将元素加入对应频率的层级中
    void push(int val) {
        int curTimes = ++valueTimes[val]; // 频率加1
        cntValues[curTimes].push_back(val); // 加入对应频率层
        topTimes = max(topTimes, curTimes); // 更新最高频率
    }

    // 出栈：从最高频率层弹出最近进入的元素
    int pop() {
        vector<int>& topStack = cntValues[topTimes];
        int ans = topStack.back(); topStack.pop_back(); // 弹出最高频层末尾元素
        if (topStack.empty()) cntValues.erase(topTimes--); // 该层空了，最高频率下降
        if (--valueTimes[ans] == 0) valueTimes.erase(ans); // 更新元素总频率
        return ans;
    }
};
