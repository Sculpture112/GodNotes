// 037 验证平衡二叉树
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 验证平衡二叉树主逻辑 (isBalanced)
// 3. 递归高度计算与平衡性判定 (height)

#include <algorithm>
#include <cmath>

using namespace std;

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Code04_BalancedBinaryTree {
public:
    // 提交如下的方法
    // balance 是成员变量，对应 Java 中的全局/静态变量逻辑
    bool balance;

    bool isBalanced(TreeNode* root) {
        // balance 是成员变量，所有递归调用过程共享
        // 每次判断开始时，设置为 true
        balance = true;
        height(root);
        return balance;
    }

private:
    // 一旦发现不平衡，返回什么高度已经不重要了
    int height(TreeNode* cur) {
        if (!balance || cur == nullptr) {
            return 0;
        }
        int lh = height(cur->left);
        int rh = height(cur->right);

        // 判断当前节点是否平衡：左右子树高度差绝对值不超过 1
        if (abs(lh - rh) > 1) {
            balance = false;
        }

        // 返回当前子树的高度
        return max(lh, rh) + 1;
    }
};
