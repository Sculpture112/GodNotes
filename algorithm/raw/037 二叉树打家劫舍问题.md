// 037 二叉树打家劫舍问题
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 核心递归函数 (f) - 计算偷与不偷当前节点的最大收益
// 3. 打家劫舍主逻辑 (rob)

# include <algorithm>

using namespace std;

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Code07_HouseRobberIII {
public:
    // 提交如下的方法
    int rob(TreeNode* root) {
        f(root);
        return max(yes, no);
    }

    // yes 变成，X子树在偷头节点的情况下，最大的收益
    // C++ 中使用成员变量来对应 Java 的静态/全局变量
    int yes;

    // no 变成，X子树在不偷头节点的情况下，最大的收益
    int no;

    void f(TreeNode* root) {
        if (root == nullptr) {
            yes = 0;
            no = 0;
        } else {
            // y 记录当前节点偷的情况下的收益，初值为当前节点的钱
            int y = root->val;
            // n 记录当前节点不偷的情况下的收益，初值为 0
            int n = 0;

            // 处理左子树
            f(root->left);
            // 如果偷当前节点，那么左子树只能选“不偷”的结果
            y += no;
            // 如果不偷当前节点，那么左子树可以选“偷”或“不偷”中较大的那个
            n += max(yes, no);

            // 处理右子树
            f(root->right);
            // 如果偷当前节点，那么右子树只能选“不偷”的结果
            y += no;
            // 如果不偷当前节点，那么右子树可以选“偷”或“不偷”中较大的那个
            n += max(yes, no);

            // 将当前子树的结果写回“全局”变量，供父节点使用
            yes = y;
            no = n;
        }
    }
};
