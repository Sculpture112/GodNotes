// 037 验证搜索二叉树
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 基于中序遍历的迭代验证 (isValidBST1)
// 3. 基于递归采集范围信息的验证 (isValidBST2)

# include <algorithm>

# include <climits>

using namespace std;

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Code05_ValidateBinarySearchTree {
public:
    // 1. 基于中序遍历的迭代实现
    // 搜索二叉树的中序遍历结果必须是严格升序的
    static const int MAXN = 10001;
    TreeNode* stack[MAXN];
    int r;

    // 提交时改名为isValidBST
    bool isValidBST1(TreeNode* head) {
        if (head == nullptr) {
            return true;
        }
        TreeNode* pre = nullptr;
        r = 0;
        while (r > 0 || head != nullptr) {
            if (head != nullptr) {
                stack[r++] = head;
                head = head->left;
            } else {
                head = stack[--r];
                // 当前节点值必须大于前驱节点值
                if (pre != nullptr && pre->val >= head->val) {
                    return false;
                }
                pre = head;
                head = head->right;
            }
        }
        return true;
    }

    // 2. 基于递归采集信息的实现
    // 使用 long long 防止节点值为 int 的极值时判断溢出
    long long min_val, max_val;

    // 提交时改名为isValidBST
    bool isValidBST2(TreeNode* head) {
        if (head == nullptr) {
            min_val = LLONG_MAX;
            max_val = LLONG_MIN;
            return true;
        }

        // 递归左树
        bool lok = isValidBST2(head->left);
        long long lmin = min_val;
        long long lmax = max_val;

        // 递归右树
        bool rok = isValidBST2(head->right);
        long long rmin = min_val;
        long long rmax = max_val;

        // 更新当前子树的最小值和最大值
        min_val = min({(long long)head->val, lmin, rmin});
        max_val = max({(long long)head->val, lmax, rmax});

        // 验证：左树合法 && 右树合法 && 左树最大值 < 头 && 头 < 右树最小值
        return lok && rok && lmax < head->val && head->val < rmin;
    }
};
