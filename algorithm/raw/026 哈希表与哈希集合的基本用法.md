// 026 哈希表与哈希集合的基本用法

# include <iostream>

# include <string>

# include <unordered_set>

# include <unordered_map>

using namespace std;

struct Student { int age; string name; Student(int a, string b) : age(a), name(b) {} };

int main() {
    // 1. 字符串比较逻辑
    string str1 = "Hello", str2 = "Hello";
    cout << (str1 == str2) << endl; // C++ string 的 == 直接比较值，返回 1 (true)

    // 2. unordered_set (哈希集合)
    unordered_set<string> set;
    set.insert(str1); // 添加元素
    cout << set.count("Hello") << endl; // count 返回出现次数（0或1），模拟 contains
    cout << set.size() << endl; // 获取大小
    set.erase(str1); // 删除元素
    set.clear(); // 清空集合

    // 3. unordered_map (哈希表)
    unordered_map<string, string> map1;
    map1[str1] = "World"; // 插入或修改
    cout << map1.count(str2) << endl; // 判断 key 是否存在
    cout << map1["Hello"] << endl; // 获取 value
    map1.erase("Hello"); // 根据 key 删除

    // 4. 笔试技巧：用数组替代哈希表 (当 key 范围固定可控时)
    int arr[100] = {0}; // 假设 key 范围在 0-99
    arr[56] = 7285; // 增/改，效率远高于哈希表

    // 5. 自定义对象作为 key
    Student *s1 = new Student(17, "张三"), *s2 = new Student(17, "张三");
    unordered_map<Student*, string> map3; // 用指针模拟 Java 默认的引用比较
    map3[s1] = "这是张三";
    cout << (map3.count(s1)) << " " << (map3.count(s2)) << endl; // s1 存在，s2 不存在（地址不同）

    return 0;
}
