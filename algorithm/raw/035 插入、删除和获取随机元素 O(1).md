// 035 插入、删除和获取随机元素 O(1)

# include <unordered_map>

# include <vector>

# include <cstdlib>

using namespace std;

class RandomizedSet {
    unordered_map<int, int> map; // key: 元素值, value: 在数组中的下标
    vector<int> arr;            // 存储实际元素，用于随机访问
public:
    RandomizedSet() {}

    // 插入：若不存在，则加到数组末尾并在哈希表记录下标
    bool insert(int val) {
        if (map.count(val)) return false;
        map[val] = arr.size(); arr.push_back(val);
        return true;
    }

    // 删除：将待删元素与末尾元素交换，更新末尾元素的下标，最后弹出
    bool remove(int val) {
        if (!map.count(val)) return false;
        int idx = map[val], lastVal = arr.back();
        arr[idx] = lastVal; map[lastVal] = idx; // 覆盖待删位置，更新末尾值下标
        arr.pop_back(); map.erase(val);         // 物理删除末尾，删除哈希记录
        return true;
    }

    // 随机获取：利用数组下标实现等概率随机
    int getRandom() { return arr[rand() % arr.size()]; }
};
