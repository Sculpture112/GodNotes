// 048 二维差分模版(洛谷)
// 1. 二维差分更新逻辑 (add)
// 2. 前缀和还原矩阵逻辑 (build)
// 3. 矩阵清空逻辑 (clear)

#include <vector>
#include <cstring>

using namespace std;

class Code03_DiffMatrixLuogu {
public:
    static const int MAXN = 1002;

    // 二维差分数组
    int diff[MAXN][MAXN];

    int n, q;

    // 1. 二维差分更新
    // 在左上角 (a, b) 到右下角 (c, d) 的矩形区域内每个位置加上 k
    void add(int a, int b, int c, int d, int k) {
        diff[a][b] += k;
        diff[c + 1][b] -= k;
        diff[a][d + 1] -= k;
        diff[c + 1][d + 1] += k;
    }

    // 2. 构建最终矩阵
    // 对差分数组求二维前缀和，从而还原出每个位置被修改后的最终值
    void build() {
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= n; j++) {
                diff[i][j] += diff[i - 1][j] + diff[i][j - 1] - diff[i - 1][j - 1];
            }
        }
    }

    // 3. 清空矩阵
    // 为处理多组测试数据，清空 diff 数组
    void clear() {
        for (int i = 1; i <= n + 1; i++) {
            for (int j = 1; j <= n + 1; j++) {
                diff[i][j] = 0;
            }
        }
    }
};
