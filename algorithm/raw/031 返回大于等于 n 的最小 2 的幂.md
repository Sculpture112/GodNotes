// 031 返回大于等于 n 的最小 2 的幂

# include <iostream>

using namespace std;

int near2power(int n) {
    if (n <= 0) return 1; // 规定非正数返回 1
    n--; // 关键：防止 n 本身就是 2 的幂时结果翻倍
    n |= (unsigned int)n >> 1;  // 将最高位的 1 拷贝到第 1 位
    n |= (unsigned int)n >> 2;  // 将连续的 1 拷贝到后续 2 位
    n |= (unsigned int)n >> 4;  // 拷贝到后续 4 位
    n |= (unsigned int)n >> 8;  // 拷贝到后续 8 位
    n |= (unsigned int)n >> 16; // 拷贝到后续 16 位，至此 32 位全满
    return n + 1; // 满位 11...11 加 1 变为 100...00
}

int main() {
    cout << near2power(100) << endl; // 输出 128
    return 0;
}
