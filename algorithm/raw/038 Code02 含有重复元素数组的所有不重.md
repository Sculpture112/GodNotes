// 038 Code02 含有重复元素数组的所有不重复组合
// 1. 核心思路：排序后，将相同的数字看作一组
// 2. 递归决策：对于每一组相同的数字，决定选 0 个, 1 个, 2 个... 直到该组上限
// 3. 这种方法天然避开了重复组合，无需使用 Set 集合

# include <iostream>

# include <vector>

# include <algorithm>

using namespace std;

/**
 * 辅助递归函数
 * @param nums  原始数组（已排序）
 * @param i     当前处理到的下标位置
 * @param path  路径数组，模拟栈
 * @param size  当前路径的有效长度
 * @param ans   结果集
 */
void f(vector<int>& nums, int i, int path[], int size, vector<vector<int>>& ans) {
    if (i == nums.size()) {
        // 收集结果
        vector<int> cur;
        for (int j = 0; j < size; j++) {
            cur.push_back(path[j]);
        }
        ans.push_back(cur);
    } else {
        // 1. 找到当前组的边界：找出下一组第一个数的位置 j
        int j = i + 1;
        while (j < nums.size() && nums[i] == nums[j]) {
            j++;
        }

        // 2. 决策：当前数字 nums[i] 选 0 个
        // 直接跳到下一组数字的位置 j
        f(nums, j, path, size, ans);

        // 3. 决策：当前数字 nums[i] 选 1个、2个...k个
        // 这里的 i 是当前组内元素的移动下标，j 是下一组的起点
        for (; i < j; i++) {
            path[size++] = nums[i]; // 每选一个，size 增加
            f(nums, j, path, size, ans);
        }
    }
}

vector<vector<int>> subsetsWithDup(vector<int>& nums) {
    vector<vector<int>> ans;
    // 必须排序，让相同的数聚在一起
    sort(nums.begin(), nums.end());

    // 使用静态数组或 vector 模拟 path，这里用动态分配模拟 Java 的 int[] path
    int* path = new int[nums.size()];

    f(nums, 0, path, 0, ans);

    delete[] path; // 释放内存
    return ans;
}

// 简单验证逻辑
int main() {
    vector<int> nums = {1, 2, 2};
    vector<vector<int>> result = subsetsWithDup(nums);

    printf("nums = [1, 2, 2] 的不重复组合:\n");
    for (const auto& subset : result) {
        printf("[ ");
        for (int x : subset) printf("%d ", x);
        printf("]\n");
    }

    return 0;
}
