// 110 Code04 线段树支持范围重置、范围查询最大值
// 1. 最大值信息的汇总 (up)
// 2. 懒信息的下发 (down)
// 3. 懒更新任务的具体执行 (lazy)
// 4. 建树 (build)
// 5. 范围重置 (update)
// 6. 查询最大值 (query)

# include <iostream>

# include <vector>

# include <algorithm>

# include <cstdio>

# include <climits>

# include <ctime>

using namespace std;

// 基础常量定义
const int MAXN = 100005;

// 全局静态数组，防止栈溢出
long long arr[MAXN];
long long max_tree[MAXN << 2];
long long change[MAXN << 2];
bool update_tag[MAXN << 2];

// 1. 最大值信息的汇总
void up(int i) {
    // 父范围的最大值 = max(左孩子最大值, 右孩子最大值)
    max_tree[i] = max(max_tree[i << 1], max_tree[i << 1 | 1]);
}

// 3. 懒更新任务的具体执行
// 范围重置操作：当前范围所有数字重置为 v
void lazy(int i, long long v) {
    max_tree[i] = v;
    change[i] = v;
    update_tag[i] = true;
}

// 2. 懒信息的下发
void down(int i) {
    if (update_tag[i]) {
        // 向左孩子发懒信息
        lazy(i << 1, change[i]);
        // 向右孩子发懒信息
        lazy(i << 1 | 1, change[i]);
        // 标记重置
        update_tag[i] = false;
    }
}

// 4. 建树
void build(int l, int r, int i) {
    if (l == r) {
        max_tree[i] = arr[l];
    } else {
        int mid = (l + r) >> 1;
        build(l, mid, i << 1);
        build(mid + 1, r, i << 1 | 1);
        up(i);
    }
    change[i] = 0;
    update_tag[i] = false;
}

// 5. 范围重置
// jobl ~ jobr 范围上每个数字重置为 jobv
void update(int jobl, int jobr, long long jobv, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        lazy(i, jobv);
    } else {
        down(i);
        int mid = (l + r) >> 1;
        if (jobl <= mid) {
            update(jobl, jobr, jobv, l, mid, i << 1);
        }
        if (jobr > mid) {
            update(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        }
        up(i);
    }
}

// 6. 查询最大值
// 查询 jobl ~ jobr 范围内的最大值
long long query(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return max_tree[i];
    }
    down(i);
    int mid = (l + r) >> 1;
    long long ans = LLONG_MIN;
    if (jobl <= mid) {
        ans = max(ans, query(jobl, jobr, l, mid, i << 1));
    }
    if (jobr > mid) {
        ans = max(ans, query(jobl, jobr, mid + 1, r, i << 1 | 1));
    }
    return ans;
}

// 验证结构的更新（对数器暴力逻辑）
void checkUpdate(vector<long long>& check, int jobl, int jobr, long long jobv) {
    for (int i = jobl; i <= jobr; i++) {
        check[i] = jobv;
    }
}

// 验证结构的查询（对数器暴力逻辑）
long long checkQuery(vector<long long>& check, int jobl, int jobr) {
    long long ans = LLONG_MIN;
    for (int i = jobl; i <= jobr; i++) {
        ans = max(ans, check[i]);
    }
    return ans;
}

int main() {
    srand(time(0));
    printf("测试开始\n");
    int n = 1000;
    int v = 2000;
    int t = 1000000;

    // 生成随机数组
    for (int i = 1; i <= n; i++) {
        arr[i] = (long long)(rand() % v);
    }

    // 建立线段树
    build(1, n, 1);

    // 建立验证数组
    vector<long long> check(n + 1);
    for (int i = 1; i <= n; i++) {
        check[i] = arr[i];
    }

    for (int i = 1; i <= t; i++) {
        int op = rand() % 2;
        int a = rand() % n + 1;
        int b = rand() % n + 1;
        int jobl = min(a, b);
        int jobr = max(a, b);

        if (op == 0) {
            long long jobv = (long long)(rand() % (v * 2) - v);
            update(jobl, jobr, jobv, 1, n, 1);
            checkUpdate(check, jobl, jobr, jobv);
        } else {
            long long ans1 = query(jobl, jobr, 1, n, 1);
            long long ans2 = checkQuery(check, jobl, jobr);
            if (ans1 != ans2) {
                printf("出错了!\n");
                break;
            }
        }
    }
    printf("测试结束\n");

    return 0;
}
