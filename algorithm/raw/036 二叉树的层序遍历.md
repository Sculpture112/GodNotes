// 036 二叉树的层序遍历
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 普通BFS解法 (使用哈希表记录层级的BFS)
// 3. 优化BFS解法 (手动数组模拟队列，分层处理的优化BFS)

# include <vector>

# include <queue>

# include <unordered_map>

using namespace std;

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Code01_LevelOrderTraversal {
public:
    // 提交时把方法名改为levelOrder，此方法为普通bfs，此题不推荐
    vector<vector<int>> levelOrder1(TreeNode* root) {
        vector<vector<int>> ans;
        if (root != nullptr) {
            queue<TreeNode*> q;
            unordered_map<TreeNode*, int> levels;
            q.push(root);
            levels[root] = 0;
            while (!q.empty()) {
                TreeNode* cur = q.front();
                q.pop();
                int level = levels[cur];
                // 如果结果数组大小等于当前层级，说明开启了新的一层
                if (ans.size() == level) {
                    ans.push_back(vector<int>());
                }
                ans[level].push_back(cur->val);
                if (cur->left != nullptr) {
                    q.push(cur->left);
                    levels[cur->left] = level + 1;
                }
                if (cur->right != nullptr) {
                    q.push(cur->right);
                    levels[cur->right] = level + 1;
                }
            }
        }
        return ans;
    }

    // 如果测试数据量变大了就修改这个值
    static const int MAXN = 2001;
    // C++ 中为了完全对应原代码逻辑，使用手动数组模拟队列
    TreeNode* q_arr[MAXN];
    int l, r;

    // 提交时把方法名改为levelOrder，此方法为每次处理一层的优化bfs，此题推荐
    vector<vector<int>> levelOrder2(TreeNode* root) {
        vector<vector<int>> ans;
        if (root != nullptr) {
            l = r = 0;
            q_arr[r++] = root;
            while (l < r) { // 队列里还有东西
                int size = r - l; // 当前层的节点个数
                vector<int> current_level;
                for (int i = 0; i < size; i++) {
                    TreeNode* cur = q_arr[l++];
                    current_level.push_back(cur->val);
                    if (cur->left != nullptr) {
                        q_arr[r++] = cur->left;
                    }
                    if (cur->right != nullptr) {
                        q_arr[r++] = cur->right;
                    }
                }
                ans.push_back(current_level);
            }
        }
        return ans;
    }
};
