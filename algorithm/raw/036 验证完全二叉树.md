```cpp
// 036 验证完全二叉树
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 验证完全二叉树实现 (使用 BFS 逻辑与叶节点阶段标记)

#include <vector>

using namespace std;

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Code08_CompletenessOfBinaryTree {
public:
    // 如果测试数据量变大了就修改这个值
    static const int MAXN = 101;
    // 使用数组模拟队列
    TreeNode* q[MAXN];
    int l, r;

    bool isCompleteTree(TreeNode* h) {
        if (h == nullptr) {
            return true;
        }
        l = r = 0;
        q[r++] = h;

        // 是否遇到过左右两个孩子不双全的节点
        // 一旦遇到不双全的节点，后续所有节点都必须是叶子节点
        bool leaf = false;

        while (l < r) {
            h = q[l++];

            // 两个错误的任一情况出现都不是完全二叉树：
            // 1) 有右孩子但没左孩子
            // 2) 已经开启了叶节点阶段，一旦遇到不双全的节点，后续所有节点都必须是叶子节点
            if ((h->left == nullptr && h->right != nullptr) || 
                (leaf && (h->left != nullptr || h->right != nullptr))) {
                return false;
            }

            if (h->left != nullptr) {
                q[r++] = h->left;
            }
            if (h->right != nullptr) {
                q[r++] = h->right;
            }

            // 如果当前节点孩子不全（只有左没右，或者左右都没），开启叶节点阶段
            if (h->left == nullptr || h->right == nullptr) {
                leaf = true;
            }
        }
        return true;
    }
};
```
