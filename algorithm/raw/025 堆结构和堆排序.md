// 025 堆结构和堆排序
#include <cstdio>
#include <algorithm>
using namespace std;

const int MAXN = 100001;
int arr[MAXN], n;

// 向上调整大根堆：用于从顶到底建堆过程
void heapInsert(int i) {
    while (arr[i] > arr[(i - 1) / 2]) { swap(arr[i], arr[(i - 1) / 2]); i = (i - 1) / 2; }
}

// 向下调整大根堆：用于从底到顶建堆或弹出堆顶后的调整
void heapify(int i, int size) {
    int l = i * 2 + 1; // 左孩子索引
    while (l < size) {
        int best = l + 1 < size && arr[l + 1] > arr[l] ? l + 1 : l; // 左右孩子选大的
        best = arr[best] > arr[i] ? best : i; // 孩子与父亲比
        if (best == i) break; // 父亲大则停止
        swap(arr[i], arr[best]); i = best; l = i * 2 + 1;
    }
}

// 堆排序改进版：从底到顶建堆O(N)，排序过程O(N*logN)
void heapSort() {
    for (int i = n - 1; i >= 0; i--) heapify(i, n); // O(N) 建堆
    int size = n;
    while (size > 1) { swap(arr[0], arr[--size]); heapify(0, size); } // 交换并调整
}

int main() {
    if (scanf("%d", &n) != EOF) {
        for (int i = 0; i < n; i++) scanf("%d", &arr[i]);
        heapSort();
        for (int i = 0; i < n; i++) printf("%d%c", arr[i], i == n - 1 ? '\n' : ' ');
    }
    return 0;
}
