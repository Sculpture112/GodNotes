// 039 Code02 含有嵌套的字符串解码
// 1. 核心思路：将字符串解码看作是另一种形式的“表达式计算”
// 2. 嵌套处理：遇到 '[' 开启递归 f，遇到 ']' 或结尾停止并汇报全局位置 where
// 3. 结果合并：通过 get 函数（或简单的循环）将递归回来的子串按 cnt 倍数翻倍

# include <iostream>

# include <string>

# include <vector>

using namespace std;

// 全局变量记录递归位置
static int where = 0;

/**
 * 辅助函数：将字符串 str 重复 cnt 次
 */
string get(int cnt, string str) {
    string res = "";
    for (int i = 0; i < cnt; i++) {
        res += str;
    }
    return res;
}

/**
 * 核心递归函数
 * @param s 原始字符数组
 * @param i 当前处理起始位置
 */
string f(const string& s, int i) {
    string path = "";
    int cnt = 0;

    while (i < s.length() && s[i] != ']') {
        if ((s[i] >= 'a' && s[i] <= 'z') || (s[i] >= 'A' && s[i] <= 'Z')) {
            // 情况1：遇到普通字母
            path += s[i++];
        } else if (s[i] >= '0' && s[i] <= '9') {
            // 情况2：遇到数字，累加倍数
            cnt = cnt * 10 + (s[i++] - '0');
        } else {
            // 情况3：遇到 '['，递归去拿括号内部解码后的结果
            // 注意：此时 cnt 已经存好了，比如 3[...
            path += get(cnt, f(s, i + 1));
            i = where + 1; // 更新 i，跳过刚才递归处理完的那个 ']'
            cnt = 0;       // 倍数归零，迎接下一组
        }
    }
    where = i; // 汇报当前处理到的下标
    return path;
}

string decodeString(string str) {
    where = 0;
    return f(str, 0);
}

// 简单验证
int main() {
    string s = "3[a2[c]]"; // a cc a cc a cc -> accaccacc
    printf("输入: %s\n", s.c_str());
    printf("结果: %s\n", decodeString(s).c_str());
    return 0;
}
