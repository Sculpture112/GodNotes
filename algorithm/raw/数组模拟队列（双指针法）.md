```cpp
// 数组模拟队列（双指针法）
class Queue2 {
    vector<int> q;
    int l, r; // l: 头指针, r: 尾指针(指向下一个空位)

public:
    Queue2(int n) : q(n), l(0), r(0) {}

    bool isEmpty() { return l == r; }
    void offer(int num) { q[r++] = num; } // 入队：存值后 r 右移
    int poll() { return q[l++]; }         // 出队：取值后 l 右移
    int head() { return q[l]; }           // 查看队头
    int size() { return r - l; }
};
```
