// 039 Code01 含有嵌套的表达式求值
// 1. 递归思路：遇到左括号 '(' 就开启下一层递归，遇到右括号 ')' 就结算并跳回
// 2. 乘除优先：通过 push 函数，遇到 * / 立即计算，保证 numbers 内部只有加减关系
// 3. 全局指针：使用 static 变量 where 记录递归处理到的位置

#include <iostream>
#include <vector>
#include <string>

using namespace std;

// 全局变量，记录递归计算到了哪个位置
static int where = 0;

// 辅助函数：负责执行乘除合并逻辑
void push(vector<int>& numbers, vector<char>& ops, int cur, char op) {
    if (numbers.empty() || (ops.back() == '+' || ops.back() == '-')) {
        numbers.push_back(cur);
        ops.push_back(op);
    } else {
        int topNumber = numbers.back();
        char topOp = ops.back();
        if (topOp == '*') {
            numbers.back() = topNumber * cur;
        } else {
            numbers.back() = topNumber / cur;
        }
        ops.back() = op;
    }
}

// 辅助函数：计算最终的加减序列
int compute(const vector<int>& numbers, const vector<char>& ops) {
    int n = numbers.size();
    int ans = numbers[0];
    for (int i = 1; i < n; i++) {
        ans += (ops[i - 1] == '+' ? numbers[i] : -numbers[i]);
    }
    return ans;
}

// 核心递归函数
int f(const string& s, int i) {
    int cur = 0;
    vector<int> numbers;
    vector<char> ops;

    while (i < s.length() && s[i] != ')') {
        if (s[i] >= '0' && s[i] <= '9') {
            cur = cur * 10 + (s[i++] - '0');
        } else if (s[i] != '(') {
            // 遇到运算符 + - * /
            push(numbers, ops, cur, s[i++]);
            cur = 0;
        } else {
            // 遇到左括号，递归进去拿结果
            cur = f(s, i + 1);
            i = where + 1; // 从递归结算出来的右括号后面继续
        }
    }
    // 最后补上一个加号，收尾结算
    push(numbers, ops, cur, '+');
    where = i; // 汇报当前位置
    return compute(numbers, ops);
}

int calculate(string str) {
    where = 0;
    return f(str, 0);
}

// 简单测试
int main() {
    string exp = "1+(2*(3+4)/2)-5"; // 1 + 7 - 5 = 3
    printf("表达式 %s 的计算结果为: %d\n", exp.c_str(), calculate(exp));
    return 0;
}
