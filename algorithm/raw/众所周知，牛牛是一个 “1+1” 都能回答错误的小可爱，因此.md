```cpp
# include <bits/stdc++.h>

using namespace std;

// 解决单组测试用例的函数
void solve() {
    string s[4];
    int len[4];

    // 1. 读入 4 个选项并记录长度
    for (int i = 0; i < 4; ++i) {
        // 虽然题目没说有空格，但在处理这种整行字符串时，getline 比 cin 更稳
        if (!(getline(cin, s[i]))) return;
        len[i] = s[i].length();
    }

    // 2. 统计长度分布
    int minL = *min_element(len, len + 4);
    int maxL = *max_element(len, len + 4);
```

段代码使用的是 C++ 标准库（STL）中的算法函数，用于在一个数组（或容器）中寻找最小值和最大值。

```cpp
简单来说，这两行代码的作用是：从长度为 4 的数组 len 中，分别找出最小的元素和最大的元素，并将它们的值赋给 minL 和 maxL。    
    int cntMin = 0, cntMax = 0;
    int minIdx = -1, maxIdx = -1;

    for (int i = 0; i < 4; ++i) {
        if (len[i] == minL) {
            cntMin++;
            minIdx = i;
        }
        if (len[i] == maxL) {
            cntMax++;
            maxIdx = i;
        }
    }

    // 3. 根据题目定义的逻辑进行判断
    // 三长一短：恰有一个最短，且不存在恰有一个最长
    bool isThreeLongOneShort = (cntMin == 1 && cntMax != 1);
    // 三短一长：恰有一个最长，且不存在恰有一个最短
    bool isThreeShortOneLong = (cntMax == 1 && cntMin != 1);

    if (isThreeLongOneShort) {
        cout << (char)('A' + minIdx) << endl;
    } else if (isThreeShortOneLong) {
        cout << (char)('A' + maxIdx) << endl;
    } else {
        cout << "C" << endl; // 参差不齐
    }
}
```
