```cpp
// 110 Code06 线段树同时支持范围重置、范围增加、范围查询最大值
// 1. 最大值信息的汇总 (up)
// 2. 懒信息的下发 (down)
// 3. 范围重置懒更新任务执行 (updateLazy)
// 4. 范围增加懒更新任务执行 (addLazy)
// 5. 建树 (build)
// 6. 范围重置修改 (update)
// 7. 范围增加修改 (add)
// 8. 范围查询最大值 (query)

#include <iostream>
#include <cstdio>
#include <algorithm>
#include <climits>

using namespace std;

// 基础常量定义，MAXN 为 1,000,001
const int MAXN = 1000001;

// 全局静态数组，long long 处理，防止溢出
long long arr[MAXN];
long long max_tree[MAXN << 2];
long long add_tag[MAXN << 2];
long long change_tag[MAXN << 2];
bool update_tag[MAXN << 2];

// 1. 最大值信息的汇总
void up(int i) {
    max_tree[i] = max(max_tree[i << 1], max_tree[i << 1 | 1]);
}

// 3. 范围重置懒更新任务执行
// 优先级：重置操作会覆盖掉之前的增加操作
void updateLazy(int i, long long v) {
    max_tree[i] = v;
    add_tag[i] = 0;
    change_tag[i] = v;
    update_tag[i] = true;
}

// 4. 范围增加懒更新任务执行
void addLazy(int i, long long v) {
    max_tree[i] += v;
    add_tag[i] += v;
}

// 2. 懒信息的下发
// 遵循先处理重置(Update)，再处理增加(Add)的原则
void down(int i) {
    if (update_tag[i]) {
        updateLazy(i << 1, change_tag[i]);
        updateLazy(i << 1 | 1, change_tag[i]);
        update_tag[i] = false;
    }
    if (add_tag[i] != 0) {
        addLazy(i << 1, add_tag[i]);
        addLazy(i << 1 | 1, add_tag[i]);
        add_tag[i] = 0;
    }
}

// 5. 建树
void build(int l, int r, int i) {
    if (l == r) {
        max_tree[i] = arr[l];
    } else {
        int mid = (l + r) >> 1;
        build(l, mid, i << 1);
        build(mid + 1, r, i << 1 | 1);
        up(i);
    }
    add_tag[i] = 0;
    change_tag[i] = 0;
    update_tag[i] = false;
}

// 6. 范围重置修改
void update(int jobl, int jobr, long long jobv, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        updateLazy(i, jobv);
    } else {
        int mid = (l + r) >> 1;
        down(i);
        if (jobl <= mid) {
            update(jobl, jobr, jobv, l, mid, i << 1);
        }
        if (jobr > mid) {
            update(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        }
        up(i);
    }
}

// 7. 范围增加修改
void add(int jobl, int jobr, long long jobv, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        addLazy(i, jobv);
    } else {
        int mid = (l + r) >> 1;
        down(i);
        if (jobl <= mid) {
            add(jobl, jobr, jobv, l, mid, i << 1);
        }
        if (jobr > mid) {
            add(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        }
        up(i);
    }
}

// 8. 范围查询最大值
long long query(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return max_tree[i];
    }
    int mid = (l + r) >> 1;
    down(i);
    long long ans = LLONG_MIN;
    if (jobl <= mid) {
        ans = max(ans, query(jobl, jobr, l, mid, i << 1));
    }
    if (jobr > mid) {
        ans = max(ans, query(jobl, jobr, mid + 1, r, i << 1 | 1));
    }
    return ans;
}

int main() {
    int n, m;
    // 使用 scanf/printf 以匹配 Java 中 StreamTokenizer 的高效 IO
    if (scanf("%d %d", &n, &m) != EOF) {
        for (int i = 1; i <= n; i++) {
            scanf("%lld", &arr[i]);
        }
        build(1, n, 1);

        int op, jobl, jobr;
        long long jobv;
        for (int i = 1; i <= m; i++) {
            scanf("%d", &op);
            if (op == 1) {
                scanf("%d %d %lld", &jobl, &jobr, &jobv);
                update(jobl, jobr, jobv, 1, n, 1);
            } else if (op == 2) {
                scanf("%d %d %lld", &jobl, &jobr, &jobv);
                add(jobl, jobr, jobv, 1, n, 1);
            } else {
                scanf("%d %d", &jobl, &jobr);
                printf("%lld\n", query(jobl, jobr, 1, n, 1));
            }
        }
    }
    return 0;
}
```
