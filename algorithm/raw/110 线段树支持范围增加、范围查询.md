// 110 线段树支持范围增加、范围查询
// 1. 累加和信息的汇总 (up)
// 2. 懒信息的下发 (down)
// 3. 懒更新任务的具体执行 (lazy)
// 4. 建树 (build)
// 5. 范围修改 (add)
// 6. 查询累加和 (query)

# include <iostream>

# include <cstdio>

# include <algorithm>

using namespace std;

// 使用常数定义最大范围
const int MAXN = 100005;

// 全局静态数组，long long 处理，防止溢出
long long arr[MAXN];
long long sum_tree[MAXN << 2];
long long add_tag[MAXN << 2];

// 1. 累加和信息的汇总
void up(int i) {
    // 父范围的累加和 = 左范围累加和 + 右范围累加和
    sum_tree[i] = sum_tree[i << 1] + sum_tree[i << 1 | 1];
}

// 3. 懒更新任务的具体执行
// 把线段树当前范围的 sum 数组和 add 数组调整好
void lazy(int i, long long v, int n) {
    sum_tree[i] += v * n;
    add_tag[i] += v;
}

// 2. 懒信息的下发
void down(int i, int ln, int rn) {
    if (add_tag[i] != 0) {
        // 发左
        lazy(i << 1, add_tag[i], ln);
        // 发右
        lazy(i << 1 | 1, add_tag[i], rn);
        // 父范围懒信息清空
        add_tag[i] = 0;
    }
}

// 4. 建树
void build(int l, int r, int i) {
    add_tag[i] = 0;
    if (l == r) {
        sum_tree[i] = arr[l];
    } else {
        int mid = (l + r) >> 1;
        build(l, mid, i << 1);
        build(mid + 1, r, i << 1 | 1);
        up(i);
    }
}

// 5. 范围修改
// jobl ~ jobr 范围上每个数字增加 jobv
void add(int jobl, int jobr, long long jobv, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        lazy(i, jobv, r - l + 1);
    } else {
        int mid = (l + r) >> 1;
        down(i, mid - l + 1, r - mid);
        if (jobl <= mid) {
            add(jobl, jobr, jobv, l, mid, i << 1);
        }
        if (jobr > mid) {
            add(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        }
        up(i);
    }
}

// 6. 查询累加和
long long query(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return sum_tree[i];
    }
    int mid = (l + r) >> 1;
    down(i, mid - l + 1, r - mid);
    long long ans = 0;
    if (jobl <= mid) {
        ans += query(jobl, jobr, l, mid, i << 1);
    }
    if (jobr > mid) {
        ans += query(jobl, jobr, mid + 1, r, i << 1 | 1);
    }
    return ans;
}

int main() {
    int n, m;
    // 使用 scanf 提升大量数据的输入效率
    while (scanf("%d %d", &n, &m) != EOF) {
        for (int i = 1; i <= n; i++) {
            scanf("%lld", &arr[i]);
        }

        build(1, n, 1);

        int op, jobl, jobr;
        long long jobv;
        for (int i = 1; i <= m; i++) {
            scanf("%d", &op);
            if (op == 1) {
                scanf("%d %d %lld", &jobl, &jobr, &jobv);
                add(jobl, jobr, jobv, 1, n, 1);
            } else {
                scanf("%d %d", &jobl, &jobr);
                printf("%lld\n", query(jobl, jobr, 1, n, 1));
            }
        }
    }
    return 0;
}
