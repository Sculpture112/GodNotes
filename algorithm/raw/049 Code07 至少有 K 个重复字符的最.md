// 049 Code07 至少有 K 个重复字符的最长子串
// 测试链接 : https://leetcode.cn/problems/longest-substring-with-at-least-k-repeating-characters/
// 核心思路：通过枚举字符种类数（1~26），强行构造滑动窗口的单调性

# include <iostream>

# include <vector>

# include <string>

# include <algorithm>

# include <cstring>

using namespace std;

int longestSubstring(string s, int k) {
    int n = s.length();
    int ans = 0;
    int cnts[256];

    // require: 当前限制子串中必须包含的字符种类数
    for (int require = 1; require <= 26; require++) {
        memset(cnts, 0, sizeof(cnts));

        // collect: 窗口内当前包含的字符种类总数
        // satisfy: 窗口内出现次数 >= k 的字符种类数
        for (int l = 0, r = 0, collect = 0, satisfy = 0; r < n; r++) {
            // [右边界进窗]
            int r_idx = (unsigned char)s[r];
            if (++cnts[r_idx] == 1) {
                collect++;
            }
            if (cnts[r_idx] == k) {
                satisfy++;
            }

            // [左边界缩窗]：当种类超过了当前限制 require
            while (collect > require) {
                int l_idx = (unsigned char)s[l];
                if (cnts[l_idx] == 1) {
                    collect--;
                }
                if (cnts[l_idx] == k) {
                    satisfy--;
                }
                cnts[l_idx]--;
                l++;
            }

            // [结算]：只有当种类数达标，且每种都满足 >= k 次时，更新最大长度
            if (satisfy == require) {
                ans = max(ans, r - l + 1);
            }
        }
    }
    return ans;
}

int main() {
    string s = "ababbc";
    int k = 2;
    // 结果应为 5 ("ababb")
    printf("最长子串长度: %d\n", longestSubstring(s, k));
    return 0;
}
