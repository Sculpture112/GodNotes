```cpp
// 049 Code04 加油站
// 测试链接 : https://leetcode.cn/problems/gas-station/
// 核心思路：滑动窗口 + 断点跳跃贪心
// 复杂度：时间 O(N)，空间 O(1)

#include <iostream>
#include <vector>

using namespace std;

int canCompleteCircuit(vector<int>& gas, vector<int>& cost) {
    int n = gas.size();

    // l 是出发点，r 是尝试到达的下一个点
    // [l, r] 维护的是当前能够走通的区间
    for (int l = 0, r = 0, sum = 0; l < n; ) {
        sum = 0;
        r = l; // 每次重新从 l 开始计算

        // 尝试从 l 走，看最远能走到哪
        // 注意：这里用 sum + gas - cost >= 0 来判断能否跨过当前 r 站
        while (sum + gas[r % n] - cost[r % n] >= 0) {
            sum += gas[r % n] - cost[r % n];
            r++;

            // 如果窗口长度达到 n，说明走完一圈了
            if (r - l == n) {
                return l;
            }
        }

        // 关键：如果从 l 走不到 r + 1，那么 l 到 r 之间的任何点都走不到 r + 1
        // 下一次尝试直接从 r + 1 开始
        l = r + 1;
    }

    return -1;
}

int main() {
    vector<int> gas = {1, 2, 3, 4, 5};
    vector<int> cost = {3, 4, 5, 1, 2};
    // 预期结果：3 (加油站编号)
    printf("起始加油站编号: %d\n", canCompleteCircuit(gas, cost));
    return 0;
}
```
