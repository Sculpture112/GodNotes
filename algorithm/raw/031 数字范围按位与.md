// 031 数字范围按位与
class Solution {
public:
    int rangeBitwiseAnd(int left, int right) {
        // 核心：只要 left < right，说明区间内必然有数使得 right 的低位 1 变 0
        // 利用 Brian Kernighan 算法不断抹掉 right 最右侧的 1(效率极高）
        while (left < right) {
            // right & -right 提取最右侧的 1，减去它即为抹掉该位
            right -= right & -right; 
        }
        return right;
    }
};
