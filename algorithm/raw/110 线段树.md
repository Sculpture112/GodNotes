// 110 线段树
// 1. 线段树维护区间增加、区间查询累加和 (add, query)
// 2. 懒更新信息下发与汇总 (down, up)
// 3. 懒更新任务具体执行 (lazy)
// 4. 递归建树 (build)

# include <iostream>

# include <vector>

# include <cstdio>

using namespace std;

namespace class110 {

class Code01_SegmentTreeAddQuerySum {
public:
    static const int MAXN = 100005;
    long long arr[MAXN];
    long long sum[MAXN << 2];
    long long add_tag[MAXN << 2];

    // 1. 建树
    // l~r 是当前节点管辖范围，i 是数组下标
    void build(int l, int r, int i) {
        add_tag[i] = 0;
        if (l == r) {
            sum[i] = arr[l];
        } else {
            int mid = (l + r) >> 1;
            build(l, mid, i << 1);
            build(mid + 1, r, i << 1 | 1);
            up(i);
        }
    }

    // 2. 范围修改
    // jobl ~ jobr 范围上每个数字增加 jobv
    void add(int jobl, int jobr, long long jobv, int l, int r, int i) {
        if (jobl <= l && r <= jobr) {
            lazy(i, jobv, r - l + 1);
        } else {
            int mid = (l + r) >> 1;
            down(i, mid - l + 1, r - mid);
            if (jobl <= mid) {
                add(jobl, jobr, jobv, l, mid, i << 1);
            }
            if (jobr > mid) {
                add(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
            }
            up(i);
        }
    }

    // 3. 查询累加和
    // jobl ~ jobr 范围上的数字总和
    long long query(int jobl, int jobr, int l, int r, int i) {
        if (jobl <= l && r <= jobr) {
            return sum[i];
        }
        int mid = (l + r) >> 1;
        down(i, mid - l + 1, r - mid);
        long long ans = 0;
        if (jobl <= mid) {
            ans += query(jobl, jobr, l, mid, i << 1);
        }
        if (jobr > mid) {
            ans += query(jobl, jobr, mid + 1, r, i << 1 | 1);
        }
        return ans;
    }

private:
    // 4. 累加和信息的汇总
    void up(int i) {
        // 父范围的累加和 = 左范围累加和 + 右范围累加和
        sum[i] = sum[i << 1] + sum[i << 1 | 1];
    }

    // 5. 懒信息的下发
    void down(int i, int ln, int rn) {
        if (add_tag[i] != 0) {
            // 发左
            lazy(i << 1, add_tag[i], ln);
            // 发右
            lazy(i << 1 | 1, add_tag[i], rn);
            // 父范围懒信息清空
            add_tag[i] = 0;
        }
    }

    // 6. 懒更新任务的具体执行
    // 把线段树当前范围的 sum 数组和 add 数组调整好，不再继续往下发
    void lazy(int i, long long v, int n) {
        sum[i] += v * n;
        add_tag[i] += v;
    }
};

} // namespace class110

// 实例化线段树（静态全局分配空间，防止栈溢出）
class110::Code01_SegmentTreeAddQuerySum st;

int main() {
    int n, m;
    // 使用高效的 scanf 处理输入，对标 Java 的 StreamTokenizer
    while (scanf("%d %d", &n, &m) != EOF) {
        for (int i = 1; i <= n; i++) {
            scanf("%lld", &st.arr[i]);
        }

        st.build(1, n, 1);

        int op, l, r;
        long long v;
        for (int i = 0; i < m; i++) {
            scanf("%d", &op);
            if (op == 1) {
                scanf("%d %d %lld", &l, &r, &v);
                st.add(l, r, v, 1, n, 1);
            } else {
                scanf("%d %d", &l, &r);
                printf("%lld\n", st.query(l, r, 1, n, 1));
            }
        }
    }
    return 0;
}
