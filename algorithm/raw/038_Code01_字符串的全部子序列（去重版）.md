// 038 Code01 字符串的全部子序列（去重版）
// 1. 递归决策：每个位置的字符选或者不选
// 2. 使用 set 进行结果去重
// 3. 两种实现：string 动态拼接与 char 数组下标控制

#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <algorithm>

using namespace std;

// 解法一：使用 string 模拟路径（对应 Java 的 StringBuilder）
void f1(const string& s, int i, string& path, unordered_set<string>& res) {
    if (i == s.length()) {
        res.insert(path);
    } else {
        // 决策1：要当前字符
        path.push_back(s[i]);
        f1(s, i + 1, path, res);
        // 回溯：弹出当前字符
        path.pop_back();

        // 决策2：不要当前字符
        f1(s, i + 1, path, res);
    }
}

vector<string> generatePermutation1(string str) {
    unordered_set<string> set;
    string path = "";
    f1(str, 0, path, set);

    vector<string> ans;
    for (const string& s : set) {
        ans.push_back(s);
    }
    return ans;
}

// 解法二：使用字符数组覆盖更新，靠 size 维护逻辑长度（效率更高）
void f2(const string& s, int i, vector<char>& path, int size, unordered_set<string>& res) {
    if (i == s.length()) {
        res.insert(string(path.begin(), path.begin() + size));
    } else {
        // 决策1：要当前字符，填入 path[size]
        path[size] = s[i];
        f2(s, i + 1, path, size + 1, res);

        // 决策2：不要当前字符，直接传 size
        // 这里不需要清理 path[size]，因为逻辑长度 size 没变
        f2(s, i + 1, path, size, res);
    }
}

vector<string> generatePermutation2(string str) {
    unordered_set<string> set;
    vector<char> path(str.length());
    f2(str, 0, path, 0, set);

    vector<string> ans;
    for (const string& s : set) {
        ans.push_back(s);
    }
    return ans;
}

// 简单测试
int main() {
    string test = "acc";

    printf("测试开始 (字符串: %s)\n", test.c_str());

    vector<string> result = generatePermutation2(test);

    printf("去重后的子序列有:\n");
    for (const string& s : result) {
        printf("\"%s\"\n", s.c_str());
    }

    return 0;
}
