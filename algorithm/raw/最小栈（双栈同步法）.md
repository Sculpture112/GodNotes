// 最小栈（双栈同步法）
class MinStack {
    stack<int> data; // 存数据的栈
    stack<int> minS; // 存最小值的栈（与 data 高度同步）

public:
    void push(int val) {
        data.push(val);
        // 核心逻辑：若新值更小（或相等），压入新值；否则重复压入当前的最小值
        // 这样 minS 的栈顶永远对应 data 当前状态下的最小值
        if (minS.empty() || val <= minS.top()) {
            minS.push(val);
        } else {
            minS.push(minS.top());
        }
    }

    void pop() {
        // 由于高度同步，直接同时弹出即可，无需判断
        data.pop();
        minS.pop();
    }

    int top() { return data.top(); }

    int getMin() { return minS.top(); }
};
