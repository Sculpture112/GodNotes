// 049 Code03 最小覆盖子串
// 测试链接 : https://leetcode.cn/problems/minimum-window-substring/
// 核心思路：欠债模型 + 滑动窗口
// 时间复杂度 O(N)，空间复杂度 O(1)

# include <iostream>

# include <vector>

# include <string>

# include <climits>

using namespace std;

string minWindow(string s, string t) {
    if (s.length() < t.length()) return "";

    // 记录每种字符的欠债情况
    // cnts[i] < 0 : 欠 i 字符多少个
    // cnts[i] > 0 : 多余了 i 字符多少个
    int cnts[256] = {0};
    for (char c : t) {
        cnts[(unsigned char)c]--;
    }

    int n = s.length();
    int minLen = INT_MAX;
    int start = 0;
    // debt 表示总共还差多少个字符才能达标
    int debt = t.length();

    for (int l = 0, r = 0; r < n; r++) {
        // [还债/进窗]
        // 如果当前字符在欠债名单里，debt 减少
        if (cnts[(unsigned char)s[r]]++ < 0) {
            debt--;
        }

        // 债还清了，开始尝试缩窗
        if (debt == 0) {
            // [收缩/去重]
            // 如果左边界的字符是盈余的 (cnts > 0)，就扔掉它
            while (cnts[(unsigned char)s[l]] > 0) {
                cnts[(unsigned char)s[l++]]--;
            }

            // 更新最小长度
            if (r - l + 1 < minLen) {
                minLen = r - l + 1;
                start = l;
            }
        }
    }

    return minLen == INT_MAX ? "" : s.substr(start, minLen);
}

// 简单验证
int main() {
    string s = "ADOBECODEBANC";
    string t = "ABC";
    // 结果应该是 "BANC"
    cout << "最小覆盖子串: " << minWindow(s, t) << endl;
    return 0;
}
