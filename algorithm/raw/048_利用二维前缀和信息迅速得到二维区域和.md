```cpp
// 048 利用二维前缀和信息迅速得到二维区域和
// 1. NumMatrix 类定义 (NumMatrix Class)
// 2. 构造函数：计算二维前缀和 (Constructor: Calculate 2D Prefix Sum)
// 3. 区域和查询 (sumRegion)

#include <vector>

using namespace std;

class NumMatrix {
public:
    // 存储二维前缀和的矩阵
    vector<vector<int>> sum;

    // 2. 构造函数
    NumMatrix(vector<vector<int>>& matrix) {
        if (matrix.empty() || matrix[0].empty()) return;
        int n = matrix.size();
        int m = matrix[0].size();

        // 初始化 n+1 行 m+1 列的矩阵，索引从 1 开始更方便计算边界
        // C++ vector 自动初始化为 0
        sum.resize(n + 1, vector<int>(m + 1, 0));

        // 将原矩阵内容填入 sum 矩阵的 [1,1] 到 [n,m] 区域
        for (int a = 1, c = 0; c < n; a++, c++) {
            for (int b = 1, d = 0; d < m; b++, d++) {
                sum[a][b] = matrix[c][d];
            }
        }

        // 核心逻辑：计算二维前缀和
        // sum[i][j] 代表从 [0,0] 到 [i-1][j-1] 区域内所有元素的总和
        for (int i = 1; i <= n; i++) {
            for (int j = 1; j <= m; j++) {
                sum[i][j] += sum[i][j - 1] + sum[i - 1][j] - sum[i - 1][j - 1];
            }
        }
    }

    // 3. 区域和查询
    // (a, b) 为左上角坐标，(c, d) 为右下角坐标
    int sumRegion(int a, int b, int c, int d) {
        // 因为 sum 矩阵是 1-indexed，所以右下角坐标需要加 1
        c++;
        d++;
        // 二维前缀和公式：右下大矩形 - 左侧长条 - 上方长条 + 左上重叠块
        return sum[c][d] - sum[c][b] - sum[a][d] + sum[a][b];
    }
};
```
