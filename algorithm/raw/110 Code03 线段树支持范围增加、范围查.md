// 110 Code03 线段树支持范围增加、范围查询最大值
// 1. 最大值信息的汇总 (up)
// 2. 懒信息的下发 (down)
// 3. 懒更新任务的具体执行 (lazy)
// 4. 建树 (build)
// 5. 范围增加 (add)
// 6. 查询最大值 (query)

# include <iostream>

# include <vector>

# include <algorithm>

# include <cstdio>

# include <climits>

# include <ctime>

using namespace std;

// 基础常量定义
const int MAXN = 100005;

// 全局静态数组，long long 处理，防止溢出
long long arr[MAXN];
long long max_tree[MAXN << 2];
long long add_tag[MAXN << 2];

// 1. 最大值信息的汇总
void up(int i) {
    // 父范围的最大值 = max(左子范围最大值, 右子范围最大值)
    max_tree[i] = max(max_tree[i << 1], max_tree[i << 1 | 1]);
}

// 3. 懒更新任务的具体执行
// 范围增加操作：当前范围所有数字增加 v
void lazy(int i, long long v) {
    max_tree[i] += v;
    add_tag[i] += v;
}

// 2. 懒信息的下发
void down(int i) {
    if (add_tag[i] != 0) {
        // 向左孩子发懒信息
        lazy(i << 1, add_tag[i]);
        // 向右孩子发懒信息
        lazy(i << 1 | 1, add_tag[i]);
        // 父范围懒信息清空
        add_tag[i] = 0;
    }
}

// 4. 建树
void build(int l, int r, int i) {
    if (l == r) {
        max_tree[i] = arr[l];
    } else {
        int mid = (l + r) >> 1;
        build(l, mid, i << 1);
        build(mid + 1, r, i << 1 | 1);
        up(i);
    }
    add_tag[i] = 0;
}

// 5. 范围增加
// jobl ~ jobr 范围上每个数字增加 jobv
void add(int jobl, int jobr, long long jobv, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        lazy(i, jobv);
    } else {
        down(i);
        int mid = (l + r) >> 1;
        if (jobl <= mid) {
            add(jobl, jobr, jobv, l, mid, i << 1);
        }
        if (jobr > mid) {
            add(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        }
        up(i);
    }
}

// 6. 查询最大值
// 查询 jobl ~ jobr 范围内的最大值
long long query(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return max_tree[i];
    }
    down(i);
    int mid = (l + r) >> 1;
    long long ans = LLONG_MIN;
    if (jobl <= mid) {
        ans = max(ans, query(jobl, jobr, l, mid, i << 1));
    }
    if (jobr > mid) {
        ans = max(ans, query(jobl, jobr, mid + 1, r, i << 1 | 1));
    }
    return ans;
}

// 验证结构的增加（暴力）
void checkAdd(vector<long long>& check, int jobl, int jobr, long long jobv) {
    for (int i = jobl; i <= jobr; i++) {
        check[i] += jobv;
    }
}

// 验证结构的查询（暴力）
long long checkQuery(vector<long long>& check, int jobl, int jobr) {
    long long ans = LLONG_MIN;
    for (int i = jobl; i <= jobr; i++) {
        ans = max(ans, check[i]);
    }
    return ans;
}

int main() {
    srand(time(0));
    printf("测试开始\n");
    int n = 1000;
    int v = 2000;
    int t = 1000000;

    // 初始化随机数组
    for (int i = 1; i <= n; i++) {
        arr[i] = rand() % v;
    }

    // 建立线段树
    build(1, n, 1);

    // 建立对数器验证结构
    vector<long long> check(n + 1);
    for (int i = 1; i <= n; i++) {
        check[i] = arr[i];
    }

    for (int i = 1; i <= t; i++) {
        int op = rand() % 2;
        int a = rand() % n + 1;
        int b = rand() % n + 1;
        int jobl = min(a, b);
        int jobr = max(a, b);

        if (op == 0) {
            long long jobv = rand() % (v * 2) - v;
            add(jobl, jobr, jobv, 1, n, 1);
            checkAdd(check, jobl, jobr, jobv);
        } else {
            long long ans1 = query(jobl, jobr, 1, n, 1);
            long long ans2 = checkQuery(check, jobl, jobr);
            if (ans1 != ans2) {
                printf("出错了!\n");
                break;
            }
        }
    }
    printf("测试结束\n");

    return 0;
}
