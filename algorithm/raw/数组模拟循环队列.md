// 循环队列（Size 变量解耦空/满判断）
class MyCircularQueue {
    vector<int> q;
    int l, r, size, limit; // l:头, r:尾(下个空位), limit:容量

public:
    MyCircularQueue(int k) : q(k), l(0), r(0), size(0), limit(k) {}

    bool enQueue(int val) {
        if (size == limit) return false;
        q[r] = val;
        // 环形步进：若到达数组末尾，回到下标 0
        r = (r == limit - 1) ? 0 : (r + 1); 
        size++;
        return true;
    }

    bool deQueue() {
        if (size == 0) return false;
        // 环形步进：若到达数组末尾，回到下标 0
        l = (l == limit - 1) ? 0 : (l + 1); 
        size--;
        return true;
    }

    int Front() { return size == 0 ? -1 : q[l]; }

    int Rear() {
        if (size == 0) return -1;
        // 计算实际尾元素：r 指向的是"下一个"空位，所以要回退一格
        // 特殊情况：如果 r 是 0，则尾元素在 limit-1
        int lastIndex = (r == 0) ? limit - 1 : r - 1;
        return q[lastIndex];
    }

    bool isEmpty() { return size == 0; }
    bool isFull() { return size == limit; }
};
