```cpp
灯泡的异或
```

```cpp
#include <iostream>
#include <cmath>

using namespace std;

int main() {
    int n;
    // 读取操作次数
    if (!(cin >> n)) return 0;

    int ans = 0; // 用于存储异或结果的变量

    for (int i = 0; i < n; ++i) {
        double a;
        int t;
        // 读取每一行的 a 和 t
        cin >> a >> t;

        for (int j = 1; j <= t; ++j) {
            // 计算灯的编号并取整
            // floor(j * a) 返回不大于 j*a 的最大整数
            int lamp_id = floor(j * a);

            // 核心逻辑：异或运算
            // 出现偶数次的编号会抵消，出现奇数次的会留下
            ans ^= lamp_id;
        }
    }

    // 输出最后剩下的那个编号
    cout << ans << endl;

    return 0;
}
```
