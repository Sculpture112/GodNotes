```cpp
// 数组模拟栈（单指针法）
class Stack2 {
    vector<int> s;
    int top; // 既代表当前栈的大小，也指向栈顶的下一个位置

public:
    Stack2(int n) : s(n), top(0) {}

    bool isEmpty() { return top == 0; }
    void push(int num) { s[top++] = num; } // 压栈：先存再移
    int pop() { return s[--top]; }         // 弹栈：先移再取
    int peek() { return s[top - 1]; }      // 查看栈顶：需回退一格
};
```
