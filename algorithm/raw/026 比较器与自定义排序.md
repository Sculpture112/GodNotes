// 026 比较器与自定义排序

# include <iostream>

# include <vector>

# include <string>

# include <algorithm>

# include <set>

using namespace std;

struct Employee {
    int company, age;
    Employee(int c, int a) : company(c), age(a) {}
    // 为了 toString 演示，转为地址字符串
    string toString() const { return to_string((long long)this); }
};

int main() {
    vector<Employee> arr = { {2,27}, {1,60}, {4,19}, {3,23}, {1,35}, {3,55} };

    // 1. 年龄升序排列
    sort(arr.begin(), arr.end(), [](const Employee& a, const Employee& b) {
        return a.age < b.age; // 返回 true 表示 a 优先级高
    });

    // 2. 年龄降序排列
    sort(arr.begin(), arr.end(), [](const Employee& a, const Employee& b) {
        return a.age > b.age;
    });

    // 3. 多级排序：公司升序；公司一样，年龄升序
    sort(arr.begin(), arr.end(), [](const Employee& a, const Employee& b) {
        if (a.company != b.company) return a.company < b.company;
        return a.age < b.age;
    });

    // 4. 有序表与去重逻辑
    // C++ set 认为 !(a < b) && !(b < a) 就是同一个元素
    auto comp = [](const Employee& a, const Employee& b) {
        if (a.company != b.company) return a.company < b.company;
        if (a.age != b.age) return a.age < b.age;
        return a.toString() < b.toString(); // 增加内存地址比较防止去重
    };
    set<Employee, decltype(comp)> treeSet(comp);
    for(auto& e : arr) treeSet.insert(e);
    treeSet.insert({2, 27}); // 插入一个新对象，地址不同，不会去重
    cout << "TreeSet Size: " << treeSet.size() << endl;

    // 5. 字符串字典序比较
    string s1 = "abcde", s2 = "ks";
    cout << (s1 < s2 ? "s1 < s2" : "s1 >= s2") << endl; // s1 字典序小

    return 0;
}
