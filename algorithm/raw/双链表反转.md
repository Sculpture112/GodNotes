// 双链表反转
DoubleListNode* reverseDoubleList(DoubleListNode* head) {
    DoubleListNode* pre = nullptr;
    DoubleListNode* next = nullptr;
    while (head != nullptr) {
        next = head->next;  // 1. 记录
        head->next = pre;   // 2. 反转 Next
        head->last = next;  // 3. 反转 Last
        pre = head;         // 4. 推进
        head = next;        // 5. 移动
    }
    return pre;
}
