# include <iostream>

# include <string>

using namespace std;

int main() {
    string n;
    // 将数字作为字符串读取，这样处理位数更方便
    if (!(cin >> n)) return 0;

    int len = n.length();

    for (int i = 0; i < len; ++i) {
        // 输出当前数字
        cout << n[i];

        // 判断是否需要插入逗号：
        // 1. (len - 1 - i) 是当前字符后面剩余的字符数
        // 2. 如果剩余字符数是 3 的倍数，且剩余数不为 0（即不是最后一位），则加逗号
        if ((len - 1 - i) % 3 == 0 && i != len - 1) {
            cout << ",";
        }
    }

    return 0;
}
