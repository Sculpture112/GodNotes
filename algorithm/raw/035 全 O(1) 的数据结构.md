# include <unordered_map>

# include <unordered_set>

# include <string>

using namespace std;

class AllOne {
private:
    // 桶结构：具有相同频率的 Key 存放在同一个桶里
    struct Bucket {
        int cnt;
        unordered_set<string> keys;
        Bucket *last, *next;

        Bucket(string s, int c) : cnt(c), last(nullptr), next(nullptr) {
            if (s != "") keys.insert(s);
        }
    };

    Bucket *head, *tail;               // 哨兵节点：头（最小频率）、尾（最大频率）
    unordered_map<string, Bucket*> map; // 快速查找 Key 对应的桶

    // 在 cur 桶后面插入一个新桶 pos
    void insert(Bucket* cur, Bucket* pos) {
        pos->next = cur->next;
        pos->last = cur;
        cur->next->last = pos;
        cur->next = pos;
    }

    // 从双向链表中移除空桶
    void remove(Bucket* cur) {
        cur->last->next = cur->next;
        cur->next->last = cur->last;
        delete cur;
    }

public:
    AllOne() {
        // 初始化哨兵，head 为最小值，tail 为最大值
        head = new Bucket("", 0);
        tail = new Bucket("", 2147483647);
        head->next = tail;
        tail->last = head;
    }

    // 增加 Key 的频率
    void inc(string key) {
        if (map.find(key) == map.end()) { 
            // 情况 1: 新增 Key，目标频率为 1
            if (head->next->cnt != 1) {
                // 如果当前不存在频率为 1 的桶，则新建
                insert(head, new Bucket(key, 1));
            } else {
                // 否则直接加入频率为 1 的桶
                head->next->keys.insert(key);
            }
            map[key] = head->next;
        } else { 
            // 情况 2: 已有 Key，频率 +1
            Bucket *cur = map[key];
            Bucket *nxt = cur->next;

            if (nxt->cnt != cur->cnt + 1) {
                // 如果不存在频率为 cur->cnt + 1 的桶，则新建
                nxt = new Bucket(key, cur->cnt + 1);
                insert(cur, nxt);
            } else {
                // 否则加入已有的频率桶
                nxt->keys.insert(key);
            }
            map[key] = nxt;

            // 从原桶移除，若原桶变空则删除原桶
            cur->keys.erase(key);
            if (cur->keys.empty()) remove(cur);
        }
    }

    // 减少 Key 的频率
    void dec(string key) {
        Bucket *cur = map[key];

        if (cur->cnt == 1) {
            // 频率降为 0，直接从哈希表中移除
            map.erase(key);
        } else {
            // 频率减少，移动到前一个频率桶
            Bucket *pre = cur->last;
            if (pre->cnt != cur->cnt - 1) {
                // 如果前一个桶频率不匹配，新建频率桶
                pre = new Bucket(key, cur->cnt - 1);
                insert(cur->last, pre);
            } else {
                pre->keys.insert(key);
            }
            map[key] = pre;
        }

        // 清理旧桶逻辑
        cur->keys.erase(key);
        if (cur->keys.empty()) remove(cur);
    }

    // 获取频率最高的 Key
    string getMaxKey() {
        // tail->last 如果是 head，说明链表为空
        return tail->last == head ? "" : *tail->last->keys.begin();
    }

    // 获取频率最低的 Key
    string getMinKey() {
        // head->next 如果是 tail，说明链表为空
        return head->next == tail ? "" : *head->next->keys.begin();
    }
};
