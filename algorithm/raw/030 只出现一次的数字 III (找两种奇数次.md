// 030 只出现一次的数字 III (找两种奇数次的数)
#include <vector>
using namespace std;

class Solution {
public:
    vector<int> singleNumber(vector<int>& nums) {
        long long eor1 = 0; // 使用 long long 防止 -eor1 溢出
        for (int n : nums) eor1 ^= n; // 最终 eor1 = a ^ b
        // 提取二进制最右侧的 1：a 和 b 在这一位上必然一个为 0，一个为 1
        long long rightOne = eor1 & (-eor1); 
        int eor2 = 0;
        for (int n : nums) {
            // 利用 rightOne 将数组分为两组，eor2 只负责其中一组
            if ((n & rightOne) == 0) eor2 ^= n; 
        }
        return {eor2, (int)(eor1 ^ eor2)}; // 两个结果分别为 eor2 和 a ^ b ^ eor2
    }
};
