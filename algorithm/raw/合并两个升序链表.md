// 合并两个升序链表
static ListNode* mergeTwoLists(ListNode* head1, ListNode* head2) {
    // 1. 创建哨兵节点 (Dummy Node)
    // 它作为一个占位符，dummy.next 才是真正合并后的链表头
    ListNode dummy(0); 
    ListNode* pre = &dummy; // pre 始终指向结果链表的最后一个节点

    // 2. 双指针 PK 过程
    while (head1 != nullptr && head2 != nullptr) {
        if (head1->val <= head2->val) {
            pre->next = head1;
            head1 = head1->next; // 移动 head1
        } else {
            pre->next = head2;
            head2 = head2->next; // 移动 head2
        }
        pre = pre->next; // pre 推进
    }

    // 3. 收尾工作：将剩下的“尾巴”直接挂载
    pre->next = (head1 != nullptr) ? head1 : head2;

    // 4. 返回真正的头节点
    return dummy.next;
}
