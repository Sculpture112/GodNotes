// 035 实现 LRU 结构

# include <unordered_map>

using namespace std;

// 双向链表节点定义
struct Node {
    int k, v;
    Node *last;
    Node *next;

    Node(int key, int val) 
        : k(key), v(val), last(nullptr), next(nullptr) {}
};

class LRUCache {
private:
    unordered_map<int, Node*> map;
    Node *head; // 头部节点，表示最久未访问 (Least Recently Used)
    Node *tail; // 尾部节点，表示最新访问 (Most Recently Used)
    int cap;

    // 将节点移动到尾部：表示该节点刚被访问，优先级提升到最高
    void moveToTail(Node* node) {
        if (tail == node) return;

        if (head == node) {
            head = node->next;
            head->last = nullptr;
        } else {
            // 从中间位置摘除节点
            node->last->next = node->next;
            node->next->last = node->last;
        }

        // 重新连接到尾部
        node->last = tail;
        node->next = nullptr;
        tail->next = node;
        tail = node;
    }

    // 移除头部节点：淘汰那个最久没被访问的节点
    Node* removeHead() {
        if (!head) return nullptr;

        Node* res = head;
        if (head == tail) {
            head = tail = nullptr;
        } else {
            head = head->next;
            head->last = nullptr;
            res->next = nullptr;
        }
        return res;
    }

public:
    LRUCache(int capacity) 
        : head(nullptr), tail(nullptr), cap(capacity) {}

    // 获取数据：若存在则移动到尾部并返回值，否则返回 -1
    int get(int key) {
        if (map.find(key) == map.end()) {
            return -1;
        }
        moveToTail(map[key]);
        return map[key]->v;
    }

    // 存入数据：若存在则更新并移到尾部；若不存在，满额则踢掉头节点再插入新节点
    void put(int key, int value) {
        if (map.count(key)) {
            // 节点已存在，更新值并移动到末尾
            map[key]->v = value;
            moveToTail(map[key]);
        } else {
            // 节点不存在，检查容量
            if (map.size() == cap) {
                Node* old = removeHead();
                map.erase(old->k);
                delete old;
            }

            // 插入新节点
            Node* node = new Node(key, value);
            map[key] = node;

            if (!head) {
                head = tail = node;
            } else {
                tail->next = node;
                node->last = tail;
                tail = node;
            }
        }
    }
};
