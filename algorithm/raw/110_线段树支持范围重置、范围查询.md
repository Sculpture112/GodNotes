```cpp
// 110 线段树支持范围重置、范围查询
// 1. 累加和信息的汇总 (up)
// 2. 懒信息的下发 (down)
// 3. 懒更新任务的具体执行 (lazy)
// 4. 建树 (build)
// 5. 范围重置 (update)
// 6. 查询累加和 (query)

#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>

using namespace std;

// 基础常量定义
const int MAXN = 100005;

// 全局静态数组，防止栈溢出
long long arr[MAXN];
long long sum_tree[MAXN << 2];
long long change[MAXN << 2];
bool update_tag[MAXN << 2];

// 1. 累加和信息的汇总
void up(int i) {
    sum_tree[i] = sum_tree[i << 1] + sum_tree[i << 1 | 1];
}

// 3. 懒更新任务的具体执行
// 范围重置操作：当前范围所有数字变为 v
void lazy(int i, long long v, int n) {
    sum_tree[i] = v * n;
    change[i] = v;
    update_tag[i] = true;
}

// 2. 懒信息的下发
void down(int i, int ln, int rn) {
    if (update_tag[i]) {
        lazy(i << 1, change[i], ln);
        lazy(i << 1 | 1, change[i], rn);
        update_tag[i] = false;
    }
}

// 4. 建树
void build(int l, int r, int i) {
    if (l == r) {
        sum_tree[i] = arr[l];
    } else {
        int mid = (l + r) >> 1;
        build(l, mid, i << 1);
        build(mid + 1, r, i << 1 | 1);
        up(i);
    }
    change[i] = 0;
    update_tag[i] = false;
}

// 5. 范围重置
// jobl ~ jobr 范围上每个数字重置为 jobv
void update(int jobl, int jobr, long long jobv, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        lazy(i, jobv, r - l + 1);
    } else {
        int mid = (l + r) >> 1;
        down(i, mid - l + 1, r - mid);
        if (jobl <= mid) {
            update(jobl, jobr, jobv, l, mid, i << 1);
        }
        if (jobr > mid) {
            update(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        }
        up(i);
    }
}

// 6. 查询累加和
long long query(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return sum_tree[i];
    }
    int mid = (l + r) >> 1;
    down(i, mid - l + 1, r - mid);
    long long ans = 0;
    if (jobl <= mid) {
        ans += query(jobl, jobr, l, mid, i << 1);
    }
    if (jobr > mid) {
        ans += query(jobl, jobr, mid + 1, r, i << 1 | 1);
    }
    return ans;
}

// 对数器使用的暴力验证逻辑
void checkUpdate(vector<long long>& check, int jobl, int jobr, long long jobv) {
    for (int i = jobl; i <= jobr; i++) {
        check[i] = jobv;
    }
}

long long checkQuery(vector<long long>& check, int jobl, int jobr) {
    long long ans = 0;
    for (int i = jobl; i <= jobr; i++) {
        ans += check[i];
    }
    return ans;
}

int main() {
    // 模拟对数器验证
    printf("测试开始\n");
    int n = 1000;
    int v = 2000;
    int t = 1000000;

    // 初始化随机数组
    for (int i = 1; i <= n; i++) {
        arr[i] = rand() % v;
    }

    build(1, n, 1);

    vector<long long> check(n + 1);
    for (int i = 1; i <= n; i++) {
        check[i] = arr[i];
    }

    for (int i = 1; i <= t; i++) {
        int op = rand() % 2;
        int a = rand() % n + 1;
        int b = rand() % n + 1;
        int jobl = min(a, b);
        int jobr = max(a, b);

        if (op == 0) {
            long long jobv = rand() % (v * 2) - v;
            update(jobl, jobr, jobv, 1, n, 1);
            checkUpdate(check, jobl, jobr, jobv);
        } else {
            long long ans1 = query(jobl, jobr, 1, n, 1);
            long long ans2 = checkQuery(check, jobl, jobr);
            if (ans1 != ans2) {
                printf("出错了!\n");
                break;
            }
        }
    }
    printf("测试结束\n");

    return 0;
}
```
