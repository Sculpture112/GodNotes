```cpp

# include <bits/stdc++.h>

using namespace std;

int main() {
    int n;
    cin>>n;

    vector<int> a(n+1);
    for(int i =1;i<=n;++i){
        cin>>a[i];
    }
    vector<int> affter_max(n+2,0);
    for(int i =n;i>=1;--i){
        affter_max[i] = max(affter_max[i+1],a[i]);

    }
    stack<int> s;
    for(int i =1 ;i<=n;++i){
        s.push(a[i]);
        while(!s.empty() && s.top() > affter_max[i+1]){
            cout<<s.top()<<" ";
            s.pop();
        }
    }
    cout<<endl;

}
