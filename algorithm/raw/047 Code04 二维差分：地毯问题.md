```cpp
// 047 Code04 二维差分：地毯问题
// 测试链接 : https://www.luogu.com.cn/problem/P3397
// 核心思路：
// 1. 差分操作：add(x1, y1, x2, y2, V)
// 2. 还原操作：二维前缀和 s[i][j] = s[i-1][j] + s[i][j-1] - s[i-1][j-1] + diff[i][j]

#include <iostream>
#include <vector>

using namespace std;

const int MAXN = 1005;
int diff[MAXN][MAXN];
int n, m;

// 二维差分：对子矩阵进行 O(1) 修改
void add(int x1, int y1, int x2, int y2, int v) {
    diff[x1][y1] += v;
    diff[x1][y2 + 1] -= v;
    diff[x2 + 1][y1] -= v;
    diff[x2 + 1][y2 + 1] += v;
}

// 二维前缀和还原
void build() {
    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            diff[i][j] += diff[i - 1][j] + diff[i][j - 1] - diff[i - 1][j - 1];
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> n >> m)) return 0;

    for (int i = 0; i < m; i++) {
        int x1, y1, x2, y2;
        cin >> x1 >> y1 >> x2 >> y2;
        add(x1, y1, x2, y2, 1);
    }

    build();

    for (int i = 1; i <= n; i++) {
        for (int j = 1; j <= n; j++) {
            cout << diff[i][j] << (j == n ? "" : " ");
        }
        cout << "\n";
    }

    return 0;
}
```
