// 随机快速排序（荷兰国旗划分版）
int arr[100001];
int first, last; // 全局变量，用于记录 ==x 区域的左右边界

// 荷兰国旗划分：将 arr[l...r] 分为 <x, ==x, >x 三部分
void partition(int l, int r, int x) {
    first = l;   // <x 区域的右边界的下一位
    last = r;    // >x 区域的左边界的上一位
    int i = l;

    while (i <= last) {
        if (arr[i] == x) {
            i++; // 等于区直接跳过
        } else if (arr[i] < x) {
            swap(arr[first++], arr[i++]); // 发配到左边，i 前进
        } else {
            swap(arr[i], arr[last--]); // 发配到右边，i 原地不动（待检查换回来的数）
        }
    }
}

// 递归主函数
void quickSort(int l, int r) {
    if (l >= r) return;

    // 1. 灵魂操作：随机选基准值，避免退化
    int x = arr[l + rand() % (r - l + 1)];

    // 2. 划分区域
    partition(l, r, x);

    // 3. 记录边界（必须存入局部变量，防递归污染）
    int left = first;
    int right = last;

    // 4. 递归左右两侧（中间 ==x 部分无需再排）
    quickSort(l, left - 1);
    quickSort(right + 1, r);
}
