// 025 堆结构和堆排序（填函数版）
#include <vector>
#include <algorithm>
using namespace std;

class Solution {
public:
    // i位置的数变小了，向下调整大根堆
    void heapify(vector<int>& arr, int i, int size) {
        int l = i * 2 + 1; // 左孩子下标
        while (l < size) {
            // 评选最强的孩子：右孩子存在且比左孩子大，则选右，否则选左
            int best = l + 1 < size && arr[l + 1] > arr[l] ? l + 1 : l;
            // 父亲与最强的孩子PK，谁大谁是新的 best
            best = arr[best] > arr[i] ? best : i;
            if (best == i) break; // 父亲最强，结构已稳
            swap(arr[i], arr[best]); i = best; l = i * 2 + 1;
        }
    }

    // 堆排序入口函数
    vector<int> sortArray(vector<int>& nums) {
        int n = nums.size();
        if (n <= 1) return nums;
        // 从底到顶建立大根堆，时间复杂度 O(N)
        for (int i = n - 1; i >= 0; i--) heapify(nums, i, n);
        int size = n;
        // 依次将堆顶（最大值）交换到末尾，并缩小堆范围进行调整
        while (size > 1) {
            swap(nums[0], nums[--size]);
            heapify(nums, 0, size);
        }
        return nums;
    }
};
