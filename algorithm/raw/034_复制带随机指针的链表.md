```cpp
// 034 复制带随机指针的链表
#include <iostream>

struct Node { int val; Node *next, *random; Node(int v) : val(v), next(nullptr), random(nullptr) {} };

class Solution {
public:
    Node* copyRandomList(Node* head) {
        if (!head) return nullptr;
        Node *cur = head, *nxt = nullptr, *copy = nullptr;
        // 1. 克隆节点并插空：1 -> 2 变为 1 -> 1' -> 2 -> 2'
        while (cur) {
            nxt = cur->next;
            cur->next = new Node(cur->val);
            cur->next->next = nxt;
            cur = nxt;
        }
        // 2. 设置克隆节点的 random 指针：copy->random = cur->random->next
        cur = head;
        while (cur) {
            nxt = cur->next->next;
            copy = cur->next;
            copy->random = cur->random ? cur->random->next : nullptr;因为next才是copy要指的地方，问号前面那个结点是条件
            cur = nxt;
```

```cpp
        }
        // 3. 拆分链表：恢复原链表并提取克隆链表
        Node* ans = head->next;
        cur = head;
        while (cur) {
            nxt = cur->next->next;
            copy = cur->next;
            cur->next = nxt; // 原链表连接
            copy->next = nxt ? nxt->next : nullptr; // 克隆链表连接
            cur = nxt;
        }
        return ans;
    }
};
```
