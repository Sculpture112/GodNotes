// 036 二叉树的最大宽度
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 最大宽度实现 (使用两个数组模拟队列，通过 ID 编号计算宽度)

# include <algorithm>

# include <vector>

using namespace std;

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Code03_WidthOfBinaryTree1 {
public:
    // 如果测试数据量变大了就修改这个值
    static const int MAXN = 3001;

    // nq 数组模拟存储节点的队列
    TreeNode* nq[MAXN];

    // iq 数组模拟存储节点对应 ID 的队列
    // 使用 long long 对应 Java 的 long，防止编号在深度较大时溢出
    long long iq[MAXN];

    int l, r;

    // 用每次处理一层的优化bfs就非常容易实现
    int widthOfBinaryTree(TreeNode* root) {
        if (root == nullptr) {
            return 0;
        }
        int ans = 1;
        l = r = 0;
        nq[r] = root;
        // 初始节点编号为 1
        iq[r++] = 1;
        while (l < r) {
            int size = r - l;
            // 每一层结算一次最大宽度
            // 当前层的宽度 = 右端点 ID - 左端点 ID + 1
            ans = max(ans, (int)(iq[r - 1] - iq[l] + 1));
            for (int i = 0; i < size; i++) {
                TreeNode* node = nq[l];
                long long id = iq[l++];
                if (node->left != nullptr) {
                    nq[r] = node->left;
                    // 左孩子编号 = 父编号 * 2
                    iq[r++] = id * 2;
                }
                if (node->right != nullptr) {
                    nq[r] = node->right;
                    // 右孩子编号 = 父编号 * 2 + 1
                    iq[r++] = id * 2 + 1;
                }
            }
        }
        return ans;
    }
};
