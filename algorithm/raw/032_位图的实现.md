```cpp
// 032 位图的实现
#include <vector>
using namespace std;

class Bitset {
public:
    vector<int> set;
    // 构造函数：n个数字需要 (n+31)/32 个 int 空间
    Bitset(int n) : set((n + 31) / 32, 0) {}

    // 添加数字：将对应位置的 bit 设为 1
    void add(int num) { set[num >> 5] |= (1 << (num & 31)); } // num/32 可写为 num>>5
num & 31：等同于 num % 32
    // 删除数字：将对应位置的 bit 设为 0
    void remove(int num) { set[num >> 5] &= ~(1 << (num & 31)); } // 使用取反掩码
```

```cpp
    // 翻转数字：1变0，0变1
    void reverse(int num) { set[num >> 5] ^= (1 << (num & 31)); }

    // 查询数字：检查对应 bit 是否为 1
    bool contains(int num) { return (set[num >> 5] >> (num & 31)) & 1; }
```
