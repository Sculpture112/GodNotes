// 030 找到缺失的数字

# include <vector>

using namespace std;

class Solution {
public:
    int missingNumber(vector<int>& nums) {
        int eorAll = 0, eorHas = 0, n = nums.size();
        for (int i = 0; i < n; i++) {
            eorAll ^= i;      // 累积异或 0 到 n-1
            eorHas ^= nums[i]; // 累积异或数组中存在的数字
        }
        return eorAll ^ n ^ eorHas; // 补上最后一个数字 n 后整体异或，剩下的即为缺失值
    }
};
