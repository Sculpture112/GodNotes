// 026 有序表与优先级队列
#include <iostream>
#include <map>
#include <set>
#include <queue>
#include <string>
#include <vector>

using namespace std;

int main() {
    // 1. TreeMap 对应 std::map，基于红黑树，Key 有序
    map<int, string> treeMap;
    treeMap[5] = "这是5"; treeMap[7] = "这是7"; treeMap[1] = "这是1";
    treeMap[2] = "这是2"; treeMap[3] = "这是3"; treeMap[4] = "这是4";
    treeMap[8] = "这是8";

    // 查找与获取
    cout << treeMap.count(1) << " " << treeMap.count(10) << endl; // count=1 代表存在
    treeMap[4] = "张三是4"; // 修改 Value
    treeMap.erase(4); // 删除 Key

    // 获取最值
    if (!treeMap.empty()) {
        cout << treeMap.begin()->first << endl;  // 最小 Key: 1
        cout << treeMap.rbegin()->first << endl; // 最大 Key: 8
    }

    // floorKey (<=4 且最近): 找第一个 >4 的元素，向前退一步
    auto it_f = treeMap.upper_bound(4);
    if (it_f != treeMap.begin()) cout << (--it_f)->first << endl; // 输出 3

    // ceilingKey (>=4 且最近): 直接找第一个 >=4 的元素
    auto it_c = treeMap.lower_bound(4);
    if (it_c != treeMap.end()) cout << it_c->first << endl; // 输出 5 (因为4被删了)

    // 2. TreeSet 对应 std::set，有序且去重
    set<int> treeSet = {3, 3, 4, 4};
    cout << "set size: " << treeSet.size() << endl; // 结果为 2

    // 3. PriorityQueue 对应 std::priority_queue
    // C++ 默认是大根堆，Java 默认是小根堆
    priority_queue<int, vector<int>, greater<int>> minHeap; // 小根堆写法
    minHeap.push(3); minHeap.push(4); minHeap.push(3);
    while (!minHeap.empty()) { cout << minHeap.top() << " "; minHeap.pop(); }

    priority_queue<int> maxHeap; // 大根堆写法
    maxHeap.push(3); maxHeap.push(4); maxHeap.push(3);
    while (!maxHeap.empty()) { cout << maxHeap.top() << " "; maxHeap.pop(); }

    return 0;
}
