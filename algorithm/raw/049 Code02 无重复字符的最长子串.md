// 049 Code02 无重复字符的最长子串
// 测试链接 : https://leetcode.cn/problems/longest-substring-without-repeating-characters/
// 核心思路：滑动窗口 + 下标记录
// 复杂度：时间 O(N)，空间 O(1)（固定 256 大小的数组）

# include <iostream>

# include <string>

# include <vector>

# include <algorithm>

# include <cstring> // for memset

using namespace std;

int lengthOfLongestSubstring(string str) {
    int n = str.length();
    if (n == 0) return 0;

    // 记录字符上一次出现的位置，初始化为 -1
    // C++ 中可以直接开 256 的数组，涵盖所有 ASCII 字符
    int last[256];
    memset(last, -1, sizeof(last));

    int ans = 0;
    // l: 当前无重复子串的左起点
    // r: 当前探测的终点
    for (int l = 0, r = 0; r < n; r++) {
        // 当前字符 s[r]
        unsigned char c = (unsigned char)str[r]; 

        // 核心逻辑：
        // 如果 c 之前出现过，那么左边界 l 必须更新
        // 更新到 max(当前l, 该字符上次出现位置+1)
        l = max(l, last[c] + 1);

        // 计算当前窗口 [l, r] 的长度并更新答案
        ans = max(ans, r - l + 1);

        // 更新字符 c 上次出现的位置为当前下标 r
        last[c] = r;
    }

    return ans;
}

// 简单测试
int main() {
    string s = "abcabcbb";
    // 预期结果：3 ("abc")
    printf("最长无重复子串长度: %d\n", lengthOfLongestSubstring(s));

    string s2 = "pwwkew";
    // 预期结果：3 ("wke")
    printf("最长无重复子串长度: %d\n", lengthOfLongestSubstring(s2));

    return 0;
}
