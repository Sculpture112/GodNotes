// 036 二叉树的锯齿形层序遍历
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 锯齿形层序遍历实现 (使用手动数组模拟队列，通过双向索引控制收集顺序)
锯齿就是从左往右，再从右往左

# include <vector>

using namespace std;

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Code02_ZigzagLevelOrderTraversal {
public:
    // 如果测试数据量变大了就修改这个值
    static const int MAXN = 2001;
    // 使用静态数组模拟队列，提高性能
    TreeNode* q[MAXN];
    int l, r;

    // 用每次处理一层的优化bfs就非常容易实现
    vector<vector<int>> zigzagLevelOrder(TreeNode* root) {
        vector<vector<int>> ans;
        if (root != nullptr) {
            l = r = 0;
            q[r++] = root;
            // false 代表从左往右
            // true 代表从右往左
            bool reverse = false;
            while (l < r) {
                int size = r - l;
                vector<int> current_level;

                // 根据 reverse 标志决定遍历方向
                // reverse == false, 左 -> 右， l....r-1, 收集size个
                // reverse == true,  右 -> 左， r-1....l, 收集size个
                // i 为起始索引，j 为步长方向，k 为计数器
                for (int i = reverse ? r - 1 : l, j = reverse ? -1 : 1, k = 0; k < size; i += j, k++) {
                    TreeNode* cur = q[i];
                    current_level.push_back(cur->val);
                }

                // 正常的队列弹出逻辑，并放入下一层的孩子节点
                for (int i = 0; i < size; i++) {
                    TreeNode* cur = q[l++];
                    if (cur->left != nullptr) {
                        q[r++] = cur->left;
                    }
                    if (cur->right != nullptr) {
                        q[r++] = cur->right;
                    }
                }

                ans.push_back(current_level);
                // 每一层结束后反转方向
                reverse = !reverse;
            }
        }
        return ans;
    }
};
