# 038 Code03 没有重复项数字的全排列

```cpp
// 038 Code03 没有重复项数字的全排列
// 1. 核心思路：i 之前的数字已经确定，i 之后的数字（包括 i）都可以交换到 i 位置上来尝试
// 2. 递归深度：每一层递归确定一个位置上的数字
// 3. 回溯恢复：交换过去之后，递归回来必须交换回来，确保上一层决策时的现场不被破坏

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 交换函数
void swap(vector<int>& nums, int i, int j) {
    int tmp = nums[i];
    nums[i] = nums[j];
    nums[j] = tmp;
}

/**
 // * 递归函数
 // * @param nums 原始数组
 // * @param i    当前来到哪个位置做选择
 // * @param ans  收集结果
 */
void f(vector<int>& nums, int i, vector<vector<int>>& ans) {
    if (i == nums.size()) {
        // 到底了，收集当前数组的快照
        ans.push_back(nums);
    } else {
        for (int j = i; j < nums.size(); j++) {
            // 把 j 位置的数换到 i 位置来
            swap(nums, i, j);
            // 去填充下一个位置
            f(nums, i + 1, ans);
            // 特别重要：递归回来，必须换回来（回溯）
            swap(nums, i, j);
        }
    }
}

vector<vector<int>> permute(vector<int>& nums) {
    vector<vector<int>> ans;
    f(nums, 0, ans);
    return ans;
}

int main() {
    vector<int> nums = {1, 2, 3};
    vector<vector<int>> ans = permute(nums);

    printf("数组 [1, 2, 3] 的全排列结果：\n");
    for (const auto& list : ans) {
        printf("[ ");
        for (int num : list) {
            printf("%d ", num);
        }
        printf("]\n");
    }

    return 0;
}
```
