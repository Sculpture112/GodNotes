# 最小栈（静态数组实现版）

```cpp
// 最小栈（静态数组实现版）
class MinStack2 {
    static const int MAXN = 8001; // 根据题目数据范围设定
    int data[MAXN];  // 数据栈数组
    int minArr[MAXN]; // 最小值栈数组（与 data 下标同步）
    int size;         // 当前栈顶指针（即元素个数）

public:
    MinStack2() : size(0) {}

    void push(int val) {
        data[size] = val;
        // 核心逻辑：若栈空或新值更小，minArr 存新值；否则沿用上一个最小值
        if (size == 0 || val <= minArr[size - 1]) {
            minArr[size] = val;
        } else {
            minArr[size] = minArr[size - 1];
        }
        size++;
    }

    // 逻辑删除：只需减少 size，无需真正清除内存
    void pop() { size--; }

    int top() { return data[size - 1]; }

    int getMin() { return minArr[size - 1]; }
};
```
