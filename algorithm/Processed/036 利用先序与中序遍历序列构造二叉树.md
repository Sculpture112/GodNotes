# 036 利用先序与中序遍历序列构造二叉树

```cpp
// 036 利用先序与中序遍历序列构造二叉树
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 构造二叉树主函数 (buildTree)
// 3. 递归构造辅助函数 (f)

#include <vector>
#include <unordered_map>

using namespace std;

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Code07_PreorderInorderBuildBinaryTree {
public:
    // 提交如下的方法
    TreeNode* buildTree(vector<int>& pre, vector<int>& in) {
        if (pre.empty() || in.empty() || pre.size() != in.size()) {
            return nullptr;
        }
        // 使用哈希表记录中序遍历中值与索引的对应关系，实现 O(1) 查找
        unordered_map<int, int> posMap;
        for (int i = 0; i < in.size(); i++) {
            posMap[in[i]] = i;
        }
        return f(pre, 0, pre.size() - 1, in, 0, in.size() - 1, posMap);
    }

    // 递归函数 f
    // pre: 先序数组, l1...r1 是当前处理范围
    // in: 中序数组, l2...r2 是当前处理范围
    // map: 中序位置索引表
    TreeNode* f(vector<int>& pre, int l1, int r1, vector<int>& in, int l2, int r2, unordered_map<int, int>& map) {
        if (l1 > r1) {
            return nullptr;

// 见下方绿色注释
        }
        TreeNode* head = new TreeNode(pre[l1]);
        if (l1 == r1) {
            return head;
        }

        // 找到当前根节点在中序遍历中的位置
        int k = map[pre[l1]];

        // pre : l1(........)[.......r1]
        // in  : (l2......)k[........r2]

左子树长度L= k-l2

k不在中序的左区间，是l2到k-1，再加上左子树（或者右子树）可能是空树，L = 0左子树 r1 = l1 + 1，l1 = l1+ L 所以l1可能会>r1

// (...) 是左子树对应范围，[...] 是右子树对应范围
        // 左子树长度为 k - l2

        // 递归构建左子树
        head->left = f(pre, l1 + 1, l1 + k - l2, in, l2, k - 1, map);
        // 递归构建右子树
        head->right = f(pre, l1 + k - l2 + 1, r1, in, k + 1, r2, map);

        return head;
    }
};
```
