# 039 Code01 含有嵌套的表达式求值

```cpp
// 039 Code01 含有嵌套的表达式求值
// 测试链接 : https://leetcode.cn/problems/basic-calculator-iii/
// 测试链接 : https://www.nowcoder.com/practice/c215ba61c8b1443b996351df929dc4d4
// 核心思路：递归处理括号 + 栈/列表处理乘除优先级

#include <iostream>
#include <vector>
#include <string>

using namespace std;

class Solution {
public:
    int where = 0;

    int calculate(string s) {
        where = 0;
        return f(s, 0);
    }

    // f(s, i) 返回从 s[i...] 开始计算的结果
    int f(const string& s, int i) {
        long cur = 0;
        vector<int> numbers;
        vector<char> ops;

        while (i < s.length() && s[i] != ')') {
            if (isdigit(s[i])) {
                cur = cur * 10 + (s[i++] - '0');
            } else if (s[i] != '(') {
                // 遇到运算符 + - * /
                push(numbers, ops, (int)cur, s[i++]);
                cur = 0;
            } else {
                // 遇到左括号，递归进去拿结果
                cur = f(s, i + 1);
                i = where + 1;
            }
        }
        // 最后一个数字或括号结果压入
        push(numbers, ops, (int)cur, '+');
        where = i;
        return compute(numbers, ops);
    }

    void push(vector<int>& numbers, vector<char>& ops, int cur, char op) {
        if (numbers.empty() || (ops.back() != '*' && ops.back() != '/')) {
            numbers.push_back(cur);
            ops.push_back(op);
        } else {
            int topNum = numbers.back();
            char topOp = ops.back();
            numbers.pop_back();
            ops.pop_back();
            if (topOp == '*') {
                numbers.push_back(topNum * cur);
            } else {
                numbers.push_back(topNum / cur);
            }
            ops.push_back(op);
        }
    }

    int compute(const vector<int>& numbers, const vector<char>& ops) {
        int ans = numbers[0];
        for (int i = 1; i < numbers.size(); i++) {
            ans += (ops[i - 1] == '+') ? numbers[i] : -numbers[i];
        }
        return ans;
    }
};

int main() {
    Solution sol;
    string exp = "1+2*(3-1)+4/2";
    // 1 + 2*2 + 2 = 7
    cout << "计算结果: " << sol.calculate(exp) << endl;
    return 0;
}
```
