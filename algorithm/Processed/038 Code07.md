// 038 Code07

// 打印 n 层汉诺塔问题的最优移动轨迹

// 核心思路：把 n 个盘子的移动拆解为三步

// 1. 将前 n-1 个盘子从 from 移动到 other (利用 to 作为辅助)

// 2. 将第 n 个盘子从 from 移动到 to

// 3. 将前 n-1 个盘子从 other 移动到 to (利用 from 作为辅助)

#include <iostream>
#include <string>

using namespace std;

/**
 // * 汉诺塔递归函数
 */
void f(int i, string from, string to, string other) {
    if (i == 1) {
        // 使用 cout 输出，直接插入变量，无需 .c_str()
        cout << "移动圆盘 1 从 " << from << " 到 " << to << endl;
    } else {
        f(i - 1, from, other, to);

        // endl 相当于 \n 并刷新缓冲区
        cout << "移动圆盘 " << i << " 从 " << from << " 到 " << to << endl;

        f(i - 1, other, to, from);
    }
}

void hanoi(int n) {
    if (n > 0) {
        f(n, "左", "右", "中");
    }
}

int main() {
    // 提高 C++ 输入输出流的效率（可选）
    ios::sync_with_stdio(false);
    cin.tie(0);

    int n = 3;
    cout << "--- " << n << " 层汉诺塔移动轨迹 ---" << endl;
    hanoi(n);

    return 0;
}
