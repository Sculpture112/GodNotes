# 047 Code02 等差数列差分

```cpp
// 047 Code02 等差数列差分
// 核心：二阶差分
// 1. 设置四个点：l, l+1, r+1, r+2
// 2. 两次前缀和加工
// 3. 注意 long long 防止溢出，以及大数据量的快读处理

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

// 10^7 + 5 的范围，必须开在全局存储区
const int MAXN = 10000005;
long long arr[MAXN]; 
int n, m;

// 二阶差分设置函数
void set(int l, int r, long long s, long long e, long long d) {
    arr[l] += s;
    arr[l + 1] += (d - s);
    arr[r + 1] -= (d + e);
    arr[r + 2] += e;
}

// 两次前缀和还原
void build() {
    // 第一次加工：还原成一阶差分数组
    for (int i = 1; i <= n; i++) {
        arr[i] += arr[i - 1];
    }
    // 第二次加工：还原成原始数值数组
    for (int i = 1; i <= n; i++) {
        arr[i] += arr[i - 1];
    }
}

int main() {
    // 优化 C++ 输入输出效率
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> n >> m)) return 0;

    for (int i = 0; i < m; i++) {
        int l, r;
        long long s, e;
        cin >> l >> r >> s >> e;
        // 计算公差 d
        long long d = (e - s) / (r - l == 0 ? 1 : (r - l)); 
        // 实际上题目保证了 r >= l，当 r == l 时 d 并不重要，因为 l+1 和 r+1 会抵消
        set(l, r, s, e, (r == l) ? 0 : d);
    }

    build();

    long long max_val = 0;
    long long xor_sum = 0;
    for (int i = 1; i <= n; i++) {
        max_val = max(max_val, arr[i]);
        xor_sum ^= arr[i];
    }

    cout << xor_sum << " " << max_val << endl;

    return 0;
}
```
