# 037 修剪搜索二叉树

```cpp
// 037 修剪搜索二叉树
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 修剪搜索二叉树递归实现 (trimBST)

#include <cstddef>

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Code06_TrimBinarySearchTree {
public:
    // 提交以下的方法
    // 修剪范围 [low, high]
    TreeNode* trimBST(TreeNode* cur, int low, int high) {
        if (cur == nullptr) {
            return nullptr;
        }

        // 如果当前节点的值小于最小值，说明该节点及其左子树都不在范围内
        // 直接返回其右子树修剪后的结果
        if (cur->val < low) {
            return trimBST(cur->right, low, high);
        }

        // 如果当前节点的值大于最大值，说明该节点及其右子树都不在范围内
        // 直接返回其左子树修剪后的结果
        if (cur->val > high) {
            return trimBST(cur->left, low, high);
        }

        // 当前节点值在范围内，保留该节点
        // 递归地修剪其左右子树，并将返回的新子树连接到当前节点上
        cur->left = trimBST(cur->left, low, high);
        cur->right = trimBST(cur->right, low, high);

        return cur;
    }
};
```
