# 050 Code02 寻找重复数

```cpp
// 050 Code02 寻找重复数
// 测试链接 : https://leetcode.cn/problems/find-the-duplicate-number/
// 核心思路：快慢指针找环入口
// 时间复杂度 O(N)，额外空间复杂度 O(1)

#include <iostream>
#include <vector>

using namespace std;

int findDuplicate(vector<int>& nums) {
    if (nums.size() < 2) {
        return -1;
    }

    // 初始位置：slow 在 0 走一步，fast 在 0 走两步
    int slow = nums[0];
    int fast = nums[nums[0]];

    // 第一阶段：找相遇点
    while (slow != fast) {
        slow = nums[slow];
        fast = nums[nums[fast]];
    }

    // 第二阶段：fast 回到起点，寻找环入口
    fast = 0;
    while (slow != fast) {
        fast = nums[fast];
        slow = nums[slow];
    }

    // 相遇处即为重复的数字
    return slow;
}

int main() {
    vector<int> nums = {1, 3, 4, 2, 2};
    // 预期结果：2
    printf("重复的数字是: %d\n", findDuplicate(nums));
    return 0;
}
```
