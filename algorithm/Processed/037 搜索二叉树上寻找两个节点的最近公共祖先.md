# 037 搜索二叉树上寻找两个节点的最近公共祖先

```cpp
// 037 搜索二叉树上寻找两个节点的最近公共祖先
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 搜索二叉树最近公共祖先实现 (lowestCommonAncestor)

#include <algorithm>

using namespace std;

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Code02_LowestCommonAncestorBinarySearch {
public:
    // 提交如下的方法
    TreeNode* lowestCommonAncestor(TreeNode* root, TreeNode* p, TreeNode* q) {
        // root从上到下
        // 如果先遇到了p，说明p是答案
        // 如果先遇到了q，说明q是答案
        // 如果root在p~q的值之间，不用管p和q谁大谁小，只要root在中间，那么此时的root就是答案
        // 如果root在p~q的值的左侧，那么root往右移动
        // 如果root在p~q的值的右侧，那么root往左移动

        // 在 C++ 中使用 std::min 和 std::max
        while (root->val != p->val && root->val != q->val) {

// 这里如果root等于p，直接就返回root了
            int minVal = min(p->val, q->val);
            int maxVal = max(p->val, q->val);

// 判断走左还是走右，为什么大小值可以判断呢。因为如果root值比两个都小，那么一定是走右边（大的值

            if (minVal < root->val && root->val < maxVal) {
                break;
            }

            // 利用搜索二叉树性质进行路径选择
            root = root->val < minVal ? root->right : root->left;

}
        return root;
    }
};
```
