# 050 Code04 救生艇

```cpp
// 050 Code04 救生艇
// 测试链接 : https://leetcode.cn/problems/boats-to-save-people/
// 核心思路：排序 + 双指针贪心
// 时间复杂度 O(N log N)，空间复杂度 O(log N) (取决于排序实现)

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    int numRescueBoats(vector<int>& people, int limit) {
        // 1. 排序是双指针的前提
        sort(people.begin(), people.end());

        int n = people.size();
        int l = 0;
        int r = n - 1;
        int ans = 0;

        while (l <= r) {
            // 判定当前最轻和最重的是否能同船
            // 如果 l == r，说明只剩一个人了，sum 就是那个人的体重
            int sum = (l == r) ? people[l] : people[l] + people[r];

            if (sum > limit) {
                // 太重了，最重的 r 必须自己坐一艘，l 等待下次机会
                r--;
            } else {
                // 能装下，l 和 r 凑一对走
                l++;
                r--;
            }
            // 无论是哪种情况，都需要消耗一艘船
            ans++;
        }
        return ans;
    }
};

int main() {
    Solution sol;
    vector<int> p = {3, 2, 2, 1};
    int limit = 3;
    // 排序后: 1, 2, 2, 3
    // (1, 2), (2), (3) -> 3艘
    printf("最小船数: %d\n", sol.numRescueBoats(p, limit));
    return 0;
}
```
