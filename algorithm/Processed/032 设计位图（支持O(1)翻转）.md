# 032 设计位图（支持O(1)翻转）

```cpp
// 032 设计位图（支持O(1)翻转）
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

class Bitset {
    vector<int> set; int sz, zeros, ones; bool rev; // 数组、总大小、0计数、1计数、翻转标志
public:
    Bitset(int n) : set((n + 31) / 32, 0), sz(n), zeros(n), ones(0), rev(false) {}

    // 将第i位设为1：根据rev状态判断当前位是否已经是逻辑上的1
    void fix(int i) {
        int idx = i >> 5, bit = i & 31;
        if (!rev) { // 正常模式：1是1。若为0则变1。
            if (!(set[idx] & (1 << bit))) { zeros--; ones++; set[idx] |= (1 << bit); }
        } else { // 翻转模式：0是1。若为1则变0。
            if (set[idx] & (1 << bit)) { zeros--; ones++; set[idx] ^= (1 << bit); }
        }
    }

    // 将第i位设为0：逻辑同上
    void unfix(int i) {
        int idx = i >> 5, bit = i & 31;//这个就是%31
        if (!rev) { // 正常模式：0是0。若为1则变0。
            if (set[idx] & (1 << bit)) { ones--; zeros++; set[idx] ^= (1 << bit); }
        } else { // 翻转模式：1是0。若为0则变1。
            if (!(set[idx] & (1 << bit))) { ones--; zeros++; set[idx] |= (1 << bit); }
        }
    }

    // O(1)翻转：切换解释权，并交换0和1的计数
    void flip() { rev = !rev; swap(ones, zeros); }

    bool all() { return ones == sz; } // 1的数量等于总大小
    bool one() { return ones > 0; }   // 1的数量大于0
    int count() { return ones; }

    string toString() {
        string res = "";
        for (int i = 0, k = 0; i < sz; k++) {
            int val = set[k];
            for (int j = 0; j < 32 && i < sz; j++, i++) {
                int s = (val >> j) & 1; // 提取原始状态
                res += (s ^ (rev ? 1 : 0)) + '0'; // 根据rev异或出逻辑状态
            }
        }
        return res;
    }
};
```
