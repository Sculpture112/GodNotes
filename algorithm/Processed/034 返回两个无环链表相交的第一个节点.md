# 034 返回两个无环链表相交的第一个节点

```cpp
// 034 返回两个无环链表相交的第一个节点
#include <iostream>

struct ListNode { int val; ListNode *next; }; // 链表节点定义

class Solution {
public:
    ListNode *getIntersectionNode(ListNode *h1, ListNode *h2) {
        if (!h1 || !h2) return nullptr;
        ListNode *a = h1, *b = h2;
        int diff = 0;
        // 1. 走到尾部并计算长度差值：a每走一步diff自增，b每走一步diff自减
        while (a->next) { a = a->next; diff++; }
        while (b->next) { b = b->next; diff--; }
        if (a != b) return nullptr; // 2. 如果尾节点地址不同，说明两个链表根本没连在一起
        a = diff >= 0 ? h1 : h2; // 3. 令 a 指向长链表的头，b 指向短链表的头
        b = a == h1 ? h2 : h1;
        diff = abs(diff); // 取差值的绝对值
        while (diff--) a = a->next; // 4. 长链表先走完多出来的部分
        while (a != b) { a = a->next; b = b->next; } // 5. 两者同步走，碰撞处即为交点
        return a;
    }
};
```
