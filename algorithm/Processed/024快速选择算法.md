# 024快速选择算法

```cpp
// 024快速选择算法
#include <vector>
#include <algorithm>
#include <cstdlib>
using namespace std;

class Solution {
public:
    int first, last; // 记录荷兰国旗划分后等于区域的左右边界

// 排序功能
    // 荷兰国旗划分：将数组分为小于x、等于x、大于x三个部分
    void partition(vector<int>& arr, int l, int r, int x) {
        first = l; last = r; int i = l;
        while (i <= last) {
            if (arr[i] == x) i++; // 等于x，指针后移
            else if (arr[i] < x) swap(arr[first++], arr[i++]); // 小于x，换到左边
            else swap(arr[i], arr[last--]); // 大于x，换到右边
        }
    }

    // 随机选择核心逻辑：平均时间复杂度 O(N)
    int findKthLargest(vector<int>& nums, int k) {
        int i = nums.size() - k; // 第K大元素即为升序排序后索引为 nums.size()-k 的数
        for (int l = 0, r = nums.size() - 1; l <= r; ) {
            // 随机选择基准值x并进行划分，保证概率上的 O(N) 性能
// 先划分
            partition(nums, l, r, nums[l + rand() % (r - l + 1)]);
            if (i < first) r = first - 1; // 目标在小于区域
            else if (i > last) l = last + 1; // 目标在大于区域
            else return nums[i]; // 目标就在等于区域，直接返回
// 找到了直接返回
        }
        return -1;
    }
};
```
