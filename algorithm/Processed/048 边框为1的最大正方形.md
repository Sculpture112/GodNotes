# 048 边框为1的最大正方形

```cpp
// 048 边框为1的最大正方形
// 1. 查找最大 1 边框正方形主函数 (largest1BorderedSquare)
// 2. 构建二维前缀和 (build)
// 3. 区域求和计算 (sum)
// 4. 边界安全访问辅助函数 (get)

#include <vector>
#include <algorithm>

using namespace std;

class Code02_LargestOneBorderedSquare {
public:
    // 时间复杂度O(n * m * min(n,m))，额外空间复杂度O(1)
    // 复杂度指标上绝对是最优解
    int largest1BorderedSquare(vector<vector<int>>& g) {
        int n = g.size();
        int m = g[0].size();
        // 预处理：原地将 g 转换为二维前缀和数组，节省空间
        build(n, m, g);

        // 如果整个矩阵和为 0，说明没有 1，直接返回 0
        if (sum(g, 0, 0, n - 1, m - 1) == 0) {
            return 0;
        }

        // 找到的最大合法正方形的边长，初始为 1（因为已经确认存在 1）
        int ans = 1;
        for (int a = 0; a < n; a++) {
            for (int b = 0; b < m; b++) {
                // (a,b) 作为所有可能的左上角点
                // (c,d) 是更大边长的右下角点，k 是当前尝试的边长
                // 从 ans + 1 开始尝试，因为我们要寻找比当前 ans 更大的正方形
                for (int c = a + ans, d = b + ans, k = ans + 1; c < n && d < m; c++, d++, k++) {
                    // 核心逻辑：
                    // 外层正方形总和 - 内部正方形总和 = 边框上 1 的数量
                    // 对于边长为 k 的正方形，若边框全是 1，则数量应为 (k-1) * 4
                    if (sum(g, a, b, c, d) - sum(g, a + 1, b + 1, c - 1, d - 1) == (k - 1) << 2) {
                        ans = k;
                    }
                }
            }
        }
        return ans * ans;
    }

private:
    // 2. 构建二维前缀和逻辑
    // 把 g 变成原始二维数组的前缀和数组 sum，复用自己
    void build(int n, int m, vector<vector<int>>& g) {
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                g[i][j] += get(g, i, j - 1) + get(g, i - 1, j) - get(g, i - 1, j - 1);
            }
        }
    }

    // 3. 区域求和计算
    int sum(vector<vector<int>>& g, int a, int b, int c, int d) {
        // 如果左上角坐标已经越过右下角（如计算 1x1 正方形的内部区域），返回 0
        return a > c ? 0 : (g[c][d] - get(g, c, b - 1) - get(g, a - 1, d) + get(g, a - 1, b - 1));
    }

    // 4. 边界安全访问辅助函数
    // 对应 Java 中的 get 静态方法，处理索引越界返回 0
    int get(vector<vector<int>>& g, int i, int j) {
        return (i < 0 || j < 0) ? 0 : g[i][j];
    }
};
```
