# 048 用邮票贴满网格图

```cpp
// 048 用邮票贴满网格图
// 1. 邮票放置判定主逻辑 (possibleToStamp)
// 2. 构建二维前缀和 (build)
// 3. 区域求和查询 (sumRegion)
// 4. 二维差分区域增加 (add)

#include <vector>

using namespace std;

class Code04_StampingTheGrid {
public:
    // 时间复杂度O(n*m)，额外空间复杂度O(n*m)
    bool possibleToStamp(vector<vector<int>>& grid, int h, int w) {
        int n = grid.size();
        int m = grid[0].size();

        // 1. 构建原始矩阵的前缀和数组 sum
        // 用于快速查询 grid 中某个区域是否全为 0（即是否可以放邮票）
        vector<vector<int>> sum(n + 1, vector<int>(m + 1, 0));
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                sum[i + 1][j + 1] = grid[i][j];
            }
        }
        build(sum);

        // 2. 准备差分矩阵 diff
        // diff 用来标记哪些格子被邮票覆盖了
        // 因为后面 add 操作会访问到 c+1, d+1，所以大小开到 n+2, m+2
        vector<vector<int>> diff(n + 2, vector<int>(m + 2, 0));

        // 遍历每一个可能的邮票左上角 (a, b)
        for (int a = 1, c = a + h - 1; c <= n; a++, c++) {
            for (int b = 1, d = b + w - 1; d <= m; b++, d++) {
                // 如果原始矩阵中该矩形区域累加和为 0，说明全是空格，可以贴邮票
                if (sumRegion(sum, a, b, c, d) == 0) {
                    add(diff, a, b, c, d);
                }
            }
        }

        // 3. 将差分矩阵还原成前缀和矩阵，此时 diff[i][j] > 0 代表被邮票覆盖过
        build(diff);

        // 4. 最后检查：所有的空格子是否都被至少一张邮票覆盖
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                // grid[i][j] == 0 说明是空格
                // diff[i+1][j+1] == 0 说明没被邮票盖住
                if (grid[i][j] == 0 && diff[i + 1][j + 1] == 0) {
                    return false;
                }
            }
        }
        return true;
    }

private:
    // 2. 构建二维前缀和
    void build(vector<vector<int>>& m) {
        for (int i = 1; i < m.size(); i++) {
            for (int j = 1; j < m[0].size(); j++) {
                m[i][j] += m[i - 1][j] + m[i][j - 1] - m[i - 1][j - 1];
            }
        }
    }

    // 3. 区域求和查询
    int sumRegion(const vector<vector<int>>& sum, int a, int b, int c, int d) {
        return sum[c][d] - sum[c][b - 1] - sum[a - 1][d] + sum[a - 1][b - 1];
    }

    // 4. 二维差分区域增加
    void add(vector<vector<int>>& diff, int a, int b, int c, int d) {
        diff[a][b] += 1;
        diff[c + 1][d + 1] += 1;
        diff[c + 1][b] -= 1;
        diff[a][d + 1] -= 1;
    }
};
```
