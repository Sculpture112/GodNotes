# 050 Code06 供暖器

```cpp
// 050 Code06 供暖器
// 测试链接 : https://leetcode.cn/problems/heaters/
// 核心思路：排序 + 双指针。对每个房屋寻找最近的供暖器，指针不回退。
// 时间复杂度 O(N log N + M log M)，额外空间复杂度 O(1)

#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>

using namespace std;

class Solution {
public:
    // 判断对于第 i 号房屋，第 j 号供暖器是否比第 j+1 号更好
    bool best(const vector<int>& houses, const vector<int>& heaters, int i, int j) {
        // 如果 j 已经是最后一台供暖器了，那没得选，j 就是最好的
        if (j == heaters.size() - 1) {
            return true;
        }
        // 比较当前供暖器和下一台供暖器到房屋的距离
        // 注意：这里用 <= 是为了在距离相等时也倾向于跳到下一个，优化逻辑
        return abs(heaters[j] - houses[i]) < abs(heaters[j + 1] - houses[i]);
    }

    int findRadius(vector<int>& houses, vector<int>& heaters) {
        // 1. 排序是双指针的前提
        sort(houses.begin(), houses.end());
        sort(heaters.begin(), heaters.end());

        int ans = 0;
        int n = houses.size();
        int m = heaters.size();

        // i 扫描房屋，j 扫描供暖器
        for (int i = 0, j = 0; i < n; i++) {
            // 如果当前的供暖器 j 不是最优的（即 j+1 离 i 更近），则 j 右移
            while (!best(houses, heaters, i, j)) {
                j++;
            }
            // 当前房屋到最近供暖器的距离
            int curDist = abs(heaters[j] - houses[i]);
            // 全局半径取所有房屋所需距离的最大值
            ans = max(ans, curDist);
        }

        return ans;
    }
};

int main() {
    Solution sol;
    vector<int> houses = {1, 2, 3, 4};
    vector<int> heaters = {1, 4};
    // 房屋1找1(0), 房屋2找1(1), 房屋3找4(1), 房屋4找4(0)
    // 最大距离是 1
    printf("最小加热半径: %d\n", sol.findRadius(houses, heaters));
    return 0;
}
```
