# 035 快速获得数据流的中位数的结构

```cpp
// 035 快速获得数据流的中位数的结构
#include <queue>
#include <vector>
#include <cmath>
using namespace std;

class MedianFinder {
    priority_queue<int> maxH; // 大根堆：存较小的一半
    priority_queue<int, vector<int>, greater<int>> minH; // 小根堆：存较大的一半
    // 维持两个堆的数量平衡，差值不能超过1
    void balance() {
        if (abs((int)maxH.size() - (int)minH.size()) == 2) {
            if (maxH.size() > minH.size()) { minH.push(maxH.top()); maxH.pop(); }
            else { maxH.push(minH.top()); minH.pop(); }
        }
    }
public:
    MedianFinder() {}
    // 新增数字：根据与大根堆顶的比较决定去向，并调平衡
    void addNum(int num) {
        if (maxH.empty() || maxH.top() >= num) maxH.push(num);
        else minH.push(num);
        balance();
    }
    // 获取中位数：数量不等取较多的堆顶，相等则取两堆顶平均值
    double findMedian() {
        if (maxH.size() == minH.size()) return (maxH.top() + minH.top()) / 2.0;
        return maxH.size() > minH.size() ? maxH.top() : minH.top();
    }
};
```
