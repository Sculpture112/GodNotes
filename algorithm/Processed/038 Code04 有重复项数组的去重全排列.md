# 038 Code04 有重复项数组的去重全排列

```cpp
// 038 Code04 有重复项数组的去重全排列
// 1. 核心思路：在 Code03 交换全排列的基础上，增加“同层去重”
// 2. 剪枝逻辑：在当前位置 i 尝试交换 j 位置的数时，如果 nums[j] 这个数值在当前层已经交换过一次了，则不再尝试
// 3. 空间优化：在 C++ 中，如果数值范围有限，可以用 boolean 数组代替 unordered_set

#include <iostream>
#include <vector>
#include <unordered_set>
#include <algorithm>

using namespace std;

// 交换函数
void swap(vector<int>& nums, int i, int j) {
    int tmp = nums[i];
    nums[i] = nums[j];
    nums[j] = tmp;
}

/**
 // * 递归函数
 // * @param nums 数组
 // * @param i    当前来到哪个位置做选择
 // * @param ans  收集结果
 */
void f(vector<int>& nums, int i, vector<vector<int>>& ans) {
    if (i == nums.size()) {
        ans.push_back(nums);
    } else {
        // 关键点：set 是局部变量，每一层递归都有自己的 set
        // 它记录了当前 i 位置已经尝试过哪些数值了
        unordered_set<int> visited;
        for (int j = i; j < nums.size(); j++) {
            // 如果 nums[j] 之前没有在这个位置（i位置）出现过，才进行交换尝试
            if (visited.find(nums[j]) == visited.end()) {
                visited.insert(nums[j]);
                swap(nums, i, j);
                f(nums, i + 1, ans);
                swap(nums, i, j); // 恢复现场
            }
        }
    }
}

vector<vector<int>> permuteUnique(vector<int>& nums) {
    vector<vector<int>> ans;
    // 交换全排列不强制要求排序，但排序能让结果更有序
    f(nums, 0, ans);
    return ans;
}

// 简单测试
int main() {
    vector<int> nums = {1, 1, 2};
    vector<vector<int>> result = permuteUnique(nums);

    printf("数组 [1, 1, 2] 的去重全排列结果：\n");
    for (const auto& list : result) {
        printf("[ ");
        for (int num : list) printf("%d ", num);
        printf("]\n");
    }

    return 0;
}
```
