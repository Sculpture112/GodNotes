# 050 Code07 缺失的第一个正数

```cpp
// 050 Code07 缺失的第一个正数
// 测试链接 : https://leetcode.cn/problems/first-missing-positive/
// 核心思路：原地哈希 / 区域划分。l 为有效区边界，r 为垃圾区边界。
// 时间复杂度 O(N)，额外空间复杂度 O(1)

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    int firstMissingPositive(vector<int>& arr) {
        int n = arr.size();
        // [0, l-1] 是有效区：i位置放着 i+1
        int l = 0;
        // [r, n-1] 是垃圾区
        int r = n;

        while (l < r) {
            if (arr[l] == l + 1) {
                // 情况1：当前数字就在它该在的位置
                l++;
            } else if (arr[l] <= l || arr[l] > r || arr[arr[l] - 1] == arr[l]) {
                // 情况2：当前数字是垃圾（负数、太大、或是重复的数）
                // 扔进垃圾区，预期上限 r 减小
                swap(arr[l], arr[--r]);
            } else {
                // 情况3：是好数但没在位，送去它该去的位置 arr[l]-1
                swap(arr[l], arr[arr[l] - 1]);
            }
        }
        // 最后有效区 l 的下一个数 l+1 就是缺失的最小正整数
        return l + 1;
    }
};

int main() {
    Solution sol;
    vector<int> nums = {3, 4, -1, 1};
    // 调整后可能为 [1, -1, 3, 4]，l停在1处（下标），返回 2
    printf("缺失的第一个正数: %d\n", sol.firstMissingPositive(nums));
    return 0;
}
```
