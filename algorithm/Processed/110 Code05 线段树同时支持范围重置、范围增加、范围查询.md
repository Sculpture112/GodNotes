# 110 Code05 线段树同时支持范围重置、范围增加、范围查询

```cpp
// 110 Code05 线段树同时支持范围重置、范围增加、范围查询
// 1. 累加和信息的汇总 (up)
// 2. 懒信息的下发 (down)
// 3. 范围重置懒更新任务执行 (updateLazy)
// 4. 范围增加懒更新任务执行 (addLazy)
// 5. 建树 (build)
// 6. 范围重置修改 (update)
// 7. 范围增加修改 (add)
// 8. 范围查询累加和 (query)

#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdio>
#include <ctime>

using namespace std;

// 基础常量定义
const int MAXN = 1000001;

// 全局静态数组，防止栈溢出
long long arr[MAXN];
long long sum_tree[MAXN << 2];
long long add_tag[MAXN << 2];
long long change_tag[MAXN << 2];
bool update_tag[MAXN << 2];

// 1. 累加和信息的汇总
void up(int i) {
    sum_tree[i] = sum_tree[i << 1] + sum_tree[i << 1 | 1];
}

// 3. 范围重置懒更新任务执行
// 重置操作优先级高于增加操作，一旦重置，之前的增加标记清空
void updateLazy(int i, long long v, int n) {
    sum_tree[i] = v * n;
    change_tag[i] = v;
    add_tag[i] = 0; // 关键：重置后之前的增加标记失效
    update_tag[i] = true;
}

// 4. 范围增加懒更新任务执行
void addLazy(int i, long long v, int n) {
    sum_tree[i] += v * n;
    add_tag[i] += v;
}

// 2. 懒信息的下发
// 先发重置标记，再发增加标记
void down(int i, int ln, int rn) {
    if (update_tag[i]) {
        updateLazy(i << 1, change_tag[i], ln);
        updateLazy(i << 1 | 1, change_tag[i], rn);
        update_tag[i] = false;
    }
    if (add_tag[i] != 0) {
        addLazy(i << 1, add_tag[i], ln);
        addLazy(i << 1 | 1, add_tag[i], rn);
        add_tag[i] = 0;
    }
}

// 5. 建树
void build(int l, int r, int i) {
    if (l == r) {
        sum_tree[i] = arr[l];
    } else {
        int mid = (l + r) >> 1;
        build(l, mid, i << 1);
        build(mid + 1, r, i << 1 | 1);
        up(i);
    }
    add_tag[i] = 0;
    change_tag[i] = 0;
    update_tag[i] = false;
}

// 6. 范围重置修改
void update(int jobl, int jobr, long long jobv, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        updateLazy(i, jobv, r - l + 1);
    } else {
        int mid = (l + r) >> 1;
        down(i, mid - l + 1, r - mid);
        if (jobl <= mid) {
            update(jobl, jobr, jobv, l, mid, i << 1);
        }
        if (jobr > mid) {
            update(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        }
        up(i);
    }
}

// 7. 范围增加修改
void add(int jobl, int jobr, long long jobv, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        addLazy(i, jobv, r - l + 1);
    } else {
        int mid = (l + r) >> 1;
        down(i, mid - l + 1, r - mid);
        if (jobl <= mid) {
            add(jobl, jobr, jobv, l, mid, i << 1);
        }
        if (jobr > mid) {
            add(jobl, jobr, jobv, mid + 1, r, i << 1 | 1);
        }
        up(i);
    }
}

// 8. 范围查询累加和
long long query(int jobl, int jobr, int l, int r, int i) {
    if (jobl <= l && r <= jobr) {
        return sum_tree[i];
    }
    int mid = (l + r) >> 1;
    down(i, mid - l + 1, r - mid);
    long long ans = 0;
    if (jobl <= mid) {
        ans += query(jobl, jobr, l, mid, i << 1);
    }
    if (jobr > mid) {
        ans += query(jobl, jobr, mid + 1, r, i << 1 | 1);
    }
    return ans;
}

// 对数器暴力验证逻辑
void checkAdd(vector<long long>& check, int jobl, int jobr, long long jobv) {
    for (int i = jobl; i <= jobr; i++) check[i] += jobv;
}

void checkUpdate(vector<long long>& check, int jobl, int jobr, long long jobv) {
    for (int i = jobl; i <= jobr; i++) check[i] = jobv;
}

long long checkQuery(vector<long long>& check, int jobl, int jobr) {
    long long ans = 0;
    for (int i = jobl; i <= jobr; i++) ans += check[i];
    return ans;
}

int main() {
    srand(time(0));
    printf("测试开始\n");
    int n = 1000;
    int v = 2000;
    int t = 1000000;

    for (int i = 1; i <= n; i++) arr[i] = rand() % v;

    build(1, n, 1);

    vector<long long> check(n + 1);
    for (int i = 1; i <= n; i++) check[i] = arr[i];

    for (int i = 1; i <= t; i++) {
        int op = rand() % 3;
        int a = rand() % n + 1, b = rand() % n + 1;
        int jobl = min(a, b), jobr = max(a, b);
        long long jobv = rand() % (v * 2) - v;

        if (op == 0) {
            add(jobl, jobr, jobv, 1, n, 1);
            checkAdd(check, jobl, jobr, jobv);
        } else if (op == 1) {
            update(jobl, jobr, jobv, 1, n, 1);
            checkUpdate(check, jobl, jobr, jobv);
        } else {
            if (query(jobl, jobr, 1, n, 1) != checkQuery(check, jobl, jobr)) {
                printf("出错了!\n");
                break;
            }
        }
    }
    printf("测试结束\n");
    return 0;
}
```
