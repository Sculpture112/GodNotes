# 027 合并K个有序链表



```cpp
// 027 合并K个有序链表
#include <vector>
#include <queue>
using namespace std;

struct ListNode { int val; ListNode *next; }; // 链表节点定义

class Solution {
public:
    // 自定义比较器：值小的排在堆顶（实现小根堆）
    struct cmp { bool operator()(ListNode* a, ListNode* b) { return a->val > b->val; } };
// 小根堆比较器

    ListNode* mergeKLists(vector<ListNode*>& lists) {
        priority_queue<ListNode*, vector<ListNode*>, cmp> heap;
        for (auto h : lists) if (h) heap.push(h); // 1. 所有链表的头节点入堆
        if (heap.empty()) return nullptr; // 若所有链表都为空

        ListNode *head = heap.top(); heap.pop(); // 2. 弹出全局最小作为合并后的头
        ListNode *pre = head;
        if (pre->next) heap.push(pre->next); // 谁弹出，谁的下一个节点就入堆补位

        while (!heap.empty()) {
            ListNode *cur = heap.top(); heap.pop(); // 3. 弹出当前最小节点
            pre->next = cur; pre = cur; // 4. 将其连在结果链表末尾
            if (cur->next) heap.push(cur->next); // 5. 该节点的后续节点入堆
        }
        return head;
    }
};
```
