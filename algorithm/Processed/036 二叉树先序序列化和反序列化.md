# 036 二叉树先序序列化和反序列化

```cpp
// 036 二叉树先序序列化和反序列化
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 序列化逻辑实现 (serialize)
// 3. 反序列化逻辑实现 (deserialize)

#include <string>
#include <vector>
#include <sstream>

using namespace std;

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

// 二叉树可以通过先序、后序或者按层遍历的方式序列化和反序列化
// 但是，二叉树无法通过中序遍历的方式实现序列化和反序列化
// 因为不同的两棵树，可能得到同样的中序序列，即便补了空位置也可能一样。
class Codec {
public:
    // 2. 序列化逻辑实现
    string serialize(TreeNode* root) {
        string res = "";
        f(root, res);
        return res;
    }

    // 先序遍历递归函数
    void f(TreeNode* root, string& res) {
        if (root == nullptr) {
            res += "#,";
        } else {
            res += to_string(root->val) + ",";
            f(root->left, res);
            f(root->right, res);
        }
    }

    // 3. 反序列化逻辑实现
    TreeNode* deserialize(string data) {
        if (data.empty()) return nullptr;

        // 模拟 Java 的 split(",")
        vector<string> vals;
        stringstream ss(data);
        string item;
        while (getline(ss, item, ',')) {
            vals.push_back(item);
        }

        int cnt = 0; // 当前数组消费到哪了
        return g(vals, cnt);
    }

    // 递归重建二叉树
    TreeNode* g(const vector<string>& vals, int& cnt) {
        string cur = vals[cnt++];
        if (cur == "#") {
            return nullptr;
        } else {
            // stoi 将字符串转换为整数，对应 Java 的 Integer.valueOf
            TreeNode* head = new TreeNode(stoi(cur));
            head->left = g(vals, cnt);
            head->right = g(vals, cnt);
            return head;
        }
    }
};
```
