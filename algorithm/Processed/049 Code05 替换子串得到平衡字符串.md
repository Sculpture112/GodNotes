# 049 Code05 替换子串得到平衡字符串

```cpp
// 049 Code05 替换子串得到平衡字符串
// 测试链接 : https://leetcode.cn/problems/replace-the-substring-for-balanced-string/
// 核心思路：滑动窗口 + 欠债模型。目标是覆盖所有“多出来”的字符。

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <climits>

using namespace std;

int balancedString(string str) {
    int n = str.length();
    int target = n / 4;

    // 1. 统计全词频
    int cnts[4] = {0}; // 0:Q, 1:W, 2:E, 3:R
    auto getIdx = [](char c) {
        if (c == 'Q') return 0;
        if (c == 'W') return 1;
        if (c == 'E') return 2;
        return 3;
    };

    vector<int> s(n);
    for (int i = 0; i < n; i++) {
        s[i] = getIdx(str[i]);
        cnts[s[i]]++;
    }

    // 2. 确定“债务”：即那些超过 target 的部分
    int debt = 0;
    int require[4] = {0};
    bool alreadyBalanced = true;
    for (int i = 0; i < 4; i++) {
        if (cnts[i] > target) {
            require[i] = target - cnts[i]; // 用负数表示欠债量
            debt -= require[i]; 
            alreadyBalanced = false;
        } else {
            require[i] = 0; // 不超过 target 的字符不需要被替换掉
        }
    }

    if (alreadyBalanced) return 0;

    // 3. 滑动窗口寻找包含所有“多余字符”的最短子串
    int ans = INT_MAX;
    for (int l = 0, r = 0; r < n; r++) {
        // 进窗：尝试还债
        if (require[s[r]]++ < 0) {
            debt--;
        }

        // 债还完了（窗口包含了所有多余字符）
        if (debt == 0) {
            // 缩窗：扔掉多余的盈余字符
            while (require[s[l]] > 0) {
                require[s[l++]]--;
            }
            ans = min(ans, r - l + 1);
        }
    }

    return ans;
}

int main() {
    // 示例："QQQW" -> n=4, target=1. Q多出2个。最短覆盖Q,Q的子串是"QQ"，长度2
    cout << balancedString("QQQW") << endl; 
    return 0;
}
```
