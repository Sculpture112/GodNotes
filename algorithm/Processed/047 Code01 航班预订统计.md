# 047 Code01 航班预订统计

```cpp
// 047 Code01 航班预订统计
// 核心思想：一维差分
// 1. 建立一个长度为 n+2 的数组 cnt（为了处理 R+1 越界问题）
// 2. 差分操作：对于 [L, R, V]，执行 cnt[L]+=V, cnt[R+1]-=V
// 3. 前缀和还原：cnt[i] = cnt[i] + cnt[i-1]
// 4. 复杂度：时间 O(N+M)，空间 O(N)

#include <iostream>
#include <vector>

using namespace std;

vector<int> corpFlightBookings(vector<vector<int>>& bookings, int n) {
    // 为什么要 n + 2？
    // 因为航班编号是 1~n，R+1 最大可能是 n+1
    // 为了防止下标越界且逻辑清晰，开到 n+2
    vector<int> cnt(n + 2, 0);

    // 1. 设置差分数组
    for (const auto& book : bookings) {
        int first = book[0];
        int last = book[1];
        int seats = book[2];

        cnt[first] += seats;
        cnt[last + 1] -= seats;
    }

    // 2. 加工前缀和
    // 此时 cnt[i] 存的是与前一个位置的差值
    // 累加后，cnt[i] 就变成了第 i 个航班的总座位数
    for (int i = 1; i <= n; i++) {
        cnt[i] += cnt[i - 1];
    }

    // 3. 结果平移到 0 下标起始的 vector
    vector<int> ans(n);
    for (int i = 0; i < n; i++) {
        ans[i] = cnt[i + 1];
    }

    return ans;
}

// 简单测试
int main() {
    vector<vector<int>> bookings = {{1, 2, 10}, {2, 3, 20}, {2, 5, 25}};
    int n = 5;
    vector<int> result = corpFlightBookings(bookings, n);

    printf("各航班预订座位总数: ");
    for (int x : result) printf("%d ", x);
    printf("\n"); // 预期输出: 10 55 45 25 25

    return 0;
}
```
