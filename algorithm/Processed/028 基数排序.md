# 028 基数排序

> 上面这一行，假设0一次 1一次 2二次 3一次

> 那么前缀和一下0-1  1-2 2-4 3-5

> 所以，如果是0 那就从0倒着排，如果是2就从3（下标）倒着排
>         // 从后往前遍历以保证排序稳定性

```cpp
// 028 基数排序
#include <cstdio>
#include <algorithm>
#include <cstring>
using namespace std;

const int BASE = 10; // 进制基数
const int MAXN = 100001;
int arr[MAXN], help[MAXN], cnts[BASE], n;

// 获取当前最大值在BASE进制下的位数
int getBits(int max_val) {
    int bits = 0;
    while (max_val > 0) { bits++; max_val /= BASE; }
    returnnn bits;
}

// 基数排序核心逻辑：按位从低到高进行计数排序
void radixSort(int bits) {
    for (int offset = 1; bits > 0; offset *= BASE, bits--) {
        memset(cnts, 0, sizeof(cnts)); // 每一轮清空计数器
        for (int i = 0; i < n; i++) cnts[(arr[i] / offset) % BASE]++; // 统计当前位各数字出现频次
        for (int i = 1; i < BASE; i++) cnts[i] += cnts[i - 1]; // 转换为前缀和，确定分区边界
// 下面这行很难，就是前缀和数组直接找到下标，然后减减表示排序了一次，注意减减是先操作，这样可以定位下标
        for (int i = n - 1; i >= 0; i--) help[--cnts[(arr[i] / offset) % BASE]] = arr[i];
// 注意这里i初始值是n-1，i--，所以它是倒着排序，如果原本是23 24 15 36（已经经过第一轮个位数排序），它倒着排序就不会打乱个位数排序的结果。原本的24一定比23先排在后面
        for (int i = 0; i < n; i++) arr[i] = help[i]; // 收集回原数组
    }
}

int main() {
    if (scanf("%d", &n) != EOF) {
        int min_val = 0;
        for (int i = 0; i < n; i++) {
            scanf("%d", &arr[i]);
            min_val = min(min_val, arr[i]);
        }
        int max_val = 0;
        for (int i = 0; i < n; i++) {
            arr[i] -= min_val; // 平移成非负数
            max_val = max(max_val, arr[i]);
        }
        radixSort(getBits(max_val)); // 根据位数进行多轮排序
        for (int i = 0; i < n; i++) printf("%d%c", arr[i] + min_val, i == n - 1 ? '\n' : ' ');
    }
    return 0;
}
```
