# 用栈实现队列（双栈“倒水”法）

```cpp
// 用栈实现队列（双栈“倒水”法）
class MyQueue {
    stack<int> in, out;

    // 核心逻辑：当 out 栈为空时，将 in 栈所有数据一次性倒入 out 栈
    void inToOut() {
        if (out.empty()) {
            while (!in.empty()) {
                out.push(in.top());
                in.pop();
            }
        }
    }

public:
    void push(int x) {
        in.push(x);
        inToOut(); // 每次 push 尝试倒数据（可选，也可只在 pop/peek 时调用）
    }

    int pop() {
        inToOut(); // 确保 out 有数据
        int val = out.top();
        out.pop();
        return val;
    }

    int peek() {
        inToOut(); // 确保 out 有数据
        return out.top();
    }

    bool empty() { return in.empty() && out.empty(); }
};
```
