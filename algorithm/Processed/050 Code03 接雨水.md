# 050 Code03 接雨水

```cpp
// 050 Code03 接雨水
// 测试链接 : https://leetcode.cn/problems/trapping-rain-water/
// 核心思路：双指针贪心（木桶原理）
// 时间复杂度 O(N)，额外空间复杂度 O(1)

#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

class Solution {
public:
    // 最优解：双指针
    int trap(vector<int>& height) {
        int n = height.size();
        if (n <= 2) return 0;

        // l, r 是当前待处理的左右指针
        // lmax, rmax 分别是 [0, l-1] 和 [r+1, n-1] 范围内的最大值
        int l = 1, r = n - 2;
        int lmax = height[0], rmax = height[n - 1];
        int ans = 0;

        while (l <= r) {
            if (lmax <= rmax) {
                // 左侧最大值较小，可以结算左侧
                // 因为位置 l 的水位受限于 min(lmax, 右侧某个>=rmax的数)
                // 既然 lmax <= rmax，那么水位上限就是 lmax
                ans += max(0, lmax - height[l]);
                lmax = max(lmax, height[l++]);
            } else {
                // 右侧最大值较小，结算右侧
                ans += max(0, rmax - height[r]);
                rmax = max(rmax, height[r--]);
            }
        }
        return ans;
    }
};

int main() {
    Solution sol;
    vector<int> h = {0, 1, 0, 2, 1, 0, 1, 3, 2, 1, 2, 1};
    // 预期结果：6
    printf("接到的雨水总量: %d\n", sol.trap(h));
    return 0;
}
```
