# 036 二叉树按层序列化和反序列化

```cpp
// 036 二叉树按层序列化和反序列化
// 1. 二叉树节点结构定义 (TreeNode Struct)
// 2. 按层序列化实现 (serialize)
// 3. 按层反序列化实现 (deserialize)

#include <string>
#include <vector>
#include <sstream>

using namespace std;

// 不提交这个类
struct TreeNode {
    int val;
    TreeNode *left;
    TreeNode *right;
    TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
};

class Codec {
public:
    // 如果测试数据量变大了就修改这个值
    static const int MAXN = 10001;
    TreeNode* q[MAXN];
    int l, r;

    // 按层序列化
    string serialize(TreeNode* root) {
        string res = "";
        if (root != nullptr) {
            res += to_string(root->val) + ",";

c++的stirng可以直接添加字符
            l = 0;
            r = 0;
            q[r++] = root;
            while (l < r) {
                root = q[l++];
                if (root->left != nullptr) {
                    res += to_string(root->left->val) + ",";
                    q[r++] = root->left;

// 把左右节点塞入节点数组
                } else {
                    res += "#,";
                }
                if (root->right != nullptr) {
                    res += to_string(root->right->val) + ",";
                    q[r++] = root->right;
                } else {
                    res += "#,";
                }
            }
        }
        return res;
    }

    // 按层反序列化
    TreeNode* deserialize(string data) {
        if (data == "") {
            return nullptr;
        }

        // 模拟 Java 的 split(",")
        vector<string> nodes;
        stringstream ss(data);
        string item;
        while (getline(ss, item, ',')) {

// 切割字符串！！！
            nodes.push_back(item);
        }

        int index = 0;
        TreeNode* root = generate(nodes[index++]);
        l = 0;
        r = 0;
        if (root != nullptr) {
            q[r++] = root;
        }

        while (l < r) {
            TreeNode* cur = q[l++];
            cur->left = generate(nodes[index++]);

// 写成函数判断空节点
            cur->right = generate(nodes[index++]);
            if (cur->left != nullptr) {
                q[r++] = cur->left;
            }
            if (cur->right != nullptr) {
                q[r++] = cur->right;
            }
        }
        return root;
    }

private:
    // 辅助方法：根据字符串生成节点
    TreeNode* generate(string val) {
        if (val == "#") {
            return nullptr;
        }
        return new TreeNode(stoi(val));
    }
};
```
