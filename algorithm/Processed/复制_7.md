```cpp
给定整数序列 a1,a2,…,aNa1​,a2​,…,aN​ 和常数 CC，统计序列中有多少有序数对 (i,j)(i,j) 满足 1≦i,j≦N1≦i,j≦N 且 ai−aj=Cai​−aj​=C。输入描述：第一行输入两个整数 N,C (1≦N≦2×105, 0≦C<230)N,C (1≦N≦2×105, 0≦C<230)；第二行输入 NN 个整数 a1,a2,…,aN (0≦ai<230)a1​,a2​,…,aN​ (0≦ai​<230)。输出描述：输出一个整数，表示满足条件的有序数对数量。示例1输入：5 2
1 3 3 4 5
// 复制

// 输出：
4
// 复制

// 说明：
在此样例中，满足条件的有序数对有 (3,1),(4,1),(4,2),(5,3)(3,1),(4,1),(4,2),(5,3)，共 44 对。

#include <iostream>
#include <vector>
#include <map>

using namespace std;

int main() {
    // 优化输入输出效率
    ios::sync_with_stdio(false);
    cin.tie(0);

    int N;
    long long C;
    if (!(cin >> N >> C)) return 0;

    vector<long long> a(N);
    map<long long, long long> counts;

    // 1. 读取数据并统计每个数字出现的频率
    for (int i = 0; i < N; ++i) {
        cin >> a[i];
        counts[a[i]]++;
    }

    long long total_pairs = 0;

    // 2. 遍历 map，寻找满足条件的 val + C
    // 遍历每一个出现过的数字作为 aj
    for (auto const& [val, count] : counts) {
        long long target = val + C;
        if (counts.count(target)) {
            // 如果存在 ai = aj + C
            // 数对数量 = (aj出现的次数) * (ai出现的次数)
            total_pairs += count * counts[target];
        }
    }

    // 3. 输出最终结果
    cout << total_pairs << endl;

    return 0;
}
```
