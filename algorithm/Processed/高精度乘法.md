# 高精度乘法

```cpp
// 高精度乘法



#include <bits/stdc++.h>
using namespace std;

int main(){
    string s1,s2;
    cin>>s1>>s2;

    if(s1 == "0" || s2=="0"){
        cout<<0<<endl;
        return 0;
    }

    int n = s1.size();
    int m = s2.size();

    vector<int> C(n+m,0);

    for(int i = 0;i<n;i++){
        int x = s1[n-1-i] -'0';
        int carry = 0;

        for(int j = 0;j<m;j++){
            int y = s2[m-1-j] - '0';

            int cur = C[i+j] + x* y+carry;
            C[i+j] = cur%10;
            carry = cur/=10;
        }

        if(carry) {
            C[i+m] += carry;
        }
    }

    int k = n+m-1;
    while(k>0 && C[k] == 0) k--;

    for(;k>=0;k--){
        cout<<C[k];
    }
    cout<<endl;
    return 0;
}
```
