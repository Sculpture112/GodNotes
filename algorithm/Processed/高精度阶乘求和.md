# 高精度阶乘求和

```cpp
// 高精度阶乘求和

#include <iostream>
#include <vector>
using namespace std;

int main() {
    int n;
    cin >> n;

    // 用数组存储大数字，每位存一个数字
    // 比如数字123，我们存为[3,2,1]
    vector<int> sum = {0}; // 存储总和，初始为0
    vector<int> fact = {1}; // 存储当前的阶乘，初始为1! = 1

    // 从1循环到n
    for (int i = 1; i <= n; i++) {
        // 计算 i! = (i-1)! * i
        if (i > 1) {
            int carry = 0; // 进位
            for (int j = 0; j < fact.size(); j++) {
                int product = fact[j] * i + carry;
                fact[j] = product % 10; // 当前位
                carry = product / 10; // 进位
            }
            // 处理最后的进位
            while (carry > 0) {
                fact.push_back(carry % 10);
                carry /= 10;
            }
        }

        // 把 i! 加到总和中
        // 思路：对齐两个数字的最低位，逐位相加
        int max_len = max(sum.size(), fact.size());
        int carry = 0; // 加法进位

        // 扩展sum的长度，确保能存放结果
        sum.resize(max_len, 0);

        // 逐位相加
        for (int j = 0; j < max_len; j++) {
            int a = (j < sum.size()) ? sum[j] : 0;
            int b = (j < fact.size()) ? fact[j] : 0;
            int total = a + b + carry;
            sum[j] = total % 10; // 当前位
            carry = total / 10; // 进位
        }

        // 处理最后的进位
        while (carry > 0) {
            sum.push_back(carry % 10);
            carry /= 10;
        }
    }

    // 输出结果（注意：我们是倒着存的，要倒序输出）
    for (int i = sum.size() - 1; i >= 0; i--) {
        cout << sum[i];
    }

    return 0;
}
```
